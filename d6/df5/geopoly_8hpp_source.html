53inline void polyheap::add(Nterm *p)
54{
55 int len = F->n_terms(p);
56 int i = 0;
57 while (len >= heap_size[i]) i++;
58
59 ring_elem tmp1 = heap[i];
60 ring_elem tmp2 = p;
61 F->add_to(tmp1, tmp2);
62 heap[i] = tmp1;
63
64 len = F->n_terms(heap[i]);
65 p = NULL;
66 while (len >= heap_size[i])
67 {
68 i++;
69
70 tmp1 = heap[i];
71 tmp2 = heap[i - 1];
72 F->add_to(tmp1, tmp2);
73 heap[i] = tmp1;
74
75 len = F->n_terms(heap[i]);
76 heap[i - 1] = NULL;
77 }
78 if (i > top_of_heap) top_of_heap = i;
79}

81inline Nterm *polyheap::remove_lead_term()
82{
83 int lead_so_far = -1;
84 for (int i = 0; i <= top_of_heap; i++)
85 {
86 if (heap[i] == NULL) continue;
87 if (lead_so_far < 0)
88 {
89 lead_so_far = i;
90 continue;
91 }
92 int cmp = EQ; // F->compare(heap[lead_so_far], heap[i]);
93 if (cmp == GT) continue;
94 if (cmp == LT)
95 {
96 lead_so_far = i;
97 continue;
98 }
99 // At this point we have equality
100 K->add_to(heap[lead_so_far]->coeff, heap[i]->coeff);
101 Nterm *tmp = heap[i];
102 heap[i] = tmp->next;
103 tmp->next = NULL;
104 F->remove(reinterpret_cast<ring_elem &>(tmp));
105
106 if (K->is_zero(heap[lead_so_far]->coeff))
107 {
108 // Remove, and start over
109 tmp = heap[lead_so_far];
110 heap[lead_so_far] = tmp->next;
111 tmp->next = NULL;
112 F->remove(reinterpret_cast<ring_elem &>(tmp));
113 lead_so_far = -1;
114 i = -1;
115 }
116 }
117 if (lead_so_far < 0) return NULL;
118 Nterm *result = heap[lead_so_far];
119 heap[lead_so_far] = result->next;
120 result->next = NULL;
121 return result;
122}

125{

134 heap[i] = NULL;
135 }
136 top_of_heap = -1;
