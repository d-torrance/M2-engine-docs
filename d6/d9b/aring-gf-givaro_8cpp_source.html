119 M2_arrayint ARingGFGivaro::findMinimalPolynomial(UTT charac, UTT dim)
120 {
121  // ARingGFGivaro tmp(charac,dim);
122  // return tmp.getModPolynomialCoeffs();
123 
124  FieldType Zp(charac, 1);
125  // typedef CyclotomicTable< GFqDom<TT>, Dense > PolDom;
126  // PolDom Pdom( Zp, e );
127  typedef Givaro::Poly1FactorDom<FieldType::Self_t, Givaro::Dense> PolDom;
128  PolDom Pdom(Zp);
129  PolDom::Element F, G, H;
130 
131 // F is irreducible of degree e over Zp
132 // G is a primitive polynomial for F
133 // Pdom.random_prim_root(F,G, Degree(e));
134 
135 // F is an irreducible factor of the
136 // (p^e-1) th cyclotomic polynomial
137 // G is a primitive polynomial for F : X
138 // Pdom.getcyclo(F);
139 // Pdom.init(G, Degree(1), Zp.one);
140 
141 // F is irreducible of degree e over Zp
142 // with X as a primitive polynomial
143 #ifndef GIVARO_RANDOM_IRREDUCTIBLE_PRIMITIVE_ROOT
144  Pdom.ixe_irreducible(F, Givaro::Degree((long)dim));
145  // Pdom.init(G, Degree(1), Zp.one);
146  // Pdom.assign(G, Degree(1), Zp.one);
147  Pdom.init(G, Givaro::Degree(1));
148 #else
149  Pdom.random_irreducible(F, Givaro::Degree((long)dim));
150  Pdom.give_random_prim_root(G, F);
151 #endif
152 
153  Pdom.assign(H, G);
154 
155  typedef Givaro::Poly1PadicDom<FieldType, Givaro::Dense> PadicDom;
156  PadicDom PAD(Pdom);
157 
158  UTT generator, irreducible;
159 
160  PAD.eval(generator, H);
161  PAD.eval(irreducible, F);
162 
163  return (representationToM2Array(irreducible, dim + 1, charac));
164 }

462 void ARingGFGivaro::set_from_long(ElementType &result, int64_t a) const
463 {
464  // givaroField.init(result, a): returns an element in GF, where
465  // 0 <= a < p^n. It injects via lexicographic order of p-adic rep:
466  // 0 --> [0,0,0]
467  // 1 --> [1,0,0]
468  // p --> [0,1,0], etc.
469  ElementType p = static_cast<ElementType>(mCharac);
470  ElementType a1 = a;
471  if (a1 < 0 or a1 >= p)
472  {
473  a1 = a1 % p;
474  if (a1 < 0) a1 = a1 + p;
475  }
476  // strange: if mCharac isn't cast away from unsigned,
477  // then in "a % mCharac" a is coerced to unsigned, and get the wrong answer!
478  // e.g:
479  // (-5) % (unsigned long)(5) == 1
480  // (-5) % (long)(5) == 0. Wow!
481  givaroField.init(result, a1);
482 }

553 void ARingGFGivaro::power(ElementType &result,
554  const ElementType a,
555  const STT n) const
556 {
557  if (givaroField.isnzero(a))
558  {
559  mpz_t mpz_a;
560  mpz_t mpz_n;
561  mpz_t mpz_tmp;
562  mpz_init(mpz_a);
563  mpz_init(mpz_n);
564  mpz_init(mpz_tmp);
565  mpz_set_si(mpz_n, n);
566  mpz_set_ui(mpz_a, a);
567  // std::cerr << "a = " << a << std::endl;
568  // std::cerr << "mpz_a = " << mpz_a << std::endl;
569  // std::cerr << "n = " << n << std::endl;
570  // std::cerr << "mpz_n = " << mpz_n << std::endl;
571  mpz_fdiv_r_ui(mpz_tmp, mpz_n, givaroField.cardinality() - 1);
572  mpz_mul(mpz_n, mpz_a, mpz_tmp);
573  STT tmp =
574  static_cast<STT>(mpz_fdiv_ui(mpz_n, givaroField.cardinality() - 1));
575  if (tmp == 0)
576  {
577  tmp += givaroField.cardinality() - 1;
578  // result=givaroField.one;
579  }
580 
581  // std::cerr << "tmp = " << tmp << std::endl;

584  result = tmp;
585  mpz_clear(mpz_a);
586  mpz_clear(mpz_n);
587  mpz_clear(mpz_tmp);
588  }
589  else

648 bool ARingGFGivaro::promote(const Ring *Rf,
649  const ring_elem f,
650  ElementType &result) const
651 {
652  if (mOriginalRing != Rf) return false;
653 
654  result = givaroField.zero;
655  int exp[1];
656  ElementType genRep;
657  givaroField.generator(genRep);
658 #ifdef DEBUG_GF
659  std::cerr << "genRep " << genRep << std::endl;
660 #endif
661  for (Nterm *t = f; t != NULL; t = t->next)
662  {
663  elem a, b;
664 
665  std::pair<bool, long> res =
666  mOriginalRing->getCoefficientRing()->coerceToLongInteger(t->coeff);
667  assert(res.first);
668  set_from_long(a, res.second);
669 
670  mOriginalRing->getMonoid()->to_expvector(t->monom, exp);
671  // exp[0] is the variable we want. Notice that since the ring is a
672  // quotient,
673  // this degree is < n (where Q_ = P^n).
674  power(b, genRep, exp[0]);
675  mult(a, a, b);
676  add(result, result, a);
677  }
678  return true;
679 }
