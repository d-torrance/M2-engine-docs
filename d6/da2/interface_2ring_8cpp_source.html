302const Ring /* or null */ *IM2_Ring_frac(const Ring *R)
303{
304 try
305 {
306 if (R == globalZZ) return globalQQ;
307 const PolyRingFlat *P = R->cast_to_PolyRingFlat();
308 if (P == nullptr)
309 {
310 ERROR("expected polynomial ring");
311 return nullptr;
312 }
313 if (P->getMonoid()->numNonTermOrderVariables() > 0)
314 {
315 ERROR(
316 "cannot currently make fraction field over a polynomial ring "
317 "with a non-global monomial order");
318 return nullptr;
319 }
320 if (P->getMonoid()->numInvertibleVariables() > 0)
321 {
322 ERROR(
323 "cannot currently make fraction field over a polynomial ring "
324 "with Laurent variables, i.e. Inverses=>true set");
325 return nullptr;
326 }
327 if (R->get_precision() > 0)
328 {
329 ERROR("cannot make fraction field over approximate field base");
330 return nullptr;
331 }
332 if (P->getCoefficients()->cast_to_FractionField() != nullptr)
333 {
334 ERROR(
335 "fraction fields over other fraction fields not yet implemented");
336 return nullptr;
337 }
338 if (P->getCoefficients()->cast_to_LocalRing() != nullptr)
339 {
340 ERROR("fraction fields over other local rings not yet implemented");
341 return nullptr;
342 }
343 return FractionField::create(P);
344 } catch (const exc::engine_error& e)
345 {
346 ERROR(e.what());
347 return nullptr;
348 }

381{

385 {
386 ERROR("expected matrix to be over the same ring");

391 return nullptr;
392 }

394 if (P != nullptr)
395 {

397 intern_polyring(result);
398 return result;
399 }
400 const M2FreeAlgebra *A = R->cast_to_M2FreeAlgebra();
401 if (A != nullptr)
402 {
403 auto result = M2FreeAlgebraQuotient::create(*A, I, -1);
404 return result;
405 }
406 ERROR("expected a polynomial ring or free algebra");
407 return nullptr;
408 } catch (const exc::engine_error& e)

616bool findConwayPolynomial(long charac,
617 long deg,
618 bool find_random_if_no_conway_poly_available,
619 std::vector<long> &result_poly)
620{
621 // returns true if result_poly is actually set
622 int ret = 1;
623 fq_nmod_ctx_t ctx;
624 fmpz_t p;
625 fmpz_init(p);
626 fmpz_set_si(p, charac);
627 if (!find_random_if_no_conway_poly_available)
628 ret = _fq_nmod_ctx_init_conway(ctx, p, deg, "a");
629 else
630 fq_nmod_ctx_init(ctx, p, deg, "a");
631
632 if (ret == 0) return false;
633
634 result_poly.resize(deg + 1);
635 for (long i = 0; i <= deg; i++) result_poly[i] = 0;
636 for (long i = 0; i < ctx->len; i++)
637 {
638 if (ctx->j[i] < 0 or ctx->j[i] > deg)
639 printf("error: encountered bad degree\n");
640 // power is ctx->j[i]
641 // coeff is ctx->a[i]
642 result_poly[ctx->j[i]] = ctx->a[i];
643 }
644
645 // printf("flint GF information:\n");
646 // fq_nmod_ctx_print(ctx);
647
648 fq_nmod_ctx_clear(ctx);
649 return true;
650}
