302 const Ring /* or null */ *IM2_Ring_frac(const Ring *R)
303 {
304  try
305  {
306  if (R == globalZZ) return globalQQ;
307  const PolyRingFlat *P = R->cast_to_PolyRingFlat();
308  if (P == 0)
309  {
310  ERROR("expected polynomial ring");
311  return 0;
312  }
313  if (P->getMonoid()->numNonTermOrderVariables() > 0)
314  {
315  ERROR(
316  "cannot currently make fraction field over a polynomial ring "
317  "with a non-global monomial order");
318  return 0;
319  }
320  if (P->getMonoid()->numInvertibleVariables() > 0)
321  {
322  ERROR(
323  "cannot currently make fraction field over a polynomial ring "
324  "with Laurent variables, i.e. Inverses=>true set");
325  return 0;
326  }
327  if (R->get_precision() > 0)
328  {
329  ERROR("cannot make fraction field over approximate field base");
330  return 0;
331  }
332  if (P->getCoefficients()->cast_to_FractionField() != 0)
333  {
334  ERROR(
335  "fraction fields over other fraction fields not yet implemented");
336  return 0;
337  }
338  if (P->getCoefficients()->cast_to_LocalRing() != 0)
339  {
340  ERROR("fraction fields over other local rings not yet implemented");
341  return 0;
342  }
343  return FractionField::create(P);
344  } catch (const exc::engine_error& e)
345  {
346  ERROR(e.what());
347  return NULL;
348  }

631 
632  if (ret == 0) return false;

638  if (ctx->j[i] < 0 or ctx->j[i] > deg)
639  printf("error: encountered bad degree\n");

642  result_poly[ctx->j[i]] = ctx->a[i];
643  }
644
