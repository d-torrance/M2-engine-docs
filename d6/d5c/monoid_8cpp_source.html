143  for (int i = 0; i < monorder_->nblocks && !get_out; i++)
144  {
145  switch (monorder_->blocks[i].typ)
146  {
147  case MO_LEX:
148  case MO_LEX2:

150  case MO_NC_LEX:

154  case MO_LAURENT:
155  case MO_LAURENT_REVLEX:
156  case MO_GREVLEX:
157  case MO_GREVLEX2:
158  case MO_GREVLEX4:
159  case MO_GREVLEX_WTS:

164  case MO_POSITION_UP:

166  case MO_POSITION_DOWN:

169  INTERNAL_ERROR("monomial order block type not handled");

171  }

175

198 void Monoid::set_degrees()
199 {
200  if (degree_monoid_ == NULL)
201  {
202  degree_of_var_.push_back(static_cast<const_monomial>(NULL));
203  return;
204  }
205 
206  // Set 'degree_of_var
207  int degvars = degree_monoid_->n_vars();
208  int *t = degvals_->array;
209 
210  heft_degree_of_var_ = M2_makearrayint(nvars_);
211  if (heftvals_->len != degvars)
212  {
213  ERROR("internal error: heftvals_->len == %d != degvars == %d",
214  heftvals_->len,

216  return;
217  }
218  if (degvars > 0)
219  for (int i = 0; i < nvars_; i++)
220  {
221  monomial m = degree_monoid_->make_one();
222  degree_monoid_->from_expvector(t, m);
223  degree_of_var_.push_back(m);
224  heft_degree_of_var_->array[i] = exponents::weight(degvars, t, heftvals_);
225  t += degvars;
226  }
227  else
228  {
229  for (int i = 0; i < nvars_; i++)
230  {
231  monomial m = degree_monoid_->make_one();
232  degree_monoid_->from_expvector(t, m);
233  degree_of_var_.push_back(m);
234  heft_degree_of_var_->array[i] = 1;
235  }
236  }
237  degree_of_var_.push_back(degree_monoid_->make_one());

269 void Monoid::set_overflow_flags()
270 {
271  overflow = newarray_atomic(enum overflow_type, monomial_size_);
272  enum overflow_type flag;
273  int i = 0, k = 0;
274  for (; i < monorder_->nblocks; i++)
275  {
276  mo_block *b = &monorder_->blocks[i];
277  switch (monorder_->blocks[i].typ)
278  {
279  case MO_REVLEX:
280  case MO_WEIGHTS:
281  case MO_LAURENT:
282  case MO_LAURENT_REVLEX:
283  case MO_NC_LEX:
284  flag = OVER;
285  goto fillin;
286  case MO_POSITION_UP:
287  case MO_POSITION_DOWN:
288  ERROR(
289  "internal error - MO_POSITION_DOWN or MO_POSITION_UP "
290  "encountered");
291  assert(0);
292  break;
293  case MO_LEX:
294  case MO_GREVLEX:
295  case MO_GREVLEX_WTS:
296  flag = OVER1;
297  goto fillin;
298  case MO_LEX2:
299  case MO_GREVLEX2:
300  case MO_GREVLEX2_WTS:
301  flag = OVER2;
302  goto fillin;
303  case MO_LEX4:
304  case MO_GREVLEX4:
305  case MO_GREVLEX4_WTS:
306  flag = OVER4;
307  goto fillin;
308  fillin:
309  assert(b->first_slot == k);
310  for (int p = b->nslots; p > 0; p--)
311  {
312  assert(k < monomial_size_);
313  overflow[k++] = flag;
314  }
315  break;
316  default:
317  ERROR("internal error - missing case");
318  assert(0);
319  break;
320  }
321  }
322  assert(k == monomial_size_);
323 }

407 void Monoid::mult(const_monomial m, const_monomial n, monomial result) const
408 {
409  overflow_type *t = overflow;
410  for (int i = monomial_size_; i != 0; i--) switch (*t++)
411  {
412  case OVER:
413  *result++ = safe::add(*m++, *n++);
414  break;
415  case OVER1:
416  *result++ = safe::pos_add(*m++, *n++);
417  break;
418  case OVER2:
419  *result++ = safe::pos_add_2(*m++, *n++);
420  break;
421  case OVER4:
422  *result++ = safe::pos_add_4(*m++, *n++);
423  break;
424  default:
425  throw(exc::internal_error("missing case"));

456 int Monoid::compare(const_monomial m,
457  int mcomp,
458  const_monomial n,
459  int ncomp) const
460 {
461  int i = n_before_component_;
462  while (1)
463  {
464  if (i == 0) break;
465  if (*m > *n) return GT;
466  if (*m < *n) return LT;
467  m++, n++;
468  --i;
469  }
470  if (component_up_)
471  {
472  if (mcomp < ncomp) return LT;
473  if (mcomp > ncomp) return GT;
474  }
475  else
476  {
477  if (mcomp < ncomp) return GT;
478  if (mcomp > ncomp) return LT;
479  }
480  i = n_after_component_;
481  while (1)
482  {
483  if (i == 0) break;
484  if (*m > *n) return GT;
485  if (*m < *n) return LT;
486  m++, n++;
487  --i;
488  }
489  return EQ;
490 }

670 {
