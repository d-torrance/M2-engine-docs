182 void Monoid::set_degrees()
183 {
184  if (mDegreeMonoid == nullptr)
185  {
186  mDegreeOfVar.push_back(static_cast<const_monomial>(nullptr));
187  return;
188  }
189 
190  auto degrk = mDegreeMonoid->n_vars();
191  auto *iter = &mDegrees.front();
192 
193  if (mHeftVector.size() != degrk)
194  {
195  ERROR("internal error: mHeftVector.size() == %d != degrk == %d",
196  mHeftVector.size(),
197  degrk);
198  return;
199  }
200  if (degrk > 0)
201  for (int i = 0; i < mVariableCount; i++)
202  {
203  auto d = exponents::weight(degrk, iter, mHeftVector);
204  auto m = mDegreeMonoid->make_one();
205  mDegreeMonoid->from_expvector(iter, m);
206  mDegreeOfVar.push_back(m);
207  mHeftDegrees.push_back(d);
208  iter += degrk;
209  }
210  else
211  {
212  // TODO: what is iter in this case?!
213  auto m = mDegreeMonoid->make_one();
214  mDegreeMonoid->from_expvector(iter, m);
215  mDegreeOfVar.assign(mVariableCount, m);
216  mHeftDegrees.assign(mVariableCount, 1);
217  }
218  // an extra entry for the degree of the zero element in the degree monoid
219  mDegreeOfVar.push_back(mDegreeMonoid->make_one());
220 }

244 void Monoid::set_overflow_flags()
245 {
246  overflow = newarray_atomic(enum overflow_type, monomial_size_);
247  enum overflow_type flag;
248  int i = 0, k = 0;
249  for (; i < monorder_->nblocks; i++)
250  {
251  mo_block *b = &monorder_->blocks[i];
252  switch (monorder_->blocks[i].typ)
253  {
254  case MO_REVLEX:
255  case MO_WEIGHTS:
256  case MO_LAURENT:
257  case MO_LAURENT_REVLEX:
258  case MO_NC_LEX:
259  flag = OVER;
260  goto fillin;
261  case MO_POSITION_UP:
262  case MO_POSITION_DOWN:
263  ERROR(
264  "internal error - MO_POSITION_DOWN or MO_POSITION_UP "
265  "encountered");
266  assert(0);
267  break;
268  case MO_LEX:
269  case MO_GREVLEX:
270  case MO_GREVLEX_WTS:
271  flag = OVER1;
272  goto fillin;
273  case MO_LEX2:
274  case MO_GREVLEX2:
275  case MO_GREVLEX2_WTS:
276  flag = OVER2;
277  goto fillin;
278  case MO_LEX4:
279  case MO_GREVLEX4:
280  case MO_GREVLEX4_WTS:
281  flag = OVER4;
282  goto fillin;
283  fillin:
284  assert(b->first_slot == k);
285  for (int p = b->nslots; p > 0; p--)
286  {
287  assert(k < monomial_size_);
288  overflow[k++] = flag;
289  }
290  break;
291  default:
292  ERROR("internal error - missing case");
293  assert(0);
294  break;
295  }
296  }
297  assert(k == monomial_size_);
298 }

414 int Monoid::compare(const_monomial m,
415  int mcomp,
416  const_monomial n,
417  int ncomp) const
418 {
419  int i = n_before_component_;
420  while (1)
421  {
422  if (i == 0) break;
423  if (*m > *n) return GT;
424  if (*m < *n) return LT;
425  m++, n++;
426  --i;
427  }
428  if (component_up_)
429  {
430  if (mcomp < ncomp) return LT;
431  if (mcomp > ncomp) return GT;
432  }
433  else
434  {
435  if (mcomp < ncomp) return GT;
436  if (mcomp > ncomp) return LT;
437  }
438  i = n_after_component_;
439  while (1)
440  {
441  if (i == 0) break;
442  if (*m > *n) return GT;
443  if (*m < *n) return LT;
444  m++, n++;
445  --i;
446  }
447  return EQ;
448 }
