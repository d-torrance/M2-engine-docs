369 void KBasis::compute()
370 // Only the lead monomials of the two matrices 'this' and 'bottom' are
371 // considered. Thus, you must perform the required GB's elsewhere.
372 // Find a basis for (image this)/(image bottom) in degree d.
373 // If 'd' is NULL, first check that (image this)/(image bottom) has
374 // finite dimension, and if so, return a basis.
375 // If 'd' is not NULL, it is an element of the degree monoid.
376 {
377  if (limit == 0) return;
378  std::vector<int> zero_vars;
379  if (weight_has_zeros)
380  {
381  for (int i = 0; i < mVariables.size(); i++)
382  {
383  if (var_wts[i] == 0) { zero_vars.push_back(i); }
384  }
385  }
386 
387  for (int i = 0; i < bottom_matrix->n_rows(); i++)
388  {
389  if (system_interrupted()) return;
390  kb_comp = i;
391 
392  // Make the monomial ideal: this should contain only
393  // monomials involving 'mVariables'.
394  kb_monideal = bottom_matrix->make_monideal(i, true);
395  // the true means: over ZZ, don't consider monomials with non-unit lead
396  // coeffs
397 
398  if (kb_monideal->is_one()) continue;
399  if (hi_degree == nullptr)
400  {
401  // check here that kb_monideal is 0-dimensional
402  // (at least for the variables being used):
403  if (!all_have_pure_powers(kb_monideal, mVariables))
404  {
405  kb_error = true;
406  ERROR("module given is not finite over the base");
407  return;
408  }
409  }
410  else if (zero_vars.size() > 0)
411  {
412  // if we have any variables with zero degrees, then kb_monideal needs
413  // to be 0-dimensional in those variables.
414  if (!all_have_pure_powers(kb_monideal, zero_vars))
415  {
416  kb_error = true;
417  ERROR(
418  "module given is not finite over the zero-degree variables");
419  return;
420  }
421  }
422 
423  const int *component_degree = bottom_matrix->rows()->degree(i);
424  D->to_expvector(component_degree, kb_exp_multidegree);
425  kb_exp_weight =
426  exponents::weight(mHeftVector.size(), kb_exp_multidegree, mHeftVector);
427 
428  // Do the recursion
429  switch (computation_type)
430  {
431  case KB_FULL:
432  basis0_full();
433  break;
434  case KB_SINGLE:
435  basis0_singly_graded();
436  break;
437  case KB_MULTI:
438  basis0_multi_graded();
439  break;
440  }
441  }
442 }

444 Matrix /* or null */ *KBasis::k_basis(const Matrix *bottom,
445  M2_arrayint lo_degree,
446  M2_arrayint hi_degree,
447  std::vector<int> heftvec,
448  std::vector<int> varlist,
449  bool do_truncation,
450  int limit)
451 {
452  // There are essentially 3 situations:
453  // (a) basis(M) -- lo_degree and hi_degree are not given
454  // in this case, only need that for each variable in 'varlist',
455  // some power is an initial term of 'bottom' (for each row of 'bottom').
456  // heft is not used here, or considered.
457  // (b) basis(lo,hi,M) -- case when the ring is singly-graded
458  // one of lo and hi must be given. (otherwise we are in case (a) above)
459  // In this case, heft is a list with one element in it.
460  // Assume: heft * deg(x) > 0, for all x in 'varlist'.
461  // In this situation: we use kb_target_lo_heft, kb_target_hi_heft
462  // (c) basis(d, d, M) -- ring is multi-graded
463  // ASSUME: deg_d(x) . heft > 0 for all vars 'x' in 'varlist'
464  // where deg_d(x) consists of the first #d components of deg(x)
465  // ASSUME: 1 <= #d <= degreeRank of the ring
466  // use kb_target_multidegree, kb_target_lo_heft
467  // and kb_exp_multidegree (of length #d).
468  // if do_truncation, then any generator with heft > kb_target_lo_heft is
469  // placed in the resulting matrix
470  //
471  // Further assumptions:
472  // 1 <= #heft <= degreeRank P
473  // #heft = #lo_degree = #hi_degree, if these are not 0.
474  //
475  //
476  // Do some checks first, return 0 if not good.
477  const PolynomialRing *P = bottom->get_ring()->cast_to_PolynomialRing();
478  if (P == nullptr) return Matrix::identity(bottom->rows());
479 
480  const PolynomialRing *D = P->get_degree_ring();
481  const int *lo = lo_degree->len > 0 ? lo_degree->array : nullptr;
482  const int *hi = hi_degree->len > 0 ? hi_degree->array : nullptr;
483 
484  if (heftvec.size() > D->n_vars())
485  {
486  ERROR("expected heft vector of length <= %d", D->n_vars());
487  return nullptr;
488  }
489 
490  if (lo && heftvec.size() != lo_degree->len)
491  {
492  ERROR("expected degrees of length %d", heftvec.size());
493  return nullptr;
494  }
495 
496  if (hi && heftvec.size() != hi_degree->len)
497  {
498  ERROR("expected degrees of length %d", heftvec.size());
499  return nullptr;
500  }
501 
502  // If heftvec.size() is > 1, and both lo and hi are non-null,
503  // they need to be the same
504  if (heftvec.size() > 1 && lo && hi)
505  for (int i = 0; i < heftvec.size(); i++)
506  if (lo_degree->array[i] != hi_degree->array[i])
507  {
508  ERROR("expected degree bounds to be equal");
509  return nullptr;
510  }
511 
512  KBasis KB(bottom, lo, hi, heftvec, varlist, do_truncation, limit);
513 
514  // If either a low degree, or high degree is given, then we require a
515  // non-negative heft vector:
516  if (lo || hi)
517  for (int i = 0; i < varlist.size(); i++)
518  if (KB.var_wts[i] < 0)
519  {
520  ERROR(
521  "basis: computation requires a heft form non-negative on the "
522  "degrees of the variables");
523  return nullptr;
524  }
525  // This next line will happen if both lo,hi degrees are given, and they are
526  // different. This can only be the singly generated case, and in that case
527  // the degrees are in the wrong order, so return with 0 basis.
528  if (lo != nullptr && hi != nullptr && lo[0] > hi[0]) return KB.value();
529 
530  KB.compute();
531  if (system_interrupted()) return nullptr;
532  return KB.value();
533 }
