204GBComputation *PolyRingQuotient::make_gb(const ring_elem g) const
205// return the GB of g
206// keeping the change matrix:
207// that is, how every element of the GB is written in terms of g.
208// e.g. if the gb is generated by 1, the change matrix has an entry c such that 1 = c*g
209{
210 MatrixConstructor mat(make_FreeModule(1), 1);
211 mat.set_entry(0, 0, g);
212 Matrix *mg = mat.to_matrix(); // {g}
213
214 M2_arrayint weights = M2_makearrayint(n_vars());
215 for (int i = 0; i < n_vars(); i++) weights->array[i] = 1;
216 GBComputation *G = GBComputation::choose_gb(mg,
217 false, // collect syz
218 -1,
219 weights,
220 false,
221 -1,
222 0,
223 0
224 /* , max_reduction_count */
225 );
226 G->set_stop_conditions(false,
227 nullptr,
228 -1,
229 -1, // syzygy limit
230 -1,
231 -1,
232 -1,
233 false,
234 nullptr);
235
236 G->start_computation();
237 return G;

241 const ring_elem g) const
242{

250 matf.set_entry(0, 0, f);

252

254

257
258 delete mrem;

381 const ring_elem b,
382 ring_elem &x,
383 ring_elem &y) const
384{
385 MatrixConstructor mat(make_FreeModule(1), 2);
386 mat.set_entry(0, 0, a);
387 mat.set_entry(0, 1, b);
388 Matrix *m = mat.to_matrix(); // {a,b}
389 M2_arrayint weights = M2_makearrayint(n_vars());
390 for (int i = 0; i < n_vars(); i++) weights->array[i] = 1;
391 GBComputation *G = GBComputation::choose_gb(m,
392 true, // collect syz
393 -1, // keep all rows
394 weights,
395 false,
396 -1,
397 0,
398 0
399 /* , max_reduction_count */
400 );
401 G->set_stop_conditions(false,
402 nullptr,
403 -1,
404 1, // syzygy limit
405 -1,
406 -1,
407 -1,
408 false,
409 nullptr);
410 G->start_computation();
411 const Matrix *s = G->get_syzygies();
412
413 // Now extract the two pieces of info
414 x = s->elem(0, 0);
415 y = s->elem(1, 0);
416 ring_elem c = preferred_associate(x);
417 ring_elem x1 = mult(c, x);
418 ring_elem y1 = mult(c, y);
419 x = x1;
420 y = y1;
