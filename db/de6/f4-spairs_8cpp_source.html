106 {

116  for (p = heap->next; p != 0; p = p->next)
117  if (p->deg > nextdeg)
118  continue;
119  else if (p->deg < nextdeg)
120  {
121  len = 1;
122  nextdeg = p->deg;
123  }
124  else
125  len++;
126  result_number = len;

130 int F4SPairSet::prepare_next_degree(int max, int &result_number)
131 // Returns the (sugar) degree being done next, and collects all (or at
132 // most 'max', if max>0) spairs in this lowest degree.
133 // Returns the degree, sets result_number.
134 {
135  this_set = 0;
136  int result_degree = determine_next_degree(result_number);
137  if (result_number == 0) return 0;
138  if (max > 0 && max < result_number) result_number = max;
139  int len = result_number;
140  spair head;
141  spair *p;
142  head.next = heap;
143  p = &head;
144  while (p->next != 0)
145  if (p->next->deg != result_degree)
146  p = p->next;
147  else
148  {
149  spair *tmp = p->next;
150  p->next = tmp->next;
151  tmp->next = this_set;
152  this_set = tmp;
153  len--;
154  if (len == 0) break;
155  }
156  heap = head.next;
157  return result_degree;
158 }

245 int F4SPairSet::construct_pairs(bool remove_disjoints)
246 {
247  if (gb.size() == 0) return 0;
248 
249  VP.reset();
250  PS.reset();
251  gbelem *me = gb[gb.size() - 1];
252  int me_component = static_cast<int>(M->get_component(me->f.monoms));
253 
254  typedef VECTOR(pre_spair *) spairs;
255 
256  VECTOR(VECTOR(pre_spair *)) bins;
257 
258  // Loop through each element of gb, and create the pre_spair
259  for (int i = 0; i < gb.size() - 1; i++)
260  {
261  if (gb[i]->minlevel == ELEM_NON_MIN_GB) continue;
262  if (me_component != M->get_component(gb[i]->f.monoms)) continue;
263  pre_spair *p = create_pre_spair(i);
264  insert_pre_spair(bins, p);
265  }
266 
268  // Now minimalize the set //
270  MonomialLookupTable *montab = new MonomialLookupTable(M->n_vars());
271 
272  PreSPairSorter C;
273  int n_new_pairs = 0;
274  for (int i = 0; i < bins.size(); i++)
275  {
276  if (bins[i].size() == 0) continue;
277  // First sort the monomials of this degree
278 
279  std::stable_sort(bins[i].begin(), bins[i].end(), C);
280 
281  // Loop through each degree and potentially insert...
282  spairs::iterator first = bins[i].begin();
283  spairs::iterator next = first;
284  spairs::iterator end = bins[i].end();
285  for (; first != end; first = next)
286  {
287  next = first + 1;
288  pre_spair *chosen = *first;
289  while (next != end)
290  {
291  pre_spair *p = *next;
292  if (!varpower_monomials::is_equal(chosen->quot, p->quot)) break;
293  next++;
294  }
295  /* At this point: [first,next) is the range of equal monomials */
296 
297  int32_t junk;
298  bool inideal = montab->find_one_divisor_vp(0, chosen->quot, junk);
299  if (!inideal)
300  {
301  // MES: Maybe choose another of the equal monomials...
302  montab->insert_minimal_vp(0, chosen->quot, 0);
303  // The following condition is that gcd is not one
304  if (!remove_disjoints || !chosen->are_disjoint)
305  {
306  insert_spair(chosen, INTSIZE(gb) - 1);
307  n_new_pairs++;
308  }
309  }
310  }
311  }
312  delete montab;
313 
314  return n_new_pairs;
315 }
