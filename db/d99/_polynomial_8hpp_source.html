16 // A monomial is an array of ints, the first of which is the length of that array (including length field).
17 // e.g. [2 0] (is currently the empty monomial, but this class knows nothing about specific monomials.
18 // each monomial is of the form:
19 // <length: n+2> <degree: (currently) n> <var1> <var2> ... <varn>

22 // 2 monomials: xzx, xy (dot is only there for readability)

125inline ModuleMonom monomToModuleMonom(const Monom& a, int comp, std::pair<int*, int*> allocated_result)
126{
127 assert(allocated_result.second-allocated_result.first >= ModuleMonom::sizeOfCorrespondingModuleMonom(a));
128 auto begin = allocated_result.first;
129 begin[0] = ModuleMonom::sizeOfCorrespondingModuleMonom(a);
130 begin[1] = 0; // index
131 begin[2] = 0; // hashval
132 begin[3] = comp;
133 std::copy(a.begin()+1, a.end(), begin+4);
