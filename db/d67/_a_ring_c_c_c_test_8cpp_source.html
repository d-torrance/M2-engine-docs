15bool almostEqual(const M2::ARingCCC& C,
16 int nbits,
17 const M2::ARingCCC::ElementType& a,
18 const M2::ARingCCC::ElementType& b)
19{
20 mpfr_t epsilon;
21 mpfr_init2(epsilon, C.get_precision());
22 mpfr_set_ui_2exp(epsilon, 1, -nbits, MPFR_RNDN); // should there be exp() ???
23
24 M2::ARingCCC::ElementType c;
25 C.init(c);
26 C.subtract(c, a, b);
27
28 bool ret = mpfr_cmpabs(&c.re, epsilon) < 0 && mpfr_cmpabs(&c.im, epsilon) < 0;
29
30 C.clear(c);
31 mpfr_clear(epsilon);
32 return ret;
33}

162TEST(ARingCCC, axioms)
163{
164 M2::ARingCCC C(100);
165 ARingElementGenerator<M2::ARingCCC> gen(C);
166 M2::ARingCCC::ElementType a, b, c, d, e;
167 C.init(a);
168 C.init(b);
169 C.init(c);
170 C.init(d);
171 C.init(e);
172 for (int i = 0; i < ntrials; i++)
173 {
174 gen.nextElement(a);
175 gen.nextElement(b);
176 gen.nextElement(c);
177 // Test commutativity
178 // test: a*b = b*a
179 // test: a+b == b+a
180 C.add(d, a, b);
181 C.add(e, b, a);
182 EXPECT_TRUE(almostEqual(C, 98, d, e));
183 C.mult(d, a, b);
184 C.mult(e, b, a);
185 EXPECT_TRUE(almostEqual(C, 98, d, e));
186
187 // Test associativity
188 // test: a+(b+c) == (a+b)+c
189 // test: a*(b*c) == (a*b)*c
190 C.add(e, b, c);
191 C.add(d, a, e); // a+(b+c)
192 C.add(e, a, b);
193 C.add(e, e, c); // (a+b)+c
194 EXPECT_TRUE(almostEqual(C, 94, d, e));
195 C.mult(e, b, c);
196 C.mult(d, a, e); // a*(b*c)
197 C.mult(e, a, b);
198 C.mult(e, e, c); // (a*b)*c
199 EXPECT_TRUE(almostEqual(
200 C, 93, d, e)); // MES: I'm not sure how equal these should be.
201
202 // Test distributivity
203 // test: a*(b+c) == a*b + a*c
204 C.add(e, b, c);
205 C.mult(d, a, e); // a*(b+c)
206 C.mult(b, a, b);
207 C.mult(c, a, c);
208 C.add(e, b, c); // a*b + a*c
209 EXPECT_TRUE(almostEqual(C, 93, d, e));
210 }
211 C.clear(e);
212 C.clear(d);
213 C.clear(c);
214 C.clear(b);
215 C.clear(a);
216}
