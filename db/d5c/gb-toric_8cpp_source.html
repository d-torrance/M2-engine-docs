18 : R(RR),

190bool binomial_ring::remove_content(binomial &f) const
191{
192 // If the monomials of f have a common monomial factor, remove it from each,
193 // and
194 // return true. Otherwise return false.
195 monomial m = f.lead;
196 monomial n = f.tail;
197 bool result = false;
198 for (int i = 0; i < nvars; i++)
199 {
200 if (m[i] > 0 && n[i] > 0)
201 {
202 if (m[i] > n[i])
203 {
204 m[i] -= n[i];
205 n[i] = 0;
206 }
207 else
208 {
209 n[i] -= m[i];
210 m[i] = 0;
211 }
212 result = true;
213 }
214 }
215 if (result)
216 {
217 set_weights(m); // This is an inefficient way to do this...
218 set_weights(n);
219 }
220 return result;

223int binomial_ring::compare(monomial m, monomial n) const
224{
225 int i;
226 if (have_weights)
227 {
228 i = nvars + 1;
229 if (m[i] < n[i]) return GT;
230 if (m[i] > n[i]) return LT;
231 i--;
232 }
233 else
234 i = nvars;
235 // check degree? For now...
236 if (m[i] < n[i]) return GT; // Remember: m[nvars] is NEGATIVE of degree
237 if (m[i] > n[i]) return LT;
238 if (revlex)
239 for (; i >= 0; i--)
240 {
241 if (m[i] > n[i]) return LT;
242 if (m[i] < n[i]) return GT;
243 }
244 else
245 for (; i >= 0; i--)
246 {
247 if (m[i] > n[i]) return GT;
248 if (m[i] < n[i]) return LT;
249 }
250 return EQ;
251}

382bool binomial_ring::normalize(binomial &f) const
383// Return false if 'f' is zero. Otherwise return true,
384// and possibly swap the terms of f so that f.lead is the lead term.
385{
386 int cmp = compare(f.lead, f.tail);
387 if (cmp == EQ) return false;
388 if (cmp == LT)
389 {
390 monomial a = f.lead;
391 f.lead = f.tail;
392 f.tail = a;
393 }
394 return true;

496{

499 head.next = q->pairs;

501 while (true)
502 {
503 if (r->next == nullptr || ((cmp = R->compare(s.lcm, r->next->lcm)) == GT))

534void binomial_s_pair_set::insert(binomial_s_pair s)
535{
536 int deg = R->degree(s.lcm);
537
538 // Statistics control
539 if (deg > _max_degree)
540 {
541 // Extend _npairs:
542 for (int i = 2 * _max_degree + 2; i < 2 * deg + 2; i++) _npairs.push_back(0);
543 _max_degree = deg;
544 }
545 _npairs[2 * deg]++;
546 _npairs[2 * deg + 1]++;
547
548 s_pair_degree_list *q = _pairs;
549 while (true)
550 {
551 if (q->next == nullptr || q->next->deg > deg)
552 {
553 // Insert new degree node
554 s_pair_degree_list *q1 = new s_pair_degree_list;
555 q1->next = q->next;
556 q1->deg = deg;
557 q1->pairs = nullptr;
558 q->next = q1;
559 break;
560 }
561 if (q->next->deg == deg) break;
562 q = q->next;
563 }
564 q = q->next;
565 insert_pair(q, s);
566 _n_elems++;
567 q->n_elems++;
568}

570bool binomial_s_pair_set::next(const int *d,
571 binomial_s_pair &result,
572 int &result_deg)
573// returns next pair in degrees <= *d, if any.
574// the caller should not free any of the three fields of the
575// s_pair!!
576{
577 if (_pairs->next == nullptr) return false;
578 if (d != nullptr && _pairs->next->deg > *d) return false;
579 s_pair_degree_list *thisdeg = _pairs->next;
580 s_pair_lcm_list *thislcm = thisdeg->pairs;
581 s_pair_elem *s = thislcm->pairs;
582 result_deg = thisdeg->deg;
583
584 thisdeg->n_elems--;
585 _n_elems--;
586 _npairs[2 * (thisdeg->deg) + 1]--;
587
588 result = binomial_s_pair(s->f1, s->f2, thislcm->lcm);
589
590 thislcm->pairs = s->next;
591 if (thislcm->pairs == nullptr)
592 {
593 // Now we must remove this set
594 thisdeg->pairs = thislcm->next;
595 R->remove_monomial(_prev_lcm);
596 _prev_lcm = thislcm->lcm;
597 thislcm->lcm = nullptr;
598 freemem(thislcm);
599
600 if (thisdeg->pairs == nullptr)
601 {
602 // Now we must remove this larger degree list
603 _pairs->next = thisdeg->next;
604 freemem(thisdeg);
605 }
606 }
607
608 freemem(s);
609 return true;
610}

681// remove elements which have lead term divisible by in(f).
682// optionally auto-reduces the other elements as well.
683{

685 gbmin_elem *fm = new gbmin_elem(f, R->mask(m));
686 gbmin_elem head, *p;
687 head.next = first;
688 int deg = R->degree(m);
689 if (deg > _max_degree) _max_degree = deg;
690 for (p = &head; p->next != nullptr; p = p->next)
691 {
692 if (R->graded_compare(m, p->next->elem->f.lead) == LT) break;
693 }
694 fm->next = p->next;
695 p->next = fm;
696 first = head.next;
697
698 p = fm;
699 while (p->next != nullptr)
700 {
701 if (R->degree(p->next->elem->f.lead) > deg &&
702 R->divides(m, p->next->elem->f.lead))
703 {
704 // remove this element
705 gbmin_elem *q = p->next;
706 binomial_gb_elem *qe = q->elem;
707 p->next = q->next;
708 q->next = nullptr;
709 qe->smaller = f;
710 }
711 else
712 {
713 reduce_monomial(p->next->elem->f.tail);
714 p = p->next;

748 deglist[d] = nl;
749 }

751
752 for (int d = 0; d <= _max_degree; d++)

764 freemem(p);
765 }
766 else if ((r = find_divisor(currentresult, p->m)))
767 {
768 gbmin_elem *p1 = new gbmin_elem(p->tag->elem, p->mask);
769 R->remove_monomial(p->m);
770 freemem(p);
771 p1->next = r->tag;
772 r->tag = p1;
773 }
774 else

776 p->next = currentresult;
777 currentresult = p;
778 }
779 }
780 if (result == nullptr)

782 else if (currentresult != nullptr)
783 {
784 monomial_list *q;
785 for (q = result; q->next != nullptr; q = q->next)
786 ;
787 q->next = currentresult;
788 }
789 currentresult = nullptr;
790 }
791 freemem(deglist);
792 return result;

795void binomialGB::make_new_pairs(binomial_s_pair_set *Pairs,
796 binomial_gb_elem *f) const
797{
798 // Compute (a minimal generating set of)
799 // the ideal quotient in(Gmin) : in(f).
800 // Remove any that satisfy certain criteria:
801 // 1. gcd(lead terms) is 1: remove.
802 // 2. if homog_prime, gcd(tail terms) is not 1: remove.
803 // Any that pass these tests, insert into Pairs.
804
805 monomial m = f->f.lead;
806 monomial_list *I = ideal_quotient(m);
807
808 for (monomial_list *q = I; q != nullptr; q = q->next)
809 {
810 gbmin_elem *ge = q->tag; // a list of possibles
811
812 binomial_gb_elem *g = ge->elem;
813
814 // Criterion 1: gcd of lead terms must be not 1.
815 if (R->gcd_is_one(R->lead_monomial(f->f), R->lead_monomial(g->f)))
816 {
817 // remove each of the elements in 'g'.
818 continue;
819 }
820
821 // Criterion 2: if a homogeneous prime,
822 // gcd of tails must be 1.
823 if (is_homogeneous_prime)
824 {
825 if (!R->gcd_is_one(f->f.tail, g->f.tail)) continue;
826 }
827
828 // Finally do the insert
829 monomial lcm = R->mult(m, q->m);
830 Pairs->insert(binomial_s_pair(f, g, lcm));
831 }
832 remove_monomial_list(I);
833}
