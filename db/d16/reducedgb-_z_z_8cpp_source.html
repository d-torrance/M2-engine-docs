103  int comp,
104  int &result_loc)
105 {

118  {

124  mpz_srcptr wc = polys[w].f->coeff.get_mpz();
125  if (mpz_cmpabs(rc, wc) > 0)

127  result_loc = w;

129  }

135 {
136  gbvector *zero = 0;

140  POLY h = f;
141  exponents_t EXP = ALLOCATE_EXPONENTS(R->exponent_byte_size());
142  gbvector *r;
143  POLY g;
144  while (!R->gbvector_is_zero(h.f))
145  {
146  int w;
147  R->gbvector_get_lead_exponents(F, h.f, EXP);

164  break;
165  }
166  frem->next = h.f;
167  frem = frem->next;
