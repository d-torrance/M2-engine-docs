28struct ReducedGB_ZZ_sorter
29{
30 GBRing *R;
31 const FreeModule *F;
32 const VECTOR(POLY) & gb;
33 ReducedGB_ZZ_sorter(GBRing *R0,
34 const FreeModule *F0,
35 const VECTOR(POLY) & gb0)
36 : R(R0), F(F0), gb(gb0)
37 {
38 }
39 bool operator()(int xx, int yy)
40 {
41 gbvector *x = gb[xx].f;
42 gbvector *y = gb[yy].f;
43 int cmp = R->gbvector_compare(F, x, y);
44 if (cmp == LT) return true;
45 if (cmp == GT) return false;
46 // Now order them in ascending order on the coeff (which should always be
47 // POSITIVE).
48 return (mpz_cmp(x->coeff.get_mpz(), y->coeff.get_mpz()) < 0);
49 }
50};

52void ReducedGB_ZZ::minimalize(const VECTOR(POLY) & polys0, bool auto_reduced)
53// I have to decide: does this ADD to the existing set?
54{
55 // First sort these elements via increasing lex order (or monomial order?)
56 // Next insert minimal elements into T, and polys
57
58 VECTOR(int) positions;
59 positions.reserve(polys0.size());
60
61 for (int i = 0; i < polys0.size(); i++) positions.push_back(i);
62
63 std::stable_sort(
64 positions.begin(), positions.end(), ReducedGB_ZZ_sorter(R, F, polys0));
65
66 // Now loop through each element, and see if the lead monomial is in T.
67 // If not, add it in , and place element into 'polys'.
68
69 for (VECTOR(int)::iterator i = positions.begin(); i != positions.end(); i++)
70 {
71 gbvector *f = polys0[*i].f;
72 exponents_t e = R->exponents_make();
73 R->gbvector_get_lead_exponents(F, f, e);
74 if ((!ringtableZZ ||
75 !ringtableZZ->find_term_divisors(1, f->coeff.get_mpz(), e, 1)) &&
76 T->find_term_divisors(1, f->coeff.get_mpz(), e, f->comp) == 0)
77 {
78 // Keep this element
79
80 POLY h;
81 ring_elem junk;
82
83 h.f = R->gbvector_copy(f);
84 h.fsyz = R->gbvector_copy(polys0[*i].fsyz);
85
86 if (auto_reduced) remainder(h, false, junk); // This auto-reduces h.
87
88 if (h.f != nullptr && mpz_sgn(h.f->coeff.get_mpz()) < 0)
89 {
90 R->gbvector_mult_by_coeff_to(h.f, globalZZ->minus_one());
91 R->gbvector_mult_by_coeff_to(h.fsyz, globalZZ->minus_one());
92 }
93
94 T->insert(h.f->coeff.get_mpz(), e, h.f->comp, INTSIZE(polys));
95 polys.push_back(h);
96 }
97 else
98 R->exponents_delete(e);
99 }
100}

102enum ReducedGB_ZZ::divisor_type ReducedGB_ZZ::find_divisor(exponents_t exp,
103 int comp,
104 int &result_loc)
105{
106 int w = T->find_smallest_coeff_divisor(exp, comp); // gives smallest coeff
107 int r = -1;
108 if (ringtableZZ) r = ringtableZZ->find_smallest_coeff_divisor(exp, 1);
109
110 if (r < 0)
111 {
112 if (w < 0) return DIVISOR_NONE;
113 result_loc = w;
114 return DIVISOR_MODULE;
115 }
116 // r >= 0
117 if (w < 0)
118 {
119 result_loc = r;
120 return DIVISOR_RING;
121 }
122 // r >= 0, w >= 0
123 mpz_srcptr rc = originalR->quotient_gbvector(r)->coeff.get_mpz();
124 mpz_srcptr wc = polys[w].f->coeff.get_mpz();
125 if (mpz_cmpabs(rc, wc) > 0)
126 {
127 result_loc = w;
128 return DIVISOR_MODULE;
129 }
130 result_loc = r;
131 return DIVISOR_RING;
132}

134void ReducedGB_ZZ::remainder(POLY &f, bool use_denom, ring_elem &denom)
135{
136 gbvector *zero = nullptr;
137 gbvector head;
138 gbvector *frem = &head;
139 frem->next = nullptr;
140 POLY h = f;
141 exponents_t EXP = ALLOCATE_EXPONENTS(R->exponent_byte_size());
142 gbvector *r;
143 POLY g;
144 while (!R->gbvector_is_zero(h.f))
145 {
146 int w;
147 R->gbvector_get_lead_exponents(F, h.f, EXP);
148 int x = h.f->comp;
149 enum divisor_type typ = find_divisor(EXP, x, w);
150 switch (typ)
151 {
152 case DIVISOR_RING:
153 r = const_cast<gbvector *>(originalR->quotient_gbvector(w));
154 if (R->gbvector_reduce_lead_term_ZZ(F, Fsyz, h.f, zero, r, zero))
155 continue;
156 break;
157 case DIVISOR_MODULE:
158 g = polys[w];
159 if (R->gbvector_reduce_lead_term_ZZ(
160 F, Fsyz, h.f, h.fsyz, g.f, g.fsyz))
161 continue;
162 break;
163 case DIVISOR_NONE:
164 break;
165 }
166 frem->next = h.f;
167 frem = frem->next;
168 h.f = h.f->next;
169 frem->next = nullptr;
170 }
171 h.f = head.next;
172 f.f = h.f;
173 originalR->get_quotient_info()->gbvector_normal_form(Fsyz, h.fsyz);
174 f.fsyz = h.fsyz;
175}
