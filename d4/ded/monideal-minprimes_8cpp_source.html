76static int alg1_reduce_exp(const int *m, const int *exp)
77// Determine whether the varpower monomial 'm'
78// can be in the monomial prime ideal 'exp'.
79// exp corresponds to the set:
80// exp[i]>0 means variable is in ideal
81// exp[i]<0 means variable is not in ideal
82// exp[i]=0 means variable may or may not be in the ideal.
83// Return: 0 if 'm' is in this ideal.
84// Return: 1 if 'm' could be in the ideal.
85// Return: -1 if 'm' cannot possibly be in this ideal.
86{
87 int is_one = 1;
88 while (*m != 0)
89 {
90 if (exp[*m] == 1) return 0;
91 if (exp[*m] == 0) is_one = 0;
92 m++;
93 }
94 if (is_one) return -1;
95 return 1;
96}

110void MinimalPrimes::alg1_min_prime_generator(int *which, int depth)
111// which: which monomial we are looking at right now
112// current depth: starts at -1, goes down from there
113// so the current codim is -depth-1
114// The following information is kept as well:
115// current_minprime: array 0..codim-1 of variables in the minprime
116// current_exp: array 0..nvars-1 of 0,1,-1's
117//
118//
119{
120 for (;;)
121 {
122 if (*which == 0)
123 {
124 alg1_grab_prime(depth);
125 return;
126 }
127 switch (alg1_reduce_exp(which + 1, exp))
128 {
129 case 0:
130 which = which + *which;
131 break;
132 case -1:
133 return;
134 case 1:
135 if (depth > depth_limit)
136 {
137 int *m = which + 1;
138 while (*m != 0)
139 {
140 int v = *m;
141 if (exp[v] == 0)
142 {
143 exp[v] = 1;
144 alg1_min_prime_generator(which + *which, depth - 1);
145 exp[v] = depth;
146 }
147 m++;
148 }
149 // This code sets the 'exp' array back to the way it was
150 m = which + 1;
151 while (*m != 0)
152 {
153 int v = *m;
154 if (exp[v] == depth) exp[v] = 0;
155 m++;
156 }
157 }
158 return;
159 }
160 }

164{
165 // First, let's write out the (radical) monomial ideal in an array.
166 // We need to know how large to make it. So, we first add up all of the
167 // degrees of the gens
168
169 depth_limit = -maxcodim - 1;
170
171 long len = 1;
172 for (Bag& a : *mi)
173 {
174 long d = varpower::simple_degree(a.monom().data());
175 len += d;
176 }
177
178 len += mi->size();
179 len += mi->size();
180 monoms = newarray_atomic(int, len);
181
182 int next_monom = 0;
183
184 for (Bag& a : *mi)

186 int *m = a.monom().data();
187 int d = varpower::simple_degree(m);
188
189 monoms[next_monom++] = d + 2;

217// Place the associated primes of minimal codimension

243// Determine whether the varpower monomial 'm'
244// can be in the monomial prime ideal 'exp'.
245// exp corresponds to the set:
246// exp[i]>0 means variable is in ideal
247// exp[i]<0 means variable is not in ideal

263void MinimalPrimes::ass_prime_generator(Nmi_node *p, int codim)
264{
265 int i = codim + 1;
266 if (exps[i] == nullptr) exps[i] = newarray_atomic(int, nvars);
267 exponents_t exp0 = exps[i];
268 for (int j = 0; j < nvars; j++) exp0[j] = exps[codim][j];
269 for (;;)
270 {
271 if (p == nullptr)
272 {
273 if (state == do_codim)
274 {
275 if (codim < codim_limit) codim_limit = codim;
276 }
277 else
278 {
279 to_prime_ideal(nvars, exp0);
280 Bag *b = new Bag(0);
281 varpower::from_expvector(nvars, exp0, b->monom());
282 primes->insert(b);
283 n_minprimes++;
284 }
285 return;
286 }
287 const int *m = p->monom().data();
288 switch (reduce_exp(m, exp0))
289 {
290 case 0:
291 p = mi->next(p);
292 break;
293 case -1:
294 return;
295 case 1:
296 if (codim < codim_limit)
297 for (index_varpower i2 = m; i2.valid(); ++i2)
298 if (exp0[i2.var()] == 0)
299 {
300 exp0[i2.var()] = 1;
301 ass_prime_generator(mi->next(p), codim + 1);
302 exp0[i2.var()] = -1;
303 if (minprime_limit > 0 && n_minprimes >= minprime_limit)
304 return;
305 }
306 return;
307 }
308 }
