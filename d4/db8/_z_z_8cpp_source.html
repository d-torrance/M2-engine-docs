197 bool RingZZ::lower_associate_divisor(ring_elem &f, const ring_elem g) const
198 {
199  // This sets f to either 0, 1 or -1.
200  // if f is 0, do f=sign(g), else f=sign(f)
201  // return whether f is zero
202  mpz_ptr result = RingZZ::new_elem();
203  mpz_srcptr a = f.get_mpz();
204  mpz_srcptr b = g.get_mpz();
205  int sa = mpz_sgn(a);
206  int sb = mpz_sgn(b);
207  int s = (sa == 0 ? sb : sa);
208 
209  mpz_set_si(result, s);
210  mpz_reallocate_limbs(result);
211 
212  f = ring_elem(result);
213  return !RingZZ::is_zero(f);
214 }

308 ring_elem RingZZ::remainderAndQuotient(const ring_elem f,
309  const ring_elem g,
310  ring_elem &quot) const
311 {
312  mpz_ptr q = new_elem();
313  mpz_ptr r = new_elem();
314  int gsign = mpz_sgn(g.get_mpz());
315  mpz_t gg, ghalf;
316  mpz_init(gg);
317  mpz_init(ghalf);
318  mpz_abs(gg, g.get_mpz());
319  mpz_fdiv_qr(q, r, f.get_mpz(), gg);
320  mpz_tdiv_q_2exp(ghalf, gg, 1);
321  if (mpz_cmp(r, ghalf) > 0) // r > ghalf
322  {
323  mpz_sub(r, r, gg);
324  mpz_add_ui(q, q, 1);
325  }
326  if (gsign < 0) mpz_neg(q, q);
327 
328  mpz_clear(gg);
329  mpz_clear(ghalf);
330  mpz_reallocate_limbs(q);
331  mpz_reallocate_limbs(r);
332  quot = ring_elem(q);
333  return ring_elem(r);
334 }

377 void RingZZ::syzygy(const ring_elem a,
378  const ring_elem b,
379  ring_elem &x,
380  ring_elem &y) const
381 {
382  assert(!is_zero(b));
383  // First check the special cases a = 0, b = 1, -1. Other cases: use gcd.
384  if (RingZZ::is_zero(a))
385  {
386  x = RingZZ::from_long(1);
387  y = RingZZ::from_long(0);
388  return;
389  }
390  mpz_srcptr bb = b.get_mpz();
391  if (mpz_cmp_ui(bb, 1) == 0)
392  {
393  x = RingZZ::from_long(1);
394  y = RingZZ::negate(a);
395  return;
396  }
397  if (mask_mpz_cmp_si(bb, -1) == 0)
398  {
399  x = RingZZ::from_long(1);
400  y = RingZZ::copy(a);
401  return;
402  }
403  ring_elem g = RingZZ::gcd(a, b);
404  y = RingZZ::divide(a, g);
405  x = RingZZ::divide(b, g);
406  RingZZ::remove(g);
407  if (mpz_sgn(x.get_mpz()) > 0)
408  y = RingZZ::negate(y);
409  else
410  x = RingZZ::negate(x);
411 }
