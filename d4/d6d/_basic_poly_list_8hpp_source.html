74void toStream(const BasicPolyList& Fs, S &str)
75{
76 str.idealBegin(Fs.size());
77 for (auto& F : Fs)
78 {
79 str.appendPolynomialBegin(F.mCoefficients.size());
80 int monomStart = 0;
81 for (auto i=0; i<F.mCoefficients.size(); ++i)
82 {
83 auto monomEnd = monomStart + F.mMonomials[monomStart];
84 if (F.mComponents.empty())
85 str.appendTermBegin(0);
86 else
87 str.appendTermBegin(F.mComponents[i]);
88 for (auto j=monomStart+1; j<monomEnd; j += 2)
89 str.appendExponent(F.mMonomials[j], F.mMonomials[j+1]);
90 str.appendTermDone(F.mCoefficients[i]);
91 monomStart = monomEnd;
92 }
93 str.appendPolynomialDone();
94 }
95 str.idealDone();
96}
