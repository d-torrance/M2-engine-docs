63inline void vecHeap::add(vecterm *p)
64{
65 mLead = -1;
66 int len = K->n_nonzero_terms(p);
67 int i = 0;
68 while (len >= heap_size[i]) i++;
69 K->add_vec_to(heap[i], p);
70 len = K->n_nonzero_terms(heap[i]);
71 p = NULL;
72 while (len >= heap_size[i])
73 {
74 i++;
75 K->add_vec_to(heap[i], heap[i - 1]);
76 len = K->n_nonzero_terms(heap[i]);
77 heap[i - 1] = NULL;
78 }
79 if (i > top_of_heap) top_of_heap = i;
80}

90inline const vecterm *vecHeap::get_lead_term()
91{
92 int lead_so_far = -1;
93 for (int i = 0; i <= top_of_heap; i++)
94 {
95 if (heap[i] == NULL) continue;
96 if (lead_so_far < 0)
97 {
98 lead_so_far = i;
99 continue;
100 }
101 int cmp = compare(heap[lead_so_far], heap[i]);
102 if (cmp == GT) continue;
103 if (cmp == LT)
104 {
105 lead_so_far = i;
106 continue;
107 }
108 // At this point we have equality
109 K->add_to(heap[lead_so_far]->coeff, heap[i]->coeff);
110 vecterm *tmp = heap[i];
111 heap[i] = tmp->next;
112 tmp->next = NULL;
113 K->remove_vec(tmp);
114
115 if (K->is_zero(heap[lead_so_far]->coeff))
116 {
117 // Remove, and start over
118 tmp = heap[lead_so_far];
119 heap[lead_so_far] = tmp->next;
120 tmp->next = NULL;
121 K->remove_vec(tmp);
122 lead_so_far = -1;
123 i = -1;
124 }
125 }
126 mLead = lead_so_far;
127 if (lead_so_far < 0) return NULL;
128 vecterm *result = heap[lead_so_far];
129 return result;
130}
131inline vecterm *vecHeap::remove_lead_term()
132{
133 if (mLead < 0) get_lead_term();
134 if (mLead < 0) return NULL;
135 vecterm *result = heap[mLead];
136 heap[mLead] = result->next;
137 result->next = NULL;
138 mLead = -1;
139 return result;
140}
