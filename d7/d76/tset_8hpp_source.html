132 template <typename MonomType>
133 void TSet<MonomType>::CollectNonMultiProlongations(typename TSet<MonomType>::Iterator& iterator, std::list<Triple<MonomType>*>& set)
134 {
135 if (iterator == TripleList.end() || !(*iterator))
136 {
137 return;
138 }
139
140 typename MonomType::Integer firstMultiVar = (**iterator).GetPolynomLm().FirstMultiVar();
141 for (typename MonomType::Integer var = 0; var < firstMultiVar; ++var)
142 {
143 if (!(**iterator).TestNmp(var))
144 {
145 Polynom<MonomType>* tmpPolynom = new Polynom<MonomType>(*(**iterator).GetPolynom());
146 (*tmpPolynom) *= var;
147
148 (**iterator).SetNmp(var);
149
150 if (!tmpPolynom->IsZero())
151 {
152 set.push_back(new Triple<MonomType>(tmpPolynom
153 , (**iterator).GetAncestor()
154 , (**iterator).GetNmp()
155 , (*iterator)
156 , var)
157 );
158 }
159 delete tmpPolynom;
160 }
161 }
