25size_t MutableComplex::complexity(const iterator &i, const size_t flags) const
26// Heuristics
27// TODO: save row/column totals in a separate vector and only update when
28// pruning a unit
29{
30 ring_elem e;
31 const size_t n = i.index();
32 const std::pair<size_t, size_t> u = *i;
33 mDifferential[n]->get_entry(u.first, u.second, e);
34 // Count terms in the column
35 int s = 0;
36 if (mLocalRing == nullptr)
37 s = mPolynomialRing->n_terms(e);
38 else
39 s = mLocalRing->n_terms(e);
40 for (size_t r = 0; r < mBetti[n]; ++r)
41 {
42 if (r == u.first) continue;
43 mDifferential[n]->get_entry(r, u.second, e);
44 if (mRing->is_zero(e)) continue;
45 if (mLocalRing == nullptr)
46 s += mPolynomialRing->n_terms(e);
47 else
48 s += mLocalRing->n_terms(e);
49 }
50 // Count terms in the row
51 for (size_t c = 0; c < mBetti[n + 1]; ++c)
52 {
53 if (c == u.second) continue;
54 mDifferential[n]->get_entry(u.first, c, e);
55 if (mRing->is_zero(e)) continue;
56 if (mLocalRing == nullptr)
57 s += mPolynomialRing->n_terms(e);
58 else
59 s += mLocalRing->n_terms(e);
60 }
61 // std::cout<<s<<std::endl;
62 // Local case: (size numerator elt) * (row/numerator/size//sum) *
63 // (col/numerator/size//sum)
64 return static_cast<size_t>(s);
65}

67bool MutableComplex::next_unit(iterator &i, const size_t flags) const
68{
69 ring_elem e;
70 for (; i < i.end(); ++i)
71 {
72 mDifferential[i.index()]->get_entry((*i).first, (*i).second, e);
73 if (mRing->is_unit(e)) return true;
74 }
75 return false;

78bool MutableComplex::find_unit(iterator &i, const size_t flags) const
79{
80 if (!(flags & FLAG_BEST_UNIT)) return next_unit(i, flags);
81
82 iterator j(*this, i.index());
83 if (!next_unit(j, flags)) return false;
84 size_t t = -1, s = complexity(j, flags);
85 while (next_unit(j, flags))
86 {
87 s = complexity(j, flags);
88 if (s < t)
89 {
90 t = s;
91 *i = *j;
92 }
93 ++j;
94 }
95 return true;
96}

116void MutableComplex::prune_unit(const iterator &i, const size_t flags)
117{
118 const size_t n = i.index();
119 const std::pair<size_t, size_t> u = *i;
120 const bool trace = flags & FLAG_TRACE_MORPHISMS;
121 /* There is some redundancy here, can set some stuff to zero.
122 TODO: Maybe check to see which is more efficient?
123 Compare with using rawReduceByPivot
124 */
125 // Move to last row
126 mDifferential[n]->interchange_rows(u.first, mBetti[n] - 1);
127 if (0 < n) mDifferential[n - 1]->interchange_columns(u.first, mBetti[n] - 1);
128 if (trace) mMorphisms[n]->interchange_columns(u.first, mBetti[n] - 1);
129 // Move to last column
130 mDifferential[n]->interchange_columns(u.second, mBetti[n + 1] - 1);
131 if (n < mDifferential.size() - 1)
132 mDifferential[n + 1]->interchange_rows(u.second, mBetti[n + 1] - 1);
133 if (trace)
134 mMorphisms[n + 1]->interchange_columns(u.second, mBetti[n + 1] - 1);
135 // Get the pivot
136 ring_elem p, q, f;
137 mDifferential[n]->get_entry(mBetti[n] - 1, mBetti[n + 1] - 1, p);
138 p = mRing->invert(p);
139 q = mRing->negate(p);
140 // Clear the column
141 for (size_t r = 0; r < mBetti[n] - 1; ++r)
142 {
143 mDifferential[n]->get_entry(r, mBetti[n + 1] - 1, f);
144 mDifferential[n]->row_op(r, mRing->mult(f, q), mBetti[n] - 1);
145 if (0 < n)
146 mDifferential[n - 1]->column_op(mBetti[n] - 1, mRing->mult(f, p), r);
147 if (trace) mMorphisms[n]->column_op(mBetti[n] - 1, mRing->mult(f, p), r);
148 }
149 // Clear the row
150 for (size_t c = 0; c < mBetti[n + 1] - 1; ++c)
151 {
152 mDifferential[n]->get_entry(mBetti[n] - 1, c, f);
153 mDifferential[n]->column_op(c, mRing->mult(f, q), mBetti[n + 1] - 1);
154 if (n < mDifferential.size() - 1)
155 mDifferential[n + 1]->row_op(mBetti[n + 1] - 1, mRing->mult(f, p), c);
156 if (trace)
157 mMorphisms[n + 1]->column_op(c, mRing->mult(f, q), mBetti[n + 1] - 1);
158 }
159 --mBetti[n];
160 --mBetti[n + 1];
161}

174void MutableComplex::prune_complex(size_t nsteps, size_t flags)
175{
176 // Initialize pruning maps
177 const bool dense = mDifferential[0]->is_dense();
178 if (flags & FLAG_TRACE_MORPHISMS)
179 for (size_t n = 0; n < mDifferential.size() + 1; ++n)
180 mMorphisms.push_back(MutableMatrix::identity(mRing, mBetti[n], dense));
181 for (size_t n = 0; n < mDifferential.size() && n < nsteps; n++)
182 {
183 if (flags & FLAG_REV_ORDER)
184 prune_matrix(mDifferential.size() - n - 1, flags);
185 else
186 prune_matrix(n, flags);
187 }
188}
