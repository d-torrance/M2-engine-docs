241 template <typename MonomType>
242 void JanetTree<MonomType>::Iterator::Build(typename MonomType::Integer degree, typename MonomType::Integer var, Triple<MonomType>* triple)
243 {
244 if (!triple)
245 {
246 return;
247 }
248
249 Node* r = new typename JanetTree<MonomType>::Node(triple->GetPolynomLm()[var]);
250 Node* j = r;
251 while(degree > triple->GetPolynomLm()[var])
252 {
253 degree -= triple->GetPolynomLm()[var];
254 ++var;
255 j->NextVariable = new typename JanetTree<MonomType>::Node(triple->GetPolynomLm()[var]);
256 j = j->NextVariable;
257 }
258 j->CurrentTriple = triple;
259
260 r->NextDegree = *CurrentNode;
261 *CurrentNode = r;
262 }

302 template <typename MonomType>
303 const Triple<MonomType>* JanetTree<MonomType>::Find(const MonomType& monom) const
304 {
305 const Triple<MonomType>* triple = 0;
306
307 if (Root)
308 {
309 ConstIterator nodeIterator = Root;
310 typename MonomType::Integer degree = monom.Degree();
311 typename MonomType::Integer var = 0;
312 do
313 {
314 if (nodeIterator.GetDegree() != monom[var] && nodeIterator.HasNextDegree())
315 {
316 nodeIterator.StepNextDegree();
317 }
318
319 if (nodeIterator.GetDegree() != monom[var])
320 {
321 break;
322 }
323 else if (nodeIterator.HasNextVariable())
324 {
325 degree -= monom[var];
326 if (!degree)
327 {
328 break;
329 }
330 ++var;
331 nodeIterator.StepNextVariable();
332 }
333 else
334 {
335 triple = nodeIterator.GetTriple();
336 break;
337 }
338 } while(true);
339 }
340 return triple;
341 }

343 template <typename MonomType>
344 void JanetTree<MonomType>::Insert(Triple<MonomType>* triple)
345 {
346 if (!triple)
347 {
348 return;
349 }
350
351 typename MonomType::Integer degree = triple->GetPolynomLm().Degree();
352 typename JanetTree<MonomType>::Iterator nodeIterator(Root);
353
354 if (!Root)
355 {
356 nodeIterator.Build(degree, 0, triple);
357 }
358 else
359 {
360 typename MonomType::Integer var = 0;
361 do
362 {
363 while(nodeIterator.GetDegree() < triple->GetPolynomLm()[var] && nodeIterator.HasNextDegree())
364 {
365 nodeIterator.StepNextDegree();
366 }
367
368 if (nodeIterator.GetDegree() > triple->GetPolynomLm()[var])
369 {
370 nodeIterator.Build(degree, var, triple);
371 break;
372 }
373 else if (nodeIterator.GetDegree() == triple->GetPolynomLm()[var])
374 {
375 degree -= triple->GetPolynomLm()[var];
376 ++var;
377 nodeIterator.StepNextVariable();
378 }
379 else
380 {
381 nodeIterator.StepNextDegree();
382 nodeIterator.Build(degree, var, triple);
383 break;
384 }
385 } while(true);

389 template <typename MonomType>
390 void JanetTree<MonomType>::Delete(const Triple<MonomType>* triple)
391 {
392 if (!triple)
393 {
394 return;
395 }
396
397 Iterator nodeIterator = Root;
398 //count bifurcations
399 typename MonomType::Integer var = 0;
400 unsigned bifurcations = 0;
401
402 if (nodeIterator.HasNextDegree() && nodeIterator.HasNextVariable())
403 {
404 ++bifurcations;
405 }
406
407 do
408 {
409 while(nodeIterator.GetDegree() < triple->GetPolynomLm()[var])
410 {
411 nodeIterator.StepNextDegree();
412 if (nodeIterator.HasNextDegree() && nodeIterator.HasNextVariable())
413 {
414 ++bifurcations;
415 }
416 }
417
418 if (nodeIterator.HasNextVariable())
419 {
420 ++var;
421 nodeIterator.StepNextVariable();
422 if (nodeIterator.HasNextDegree() && nodeIterator.HasNextVariable())
423 {
424 ++bifurcations;
425 }
426 }
427 else
428 {
429 break;
430 }
431 } while(true);
432
433 //deletion
434 nodeIterator = Root;
435 var = 0;
436 bool varDirection = false;
437
438 if (nodeIterator.HasNextDegree() && nodeIterator.HasNextVariable())
439 {
440 --bifurcations;
441 }
442 if (!bifurcations)
443 {
444 if (nodeIterator.GetDegree() < triple->GetPolynomLm()[var])
445 {
446 nodeIterator.StepNextDegree();
447 }
448 else
449 {
450 varDirection = true;
451 }
452 }
453
454 while (bifurcations > 0)
455 {
456 while(nodeIterator.GetDegree() < triple->GetPolynomLm()[var] && bifurcations > 0)
457 {
458 nodeIterator.StepNextDegree();
459 if (nodeIterator.HasNextDegree() && nodeIterator.HasNextVariable())
460 {
461 --bifurcations;
462 }
463 }
464
465 if (!bifurcations)
466 {
467 if (nodeIterator.GetDegree() < triple->GetPolynomLm()[var])
468 {
469 nodeIterator.StepNextDegree();
470 break;
471 }
472 else
473 {
474 varDirection = true;
475 break;
476 }
477 }
478
479 ++var;
480 nodeIterator.StepNextVariable();
481 if (nodeIterator.HasNextDegree() && nodeIterator.HasNextVariable())
482 {
483 --bifurcations;
484 }
485 if (!bifurcations)
486 {
487 if (nodeIterator.GetDegree() < triple->GetPolynomLm()[var])
488 {
489 nodeIterator.StepNextDegree();
490 break;
491 }
492 else
493 {
494 varDirection = true;
495 break;
496 }
497 }
498 }
499
500 if (varDirection)
501 {
502 Iterator tmpIterator = nodeIterator;
503 tmpIterator.StepNextVariable();
504 tmpIterator.Clear();

506 }
