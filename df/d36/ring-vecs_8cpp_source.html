96int Ring::compare_vecs(vec v, vec w) const
97{
98 for (;; v = v->next, w = w->next)
99 {
100 if (v == nullptr)
101 {
102 if (w == nullptr) return 0;
103 return -1;
104 }
105 if (w == nullptr) return 1;
106 int cmp = v->comp - w->comp;
107 if (cmp > 0) return cmp;
108 if (cmp < 0) return cmp;
109 cmp = this->compare_elems(v->coeff, w->coeff);
110 if (cmp > 0) return cmp;
111 if (cmp < 0) return cmp;
112 }

220vec Ring::sub_vector(const vecterm *v, M2_arrayint r) const
221{
222 if (v == nullptr) return nullptr;
223 // Largest component which occurs in v occurs first.
224 VECTOR(int) trans(v->comp + 1);
225 for (int i = 0; i < v->comp; i++) trans.push_back(-1);
226
227 for (unsigned j = 0; j < r->len; j++)
228 if (r->array[j] >= 0 && r->array[j] <= v->comp) trans[r->array[j]] = j;
229
230 vecterm head;
231 vecterm *result = &head;
232 for (; v != nullptr; v = v->next)
233 if (trans[v->comp] != -1)
234 {
235 result->next = new_vec();
236 result = result->next;
237 result->next = nullptr;
238 result->coeff = v->coeff;
239 result->comp = trans[v->comp];
240 }
241 result->next = nullptr;
242 result = head.next;
243
244 vec_sort(result);
245 return result;
246}

314vec Ring::vec_eval(const RingMap *map, const FreeModule *F, const vec v) const
315// v is a vector over 'this'
316{
317 const Ring *targetRing = map->get_ring();
318
319 vecterm head;
320 vec result = &head;
321
322 for (vec t = v; t != nullptr; t = t->next)
323 {
324 ring_elem a = eval(map, t->coeff, 0); // a is now in the target ring
325 if (!targetRing->is_zero(a))
326 {
327 result->next = targetRing->make_vec(t->comp, a);
328 result = result->next;
329 }
330 }
331 result->next = nullptr;
332 return head.next;

367void Ring::mult_vec_to(vec &v, const ring_elem r, bool opposite_mult) const
368{
369 if (this->is_zero(r))
370 {
371 remove_vec(v);
372 v = nullptr;
373 return;
374 }
375 vecterm head;
376 head.next = v;
377 vec p = &head;
378 while (p->next != nullptr)
379 {
380 // old version: this->mult_to(p->next->coeff, a);
381 ring_elem c;
382 if (opposite_mult)
383 c = this->mult(p->next->coeff, r);
384 else
385 c = this->mult(r, p->next->coeff);
386 p->next->coeff = c;
387 if (this->is_zero(p->next->coeff))
388 {
389 vec tmp = p->next;
390 p->next = tmp->next;
391 remove_vec_node(tmp);
392 }
393 else
394 p = p->next;
395 }
396 v = head.next;
397}

425vec Ring::mult_vec_matrix(const Matrix *m, vec v, bool opposite_mult) const
426{
427 // Multiply m * v, using left or right mult for each scalar mult.
428
429 // Each loop below should read
430 // result = 0
431 // for each non-zero term f e[component] of the vector v
432 // result += f M[v]
433 vec result = nullptr;
434 for (; v != nullptr; v = v->next)
435 {
436 vec w = this->copy_vec(m->elem(v->comp));
437 mult_vec_to(w, v->coeff, !opposite_mult);
438 this->add_vec_to(result, w);
439 }
440 return result;
441}

509void Ring::add_vec_to(vec &v, vec &w) const
510{
511 if (w == nullptr) return;
512 if (v == nullptr)
513 {
514 v = w;
515 w = nullptr;
516 return;
517 }
518 vecterm head;
519 vec result = &head;
520 while (true)
521 if (v->comp < w->comp)
522 {
523 result->next = w;
524 result = result->next;
525 w = w->next;
526 if (w == nullptr)
527 {
528 result->next = v;
529 v = head.next;
530 return;
531 }
532 }
533 else if (v->comp > w->comp)
534 {
535 result->next = v;
536 result = result->next;
537 v = v->next;
538 if (v == nullptr)
539 {
540 result->next = w;
541 v = head.next;
542 w = nullptr;
543 return;
544 }
545 }
546 else
547 {
548 vec tmv = v;
549 vec tmw = w;
550 v = v->next;
551 w = w->next;
552 tmv->coeff = this->add(tmv->coeff, tmw->coeff);
553 if (this->is_zero(tmv->coeff))
554 {
555 remove_vec_node(tmv);
556 }
557 else
558 {
559 result->next = tmv;
560 result = result->next;
561 }
562 remove_vec_node(tmw);
563 if (w == nullptr)
564 {
565 result->next = v;
566 v = head.next;
567 return;
568 }
569 if (v == nullptr)
570 {
571 result->next = w;
572 v = head.next;
573 w = nullptr;
574 return;
575 }
576 }
577}

579ring_elem Ring::dot_product(const vecterm *v, const vecterm *w) const
580{
581 ring_elem result = this->from_long(0);
582 while (true)
583 {
584 if (v == nullptr) return result;
585 if (w == nullptr) return result;
586 if (v->comp > w->comp)
587 v = v->next;
588 else if (v->comp < w->comp)
589 w = w->next;
590 else
591 {
592 ring_elem a = this->mult(v->coeff, w->coeff);
593 result = this->add(result, a);
594 v = v->next;
595 w = w->next;
596 }
597 }

600void Ring::set_entry(vec &v, int r, ring_elem a) const
601{
602 vec p;
603 bool iszero = this->is_zero(a);
604 vecterm head;
605 head.next = v;
606 for (p = &head; p->next != nullptr; p = p->next)
607 if (p->next->comp <= r) break;
608
609 if (p->next == nullptr || p->next->comp < r)
610 {
611 if (iszero) return;
612 vec w = new_vec();
613 w->next = p->next;
614 w->comp = r;
615 w->coeff = a;
616 p->next = w;
617 }
618 else if (p->next->comp == r)
619 {
620 if (iszero)
621 {
622 // delete node
623 vec tmp = p->next;
624 p->next = tmp->next;
625 remove_vec_node(tmp);
626 }
627 else
628 p->next->coeff = a;
629 }
630 v = head.next;
631}

633void Ring::vec_sort(vecterm *&f) const
634{
635 // Internal routine to place the elements back in order after
636 // an operation such as subvector.
637 // Divide f into two lists of equal length, sort each,
638 // then add them together. This allows the same monomial
639 // to appear more than once in 'f'.
640
641 if (f == nullptr || f->next == nullptr) return;
642 vecterm *f1 = nullptr;
643 vecterm *f2 = nullptr;
644 while (f != nullptr)
645 {
646 vecterm *t = f;
647 f = f->next;
648 t->next = f1;
649 f1 = t;
650
651 if (f == nullptr) break;
652 t = f;
653 f = f->next;
654 t->next = f2;
655 f2 = t;
656 }
657
658 vec_sort(f1);
659 vec_sort(f2);
660 add_vec_to(f1, f2);
661 f = f1;
662}

671vec Ring::vec_diff(vec v, int rankFw, vec w, int use_coeff) const
672// rankFw is the rank of the free module corresponding to w.
673{
674 vec result = nullptr;
675 for (; v != nullptr; v = v->next)
676 for (vecterm *p = w; p != nullptr; p = p->next)
677 {
678 ring_elem a = diff(v->coeff, p->coeff, use_coeff);
679 if (is_zero(a))
680 {
681 remove(a);
682 continue;
683 }
684 vecterm *t = new_vec();
685 t->comp = rankFw * v->comp + p->comp;
686 t->coeff = a;
687 t->next = result;
688 result = t;
689 }
690 vec_sort(result);
691 return result;
692}

761bool Ring::vec_multi_degree(const FreeModule *F, const vec f, monomial degf) const
762// Returns true if the element is homogeneous
763// Sets degf to be the highest degree found (actually, the join of the
764// degree vectors occurring).
765{
766 monomial degv;
767 degree_monoid()->one(degf);
768 if (f == nullptr) return true;
769 bool result = multi_degree(f->coeff, degf);
770 degree_monoid()->mult(degf, F->degree(f->comp), degf);
771 degv = degree_monoid()->make_one();
772
773 for (vec v = f->next; v != nullptr; v = v->next)
774 {
775 bool ishom = multi_degree(v->coeff, degv);
776 result = result && ishom;
777 degree_monoid()->mult(degv, F->degree(v->comp), degv);
778
779 if (0 != degree_monoid()->compare(degf, degv))
780 {
781 result = false;
782 degree_monoid()->lcm(degf, degv, degf);
783 }
784 }
785 degree_monoid()->remove(degv);
786 return result;
787}

789bool Ring::vec_is_homogeneous(const FreeModule *F, const vec f) const
790{
791 if (!this->is_graded()) return false;
792 if (f == nullptr) return true;
793 monomial d = degree_monoid()->make_one();
794 monomial e = degree_monoid()->make_one();
795 bool result = multi_degree(f->coeff, d);
796 if (result)
797 {
798 degree_monoid()->mult(d, F->degree(f->comp), d);
799 for (vecterm *t = f->next; (t != nullptr) && result; t = t->next)
800 {
801 bool ishom = multi_degree(t->coeff, e);
802 result = result && ishom;
803 if (result)
804 {
805 degree_monoid()->mult(e, F->degree(t->comp), e);
806 if (0 != degree_monoid()->compare(d, e)) result = false;
807 }
808 }
809 }
810 degree_monoid()->remove(d);
811 degree_monoid()->remove(e);
812 return result;
813}

815void Ring::vec_degree_weights(const FreeModule *F,
816 const vec f,
817 const std::vector<int> &wts,
818 int &lo,
819 int &hi) const
820{
821 vecterm *t = f;
822 if (t == nullptr)
823 {
824 lo = hi = 0;
825 return;
826 }
827 degree_weights(t->coeff, wts, lo, hi);
828 lo += F->primary_degree(t->comp);
829 hi += F->primary_degree(t->comp);
830 for (t = t->next; t != nullptr; t = t->next)
831 {
832 int lo1, hi1;
833 degree_weights(t->coeff, wts, lo1, hi1);
834 lo1 += F->primary_degree(t->comp);
835 hi1 += F->primary_degree(t->comp);
836 if (hi1 > hi) hi = hi1;
837 if (lo1 < lo) lo = lo1;
838 }

841vec Ring::vec_homogenize(const FreeModule *F,
842 const vec f,
843 int v,
844 int d,
845 const std::vector<int> &wts) const
846// Any terms which can't be homogenized are silently set to 0
847{
848 vecterm head;
849 vecterm *result = &head;
850 assert(wts[v] != 0);
851 // If an error occurs, then return 0, and set ERROR
852
853 for (vec w = f; w != nullptr; w = w->next)
854 {
855 int e = F->primary_degree(w->comp);
856 ring_elem a = homogenize(w->coeff, v, d - e, wts);
857 if (!is_zero(a))
858 {
859 result->next = make_vec(w->comp, a);
860 result = result->next;
861 }
862 }
863 result->next = nullptr;
864 return head.next;
865}

868 const vec f,
869 int v,
870 const std::vector<int> &wts) const
871{

873 if (f == nullptr) return result;
874 int lo, hi;
875 vec_degree_weights(F, f, wts, lo, hi);
876 assert(wts[v] != 0);
877 int d = (wts[v] > 0 ? hi : lo);
878 return vec_homogenize(F, f, v, d, wts);

886bool static check_nterm_multiples(const PolyRing *R,
887 ring_elem f1, // in R
888 ring_elem g1, // in R
889 ring_elem c, // in flat coeffs of R
890 ring_elem d) // in flat coeffs of R
891{
892 Nterm *f;
893 Nterm *g;
894 const Monoid *M = R->getMonoid();
895 const Ring *K = R->getCoefficients();
896 for (f = f1, g = g1; f != nullptr && g != nullptr; f = f->next, g = g->next)
897 {
898 if (M->compare(f->monom, g->monom) != 0) return false;
899 ring_elem c1 = K->mult(c, g->coeff);
900 ring_elem d1 = K->mult(d, f->coeff);
901 int isequal = K->is_equal(c1, d1);
902 if (!isequal) return false;
903 }
904 if (f == nullptr && g == nullptr) return true;
905 return false;
906}

908bool Ring::vec_is_scalar_multiple(vec f, vec g) const
909// is df = cg, some scalars c,d?
910// These scalars are over the very bottom base field/ZZ.
911{
912 if (f == nullptr) return true;
913 if (g == nullptr) return true;
914 const PolynomialRing *PR = cast_to_PolynomialRing();
915 if (PR == nullptr) return true;
916 const PolyRing *PR1 = PR->getNumeratorRing();
917#ifdef DEVELOPMENT
918#warning "use numerator only"
919#endif
920 if (f->comp != g->comp) return false;
921 Nterm *f1 = f->coeff;
922 Nterm *g1 = g->coeff;
923 ring_elem c = f1->coeff;
924 ring_elem d = g1->coeff;
925 vec p, q;
926 for (p = f, q = g; p != nullptr && q != nullptr; p = p->next, q = q->next)
927 {
928 if (p->comp != q->comp) return 0;
929 if (!check_nterm_multiples(PR1, p->coeff, q->coeff, c, d)) return false;
930 }
931 if (q == nullptr && p == nullptr) return true;
932 return false;
933}

935vec Ring::vec_remove_monomial_factors(vec f, bool make_squarefree_only) const
936{
937 const PolynomialRing *PR = cast_to_PolynomialRing();
938 if (PR == nullptr) return copy_vec(f);
939 if (f == nullptr) return nullptr;
940
941 exponents_t exp = newarray_atomic(int, PR->n_vars());
942
943 Nterm *t = f->coeff;
944 PR->getMonoid()->to_expvector(t->monom, exp); // Get the process started
945
946 for (vec a = f; a != nullptr; a = a->next) monomial_divisor(a->coeff, exp);
947
948 if (make_squarefree_only)
949 // Now divide each term by exp[i]-1, if exp[i] >= 2
950 for (int i = 0; i < PR->n_vars(); i++)
951 if (exp[i] >= 1) exp[i]--;
952
953 vec result = vec_divide_by_expvector(exp, f);
954
955 freemem(exp);
956 return result;
957}
