175 void power(ElementType &result, ElementType a, long n) const
176 {
177 if (a != 0)
178 {
179 result = n_powmod2_preinv(a, n, mModulus.n, mModulus.ninv);
180 }
181 else
182 {
183 // case a == 0
184 if (n < 0) throw exc::division_by_zero_error();
185 if (n == 0) return set_from_long(result, 1);
186 if (n > 0) return set_zero(result);
187 }
188 }
189
190 void power_mpz(ElementType &result, ElementType a, mpz_srcptr n) const
191 {
192 if (a != 0)
193 {
194 unsigned long nbar = mpz_fdiv_ui(n, mCharac - 1);
195 result = n_powmod2_ui_preinv(a, nbar, mModulus.n, mModulus.ninv);
196 }
197 else
198 {
199 // case a == 0
200 if (mpz_sgn(n) == 0)
201 set_from_long(result, 1);
202 else if (mpz_sgn(n) > 0)
203 set_zero(result);
204 else
205 throw exc::division_by_zero_error();
206 }
207 }
