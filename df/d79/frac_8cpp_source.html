109 {
110  ring_elem x, y;
111  if (use_gcd_simplify)
112  {
113  y = f->denom;
114  if (R_->is_equal(y, R_->one())) return;
115  x = f->numer;
116  const RingElement *a = RingElement::make_raw(R_, x);
117  const RingElement *b = RingElement::make_raw(R_, y);
118  const RingElement *c = rawGCDRingElement(a, b, nullptr, false);
119 
120 #if 0
121  // Debugging code
122  buffer o;
123  o << newline;
124  o << "a = ";
125  a->text_out(o);
126  o << " b = ";
127  b->text_out(o);
128  o << " gcd = ";
129  c->text_out(o);
130  o << newline;
131  emit(o.str());
132 #endif
133  if (!R_->is_equal(c->get_value(), R_->one()))
134  {
135  f->numer = R_->divide(f->numer, c->get_value());
136  f->denom = R_->divide(f->denom, c->get_value());
137  }
138  // Now, let's take the content of the denominator, and divide the
139  // numerator
140  // and denominator by this value.
141  ring_elem ct = R_->content(
142  f->denom, f->numer); // result is in R_->getCoefficients()
143 
144 #if 0
145  o.reset();
146  o << "f->numer = ";
147  R_->elem_text_out(o,f->numer);
148  o << " f->denom = ";
149  R_->elem_text_out(o,f->denom);
150  o << " ass= ";

152  o << newline;
153  emit(o.str());
154 #endif
155 
156  if (!R_->getCoefficients()->is_equal(ct, R_->getCoefficients()->one()))
157  {
158  f->numer = R_->divide_by_given_content(f->numer, ct);
159  f->denom = R_->divide_by_given_content(f->denom, ct);
160  }
161  }
162  else
163  {
164  R_->syzygy(f->numer, f->denom, x, y);
165  if (R_->is_zero(x))
166  {

168  set_non_unit_frac(f->denom);
169  f->numer = R_->zero();

190 void FractionField::lower_content(ring_elem &c, const ring_elem g) const
191 {
192  if (!use_gcd_simplify) return;
193  if (is_zero(c))

202  const RingElement *c2 = RingElement::make_raw(R_, cf->denom);
203  const RingElement *g1 = RingElement::make_raw(R_, gf->numer);
204  const RingElement *g2 = RingElement::make_raw(R_, gf->denom);
205 
206  c1 = rawGCDRingElement(c1, g1, nullptr, false);
207 
208  const RingElement *cc2 = rawGCDRingElement(c2, g2, nullptr, false);
209  const RingElement *cc3 = (*c2) * (*g2);

212  frac_elem *result = new_frac_elem();
