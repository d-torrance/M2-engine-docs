83int M2FreeAlgebra::index_of_var(const ring_elem a) const
84{
85 // This function is needed at top level to be able to determine if a ring element is a variable
86 //const Poly* f = reinterpret_cast<Poly*>(a.poly_val);
87 auto f = reinterpret_cast<const Poly*>(a.get_Poly());
88
89 // f is a variable iff: #terms is 1, monomial is [3,1,v], coeff is 1

141bool M2FreeAlgebra::lift(const Ring *R, const ring_elem f1, ring_elem &result) const
142{
143 // R is the target ring
144 // f1 is an element of 'this'.
145 // set result to be the "lift" of f in the ring R, return true if this is possible.
146 // otherwise return false.
147
148 // case: R is the coefficient ring of 'this'.
149 std::cout << "called lift NC case" << std::endl;
150 if (R == coefficientRing())
151 {
152 auto f = reinterpret_cast<const Poly*>(f1.get_Poly());
153 if (f->numTerms() != 1) return false;
154 auto i = f->cbegin();
155 if (monoid().is_one(i.monom()))
156 {
157 result = coefficientRing()->copy(i.coeff());
158 return true;
159 }
160 }
161
162 // at this point, we can't lift it.
163 return false;
164}

259ring_elem M2FreeAlgebra::invert(const ring_elem f) const
260{
261 if (not is_unit(f))
262 throw exc::engine_error("attempting to divide by a non-unit");
263
264 ring_elem cf = lead_coefficient(coefficientRing(), f);
265 // At this point, f should be an element of the coefficient ring, times the monomial 1.
266 // The following will throw an error if it cannot invert the element.
267 ring_elem finv = coefficientRing()->invert(cf);
268 return from_coefficient(finv);
269}

343engine_RawArrayPairOrNull M2FreeAlgebra::list_form(const Ring *coeffR, const ring_elem ff) const
344{
345 // Either coeffR should be the actual coefficient ring (possible a "toField"ed ring)
346 // or a polynomial ring. If not, NULL is returned and an error given
347 // In the latter case, the last set of variables are part of
348 // the coefficients.
349
350 auto f = reinterpret_cast<const Poly*>(ff.get_Poly());
351 if (coeffR != coefficientRing())
352 {
353 ERROR("expected coefficient ring");
354 return nullptr;
355 }
356 int nterms = static_cast<int>(f->numTerms());
357 engine_RawMonomialArray monoms = GETMEM(engine_RawMonomialArray, sizeofarray(monoms,nterms));
358 engine_RawRingElementArray coeffs = GETMEM(engine_RawRingElementArray, sizeofarray(coeffs,nterms));
359 engine_RawArrayPair result = newitem(struct engine_RawArrayPair_struct);
360 monoms->len = nterms;
361 coeffs->len = nterms;
362 result->monoms = monoms;
363 result->coeffs = coeffs;
364
365 // fill result
366 std::vector<int> vp;
367 int next = 0;
368 for (auto i=f->cbegin(); i != f->cend(); ++i, ++next)
369 {
370 ring_elem c = coefficientRing()->copy(i.coeff());
371 vp.resize(0);
372 monoid().getMonomialReversed(i.monom(), vp); // should this instead reverse the monomial?
373 coeffs->array[next] = RingElement::make_raw(coeffR, c);
374 monoms->array[next] = EngineMonomial::make(vp); // reverses the monomial
375 }
376
377 return result;
378}
