343 engine_RawArrayPairOrNull M2FreeAlgebra::list_form(const Ring *coeffR, const ring_elem ff) const
344 {
345  // Either coeffR should be the actual coefficient ring (possible a "toField"ed ring)
346  // or a polynomial ring. If not, NULL is returned and an error given
347  // In the latter case, the last set of variables are part of
348  // the coefficients.
349 
350  auto f = reinterpret_cast<const Poly*>(ff.get_Poly());
351  if (coeffR != coefficientRing())
352  {
353  ERROR("expected coefficient ring");
354  return nullptr;
355  }
356  int nterms = static_cast<int>(f->numTerms());
357  engine_RawMonomialArray monoms = GETMEM(engine_RawMonomialArray, sizeofarray(monoms,nterms));
358  engine_RawRingElementArray coeffs = GETMEM(engine_RawRingElementArray, sizeofarray(coeffs,nterms));
359  engine_RawArrayPair result = newitem(struct engine_RawArrayPair_struct);
360  monoms->len = nterms;
361  coeffs->len = nterms;
362  result->monoms = monoms;
363  result->coeffs = coeffs;
364 
365  // fill result
366  std::vector<int> vp;
367  int next = 0;
368  for (auto i=f->cbegin(); i != f->cend(); ++i, ++next)
369  {
370  ring_elem c = coefficientRing()->copy(i.coeff());
371  vp.resize(0);
372  monoid().getMonomialReversed(i.monom(), vp); // should this instead reverse the monomial?
373  coeffs->array[next] = RingElement::make_raw(coeffR, c);
374  monoms->array[next] = EngineMonomial::make(vp); // reverses the monomial
375  }
376 
377  return result;
378 }
