41 struct ReducedGB_Field_Local_sorter
42 {
43  GBRing *R;
44  const FreeModule *F;
45  const VECTOR(POLY) & gb;
46  std::vector<int> degs;
47  ReducedGB_Field_Local_sorter(GBRing *R0,
48  const FreeModule *F0,
49  const VECTOR(POLY) & gb0)
50  : R(R0), F(F0), gb(gb0)
51  {
52  auto M = R->get_flattened_monoid();
53  for (size_t i = 0; i < gb0.size(); i++)
54  {
55  gbvector *f = gb0[i].f;
56  degs.push_back(M->simple_degree(f->monom));
57  }
58  }
59  bool operator()(int xx, int yy)
60  {
61  // this is the < operation
62  if (degs[xx] < degs[yy]) return true;
63  if (degs[xx] > degs[yy]) return false;
64  gbvector *x = gb[xx].f;
65  gbvector *y = gb[yy].f;
66  return R->gbvector_compare(F, x, y) == LT;
67  }
68 };
