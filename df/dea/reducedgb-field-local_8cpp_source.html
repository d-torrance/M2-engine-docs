41struct ReducedGB_Field_Local_sorter
42{
43 GBRing *R;
44 const FreeModule *F;
45 const VECTOR(POLY) & gb;
46 std::vector<int> degs;
47 ReducedGB_Field_Local_sorter(GBRing *R0,
48 const FreeModule *F0,
49 const VECTOR(POLY) & gb0)
50 : R(R0), F(F0), gb(gb0)
51 {
52 auto M = R->get_flattened_monoid();
53 for (size_t i = 0; i < gb0.size(); i++)
54 {
55 gbvector *f = gb0[i].f;
56 degs.push_back(M->simple_degree(f->monom));
57 }
58 }
59 bool operator()(int xx, int yy)
60 {
61 // this is the < operation
62 if (degs[xx] < degs[yy]) return true;
63 if (degs[xx] > degs[yy]) return false;
64 gbvector *x = gb[xx].f;
65 gbvector *y = gb[yy].f;
66 return R->gbvector_compare(F, x, y) == LT;
67 }
68};

70void ReducedGB_Field_Local::minimalize(const VECTOR(POLY) & polys0,
71 bool auto_reduced)
72{
73 // First sort these elements via increasing lex order (or monomial order?)
74 // Next insert minimal elements into T, and polys
75
76 VECTOR(int) positions;
77 positions.reserve(polys0.size());
78
79 for (int i = 0; i < polys0.size(); i++) positions.push_back(i);
80
81 // displayElements("-- before sort --", R, polys0, [](auto& g) { return g.f;
82 // } );
83
84 std::stable_sort(positions.begin(),
85 positions.end(),
86 ReducedGB_Field_Local_sorter(R, F, polys0));
87
88 // VECTOR(gbvector*) sorted_elements_debug_only;
89 // for (int i=0; i<positions.size(); i++)
90 // sorted_elements_debug_only.push_back(polys0[positions[i]].f);
91 // displayElements("-- after sort --", R, sorted_elements_debug_only,
92 // [](auto& g) { return g; } );
93
94 // Now loop through each element, and see if the lead monomial is in T.
95 // If not, add it in , and place element into 'polys'.
96
97 for (VECTOR(int)::iterator i = positions.begin(); i != positions.end(); i++)
98 {
99 Bag *not_used;
100 gbvector *f = polys0[*i].f;
101 exponents_t e = R->exponents_make();
102 R->gbvector_get_lead_exponents(F, f, e);
103 if ((!Rideal || !Rideal->search_expvector(e, not_used)) &&
104 T->find_divisors(1, e, f->comp) == 0)
105 {
106 // Keep this element
107
108 POLY h;
109 ring_elem junk;
110
111 h.f = R->gbvector_copy(f);
112 h.fsyz = R->gbvector_copy(polys0[*i].fsyz);
113
114 if (false and auto_reduced)
115 remainder(h, false, junk); // This auto-reduces h.
116
117 R->gbvector_remove_content(h.f, h.fsyz);
118
119 T->insert(e, f->comp, INTSIZE(polys));
120 polys.push_back(h);
121 }
122 else
123 R->exponents_delete(e);
124 }
125
126 for (int i = 0; i < polys.size(); i++)
127 {
128 int f_lead_wt;
129 gbvector *f = polys[i].f;
130 int d = wt->gbvector_weight(f, f_lead_wt);
131 int a = d - f_lead_wt;
132
133 divisor_info t;
134 t.g = polys[i];
135 t.size = R->gbvector_n_terms(f);
136 t.alpha = a;
137 gb_elems.push_back(t);
138 }
139}

481{
482 if (M2_gbTrace >= 4) {
483 buffer o;
484 text_out(o);
485 emit(o.str());

487 if (f.f == nullptr) return;
488 T1 = MonomialTable::make(R->n_vars());
489 gbvector head;
490 gbvector *frem = &head;
491 frem->next = nullptr;
492 POLY h = f;
493 exponents_t h_exp = R->exponents_make();
494 int h_alpha, g_alpha;
495 int h_deg = wt->gbvector_weight(f.f);

550 frem->next = h.f;
551 frem = frem->next;
552 h.f = h.f->next;
553 frem->next = nullptr;
554 }
555 }
556

559 R->exponents_delete(h_exp);
