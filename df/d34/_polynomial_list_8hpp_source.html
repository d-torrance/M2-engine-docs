261template<typename S>
262void toStream(const PolynomialList& Fs, S &str)
263{
264 str.idealBegin(Fs.size());
265 for (auto i=0; i<Fs.size(); ++i)
266 {
267 auto& F = Fs[i];
268 str.appendPolynomialBegin(F.mMonomials.size());
269 for (auto i=0; i<F.mMonomials.size(); ++i)
270 {
271 // get monomial
272 // write it out here using appendTermBegin, appendExponent, appendTermDone.
273 if (F.mComponents.empty())
274 str.appendTermBegin(0);
275 else
276 str.appendTermBegin(F.mComponents[i]);
277 MonomialView monom = Fs.monomialHashTable().monomialAt(F.mMonomials[i]);
278 for (auto ve = monom.begin(); ve != monom.end(); ++ve)
279 {
280 str.appendExponent(ve.var(), ve.power());
281 }
282 long val = Fs.vectorArithmetic().to_modp_long(F.mCoefficients, i);
283 str.appendTermDone(val);
284 }
285 str.appendPolynomialDone();
286 }
287 str.idealDone();
288}
