118 void ResGausserZZp::sparseCancel(CoefficientVector r,
119  CoefficientVector sparse,
120  ComponentIndex* comps,
121  CoefficientVector result_loc) const
122 // dense += c * sparse, where c is chosen to cancel column comps[0].
123 // ASSUMPTION: the lead coeff of 'sparse' is 1 or -1 (in the field)
124 // The value of c is appended to result_
125 {
126  // r += a*sparse
127  // ASSUMPTIONS:
128  // len > 0,
129  // sparse[0] = 1 or -1 (in the field)
130  // where a is
131  // r[comps[0]], if sparse[0]==1
132  // -r[comps[0]], if sparse[0]==-1
133  // r = [...., b, .....]
134  // sparse = [1,...] then a = -b
135  // sparse = [-1,...] then a = b
136 
137  auto& vec = coefficientVector(r);
138  auto elems = vec.data();
139  auto& svec = coefficientVector(sparse);
140  auto sparseelems = svec.data();
141  ComponentIndex len = static_cast<ComponentIndex>(svec.size());
142 
143  auto& result = coefficientVector(result_loc);
144 
145  int one;
146  set_one(one);
147 
148  // Basically, over ZZ/p, we are doing: r += a*sparse,
149  // where sparse is monic, and a is -r.coeffs[*comps].
150 
151  n_dense_row_cancel++;
152  n_subtract_multiple += len;
153  FieldElement a = elems[*comps];
154  if (sparseelems[0] != one) // should be minus_one
155  Kp->negate(a, a);
156  for (ComponentIndex i = len; i > 0; i--)
157  Kp->subtract_multiple(elems[*comps++], a, *sparseelems++);
158  Kp->negate(a, a);
159  result.push_back(a);
160 }
