122 bool M2FreeAlgebraQuotient::lift(const Ring *R, const ring_elem f1, ring_elem &result) const
123 {
124  // R is the target ring
125  // f1 is an element of 'this'.
126  // set result to be the "lift" of f in the ring R, return true if this is possible.
127  // otherwise return false.
128 
129  // case: R is the coefficient ring of 'this'.
130  if (R == coefficientRing())
131  {
132  auto f = reinterpret_cast<const Poly*>(f1.get_Poly());
133  if (f->numTerms() != 1) return false;
134  auto i = f->cbegin();
135  if (monoid().is_one(i.monom()))
136  {
137  result = coefficientRing()->copy(i.coeff());
138  return true;
139  }
140  return false;
141  }
142  if (R == &m2FreeAlgebra())
143  {
144  // just copy the element into result, considered in the free algebra.
145  auto f = reinterpret_cast<const Poly*>(f1.get_Poly());
146  auto resultf = new Poly;
147  freeAlgebra().copy(*resultf, *f);
148  result = ring_elem(reinterpret_cast<void *>(resultf));
149  return true;
150  }
151 
152  // at this point, we can't lift it.
153  return false;
154 }
