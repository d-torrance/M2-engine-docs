12ConstPolyList copyMatrixToVector(const M2FreeAlgebra& F,
13 const Matrix* input)
14{
15 ConstPolyList result;
16 result.reserve(input->n_cols());
17 for (int i=0; i<input->n_cols(); i++)
18 {
19 ring_elem a = input->elem(0,i);
20 auto f = reinterpret_cast<const Poly*>(a.get_Poly());
21 auto g = new Poly;
22 F.freeAlgebra().copy(*g, *f);
23 result.push_back(g);
24 }
25 return result;

101bool M2FreeAlgebraQuotient::promote(const Ring *R, const ring_elem f1, ring_elem &result) const
102{
103 // std::cout << "called promote NC case" << std::endl;
104 // Currently the only case to handle is R = A --> this, and A is the coefficient ring of this.
105 if (R == coefficientRing())
106 {
107 result = from_coefficient(f1);
108 return true;
109 }
110 if (R == &m2FreeAlgebra())
111 {
112 auto f = reinterpret_cast<const Poly*>(f1.get_Poly());
113 auto resultf = new Poly;
114 freeAlgebraQuotient().copy(*resultf, *f);
115 freeAlgebraQuotient().normalizeInPlace(*resultf);
116 result = ring_elem(reinterpret_cast<void *>(resultf));
117 return true;
118 }
119 return false;
120}

122bool M2FreeAlgebraQuotient::lift(const Ring *R, const ring_elem f1, ring_elem &result) const
123{
124 // R is the target ring
125 // f1 is an element of 'this'.
126 // set result to be the "lift" of f in the ring R, return true if this is possible.
127 // otherwise return false.
128
129 // case: R is the coefficient ring of 'this'.
130 if (R == coefficientRing())
131 {
132 auto f = reinterpret_cast<const Poly*>(f1.get_Poly());
133 if (f->numTerms() != 1) return false;
134 auto i = f->cbegin();
135 if (monoid().is_one(i.monom()))
136 {
137 result = coefficientRing()->copy(i.coeff());
138 return true;
139 }
140 return false;
141 }
142 if (R == &m2FreeAlgebra())
143 {
144 // just copy the element into result, considered in the free algebra.
145 auto f = reinterpret_cast<const Poly*>(f1.get_Poly());
146 auto resultf = new Poly;
147 freeAlgebra().copy(*resultf, *f);
148 result = ring_elem(reinterpret_cast<void *>(resultf));
149 return true;
150 }
151
152 // at this point, we can't lift it.
153 return false;
154}
