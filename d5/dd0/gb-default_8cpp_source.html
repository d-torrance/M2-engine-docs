292 static void exponents_lcm(int nvars,
293  int dega,
294  exponents a,
295  exponents b,
296  exponents result,
297  M2_arrayint weights,
298  int &result_degree)
299 // can handle the case when a == result or b == result
300 {
301  int i;
302  int deg = dega;
303  for (i = 0; i < nvars; i++)
304  {
305  int diff = b[i] - a[i];
306  if (diff <= 0)
307  result[i] = a[i];
308  else
309  {
310  result[i] = b[i];
311  deg += diff * weights->array[i];
312  }
313  }
314  result_degree = deg;
315 }

577 static unsigned long ncalls = 0;
578 static unsigned long nloops = 0;
579 static unsigned long nsaved_unneeded = 0;
580 bool gbA::pair_not_needed(spair *p, gbelem *m)
581 {
582  /* Check the criterion: in(m) divides lcm(p).
583  * If so: check if lcm(p1,m) == lcm(p) (if so, return false)
584  * check if lcm(p2,m) == lcm(p) (if so, return false)
585  * If still here, return true.
586  */
587  int i, first, second;
588  bool firstok;
589  exponents mexp, lcm, p1exp, p2exp;
590  if (p->type != SPAIR::SPAIR_SPAIR && p->type != SPAIR::SPAIR_RING)
591  return false;
592  mexp = m->lead;
593  lcm = p->lcm;
594  if (gbelem_COMPONENT(m) != spair_COMPONENT(p)) return false;
595 
596  first = p->x.pair.i;
597  second = p->x.pair.j;
598  p1exp = gb[first]->lead;
599  p2exp =
600  gb[second]->lead; /* If a ring pair, this should index into gb array */
601 
602  ncalls++;
603  for (i = 0; i < _nvars; i++, nloops++)
604  if (mexp[i] > lcm[i]) return false;
605 
606  firstok = false;
607  for (i = 0; i < _nvars; i++)
608  {
609  if (mexp[i] == lcm[i]) continue;
610  if (p1exp[i] == lcm[i]) continue;
611  firstok = true;
612  break;
613  }
614  if (!firstok) return false;
615  for (i = 0; i < _nvars; i++)
616  {
617  if (mexp[i] == lcm[i]) continue;
618  if (p2exp[i] == lcm[i]) continue;
619  return true;
620  }
621  return false;
622 }

669 gbA::spairs::iterator gbA::choose_pair(gbA::spairs::iterator first,
670  gbA::spairs::iterator next)
671 {
672  /* a is an array of spair's, and a[first], ..., a[next-1] all have the
673  same lcm, which is a minimal monomial generator of all such lcm's.
674  Our goal is to choose a nice one, and throw away the others.

1966 void gbA::remainder_non_ZZ(POLY &f, int degf, bool use_denom, ring_elem &denom)
1967 // find the remainder of f = [g,gsyz] wrt the GB,
1968 // i.e. replace f with h[h,hsyz], st
1969 // base not ZZ:
1970 // h = f - sum(a_i * g_i), in(f) not in in(G)
1971 // hsyz = fsyz - sum(a_i * gsyz_i)
1972 // denom is unchanged
1973 // base is ZZ:
1974 // h = c*f - sum(a_i * g_i), in(f) not in in(G),
1975 // hsyz = c*fsyz - sum(a_i * gsyz_i)
1976 // but a_i,h are all polynomials with ZZ coefficients (not QQ).
1977 // denom *= c
1978 // (Here: G = (g_i) is the GB, and a_i are polynomials generated
1979 // during division).
1980 // c is an integer, and is returned as 'denom'.
1981 // Five issues:
1982 // (a) if gcd(c, coeffs(f)) becomes > 1, can we divide
1983 // c, f, by this gcd? If so, how often do we do this?
1984 // (b) do we reduce by any element of the GB, or only those whose
