98 {
99  const PolynomialRing *P = R->cast_to_PolynomialRing();
100  if (!schreyer || P == 0) return Matrix::zero(R->make_FreeModule(0), this);
101  const SchreyerOrder *S = schreyer;

103  int maxtie = 0;

106  const FreeModule *F = R->make_FreeModule(maxtie + 1);
107  MatrixConstructor mat(F, this, 0);

288 FreeModule *FreeModule::exterior(int pp) const
289 // p th exterior power
290 {
291  FreeModule *result;
292  if (pp == 0)
293  return get_ring()->make_FreeModule(1);
294  else
295  result = new_free();
296 
297  int rk = rank();
298  if (pp > rk || pp < 0) return result;
299 
300  size_t p = static_cast<size_t>(pp);
301 
302  Subset a(p, 0);
303  for (size_t i = 0; i < p; i++) a[i] = i;
304 
305  int *deg = degree_monoid()->make_one();
306  do
307  {
308  degree_monoid()->one(deg);
309 
310  for (size_t r = 0; r < p; r++)
311  degree_monoid()->mult(deg, degree(static_cast<int>(a[r])), deg);
312 
313  result->append(deg);
314  }
315  while (Subsets::increment(rk, a));
316 
317  degree_monoid()->remove(deg);
318 
319  if (schreyer != NULL) result->schreyer = schreyer->exterior(pp);
320  return result;
321 }

332  void symm1(int lastn, // can use lastn..rank()-1 in product
333  int pow) const // remaining power to take
334  {
335  if (pow == 0)
336  symm1_result->append(symm1_deg);
337  else
338  {
339  for (int i = lastn; i < F->rank(); i++)
340  {
341  // increase symm1_deg, with e_i
342  D->mult(symm1_deg, F->degree(i), symm1_deg);
343 
344  symm1(i, pow - 1);
345 
346  // decrease symm1_deg back
347  D->divide(symm1_deg, F->degree(i), symm1_deg);
348  }
349  }
350  }
