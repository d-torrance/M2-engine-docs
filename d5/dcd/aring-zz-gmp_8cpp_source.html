45void ARingZZGMP::syzygy(const ElementType& a,
46 const ElementType& b,
47 ElementType& x,
48 ElementType& y) const
49{
50 assert(!is_zero(b));
51 // First check the special cases a = 0, b = 1, -1. Other cases: use gcd.
52 if (is_zero(a))
53 {
54 set_from_long(x, 1);
55 set_zero(y);
56 return;
57 }
58 if (mpz_cmp_ui(&b, 1) == 0)
59 {
60 set_from_long(x, 1);
61 negate(y, a);
62 return;
63 }
64 if (mpz_cmp_si(&b, -1) == 0)
65 {
66 set_from_long(x, 1);
67 set(y, a);
68 return;
69 }

72 mpz_gcd(&g, &a, &b);
73 divide(y, a, g);
74 divide(x, b, g);
75 if (mpz_sgn(&x) > 0)
76 negate(y, y);
77 else
78 negate(x, x);
79 clear(g);
80}
