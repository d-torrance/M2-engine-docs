13 // Format for monomials here:
14 // a. length (in bytes) (int32)
15 // b. hash value (int32)
16 // c. component (int32)
17 // d. v1 v2 ... vd
18 // where d = length-2
19 // v1 >= v2 >= ... >= vd >= 0 are indices of variables.
20 // or, maybe also have degree before this, and other weight values...
21 // SO, general form:
22 // [length, hashval, component, w1, ..., wr, v, ..., vd]

97  void setWeightAndHash(res_packed_monomial result) const
98  {
99  // assumes the following is set already:
100  // result[0] -- length
101  // result[2] -- component
102  // result[mFirstVar..len-1] -- monomial part
103  // sets:
104  // result[1] -- hash code
105  // result[FirstWeight..mFirstVar-1] -- weight vector values
106  const int* wt = mWeightVectors.data();
107  for (int j = 0; j < mNumWeights; j++, wt += mNumVars)
108  {
109  res_monomial_word val = 0;
110  for (int* v = result + mFirstVar; v != result + *result; ++v)
111  val += wt[*v];
112  result[mFirstWeight + j] = val;
113  }
114  res_monomial_word val = 0;
115  for (int* v = result + mFirstVar; v != result + *result; ++v)
116  val += hashfcn[*v];
117  result[1] = val;
118  }
119  bool from_expvector(res_const_ntuple_monomial e,
120  component_index comp,
121  res_packed_monomial result) const
122  {
123  // Pack the vector e[0]..e[mNumVars-1],comp. Create the hash value at the
124  // same time.
125  ncalls_from_expvector++;
126  result[0] = 0; // length
127  result[1] = 0; // hash value, not set here (other than this...)
128  result[2] = comp; // component
129 
130  int next_var_slot = mFirstVar;
131 
132  for (int i = mNumVars - 1; i >= 0; --i)
133  {
134  if (e[i] > 0)
135  {
136  for (int j = 0; j < e[i]; j++)
137  {
138  result[1] += hashfcn[i];
139  result[next_var_slot] = i;
140  next_var_slot++;
141  }
142  }
143  }
144  result[0] = next_var_slot; // this is the length of this monomial
145  setWeightAndHash(result);
146  return true;
147  }

186  void to_varpower_monomial(res_const_packed_monomial m,
187  res_varpower_monomial result) const
188  {
189  // 'result' must have enough space allocated
190  ncalls_to_varpower++;
191  int len = 0;
192  res_varpower_word* r = result + 1;
193 
194  if (mFirstVar != *m) // ie, m != 1.
195  {
196  int currentvar = m[mFirstVar];
197  int deg = 1;
198  const int* mend = m + *m;
199  for (auto p = m + mFirstVar + 1; p != mend; ++p)
200  {
201  if (*p == currentvar)
202  deg++;
203  else
204  {
205  *r++ = currentvar;
206  *r++ = deg;
207  len++;
208  currentvar = *p;
209  deg = 1;
210  }
211  }
212  }
213  result[0] = len;

324  bool divide(res_const_packed_monomial m,
325  res_const_packed_monomial n,
326  res_packed_monomial result) const
327  {
328  // computes m/n, or tries to. If n divides m, then return true and set
329  // 'result' to m/n
330  // otherwise return false. In the latter case, it is possible that result
331  // will have been written into partially,
332  // but the result is not a well-formed monomial.
333  ncalls_divide++;
334  if (*m < *n) return false;
335  const int* vm = m + mFirstVar;
336  const int* vn = n + mFirstVar;
337  const int* vend = n + *n;
338  int* vresult = result + mFirstVar;
339  for (; vn != vend; ++vn)
340  {
341  if (*vn > *vm) return false;
342  if (*vn == *vm) ++vm;
343  *vresult++ = *vm++;
344  }
345  for (int i = 1; i < mFirstVar; ++i) result[i] = m[i] - n[i];
346  result[0] = m[0] - n[0] + mFirstVar;
347  return true;
348  }
