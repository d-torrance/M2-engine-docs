319 void MonomialIdeal::insert1(Nmi_node *&top, Bag *b)
320 {
321  Nmi_node **p = &top, *up = nullptr;
322  int one_element = 1;
323 
324  for (index_varpower i = b->monom().raw(); i.valid();)
325  {
326  one_element = 0;
327  int insert_var = i.var();
328  int insert_exp;
329 
330  if (*p == nullptr)
331  {
332  // make a new header node
333  *p = new_internal_mi_node(insert_var, 0, up);
334  (*p)->header = (*p)->left = (*p)->right = *p;
335  }
336  else if ((*p)->var < insert_var)
337  {
338  // make a new layer
339  Nmi_node *header_node, *zero_node;
340  header_node = new_internal_mi_node(insert_var, 0, up);
341  zero_node = new_internal_mi_node(insert_var, 0, *p);
342 
343  header_node->left = header_node->right = zero_node;
344  (*p)->down() = zero_node;
345  *p = header_node->header = zero_node->header = zero_node->left =
346  zero_node->right = header_node;
347  }
348 
349  if ((*p)->var > insert_var)
350  {
351  insert_var = (*p)->var;
352  insert_exp = 0;
353  }
354  else
355  {
356  insert_exp = i.exponent();
357  ++i;
358  }
359 
360  Nmi_node *q = (*p)->right;
361  while ((q != q->header) && (q->exp < insert_exp)) q = q->right;
362  if (q->exp != insert_exp)
363  {
364  Nmi_node *insert_node;
365 
366  if (i.valid())
367  {
368  insert_node = new_internal_mi_node(
369  insert_var, insert_exp, static_cast<Nmi_node*>(nullptr));
370  q->insert_to_left(insert_node);
371  q = insert_node;
372  }
373  else
374  {
375  insert_node = new_leaf_mi_node(insert_var, insert_exp, b);
376  q->insert_to_left(insert_node);
377  return;
378  }
379  }
380 
381  up = q;
382  p = &(q->down());
383  }
384  if (one_element)
385  {
386  // insert a header node and a var/exp = 0/0 leaf
387  top = new_internal_mi_node(0, 0, static_cast<Nmi_node *>(nullptr));
388  Nmi_node *leaf_node = new_leaf_mi_node(0, 0, b);
389  top->left = top->right = leaf_node;
390  top->header = leaf_node->header = leaf_node->left = leaf_node->right =
391  top;
392  }
393 }

395 void MonomialIdeal::remove1(Nmi_node *p)
396 {
397  assert(p != nullptr);
398  assert(p->tag == Nmi_node::leaf);
399  p->baggage() = nullptr;
400  count -= 2;
401 
402  for (; p != nullptr;)
403  {
404  p->left->right = p->right;
405  p->right->left = p->left;
406  Nmi_node *q = p->header;
407  p->left = p->right = nullptr;
408  delete_mi_node(p);
409 
410  if (q->right == q->header) // only the header is left, so delete it
411  {
412  p = q->down();
413  q->down() = nullptr;
414  if (p != nullptr) p->down() = nullptr;
415  delete_mi_node(q);
416  continue;
417  }
418 
419  if (q->left != q->right) return;
420 
421  if (q->left->exp > 0) return;
422 
423  Nmi_node *dad = q->down();
424  if (q->left->tag == Nmi_node::leaf)
425  {
426  // set parent of q to be a leaf with baggage of q->left
427  // since this is a leaf, dad should be non null
428  assert(dad != nullptr);
429  dad->tag = Nmi_node::leaf;
430  dad->baggage() = q->left->baggage();
431  }
432  else
433  {
434  // set parent of q to be node pointing to q->left->down
435  q->left->down()->down() = dad;
436  if (dad != nullptr)
437  dad->down() = q->left->down();
438  else
439  mi = q->left->down();
440  q->left->down() = nullptr;
441  }
442  q->down() = nullptr;
443  delete_mi_node(q); // Deletes both nodes q, q->left.
444  return;
445  }
446  if (p == nullptr) mi = nullptr;
447 }

522 int MonomialIdeal::debug_check(Nmi_node *const p,
523  const Nmi_node *const up) const
524 // Returns the number of leaves at tree with root p.
525 // Make sure that the list header is constructed ok, that the
526 // left/right pointers are ok on this level, that the
527 // var, exp, values in this train are correct.
528 // Then loop through, checking each node (recursively) and each leaf
529 {
530  Nmi_node *q;
531  // First check the node 'p' itself
532  assert(p != nullptr);
533  assert(p->var >= 0);
534  if (up != nullptr) assert(p->var < up->var);
535  assert(p->header == p);
536  assert(p->tag == Nmi_node::node);
537  assert(p->down() == up);
538  assert(p->left != nullptr);
539  assert(p->right != nullptr);
540 
541  // Now loop through each element in left/right chain, checking that
542  // v, e, left, right values are consistent.
543  for (q = p->left; q != p; q = q->left)
544  {
545  assert(q->left != nullptr);
546  assert(q->right != nullptr);
547  assert(q->header == p);
548  assert(q->right->left == q);
549  assert(q->left->right == q);
550  assert(q->var == p->var);
551  assert((q->right == p) || (q->exp < q->right->exp));
552  assert(q->exp >= 0);
553  }
554 
555  // Now loop through again, this time descending into nodes
556  int c = 0;
557  for (q = p->right; q != p; q = q->right)
558  if (q->tag == Nmi_node::node)
559  c += debug_check(q->down(), q);
560  else
561  c++;
562  return c;
563 }

576 bool MonomialIdeal::isWellFormed() const
577 {
578  if (mi == nullptr) return true; // nothing else to check.
579 
580  Nmi_node* p = mi;
581  while (p != nullptr)
582  {
583  // check the current node
584  if (p->left == nullptr) throw exc::engine_error("left node link is null");
585  if (p->right == nullptr) throw exc::engine_error("right node link is null");
586  if (p->header == nullptr) throw exc::engine_error("header node link is null");
587  if (p->tag == Nmi_node::node and p != mi and p->val.down == nullptr) throw exc::engine_error("down node link is null");
588 
589  // now go to the next node
590  // if this is a leaf, go right.
591  // if this is a header, go down (up), and one left
592  // if this is an internal node, go down (to subtree).
593  if (p->tag == Nmi_node::node and p->header != p)
594  p = p->val.down;
595  else if (p->tag == Nmi_node::node and p->header == p)
596  {
597  // this is a header node (head of the double linked list at this level)
598  // Let's check all of the elements in the double ring at this level.
599  for (Nmi_node* q = p->right; q != p; q = q->right)
600  {
601  if (p->var != q->var) throw exc::engine_error("variable index is not consistent");
602  if (q->left->right != q) throw exc::engine_error("the double link list is inconsistent");
603  if (q->left != p and q->left->exp >= q->exp) throw exc::engine_error("exponents are not increasing going to the right");
604  }
605 
606  // Now we continue
607  p = p->val.down;
608  if (p != nullptr)
609  p = p->right;
610  }
611  else if (p->tag == Nmi_node::leaf)
612  {
613  p = p->right;
614  }
615  }
616 
617  return true;
618 }

666 MonomialIdeal *MonomialIdeal::intersect(const MonomialIdeal &J) const
667 {
668  // The idea: take the elements of 'this'
669  // for each: if the element is in J, then keep it directly.
670  // otherwie compute the lcm's.
671  VECTOR(Bag*) new_elems;
672  for (Bag& a : *this)
673  {
674  Bag *c;
675  if (J.search(a.monom().raw(), c))
676  {
677  new_elems.push_back(new Bag(a));
678  }
679  else
680  for (Bag& b : J)
681  {
682  Bag *new_elem = new Bag(a.basis_elem());
683  varpower::lcm(a.monom().raw(),
684  b.monom().raw(),
685  new_elem->monom());
686  new_elems.push_back(new_elem);
687  }
688  }
689  MonomialIdeal *result = new MonomialIdeal(get_ring(), new_elems);
690  GC_reachable_here(&J);
691  return result;
692 }

918 static void borel1(VECTOR(Bag *) &result, int *m, int loc, int nvars)
919 {
920  if (loc == 0)
921  {
922  Bag *b = new Bag();
923  varpower::from_ntuple(nvars, m, b->monom());
924  result.push_back(b);
925  }
926  else
927  {
928  int a = m[loc];
929  for (int i = 0; i <= a; i++)
930  {
931  borel1(result, m, loc - 1, nvars);
932  m[loc]--;
933  m[loc - 1]++;
934  }
935  m[loc] += a + 1;
936  m[loc - 1] -= a + 1;
937  }
