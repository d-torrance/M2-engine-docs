192  void invert(ElementType& result, const ElementType& a) const
193  // we silently assume that a != 0. If it is, result is set to a^0, i.e. 1
194  {
195  mpfr_t p, denom;
196  mpfr_init2(p, get_precision());
197  mpfr_init2(denom, get_precision());
198 
199  if (mpfr_cmpabs(&a.re, &a.im) >= 0)
200  {
201  // double p = &a.im/&a.re;
202  // double denom = &a.re + p * &a.im;
203  // &result.re = 1.0/denom;
204  // &result.im = - p/denom;
205 
206  mpfr_div(p, &a.im, &a.re, MPFR_RNDN);
207  mpfr_mul(denom, p, &a.im, MPFR_RNDN);
208  mpfr_add(denom, denom, &a.re, MPFR_RNDN);
209  mpfr_si_div(&result.re, 1, denom, MPFR_RNDN);
210  mpfr_div(&result.im, p, denom, MPFR_RNDN);
211  mpfr_neg(&result.im, &result.im, MPFR_RNDN);
212  }
213  else
214  {
215  // double p = &a.re/&a.im;
216  // double denom = &a.im + p * &a.re;
217  // &result.re = p/denom;
218  // &result.im = -1.0/denom;
219 
220  mpfr_div(p, &a.re, &a.im, MPFR_RNDN);
221  mpfr_mul(denom, p, &a.re, MPFR_RNDN);
222  mpfr_add(denom, denom, &a.im, MPFR_RNDN);
223  mpfr_si_div(&result.im, 1, denom, MPFR_RNDN);
224  mpfr_neg(&result.im, &result.im, MPFR_RNDN);
225  mpfr_div(&result.re, p, denom, MPFR_RNDN);
226  }
227 
228  mpfr_clear(p);
229  mpfr_clear(denom);
230  }

335  void divide(ElementType& res,
336  const ElementType& a,
337  const ElementType& b) const
338  {
339  mpfr_t p, denom;
340  mpfr_init2(p, get_precision());
341  mpfr_init2(denom, get_precision());
342  Element result(*this);
343 
344  if (mpfr_cmpabs(&b.re, &b.im) >= 0)
345  {
346  // for v = c + d*i,
347  // p = d/c
348  // c+di = c(1+p*i), so denom is c(1+p^2)
349  // which is c + d*p
350 
351  // double p = v.im/v.re;
352  // double denom = v.re + p * v.im;
353  // result.re = (u.re + p*u.im)/denom;
354  // result.im = (u.im - p*u.re)/denom;
355 
356  mpfr_div(p, &b.im, &b.re, MPFR_RNDN);
357  mpfr_mul(denom, p, &b.im, MPFR_RNDN);
358  mpfr_add(denom, denom, &b.re, MPFR_RNDN);
359 
360  mpfr_mul(&result.value().re, p, &a.im, MPFR_RNDN);
361  mpfr_add(&result.value().re, &result.value().re, &a.re, MPFR_RNDN);
362  mpfr_div(&result.value().re, &result.value().re, denom, MPFR_RNDN);
363 
364  mpfr_mul(&result.value().im, p, &a.re, MPFR_RNDN);
365  mpfr_neg(&result.value().im, &result.value().im, MPFR_RNDN);
366  mpfr_add(&result.value().im, &result.value().im, &a.im, MPFR_RNDN);
367  mpfr_div(&result.value().im, &result.value().im, denom, MPFR_RNDN);
368  }
369  else
370  {
371  // double p = v.re/v.im;
372  // double denom = v.im + p * v.re;
373  // result.re = (u.re * p + u.im)/denom;
374  // result.im = (-u.re + p * u.im)/denom;
375 
376  mpfr_div(p, &b.re, &b.im, MPFR_RNDN);
377  mpfr_mul(denom, p, &b.re, MPFR_RNDN);
378  mpfr_add(denom, denom, &b.im, MPFR_RNDN);
379 
380  mpfr_mul(&result.value().re, p, &a.re, MPFR_RNDN);
381  mpfr_add(&result.value().re, &result.value().re, &a.im, MPFR_RNDN);
382  mpfr_div(&result.value().re, &result.value().re, denom, MPFR_RNDN);
383 
384  mpfr_mul(&result.value().im, p, &a.im, MPFR_RNDN);
385  mpfr_sub(&result.value().im, &result.value().im, &a.re, MPFR_RNDN);
386  mpfr_div(&result.value().im, &result.value().im, denom, MPFR_RNDN);
387  }
388  mpfr_clear(p);
389  mpfr_clear(denom);
390  set(res, result);
391  }

407  void power(ElementType& result, const ElementType& a, int n) const
408  {
409  ElementType curr_pow;
410  init(curr_pow);
411  set_from_long(result, 1);
412  if (n == 0)
413  {
414  }
415  else if (n < 0)

417  n = -n;
418  invert(curr_pow, a);
419  }
420  else
421  {
422  set(curr_pow, a);
423  }
424  while (n > 0)
425  {
426  if (n % 2)

428  mult(result, result, curr_pow);
