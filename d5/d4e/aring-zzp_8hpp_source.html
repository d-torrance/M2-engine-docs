219 void divide(elem &result, elem a, elem b) const
220 {
221 if (b == 0)
222 throw exc::division_by_zero_error();

224 {
225 int c = a - b;
226 if (c <= 0) c += p1;
227 result = c;
228 }
229 else
230 result = 0;

233 void power(elem &result, elem a, int n) const
234 {
235 if (a != 0)
236 {
237 result = (a * n) % p1;
238 if (result <= 0) result += p1;
239 }
240 else
241 {
242 // a == 0
243 if (n == 0) result = p1; // the element 1 in this ring.
244 else if (n > 0) result = 0;
245 else throw exc::division_by_zero_error();
246 }
247 }

249 void power_mpz(elem &result, elem a, mpz_srcptr n) const
250 {
251 if (a != 0)
252 {
253 int n1 = static_cast<int>(mpz_fdiv_ui(n, p1));
254 power(result, a, n1);
255 }
256 else
257 {
258 if (mpz_sgn(n) == 0) result = p1; // the element 1 in this ring.
259 else if (mpz_sgn(n) < 0) throw exc::division_by_zero_error();
260 else result = 0; // result is 0 in the ring.
