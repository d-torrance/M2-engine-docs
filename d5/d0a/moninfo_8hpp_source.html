440  const_packed_monomial lcm) const
441  // Returns true if the corresponding pair (p1,p2) could be removed
442  // This is essentially the Buchberger-Moeller criterion
443  // Assumptions: lcm(p1,p2) = lcm.
444  // Here is the criterion:
445  // (a) if component(lcm) != component(m) return false
446  // (b) if m does not divide lcm, return false
447  // (c) need that (A) lcm(p1,m) != lcm and that (B) lcm(p2,m) != lcm
448  // (in these two cases, we will have already removed one of the other
449  // two pairs: (p1,m), (p2,m). Note that in any case,
450  // if (b) holds, then lcm(p1,m) divides lcm, same with lcm(p2,m).
451  // if A and B then return true
452  {
453  // TODO: MES: this maybe should check that lead components of m and p1 and
454  // p2 are the same...
455  ncalls_unneccesary++;
456  bool A = false;
457  bool B = false;
458  m += firstvar;
459  p1 += firstvar;
460  p2 += firstvar;
461  lcm += firstvar;
462  for (int i = 0; i < nvars; i++)
463  {
464  if (m[i] > lcm[i]) return false;
465  if (m[i] == lcm[i]) continue;
466  if (!A && p1[i] < lcm[i])
467  {
468  A = true;
469  continue;
470  }
471  if (!B && p2[i] < lcm[i])
472  {
473  B = true;
474  }
475  }
476  return (A && B);
477  }
