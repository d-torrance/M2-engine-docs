35ARingZZpFFPACK::ElementType ARingZZpFFPACK::computeGenerator() const
36{
37 for (UTT currIntElem = 2; currIntElem < mCharac; currIntElem++)
38 {
39 ElementType currElem;
40 set_from_long(currElem, currIntElem);
41 bool found = true;
42 ElementType tmpElem = currElem;
43 for (UTT count = 0; count < mCharac - 2; count++)
44 {
45 mult(tmpElem, tmpElem, currElem);
46 if (is_equal(currElem, tmpElem)) found = false;
47 }
48 if (found)
49 {
50 std::cerr << "generator = " << currElem << std::endl;
51 return currElem;
52 }
53 }
54 throw exc::internal_error("program logic in ffpack code is wrong if we get to this error");
55}

185void ARingZZpFFPACK::power(ElementType &result,
186 const ElementType a,
187 STT n) const
188{
189 if (is_zero(a))
190 {
191 if (n < 0)
192 throw exc::division_by_zero_error();
193 else if (n == 0)
194 set_from_long(result, 1);
195 else
196 set_zero(result);
197 return;
198 }
199 ElementType base;
200 set(base, a);
201 if (n < 0)
202 {
203 invert(base, base);
204 n = -n;
205 }
206 n = n % (mCharac - 1);
207 set_from_long(result, 1);
208 if (n == 0) return;
209
210 // Now use doubling algorithm
211 for (;;)
212 {
213 if ((n % 2) != 0) mFfpackField.mulin(result, base); // result *= base
214 n >>= 1;
215 if (n == 0)
216 return;
217 else
218 mFfpackField.mulin(base, base); // base = base^2
219 }
220}

224void ARingZZpFFPACK::power_mpz(ElementType &result,
225 const ElementType a,
226 mpz_srcptr n) const
227{
228 if (is_zero(a))
229 {
230 if (mpz_sgn(n) < 0)
231 throw exc::division_by_zero_error();
232 else if (mpz_sgn(n) == 0)
233 set_from_long(result, 1);
234 else
235 set_zero(result);
236 }
237 else
238 {
239 // a != 0
240 STT n1 = static_cast<STT>(mpz_fdiv_ui(n, mFfpackField.cardinality() - 1));
241 power(result, a, n1);
242 }
243}
