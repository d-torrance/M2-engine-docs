20 void F4toM2Interface::from_M2_vec(const VectorArithmetic* VA,
21  const MonomialInfo *MI,
22  const FreeModule *F,
23  vec v,
24  GBF4Polynomial &result)
25 {
26  const PolynomialRing *R = F->get_ring()->cast_to_PolynomialRing();
27  const Monoid *M = R->getMonoid();
28 
29  ring_elem denom;
30  gbvector *f = R->translate_gbvector_from_vec(F, v, denom);
31  GBRing *GR = R->get_gb_ring();
32  int n = GR->gbvector_n_terms(f);
33 
34  int *exp = new int[M->n_vars() + 1]; // newarray_atomic(int, M->n_vars() + 1);
35  ntuple_word *lexp = new ntuple_word[M->n_vars() + 1]; // newarray_atomic(ntuple_word, M->n_vars() + 1);
36 
37  result.len = n;
38  std::vector<ring_elem> relem_array;
39  result.monoms = newarray_atomic(monomial_word, n * MI->max_monomial_size());
40  n = 0;
41  monomial_word *nextmonom = result.monoms;
42  for (gbvector *t = f; t != nullptr; t = t->next)
43  {
44  relem_array.push_back(t->coeff);
45  M->to_expvector(t->monom, exp);
46  for (int a = 0; a < M->n_vars(); a++) lexp[a] = exp[a];
47  MI->from_expvector(
48  lexp,
49  t->comp - 1,
50  nextmonom); // gbvector components are shifted up by one
51  nextmonom += MI->monomial_size(nextmonom);
52  n++;
53  }
54  result.coeffs = VA->elementArrayFromContainer(relem_array);
55  delete [] exp;
56  delete [] lexp;
57 }

59 void F4toM2Interface::poly_set_degrees(const VectorArithmetic* VA,
60  const MonomialInfo *MI,
61  const M2_arrayint wts,
62  const GBF4Polynomial &f,
63  int &deg_result,
64  int &alpha)
65 {
66  const monomial_word *w = f.monoms;
67  monomial_word leaddeg = MI->monomial_weight(w, wts);
68  monomial_word deg = leaddeg;
69 
70  for (int i = 1; i < f.len; i++)
71  {
72  w = w + MI->monomial_size(w);
73  monomial_word degi = MI->monomial_weight(w, wts);
74  if (degi > deg) deg = degi;
75  }
76  alpha = static_cast<int>(deg - leaddeg);
77  deg_result = static_cast<int>(deg);
78 }

96 vec F4toM2Interface::to_M2_vec(const VectorArithmetic* VA,
97  const MonomialInfo *MI,
98  const GBF4Polynomial &f,
99  const FreeModule *F)
100 {
101  const PolynomialRing *R = F->get_ring()->cast_to_PolynomialRing();
102  const Monoid *M = R->getMonoid();
103 
104  monomial m1 = M->make_one();
105 
106  Nterm **comps = newarray(Nterm *, F->rank());
107  Nterm **last = newarray(Nterm *, F->rank());
108  for (int i = 0; i < F->rank(); i++)
109  {
110  comps[i] = 0;
111  last[i] = 0;
112  }
113 
114  int *exp = newarray_atomic(int, M->n_vars() + 1);
115  ntuple_word *lexp = newarray_atomic(ntuple_word, M->n_vars() + 1);
116 
117  const monomial_word *w = f.monoms;
118  for (int i = 0; i < f.len; i++)
119  {
120  long comp;
121  MI->to_expvector(w, lexp, comp);
122  w = w + MI->monomial_size(w);
123  for (int a = 0; a < M->n_vars(); a++) exp[a] = static_cast<int>(lexp[a]);
124  M->from_expvector(exp, m1);
125  ring_elem a = VA->ringElemFromElementArray(f.coeffs, i);
126  Nterm *g = R->make_flat_term(a, m1);
127  g->next = nullptr;
128  if (last[comp] == 0)
129  {
130  comps[comp] = g;
131  last[comp] = g;
132  }
133  else
134  {
135  last[comp]->next = g;
136  last[comp] = g;
137  }
138  }
139  vec result = nullptr;
140  for (int i = 0; i < F->rank(); i++)
141  {
142  if (comps[i] != 0)
143  {
144  vec v = R->make_vec(i, comps[i]);
145  R->add_vec_to(result, v);
146  comps[i] = 0;
147  last[i] = 0;
148  }
149  }
150 
151  return result;
152 }
