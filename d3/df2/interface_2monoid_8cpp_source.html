59 static void write_weights(std::vector<int>& grading,
60  int nvars,
61  int firstvar,
62  int* wts,
63  int nwts)
64 // place nvars ints into grading: 0 ... 0 wts[0] wts[1] ... wts[nwts-1] 0 ....
65 // 0
66 // where wts[0] is in the 'firstvar' location. If wts is NULL, treat it as the
67 // vector with nwts '1's.
68 {
69  for (int i = 0; i < firstvar; i++) grading.push_back(0);
70  if (wts == nullptr)
71  for (int i = 0; i < nwts; i++) grading.push_back(1);
72  else
73  for (int i = 0; i < nwts; i++) grading.push_back(wts[i]);
74  for (int i = firstvar + nwts; i < nvars; i++) grading.push_back(0);
75 }

77 bool monomialOrderingToMatrix(
78  const struct MonomialOrdering& mo,
79  std::vector<int>& mat,
80  bool& base_is_revlex,
81  int& component_direction, // -1 is Down, +1 is Up, 0 is not present
82  int& component_is_before_row // -1 means: at the end. 0 means before the
83  // order.
84  // and r means considered before row 'r' of the matrix.
85  )
86 {
87  // a false return value means an error has occurred.
88  int nvars = rawNumberOfVariables(&mo);
89  base_is_revlex = true;
90  enum LastBlock { LEX, REVLEX, WEIGHTS, NONE };
91  LastBlock last = NONE;
92  int nwts = 0; // local var used in MO_WEIGHTS section
93  int nrows = 0;
94  int firstvar = 0;
95  component_direction = 0;
96  component_is_before_row =
97  -2; // what should the default value be? Probably: -1.
98  size_t last_element = 0; // The vector 'mat' will be resized back to this
99  // value if the last part of the order is lex or
100  // revlex.
101  for (int i = 0; i < mo.len; i++)
102  {
103  mon_part p = mo.array[i];
104  switch (p->type)
105  {
106  case MO_LEX:
107  case MO_LEX2:
108  case MO_LEX4:
109  // printf("lex %d\n", p->nvars);
110  last_element = mat.size();
111  for (int j = 0; j < p->nvars; j++)
112  {
113  write_row(mat, nvars, firstvar + j, 1);
114  }
115  last = LEX;
116  firstvar += p->nvars;
117  nrows += p->nvars;
118  break;
119  case MO_GREVLEX:
120  case MO_GREVLEX2:
121  case MO_GREVLEX4:
122  // printf("grevlex %d %ld\n", p->nvars, p->wts);
123  write_weights(mat, nvars, firstvar, p->wts, p->nvars);
124  last_element = mat.size();
125  for (int j = p->nvars - 1; j >= 1; --j)
126  {
127  write_row(mat, nvars, firstvar + j, -1);
128  }
129  last = REVLEX;
130  firstvar += p->nvars;
131  nrows += p->nvars;
132  break;
133  case MO_GREVLEX_WTS:
134  case MO_GREVLEX2_WTS:
135  case MO_GREVLEX4_WTS:
136  // printf("grevlex_wts %d %ld\n", p->nvars, p->wts);
137  write_weights(mat, nvars, firstvar, p->wts, p->nvars);
138  last_element = mat.size();
139  for (int j = p->nvars - 1; j >= 1; --j)
140  {
141  write_row(mat, nvars, firstvar + j, -1);
142  }
143  last = REVLEX;
144  firstvar += p->nvars;
145  nrows += p->nvars;
146  break;
147  case MO_REVLEX:
148  // printf("revlex %d\n", p->nvars);
149  last_element = mat.size();
150  for (int j = p->nvars - 1; j >= 0; --j)
151  {
152  write_row(mat, nvars, firstvar + j, -1);
153  }
154  last = REVLEX;
155  firstvar += p->nvars;
156  nrows += p->nvars;
157  break;
158  case MO_WEIGHTS:
159  // printf("matsize= %d weights %d p->wts=%lu\n", mat.size(),
160  // p->nvars, p->wts);
161  nwts = (p->nvars > nvars ? nvars : p->nvars);
162  write_weights(mat, nvars, 0, p->wts, nwts);
163  nrows++;
164  last_element = mat.size();
165  last = WEIGHTS;
166  break;
167  case MO_LAURENT:
168  case MO_LAURENT_REVLEX:
169  case MO_NC_LEX:
170  return false;
171  break;
172  case MO_POSITION_UP:
173  component_direction = 1;
174  component_is_before_row = nrows;
175  break;
176  case MO_POSITION_DOWN:
177  component_direction = -1;
178  component_is_before_row = nrows;
179  break;
180  default:
181  // DO nothing
182  break;
183  }
184  }
185  if (last == LEX)
186  {
187  // last block was lex, so use lex tie-breaker
188  mat.resize(last_element);
189  if (nrows == component_is_before_row) component_is_before_row = -1;
190  base_is_revlex = false;
191  }
192  else if (last == REVLEX)
193  {
194  // last block was revlex, so use revlex tie-breaker
195  if (nrows == component_is_before_row) component_is_before_row = -1;
196  mat.resize(last_element);
197  }
198  else
199  {
200  // last block is a weight vector, so use revlex as the tie-breaker.
201  // nothing to change here.
202  }
203  return true;
204 }

206 M2_arrayint rawMonomialOrderingToMatrix(const struct MonomialOrdering* mo)
207 {
208  bool base;
209  std::vector<int> mat;
210  M2_arrayint result = nullptr;
211  int component_is_before_row = 0;
212  int component_direction = 0;
213  if (monomialOrderingToMatrix(
214  *mo, mat, base, component_direction, component_is_before_row))
215  {
216  int top = static_cast<int>(mat.size());
217  result = M2_makearrayint(top + 3);
218  for (int i = 0; i < top; i++) result->array[i] = mat[i];
219  result->array[top] = (base ? 1 : 0);
220  result->array[top + 1] = component_direction;
221  result->array[top + 2] = component_is_before_row;
222  }
223  return result;
224 }
