224  // insert (m,comp) into the set. Returns true if it is inserted, i.e. it isn't already in the set.
225  bool insert(Monom m, int comp)
226  {
227  size_t sz = ModuleMonom::sizeOfCorrespondingModuleMonom(m);
228  std::pair<int*, int*> mon { mArena.allocArrayNoCon<int>(sz) };
229  ModuleMonom a = monomToModuleMonom(m, comp, mon);
230  auto result = mHash.insert(a);
231  bool new_elem = result.second;
232  if (new_elem)
233  {
234  a.setIndex(mElements.size());
235  mElements.push_back(a);
236  }
237  else
238  {
239  mArena.freeTop(a + 0);
240  }
241  return new_elem;
242  }

244  // if (m,comp) is found (and is a monomial with index idx), return {idx, true}
245  // if it is not found, return {-1, false}
246  std::pair<int,bool> find(Monom m, int comp)
247  {
248  size_t sz = ModuleMonom::sizeOfCorrespondingModuleMonom(m);
249  std::pair<int*, int*> mon { mArena.allocArrayNoCon<int>(sz) };
250  ModuleMonom a = monomToModuleMonom(m, comp, mon);
251  auto result = mHash.find(a); // returns iterator pointing to value, or mHash.end()
252  bool found = (result != mHash.end());
253  int idx = (found ? result->index() : -1);
254  mArena.freeTop(mon.first);
255  return {idx, found};
256  }
