84// Hash table for monomials (or ...)
85// Functions:
86// insert(monomial,comp): returns index if (monomial,comp) is in table. Otherwise sets value to next size.
87// lookup(monomial,comp): returns index.
88// A function elsewhere will loop through the monomials in a matrix, creating the hash table of these monomials.
89// actually: there might be translation from the monomials.
90// basically: (1) loop through monomials, add (transformed valued) one by one, when needed.
91// (2) loop through monomials, add monomial at row given by 'value' in hash table.
92// Structure itself:
93// hash table (probably from stl)
94// memtailor memory area for monomials.
95// number of elements contained in here
96// Memory management:
97// reserve space for a monomial (and/or copy it in).
98// jettison last one if not needed.

224 // insert (m,comp) into the set. Returns true if it is inserted, i.e. it isn't already in the set.
225 bool insert(Monom m, int comp)
226 {
227 size_t sz = ModuleMonom::sizeOfCorrespondingModuleMonom(m);
228 std::pair<int*, int*> mon { mArena.allocArrayNoCon<int>(sz) };
229 ModuleMonom a = monomToModuleMonom(m, comp, mon);
230 auto result = mHash.insert(a);
231 bool new_elem = result.second;
232 if (new_elem)
233 {
234 a.setIndex(mElements.size());
235 mElements.push_back(a);
236 }
237 else
238 {
239 mArena.freeTop(a + 0);
240 }
241 return new_elem;
242 }

244 // if (m,comp) is found (and is a monomial with index idx), return {idx, true}
245 // if it is not found, return {-1, false}
246 std::pair<int,bool> find(Monom m, int comp)
247 {
248 size_t sz = ModuleMonom::sizeOfCorrespondingModuleMonom(m);
249 std::pair<int*, int*> mon { mArena.allocArrayNoCon<int>(sz) };
250 ModuleMonom a = monomToModuleMonom(m, comp, mon);
251 auto result = mHash.find(a); // returns iterator pointing to value, or mHash.end()
252 bool found = (result != mHash.end());
253 int idx = (found ? result->index() : -1);
254 mArena.freeTop(mon.first);
255 return {idx, found};
256 }

280 // hash table
281 // monomial arena
282 // list of pointers to these 'monomials' (so we can sort them?)
283 //
284 // data members:
285 // 1. arena
286 // 2. std::vector of int*'s into arena, unique monomials
287 // 3. unordered_set
288 // each monomial has a spot for its value. When inserted into the hash table, this value is set.
289 //
290 // Conf:
291 // hash function
292 // equality function
293 // comparison function on monomials (needs to be a different class, I think)
294 // Monomial ops:
295 // (monomial,comp) --> moduleMonomial (and back), put in space for int:
296 // --> [len, value, comp, v1, ..., vn] (len includes value)
297 // --> or [value, comp, len, deg, v1, ..., vr] (hash, length, equality, compare) all work on these.
298 //
299 // ops:
300 // H[mon,comp]: first copy monomial to arena, insert it. If there, pop monom, otherwise set 'value' field
301 // lookup: first copy monomial,comp to arena, lookup, then pop copied monomial.
302 // sortMonomials: sorts mons, sets 'value' field in each monomial.
303 // value(H[mon]) = 0-th int.
304private:
305 Conf mConf;
306 memt::Arena mArena; // memory area for monomials
307 Set mHash; // will need to put in hash function, compare fcn from Conf.
308 std::vector<ModuleMonom> mElements; // each monomial points into mArena.
309public:
310 auto begin() const -> decltype(mElements.begin()) { return mElements.begin(); }
311 auto end() const -> decltype(mElements.end()) { return mElements.end(); }
312};
