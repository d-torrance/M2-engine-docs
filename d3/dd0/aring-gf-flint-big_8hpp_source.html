311  void power_mpz(ElementType& result, const ElementType& a, mpz_srcptr n) const
312  {
313  if (mpz_sgn(n) < 0 and is_zero(a))
314  throw exc::division_by_zero_error();
315 
316  ElementType base;
317  init(base);
318  if (mpz_sgn(n) < 0)
319  invert(base, a);
320  else
321  copy(base, a);
322 
323  mpz_t abs_n;
324  mpz_init(abs_n);
325  mpz_abs(abs_n, n);
326 
327  fmpz_t fn;
328  fmpz_init_set_readonly(fn, abs_n);
329  fq_nmod_pow(&result, &base, fn, mContext);
330  fmpz_clear_readonly(fn);
331  mpz_clear(abs_n);
332  clear(base);
333  }
