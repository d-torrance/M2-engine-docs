310  void power_mpz(ElementType& result, const ElementType& a, mpz_srcptr n) const
311  {
312  if (mpz_sgn(n) < 0 and is_zero(a))
313  throw exc::division_by_zero_error();
314 
315  ElementType base;
316  init(base);
317  if (mpz_sgn(n) < 0)
318  invert(base, a);
319  else
320  copy(base, a);
321 
322  mpz_t abs_n;
323  mpz_init(abs_n);
324  mpz_abs(abs_n, n);
325 
326  fmpz_t fn;
327  fmpz_init_set_readonly(fn, abs_n);
328  fq_nmod_pow(&result, &base, fn, mContext);
329  fmpz_clear_readonly(fn);
330  mpz_clear(abs_n);
331  clear(base);
332  }
