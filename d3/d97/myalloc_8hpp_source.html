59template <typename T>
60class StatsAllocator {
61public:
62 typedef T value_type;
63
64 StatsAllocator() noexcept {}
65
66 template <typename U>
67 StatsAllocator(const StatsAllocator<U>&) noexcept {}
68
69 T* allocate(size_t num)
70 {
71 AllocLogger::logAlloc(num, sizeof(T));
72 return static_cast<T*>(::operator new (num * sizeof(T)));
73 }
74
75 void deallocate(T* p, std::size_t num)
76 {
77 AllocLogger::logDealloc(num, sizeof(T));

79 }
