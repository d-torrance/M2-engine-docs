129 int F4GB::find_or_append_column(packed_monomial m)
130 {
131  packed_monomial new_m;
132  if (H.find_or_insert(m, new_m)) return static_cast<int>(new_m[-1]);
133  // At this point, m is a new monomial to be placed as a column
134  m = next_monom;
135  B.intern(1 + M->monomial_size(m));
136  next_monom = B.reserve(1 + M->max_monomial_size());
137  next_monom++;
138  return new_column(m);

141 int F4GB::mult_monomials(packed_monomial m, packed_monomial n)
142 {
143  // We already have allocated space for a monomial
144  // Do the multiply
145  // Look it up in the hashtable
146  // If it is there, return its column
147  // If not, increment our memory block, and insert a new column.
148  packed_monomial new_m;
149  M->unchecked_mult(m, n, next_monom);
150  if (H.find_or_insert(next_monom, new_m))
151  return static_cast<int>(
152  new_m[-1]); // monom exists, don't save monomial space
153  m = next_monom;
154  B.intern(1 + M->monomial_size(m));
155  next_monom = B.reserve(1 + M->max_monomial_size());
156  next_monom++;
157  return new_column(m);
158 }

160 void F4GB::load_gen(int which)
161 {
162  GBF4Polynomial &g = gens[which]->f;
163 
164  row_elem r{};
165  r.monom = nullptr; // This says that this element corresponds to a generator
166  r.elem = which;
167  r.len = g.len;
168  r.comps = Mem->components.allocate(g.len);
169  // r.coeffs is already initialized to [nullptr].
170 
171  monomial_word *w = g.monoms;
172  for (int i = 0; i < g.len; i++)
173  {
174  M->copy(w, next_monom);
175  r.comps[i] = find_or_append_column(next_monom);
176  w += M->monomial_size(w);
177  }
178 
179  mat->rows.push_back(r);
180 }

203 void F4GB::process_column(int c)
204 {
205  /* If this column has been handled before, return.
206  Otherwise, find a GB element whose lead term
207  divides this monomial, and either mark this column
208  as not an initial element, OR append a row
209  */
210 
211  column_elem &ce = mat->columns[c];
212  if (ce.head >= -1) return;
213  int32_t which;
214  bool found = lookup->find_one_divisor_packed(M, ce.monom, which);
215  if (found)
216  {
217  packed_monomial n = next_monom;
218  M->unchecked_divide(ce.monom, gb[which]->f.monoms, n);
219  B.intern(1 + M->monomial_size(n));
220  next_monom = B.reserve(1 + M->max_monomial_size());
221  next_monom++;
222  // M->set_component(which, n);
223  ce.head = INTSIZE(mat->rows);
224  load_row(n, which);
225  }
226  else
227  ce.head = -1;
228 }

230 void F4GB::process_s_pair(spair *p)
231 {
232  int c;
233 
234  switch (p->type)
235  {
236  case F4_SPAIR_SPAIR:
237  {
238  packed_monomial n = next_monom;
239  M->unchecked_divide(p->lcm, gb[p->i]->f.monoms, n);
240  B.intern(1 + M->monomial_size(n));
241  next_monom = B.reserve(1 + M->max_monomial_size());
242  next_monom++;
243 
244  load_row(n, p->i);
245  c = mat->rows[mat->rows.size() - 1].comps[0];
246 
247  if (mat->columns[c].head >= -1)
248  n_lcmdups++;
249  else
250  {
251  // In this situation, we load the other half as a reducer
252  n = next_monom;
253  M->unchecked_divide(p->lcm, gb[p->j]->f.monoms, n);
254  B.intern(1 + M->monomial_size(n));
255  next_monom = B.reserve(1 + M->max_monomial_size());
256  next_monom++;
257  load_row(n, p->j);
258 
259  mat->columns[c].head = INTSIZE(mat->rows) - 1;
260  }
261  break;
262  }
263  case F4_SPAIR_GEN:
264  load_gen(p->i);
265  break;
266  default:
267  break;
268  }

274 void F4GB::reorder_columns()
275 {
276  // Set up to sort the columns.
277  // Result is an array 0..ncols-1, giving the new order.
278  // Find the inverse of this permutation: place values into "ord" column
279  // fields.
280  // Loop through every element of the matrix, changing its comp array.
281 
282  int nrows = INTSIZE(mat->rows);
283  int ncols = INTSIZE(mat->columns);
284 
285  // sort the columns
286 
287  int *column_order = Mem->components.allocate(ncols);
288  int *ord = Mem->components.allocate(ncols);
289 
290  ColumnsSorter C(M, mat);
291 
292 // Actual sort of columns /////////////////
293 
294  C.reset_ncomparisons();
295 
296  clock_t begin_time0 = clock();
297  for (int i = 0; i < ncols; i++)
298  {
299  column_order[i] = i;
300  }
301 
302  if (M2_gbTrace >= 2) fprintf(stderr, "ncomparisons = ");
303 
304  std::stable_sort(column_order, column_order + ncols, C);
305 
306  clock_t end_time0 = clock();
307  if (M2_gbTrace >= 2) fprintf(stderr, "%ld, ", C.ncomparisons0());
308  double nsecs0 = (double)(end_time0 - begin_time0) / CLOCKS_PER_SEC;
309  clock_sort_columns += nsecs0;
310  if (M2_gbTrace >= 2) fprintf(stderr, " time = %f\n", nsecs0);
311 
313 
314  for (int i = 0; i < ncols; i++)
315  {
316  ord[column_order[i]] = i;
317  }
318 
319  // Now move the columns into position
320  coefficient_matrix::column_array newcols;
321  newcols.reserve(ncols);
322  for (int i = 0; i < ncols; i++)
323  {
324  long newc = column_order[i];
325  newcols.push_back(mat->columns[newc]);
326  }
327 
328  // Now reset the components in each row
329  for (int r = 0; r < nrows; r++)
330  {
331  row_elem &row = mat->rows[r];
332  for (int i = 0; i < row.len; i++)
333  {
334  int oldcol = row.comps[i];
335  int newcol = ord[oldcol];
336  row.comps[i] = newcol;
337  }
338  for (int i = 1; i < row.len; i++)
339  {
340  if (row.comps[i] <= row.comps[i - 1])
341  {
342  fprintf(stderr, "Internal error: array out of order\n");
343  break;
344  }
345  }
346  }
347 
348  std::swap(mat->columns, newcols);
349  Mem->components.deallocate(column_order);
350  Mem->components.deallocate(ord);
351 }

353 void F4GB::reorder_rows()
354 {
355  int nrows = INTSIZE(mat->rows);
356  int ncols = INTSIZE(mat->columns);
357  coefficient_matrix::row_array newrows;
358  VECTOR(long) rowlocs; // 0..nrows-1 of where row as been placed
359 
360  newrows.reserve(nrows);
361  rowlocs.reserve(nrows);
362  for (int r = 0; r < nrows; r++) rowlocs.push_back(-1);
363 
364  for (int c = 0; c < ncols; c++)
365  {
366  int oldrow = mat->columns[c].head;
367  if (oldrow >= 0)
368  {
369  // Move the row into place
370  int newrow = INTSIZE(newrows);
371  newrows.push_back(mat->rows[oldrow]);
372  rowlocs[oldrow] = newrow;
373  mat->columns[c].head = newrow;
374  if (mat->columns[c].head == oldrow) mat->columns[c].head = newrow;
375  }
376  }
377  for (int r = 0; r < nrows; r++)
378  if (rowlocs[r] < 0) newrows.push_back(mat->rows[r]);
379  std::swap(mat->rows, newrows);
380 }

473 void F4GB::gauss_reduce(bool diagonalize)
474 // This reduces the matrix to a triangular form
475 {
476  // For each row which is a non-pivot row:
477  // note that the row must be reducible, since the lead term corresponds to an
478  // spair cancellation
479  // actually: not true: generators will often not be reducible...
480  // also each such row must be non-zero, for the same reason
481  int nrows = INTSIZE(mat->rows);
482  int ncols = INTSIZE(mat->columns);
483 
484  int n_newpivots = -1; // the number of new GB elements in this degree
485  int n_zero_reductions = 0;
486  if (hilbert)
487  {
488  n_newpivots = hilbert->nRemainingExpected();
489  if (n_newpivots == 0) return;
490  }
491 
492  ElementArray gauss_row { mVectorArithmetic->allocateElementArray(ncols) };
493  for (int i = 0; i < nrows; i++)
494  {
495  row_elem &r = mat->rows[i];
496  if (r.len == 0) continue; // could happen once we include syzygies...
497  int pivotcol = r.comps[0];
498  int pivotrow = mat->columns[pivotcol].head;
499  if (pivotrow == i) continue; // this is a pivot row, so leave it alone
500 
501  const ElementArray& rcoeffs = get_coeffs_array(r);
502  n_pairs_computed++;
503  mVectorArithmetic->fillDenseArray(gauss_row, rcoeffs, Range(r.comps, r.comps + r.len));
504 
505  int firstnonzero = ncols;
506  int first = r.comps[0];
507  int last = r.comps[r.len - 1];
508  do
509  {
510  pivotrow = mat->columns[first].head;
511  if (pivotrow >= 0)
512  {
513  row_elem &pivot_rowelem = mat->rows[pivotrow];
514  auto& pivot_coeffs = get_coeffs_array(pivot_rowelem);
515  n_reduction_steps++;
516  mVectorArithmetic->denseCancelFromSparse(gauss_row,
517  pivot_coeffs,
518  Range(pivot_rowelem.comps,
519  pivot_rowelem.comps + pivot_rowelem.len)
520  );
521  int last1 = pivot_rowelem.comps[pivot_rowelem.len - 1];
522  if (last1 > last) last = last1;
523  }
524  else if (firstnonzero == ncols)
525  firstnonzero = first;
526  first = mVectorArithmetic->denseNextNonzero(gauss_row, first+1, last);
527  }
528  while (first <= last);
529  if (not r.coeffs.isNull())
530  mVectorArithmetic->deallocateElementArray(r.coeffs);
531  Mem->components.deallocate(r.comps);
532  r.len = 0;
533  //Range<int> monomRange;
534  //mVectorArithmetic->denseToSparse(gauss_row, r.coeffs, monomRange, firstnonzero, last, mComponentSpace);
535  mVectorArithmetic->denseToSparse(gauss_row, r.coeffs, r.comps, firstnonzero, last, Mem->components);
536  // TODO set r.comps from monomRange. Question: who has allocated this space??
537  // Maybe for now it is just the following line:
538  r.len = mVectorArithmetic->size(r.coeffs);
539  //r.len = monomRange.size();
540  //r.comps = Mem->components.allocate(r.len);
541  //std::copy(monomRange.begin(), monomRange.end(), r.comps);
542  //mComponentSpace.freeTopArray(monomRange.begin(), monomRange.end());
543  //assert(r.len == mVectorArithmetic->size(r.coeffs));
544 
545  // the above line leaves gauss_row zero, and also handles the case when
546  // r.len is 0 (TODO: check this!!)
547  // it also potentially frees the old r.coeffs and r.comps (TODO: ?? r.comps??)
548  if (r.len > 0)
549  {
550  mVectorArithmetic->makeMonic(r.coeffs);
551  mat->columns[r.comps[0]].head = i;
552  if (--n_newpivots == 0) break;
553  }
554  else
555  n_zero_reductions++;
556  }
557  mVectorArithmetic->deallocateElementArray(gauss_row);
558 
559  if (M2_gbTrace >= 3)
560  fprintf(stderr, "-- #zeroreductions %d\n", n_zero_reductions);
561 
562  if (diagonalize) tail_reduce();
563 }

565 void F4GB::tail_reduce()
566 {
567  int nrows = INTSIZE(mat->rows);
568  int ncols = INTSIZE(mat->columns);
569 
570  ElementArray gauss_row { mVectorArithmetic->allocateElementArray(ncols) };
571  for (int i = nrows - 1; i >= 0; i--)
572  {
573  row_elem &r = mat->rows[i];
574  if (r.len <= 1 || r.coeffs.isNull())
575  continue; // row reduced to zero, ignore it.
576  // At this point, we should have an element to reduce
577  bool anychange = false;
578  mVectorArithmetic->fillDenseArray(gauss_row, r.coeffs, Range(r.comps, r.comps + r.len));
579  int firstnonzero = r.comps[0];
580  int first = (r.len == 1 ? ncols : r.comps[1]);
581  int last = r.comps[r.len - 1];
582  while (first <= last)
583  {
584  int pivotrow = mat->columns[first].head;
585  if (pivotrow >= 0)
586  {
587  anychange = true;
588  row_elem &pivot_rowelem =
589  mat->rows[pivotrow]; // pivot_rowelems.coeffs is set at this
590  mVectorArithmetic->denseCancelFromSparse(gauss_row,
591  pivot_rowelem.coeffs,
592  Range(pivot_rowelem.comps, pivot_rowelem.comps + pivot_rowelem.len));
593  int last1 = pivot_rowelem.comps[pivot_rowelem.len - 1];
594  if (last1 > last) last = last1;
595  }
596  else if (firstnonzero == ncols)
597  firstnonzero = first;
598  first = mVectorArithmetic->denseNextNonzero(gauss_row, first+1, last);
599  };
600  if (anychange)
601  {
602  Mem->components.deallocate(r.comps);
603  mVectorArithmetic->deallocateElementArray(r.coeffs);
604  r.len = 0;
605  //Range<int> monomRange;
606  //mVectorArithmetic->denseToSparse(gauss_row,
607  // r.coeffs,
608  // monomRange,
609  // firstnonzero, last,
610  // mComponentSpace);
611  mVectorArithmetic->denseToSparse(gauss_row,
612  r.coeffs,
613  r.comps,
614  firstnonzero, last,
615  Mem->components);
616  r.len = mVectorArithmetic->size(r.coeffs);
617  //r.len = monomRange.size();
618  //r.comps = Mem->components.allocate(r.len);
619  //std::copy(monomRange.begin(), monomRange.end(), r.comps);
620  //mComponentSpace.freeTopArray(monomRange.begin(), monomRange.end());
621  assert(r.len == mVectorArithmetic->size(r.coeffs));
622  }
623  else
624  {
625  mVectorArithmetic->setZeroInRange(gauss_row, firstnonzero, last);
626  }
627  if (r.len > 0)
628  {
629  mVectorArithmetic->makeMonic(r.coeffs);
630  }
631  }
632 
633  mVectorArithmetic->deallocateElementArray(gauss_row);
634 }

640 void F4GB::insert_gb_element(row_elem &r)
641 {
642  // Insert row as gb element.
643  // Actions to do:
644  // translate row to a gbelem + poly
645  // set degrees as needed
646  // insert the monomial into the lookup table
647  // find new pairs associated to this new element
648 
649  int nslots = M->max_monomial_size();
650  int nlongs = r.len * nslots;
651 
652  gbelem *result = new gbelem;
653  result->f.len = r.len;
654 
655  // If the coeff array is null, then that means the coeffs come from the
656  // original array
657  // Here we copy it over.
658 
659  // const ElementArray& v = (not r.coeffs.isNull() ? r.coeffs : mVectorArithmetic->copyElementArray(get_coeffs_array(r)));
660  // result->f.coeffs.swap(v);
661  if (r.coeffs.isNull())
662  {
663  // this means the actual coeff vector is coming from a GB element. Copy it into result->f.coeffs
664  ElementArray v { mVectorArithmetic->copyElementArray(get_coeffs_array(r)) };
665  result->f.coeffs.swap(v);
666  }
667  else
668  {
669  result->f.coeffs.swap(r.coeffs);
670  }
671  result->f.monoms = Mem->allocate_monomial_array(nlongs);
672 
673  monomial_word *nextmonom = result->f.monoms;
674  for (int i = 0; i < r.len; i++)
675  {
676  M->copy(mat->columns[r.comps[i]].monom, nextmonom);
677  nextmonom += nslots;
678  }
679  Mem->components.deallocate(r.comps);
680  r.len = 0;
681  result->deg = this_degree;
682  result->alpha = static_cast<int>(M->last_exponent(result->f.monoms));
683  result->minlevel = ELEM_MIN_GB; // MES: How do
684  // we distinguish between ELEM_MIN_GB, ELEM_POSSIBLE_MINGEN?
685 
686  int which = INTSIZE(gb);
687  gb.push_back(result);
688 
689  if (hilbert)
690  {
691  int x;
692  int *exp = newarray_atomic(int, M->n_vars());
693  M->to_intstar_vector(result->f.monoms, exp, x);
694  hilbert->addMonomial(exp, x + 1);
695  freemem(exp);
696  }
697  // now insert the lead monomial into the lookup table
698  varpower_monomial vp = newarray_atomic(varpower_word, 2 * M->n_vars() + 1);
699  M->to_varpower_monomial(result->f.monoms, vp);
700  lookup->insert_minimal_vp(M->get_component(result->f.monoms), vp, which);
701  freemem(vp);
702  // now go forth and find those new pairs
703  S->find_new_pairs(is_ideal);
704 }

841  clock_make_matrix = 0;

843 
844  // test_spair_code();
