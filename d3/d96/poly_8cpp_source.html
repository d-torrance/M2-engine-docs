307int PolyRing::index_of_var(const ring_elem a) const
308{
309 Nterm *f = a;
310 if (f == nullptr || f->next != nullptr) return -1;
311 if (!K_->is_equal(f->coeff, K_->from_long(1))) return -1;
312 int result = -1;
313 exponents_t EXP1 = ALLOCATE_EXPONENTS(exp_size);
314 M_->to_expvector(f->monom, EXP1);
315 for (int i = 0; i < n_vars(); i++)
316 if (EXP1[i] > 1)
317 return -1;
318 else if (EXP1[i] == 1)
319 {
320 if (result >= 0) return -1;
321 result = i;
322 }
323 return result;
324}

328{

342 for (int i = 0; i < n_vars(); i++)
343 if (EXP1[i] > 0) result->array[next++] = i;
344 return result;

347int check_coeff_ring(const Ring *coeffR, const PolyRing *P)
348// returns the following:
349// if coeffR is K_, then n_vars()
350// if coeffR is a poly ring, then the difference in #vars
351// otherwise returns -1.
352// If a negative number is returned, then an error is set.
353{
354 int nvars = P->n_vars();
355 if (coeffR != P->getCoefficients())
356 {
357 const PolynomialRing *P0 = coeffR->cast_to_PolynomialRing();
358 if (P0 == nullptr)
359 nvars = -1;
360 else
361 nvars -= P0->n_vars();
362 }
363 if (nvars < 0) ERROR("expected coefficient ring");
364 return nvars;

367bool PolyRing::promote(const Ring *Rf,
368 const ring_elem f,
369 ring_elem &result) const
370{
371 // case 1: Rf = A[x]/J ---> A[x]/I is one of the 'base_ring's of 'this'.
372 // case 2: Rf = A ---> A[x]/I is the ring of scalars
373
374 // Cases:
375 // Rf is a poly ring (or quotient of one), with the same coeff ring as this.
376 // Rf is a poly ring (or quotient) with the same monoid as this (but coeffs
377 // are maybe different).
378 // Rf is the coeff ring of this
379 // Rf is not the coeff ring K of this, but is a base ring of K (e.g. ZZ -->
380 // QQ, A --> frac(A), any others?)
381 // Rf is a poly ring in a smaller number of variables
382
383 int nvars0 = n_vars();
384 if (Rf != K_)
385 {
386 const PolynomialRing *Rf1 = Rf->cast_to_PolynomialRing();
387 if (Rf1 != nullptr)
388 nvars0 -= Rf1->n_vars();
389 else
390 return false;
391 if (Rf1 && nvars0 == 0)
392 {
393 result = copy(f);
394 return true;
395 }
396 }
397
398 exponents_t exp = newarray_atomic_clear(int, nvars0);
399 result = make_logical_term(Rf, f, exp);
400 return true;
401}

403bool PolyRing::lift(const Ring *Rg, const ring_elem f, ring_elem &result) const
404{
405 // case 1: Rf = A[x]/J ---> A[x]/I is one of the 'base_ring's of 'this'.
406 // case 2: Rf = A ---> A[x]/I is the ring of scalars
407
408 // We assume that Rg is one of the coefficient rings of 'this'
409
410 Nterm *t = f;
411 if (t == nullptr)
412 {
413 result = Rg->zero();
414 return true;
415 }
416
417 int nvars0 = n_vars();
418 if (Rg != K_)
419 {
420 const PolynomialRing *Rg1 = Rg->cast_to_PolynomialRing();
421 if (Rg1 != nullptr)
422 nvars0 -= Rg1->n_vars();
423 else
424 return false;
425 }
426
427 exponents_t exp = newarray_atomic(int, nvars0);
428 lead_logical_exponents(nvars0, f, exp);
429 if (!exponents::is_one(nvars0, exp)) return false;
430 if (n_logical_terms(nvars0, f) > 1) return false;
431 result = lead_logical_coeff(Rg, f);
432 return true;
433}

447ring_elem PolyRing::preferred_associate_divisor(ring_elem f) const
448// f is an element of 'this'.
449// result is in the coefficient ring
450// If the coefficient ring of this is
451// ZZ -- gcd of all, same sign as lead coeff
452// QQ -- gcd(numerators)/lcm(denominators)
453// basic field -- lead coeff

455// frac(quotient of a poly ring) -- error

458 assert(getCoefficients()->has_associate_divisors());
459 for (Nterm& t : f)
460 if (!getCoefficients()->lower_associate_divisor(result, t.coeff))
461 // ie it cannot change, no matter what next coeff is
462 return result;
463 return result;

522{

591ring_elem PolyRing::homogenize(const ring_elem f,
592 int v,
593 int d,
594 const std::vector<int> &wts) const
595{
596 // assert(wts[v] != 0);
597 // If an error occurs, then return 0, and set gError.
598
599 exponents_t exp = newarray_atomic(int, nvars_);
600 Nterm head;
601 Nterm *result = &head;
602 for (Nterm& a : f)
603 {
604 M_->to_expvector(a.monom, exp);
605 auto e = exponents::weight(n_vars(), exp, wts);
606 if (((d - e) % wts[v]) != 0)
607 {
608 // We cannot homogenize, so clean up and exit.
609 result->next = nullptr;
610 ERROR("homogenization impossible");
611 result = nullptr;
612 return result;
613 }
614 exp[v] += (d - e) / wts[v];
615 if (is_skew_ && skew_.is_skew_var(v) && exp[v] > 1) continue;
616 result->next = new_term();
617 result = result->next;
618 result->coeff = K_->copy(a.coeff);
619 M_->from_expvector(exp, result->monom);
620 }
621 result->next = nullptr;
622 sort(head.next); // The monomial order, etc. might all have changed.
623 // Some terms might even drop out
624 freemem(exp);
625 return head.next;
626}

663void PolyRing::internal_add_to(ring_elem &ff, ring_elem &gg) const
664{
665 Nterm *f = ff;
666 Nterm *g = gg;
667 gg = ZERO_RINGELEM;
668 if (g == nullptr) return;
669 if (f == nullptr)
670 {
671 ff = g;
672 return;
673 }
674 Nterm head;
675 Nterm *result = &head;
676 while (1) switch (M_->compare(f->monom, g->monom))
677 {
678 case -1:
679 result->next = g;
680 result = result->next;
681 g = g->next;
682 if (g == nullptr)
683 {
684 result->next = f;
685 ff = head.next;
686 return;
687 }
688 break;
689 case 1:
690 result->next = f;
691 result = result->next;
692 f = f->next;
693 if (f == nullptr)
694 {
695 result->next = g;
696 ff = head.next;
697 return;
698 }
699 break;
700 case 0:
701 Nterm *tmf = f;
702 Nterm *tmg = g;
703 f = f->next;
704 g = g->next;
705 K_->add_to(tmf->coeff, tmg->coeff);
706 if (is_ZZ_quotient_)
707 {
708 ring_elem t = globalZZ->remainder(tmf->coeff, ZZ_quotient_value_);
709 tmf->coeff = t;
710 }
711 if (!K_->is_zero(tmf->coeff))
712 {
713 result->next = tmf;
714 result = result->next;
715 }
716 if (g == nullptr)
717 {
718 result->next = f;
719 ff = head.next;
720 return;
721 }
722 if (f == nullptr)
723 {
724 result->next = g;
725 ff = head.next;
726 return;
727 }
728 break;
729 }

856ring_elem PolyRing::power_direct(const ring_elem ff, int n) const
857{
858 ring_elem result, g, rest, h, tmp;
859 ring_elem coef1, coef2, coef3;
860
861 Nterm *lead = ff;
862 if (lead == nullptr) return ZERO_RINGELEM;
863
864 rest = lead->next;
865 g = from_long(1);
866
867 // Start the result with the n th power of the lead term
868 Nterm *t = new_term();
869 t->coeff = K_->power(lead->coeff, n);
870 M_->power(lead->monom, n, t->monom);
871 t->next = nullptr;
872 // if (_base_ring != NULL) normal_form(t); NOT NEEDED
873 result = t;
874
875 if (POLY(rest) == nullptr) return result;
876 monomial m = M_->make_one();
877
878 mpz_t bin_c;
879
880 mpz_init_set_ui(bin_c, 1);
881
882 for (int i = 1; i <= n; i++)
883 {
884 tmp = mult(g, rest);
885 g = tmp;
886
887 mpz_mul_ui(bin_c, bin_c, n - i + 1);
888 mpz_fdiv_q_ui(bin_c, bin_c, i);
889
890 coef1 = K_->from_int(bin_c);
891
892 if (!K_->is_zero(coef1))
893 {
894 coef2 = K_->power(lead->coeff, n - i);
895 coef3 = K_->mult(coef1, coef2);
896 M_->power(lead->monom, n - i, m);
897
898 h = mult_by_term(g, coef3, m);
899 add_to(result, h);
900 }
901 }
902 mpz_clear(bin_c);
903 return result;
904}

906ring_elem PolyRing::power(const ring_elem f0, int n) const
907{
908 ring_elem ff;
909
910 if (n > 0)
911 ff = f0;
912 else if (n < 0)
913 {
914 ff = invert(f0);
915 n = -n;
916 }
917 else
918 return from_long(1);
919
920 if(!characteristic())
921 {
922 return power_direct(ff, n);
923 }
924 else
925 {
926 long p=characteristic(), pk=1;
927 ring_elem result = from_long(1);
928 ring_elem gg = copy(ff), temp; // no need to copy, just the correct number of terms
929
930 while(n)
931 {
932 for(Nterm *it=ff, *jt=gg; it!=nullptr; it=it->next, jt=jt->next)
933 {
934 jt->coeff = K_->power(it->coeff, pk);
935 M_->power(it->monom, pk, jt->monom);
936 }
937
938 temp = power_direct(gg, n%p);
939 result = mult(result, temp);
940
941 pk *= p;
942 n /= p;
943 }
944
945 return result;
946 }

1071ring_elem PolyRing::gcd_extended(const ring_elem f,
1072 const ring_elem g,
1073 ring_elem &u,
1074 ring_elem &v) const
1075// result == gcd(f,g) = u f + v g
1076{
1077 u = from_long(1);
1078 ring_elem result = copy(f);
1079
1080 if (is_zero(g))
1081 {
1082 v = from_long(0);
1083 return result;
1084 }
1085 ring_elem v1 = ZERO_RINGELEM;
1086 ring_elem v3 = copy(g);
1087 ring_elem t1, t3;
1088 ring_elem temp1, temp2, temp3;
1089 while (!is_zero(v3))
1090 {
1091 ring_elem q;
1092 t3 = remainderAndQuotient(result, v3, q);
1093
1094 // The following is: t1 = u - q*v1
1095 temp1 = mult(q, v1);
1096 subtract_to(u, temp1);
1097 t1 = u;
1098
1099 u = v1;
1100 result = v3;
1101 v1 = t1;
1102 v3 = t3;
1103 }
1104
1105 // make 'result' monic. (and divide 'u' by this as well)
1106 if (!is_zero(result))
1107 {
1108 Nterm *t = result;
1109 ring_elem c = K_->invert(t->coeff);
1110 mult_coeff_to(c, result);
1111 mult_coeff_to(c, u);
1112 }
1113
1114 // The following is v = (result - f*u)/g
1115 temp1 = mult(f, u);
1116 temp2 = subtract(result, temp1);
1117 temp3 = remainderAndQuotient(temp2, g, v);
1118
1119 return result;
1120}

1410Nterm *PolyRing::division_algorithm_with_laurent_variables(Nterm *f, Nterm *g, Nterm *&quot) const
1411{
1412 // Step 1: replace f with f1, m1 (f = m1 * f1, f1 has only exponents >= 0, and no monomial factors)
1413 // Same: g = n1 * g1.
1414 // then if f1 = q * g1 + r, (q, r polynomials with only non-negative exponents)
1415 // then f = (q*m1*n1^(-1)) * g + m1*r
1416 //std::pair<Nterm*, int*> factor_out_inverse_variables(f);
1417 auto expf = setNegativeExponentMonomial(f);
1418 auto expg = setNegativeExponentMonomial(g);
1419 monomial m = getMonoid()->make_one();
1420 monomial n = getMonoid()->make_one();
1421 getMonoid()->from_expvector(expf.data(), m);
1422 getMonoid()->from_expvector(expg.data(), n);
1423
1424 for (auto& a : expf) a = -a;
1425 for (auto& a : expg) a = -a;
1426 monomial minv = getMonoid()->make_one();
1427 monomial ninv = getMonoid()->make_one();
1428 getMonoid()->from_expvector(expf.data(), minv);
1429 getMonoid()->from_expvector(expg.data(), ninv);
1430
1431 for (int i=0; i<n_vars(); ++i)
1432 expf[i] = -expf[i] + expg[i];
1433 monomial mninv = getMonoid()->make_one();
1434 getMonoid()->from_expvector(expf.data(), mninv);
1435
1436 ring_elem c = getCoefficientRing()->from_long(1);
1437 Nterm* f1 = mult_by_term(f, c, minv); // f1 = m^-1 * f, m only involves the variables which allow negative exponents
1438 Nterm* g1 = mult_by_term(g, c, ninv); // g1 = n^-1 * g.
1439 Nterm* quot1 = nullptr;
1440 Nterm* rem1 = division_algorithm(f1, g1, quot1);
1441 Nterm* rem = mult_by_term(rem1, c, m);
1442 quot = mult_by_term(quot1, c, mninv);
1443 return rem;
1444}
