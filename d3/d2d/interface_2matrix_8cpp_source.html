230 const Matrix /* or null */ *IM2_Matrix_concat(const engine_RawMatrixArray Ms)
231 {
232  try
233  {
234  unsigned int n = Ms->len;
235  if (n == 0)
236  {
237  ERROR("matrix concat: expects at least one matrix");
238  return 0;
239  }
240  const FreeModule *F = Ms->array[0]->rows();
241  const Ring *R = F->get_ring();
242  MatrixConstructor mat(Ms->array[0]->rows(), 0);
243  int next = 0;
244  for (unsigned int i = 0; i < n; i++)
245  {
246  const Matrix *M = Ms->array[i];
247  if (F->get_ring() != M->get_ring())
248  {
249  ERROR("matrix concat: different base rings");
250  return 0;
251  }
252  if (F->rank() != M->n_rows())
253  {
254  ERROR("matrix concat: row sizes are not equal");
255  return 0;
256  }
257  for (int j = 0; j < M->n_cols(); j++)
258  {
259  mat.append(R->copy_vec(M->elem(j)));
260  mat.set_column_degree(next++, M->cols()->degree(j));
261  }
262  }
263  return mat.to_matrix();
264  } catch (const exc::engine_error& e)
265  {
266  ERROR(e.what());
267  return NULL;
268  }
269 }

271 const Matrix /* or null */ *IM2_Matrix_direct_sum(
272  const engine_RawMatrixArray Ms)
273 {
274  try
275  {
276  // Check that the matrices all have the same ring, and that there is
277  // at least one matrix.
278  unsigned int n = Ms->len;
279  if (n == 0)
280  {
281  ERROR("matrix direct sum: expects at least one matrix");
282  return 0;
283  }
284  const Matrix *result = Ms->array[0];
285  const Ring *R = result->get_ring();
286  for (unsigned int i = 1; i < n; i++)
287  if (R != Ms->array[i]->get_ring())
288  {
289  ERROR("matrix direct sum: different base rings");
290  return 0;
291  }
292  for (unsigned int i = 1; i < n; i++)
293  result = result->direct_sum(Ms->array[i]);
