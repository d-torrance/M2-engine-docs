262const Matrix /* or null */ *IM2_Matrix_concat(const engine_RawMatrixArray Ms)
263{
264 try
265 {
266 unsigned int n = Ms->len;
267 if (n == 0)
268 {
269 ERROR("matrix concat: expects at least one matrix");
270 return nullptr;
271 }
272 const FreeModule *F = Ms->array[0]->rows();
273 const Ring *R = F->get_ring();
274 MatrixConstructor mat(Ms->array[0]->rows(), 0);
275 int next = 0;
276 for (unsigned int i = 0; i < n; i++)
277 {
278 const Matrix *M = Ms->array[i];
279 if (R != M->get_ring())
280 {
281 ERROR("matrix concat: different base rings");
282 return nullptr;
283 }
284 if (F->rank() != M->n_rows())
285 {
286 ERROR("matrix concat: row sizes are not equal");
287 return nullptr;
288 }
289 for (int j = 0; j < M->n_cols(); j++)
290 {
291 mat.append(R->copy_vec(M->elem(j)));
292 mat.set_column_degree(next++, M->cols()->degree(j));
293 }
294 }
295 return mat.to_matrix();
296 } catch (const exc::engine_error& e)
297 {
298 ERROR(e.what());
299 return nullptr;
300 }

303const Matrix /* or null */ *IM2_Matrix_direct_sum(
304 const engine_RawMatrixArray Ms)
305{
306 try
307 {
308 // Check that the matrices all have the same ring, and that there is
309 // at least one matrix.
310 unsigned int n = Ms->len;
311 if (n == 0)
312 {
313 ERROR("matrix direct sum: expects at least one matrix");
314 return nullptr;
315 }
316 const Matrix *result = Ms->array[0];
317 const Ring *R = result->get_ring();
318 for (unsigned int i = 1; i < n; i++)
319 if (R != Ms->array[i]->get_ring())
320 {
321 ERROR("matrix direct sum: different base rings");
322 return nullptr;
323 }
324 for (unsigned int i = 1; i < n; i++)
325 result = result->direct_sum(Ms->array[i]);
326
327 return result;
328 } catch (const exc::engine_error& e)
329 {
330 ERROR(e.what());
331 return nullptr;
332 }

692const Matrix /* or null */ *IM2_Matrix_promote(const FreeModule *newTarget,
693 const Matrix *f)
694{
695 try
696 {
697 ring_elem a;
698 const Ring *R = f->get_ring();
699 const Ring *S = newTarget->get_ring();
700 MatrixConstructor mat(newTarget, f->n_cols());
701 Matrix::iterator i(f);
702 for (int c = 0; c < f->n_cols(); c++)
703 for (i.set(c); i.valid(); i.next())
704 if (S->promote(R, i.entry(), a))
705 mat.set_entry(i.row(), c, a);
706 else
707 {
708 ERROR("cannot promote given matrix");
709 return nullptr;
710 }
711 mat.compute_column_degrees();
712 return mat.to_matrix();
713 } catch (const exc::engine_error& e)
714 {
715 ERROR(e.what());
716 return nullptr;
717 }
718}
