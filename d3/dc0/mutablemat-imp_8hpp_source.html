120template <typename T>
121MutableMatrix /* or null */* MutableMat<T>::mult(const MutableMatrix* B) const
122{
123 // First, make sure B has the same ring/type as 'this'.
124 const T* B1 = B->coerce_const<T>();
125 if (B1 == 0)
126 {
127 ERROR(
128 "mutable matrix/ring type for (mutable) matrix multiplication "
129 "required to be the same");
130 return 0;
131 }
132 // Second, make sure the sizes are correct.
133 if (mat.numColumns() != B1->numRows())
134 {
135 ERROR("matrix sizes do not match in matrix multiplication");
136 return 0;
137 }
138 // create the result matrix
139 MutableMat<T>* result = makeZeroMatrix(n_rows(), B1->numColumns());
140 MatrixOps::mult(mat, *B1, result->mat);
141
142 return result;
143}

145template <typename T>
146void MutableMat<T>::addMultipleTo(const MutableMatrix* A,
147 const MutableMatrix* B)
148{
149 // First: make sure that A, B have the right ring/matrix type
150 const T* A1 = A->coerce_const<T>();
151 const T* B1 = B->coerce_const<T>();
152 if (A1 == 0 or B1 == 0)
153 throw exc::engine_error(
154 "mutable matrix/ring type for (mutable) matrix multiplication required "
155 "to be the same");
156 if (mat.numRows() != A1->numRows() or mat.numColumns() != B1->numColumns())
157 throw exc::engine_error(
158 "expected matrix sizes to be compatible with matrix multiplication");
159 MatrixOps::addMultipleTo(mat, *A1, *B1);
