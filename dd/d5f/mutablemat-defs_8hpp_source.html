88template <typename CoeffRing>
89Matrix* toMatrix(const Ring* R, const DMat<CoeffRing>& A)
90{
91 int nrows = static_cast<int>(A.numRows());
92 int ncols = static_cast<int>(A.numColumns());
93 FreeModule* F = R->make_FreeModule(nrows);
94 MatrixConstructor result(F, ncols);
95 if (nrows == 0 || ncols == 0) return result.to_matrix();
96
97 for (int c = 0; c < ncols; c++)
98 {
99 int r = 0;
100 auto end = A.columnEnd(c);
101 for (auto j = A.columnBegin(c); not(j == end); ++j, ++r)
102 {
103 if (not A.ring().is_zero(*j))
104 {
105 ring_elem a;
106 A.ring().to_ring_elem(a, *j);
107 result.set_entry(r, c, a);
108 }
109 }
110 }
111 result.compute_column_degrees();
112 return result.to_matrix();
113}
114template <typename CoeffRing>
115Matrix* toMatrix(const Ring* R, const SMat<CoeffRing>& A)
116{
117 int nrows = static_cast<int>(A.numRows());
118 int ncols = static_cast<int>(A.numColumns());
119 FreeModule* F = R->make_FreeModule(nrows);
120 MatrixConstructor result(F, ncols);
121 if (nrows == 0 || ncols == 0) return result.to_matrix();
122 ring_elem f;
123 auto i = A.begin();
124 for (int c = 0; c < ncols; c++)
125 {
126 ring_elem a;
127 for (i.set(c); i.valid(); i.next())
128 {
129 i.copy_elem(a);
130 int r = static_cast<int>(i.row());
131 result.set_entry(r, c, a);
132 }
133 }
134 result.compute_column_degrees();
135 return result.to_matrix();
136}

259 virtual size_t lead_row(size_t col, ring_elem& result) const
260 /* returns the largest index row which has a non-zero value in column 'col'.
261 Also sets result to be the entry at this index.
262 returns -1 if the column is 0 */
263 {
264 Element b(mat.ring());
265 mat.ring().set_zero(b);
266
267 size_t ret = MatOps::lead_row(mat, col, b);
268 if (ret != static_cast<size_t>(-1)) mat.ring().to_ring_elem(result, b);
269
270 return ret;

273 virtual bool get_entry(size_t r, size_t c, ring_elem& result) const
274 // Returns false if (r,c) is out of range or if result is 0. No error
275 // is returned. result <-- this(r,c), and is set to zero if false is returned.
276 {
277 if (r < n_rows() && c < n_cols())
278 {
279 Element a(mat.ring());
280
281 MatOps::getEntry(mat, r, c, a);
282 bool is_nonzero = not mat.ring().is_zero(a);
283 mat.ring().to_ring_elem(result, a);
284 return is_nonzero;
285 }
286
287 result = get_ring()->zero();
288 return false;
289 }

506 virtual MutableMatrix* submatrix(M2_arrayint rows, M2_arrayint cols) const
507 {
508 for (size_t r = 0; r < rows->len; r++)
509 if (rows->array[r] < 0 || rows->array[r] >= n_rows())
510 {
511 ERROR(
512 "row index %d out of bounds 0..%d", rows->array[r], n_rows() - 1);
513 return 0;
514 }
515 for (size_t c = 0; c < cols->len; c++)
516 if (cols->array[c] < 0 || cols->array[c] >= n_cols())
517 {
518 ERROR("column index %d out of bounds 0..%d",
519 cols->array[c],
520 n_cols() - 1);
521 return 0;
522 }
523 MutableMat* result =
524 new MutableMat(*this); // zero matrix, over the same ring
525 MatOps::setFromSubmatrix(getMat(), rows, cols, result->getMat());
526 return result;
