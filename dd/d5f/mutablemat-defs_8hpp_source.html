88 template <typename CoeffRing>
89 Matrix* toMatrix(const Ring* R, const DMat<CoeffRing>& A)
90 {
91  int nrows = static_cast<int>(A.numRows());
92  int ncols = static_cast<int>(A.numColumns());
93  FreeModule* F = R->make_FreeModule(nrows);
94  MatrixConstructor result(F, ncols);
95  if (nrows == 0 || ncols == 0) return result.to_matrix();
96 
97  for (int c = 0; c < ncols; c++)
98  {
99  int r = 0;
100  auto end = A.columnEnd(c);
101  for (auto j = A.columnBegin(c); not(j == end); ++j, ++r)
102  {
103  if (not A.ring().is_zero(*j))
104  {
105  ring_elem a;
106  A.ring().to_ring_elem(a, *j);
107  result.set_entry(r, c, a);
108  }
109  }
110  }
111  result.compute_column_degrees();
112  return result.to_matrix();
113 }
114 template <typename CoeffRing>
115 Matrix* toMatrix(const Ring* R, const SMat<CoeffRing>& A)
116 {
117  int nrows = static_cast<int>(A.numRows());
118  int ncols = static_cast<int>(A.numColumns());
119  FreeModule* F = R->make_FreeModule(nrows);
120  MatrixConstructor result(F, ncols);
121  if (nrows == 0 || ncols == 0) return result.to_matrix();
122  ring_elem f;
123  auto i = A.begin();
124  for (int c = 0; c < ncols; c++)
125  {
126  ring_elem a;
127  for (i.set(c); i.valid(); i.next())
128  {
129  i.copy_elem(a);
130  int r = static_cast<int>(i.row());
131  result.set_entry(r, c, a);
132  }
133  }
134  result.compute_column_degrees();
135  return result.to_matrix();
136 }
