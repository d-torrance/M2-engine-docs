107 {
108  ring_elem ff = gg;
109  int cmp = mpz_sgn(m); // the sign of m, <0, ==0, >0
110  if (cmp == 0) return one();
111  mpz_t n;
112  mpz_init_set(n, m);
113  // @TODO MES: rewrite this so it inverts before creating n.
114  // That way we don't have to catch any exceptions here.
115  // e.g. as
116 #if 0
117  if (cmp < 0)
118  ff = invert(ff);
119  mpz_init_set(n, m);
120  mpz_t n;
121  if (cmp < 0)
122  mpz_neg(n, n);
123 #endif
124  if (cmp < 0)
125  {
126  mpz_neg(n, n);
127  ff = invert(
128  ff); // this can raise an exception, in which case we need to free n.
129  if (is_zero(ff))
130  {
131  ERROR(
132  "either element not invertible, or no method available to "
133  "compute its inverse");
134  mpz_clear(n);
135  return ff;
136  }
137  }
138  ring_elem prod = from_long(1);
139  ring_elem base = copy(ff);
140  ring_elem tmp;
141 
142  for (;;)
143  {
144  if (RingZZ::mod_ui(n, 2) == 1)
145  {
146  tmp = mult(prod, base);
147  prod = tmp;
148  }
149  mpz_tdiv_q_2exp(n, n, 1);
150  if (mpz_sgn(n) == 0)
151  {
152  mpz_clear(n);
153  return prod;
154  }
155  else
156  {
157  tmp = mult(base, base);
158  base = tmp;
159  }

163 ring_elem Ring::power(const ring_elem gg, int n) const
164 {
165  // TODO: reorganize to match the above routine (but using an int).
166  ring_elem ff = gg;
167  if (n == 0) return one();
168  if (n < 0)
169  {
170  n = -n;
171  ff = invert(ff);
172  if (is_zero(ff))
173  {
174  ERROR("negative power of noninvertible element requested");
175  return ff;
176  }
177  }
178 
179  // The exponent 'n' should be > 0 here.
180  ring_elem prod = from_long(1);
181  ring_elem base = copy(ff);
182  ring_elem tmp;
183 
184  for (;;)
185  {
186  if ((n % 2) != 0)
187  {
188  tmp = mult(prod, base);
189  prod = tmp;
190  }
191  n >>= 1;
192  if (n == 0) { return prod; }
193  else
194  {
195  tmp = mult(base, base);
196  base = tmp;
197  }
198  }
199 }
