470 bool lower_associate_divisor(ring_elem &f, const ring_elem g) const
471 {
472 mpq_srcptr a = MPQ_VAL(f);
473 mpq_srcptr b = MPQ_VAL(g);
474 int sa = mpq_sgn(a);
475 int sb = mpq_sgn(b);
476 int s = (sa == 0 ? sb : sa);
477 mpq_ptr result = getmemstructtype(mpq_ptr);
478 mpq_init(result);
479
480 mpz_gcd(mpq_numref(result), mpq_numref(a), mpq_numref(b));
481 mpz_lcm(mpq_denref(result), mpq_denref(a), mpq_denref(b));
482 if (s != mpq_sgn(result)) mpq_neg(result, result);
483 mpz_reallocate_limbs(mpq_numref(result));
484 mpz_reallocate_limbs(mpq_denref(result));
485 f = ring_elem(result);
486 return true; // the answer could become lower, if a newer g has a larger
487 // denom
488 }

490 void lower_content(ring_elem &c, const ring_elem g) const
491 {
492 if (is_zero(c))
493 {
494 c = g;
495 return;
496 }
497 mpq_srcptr a = MPQ_VAL(c);
498 mpq_srcptr b = MPQ_VAL(g);
499 int sa = mpq_sgn(a);
500 mpq_ptr result = getmemstructtype(mpq_ptr);
501 mpq_init(result);
502
503 mpz_gcd(mpq_numref(result), mpq_numref(a), mpq_numref(b));
504 mpz_lcm(mpq_denref(result), mpq_denref(a), mpq_denref(b));
505 if (sa != mpq_sgn(result)) mpq_neg(result, result);
506 mpz_reallocate_limbs(mpq_numref(result));
507 mpz_reallocate_limbs(mpq_denref(result));
508 c = ring_elem(result);
509 }

542namespace RingPromoter {
543 template <typename SourceRing, typename TargetRing>
544 bool promoter(const Ring *R,
545 const Ring *S,
546 const ring_elem fR,
547 ring_elem &resultS)
548 {
549 assert(dynamic_cast<const ConcreteRing<SourceRing> *>(R) != 0);
550 assert(dynamic_cast<const ConcreteRing<TargetRing> *>(S) != 0);
551 const SourceRing &R1 =
552 dynamic_cast<const ConcreteRing<SourceRing> *>(R)->ring();
553 const TargetRing &S1 =
554 dynamic_cast<const ConcreteRing<TargetRing> *>(S)->ring();
555
556 typename SourceRing::Element fR1(R1);
557 typename TargetRing::Element gS1(S1);
558
559 R1.from_ring_elem(fR1, fR);
560 bool retval = mypromote(R1, S1, fR1, gS1);
561 if (retval) S1.to_ring_elem(resultS, gS1);
562 return retval;
563 }
564
565 template <typename SourceRing, typename TargetRing>
566 bool lifter(const Ring *R,
567 const Ring *S,
568 ring_elem &result_gR,
569 const ring_elem gS)
570 {
571 assert(dynamic_cast<const ConcreteRing<SourceRing> *>(R) != 0);
572 assert(dynamic_cast<const ConcreteRing<TargetRing> *>(S) != 0);
573 const SourceRing &R1 =
574 dynamic_cast<const ConcreteRing<SourceRing> *>(R)->ring();
575 const TargetRing &S1 =
576 dynamic_cast<const ConcreteRing<TargetRing> *>(S)->ring();
577
578 typename SourceRing::Element fR1(R1);
579 typename TargetRing::Element gS1(S1);
580
581 S1.from_ring_elem(gS1, gS);
582 bool retval = mylift(R1, S1, fR1, gS1); // sets fR1.
583 if (retval) R1.to_ring_elem(result_gR, fR1);
584 return retval;
585 }

588template <typename RingType>
589bool ConcreteRing<RingType>::promote(const Ring *R,
590 const ring_elem fR,
591 ring_elem &resultS) const
592{
593 const Ring *S = this;
594 // fprintf(stderr, "calling promote\n");
595 namespace RP = RingPromoter;
596 if (R == globalZZ)
597 {
598 resultS = S->from_int(fR.get_mpz());
599 return true;
600 }
601 if (R == S)
602 {
603 resultS = copy(fR);
604 return true;
605 }
606 switch (R->ringID())
607 {
608 case M2::ring_ZZp:
609 switch (S->ringID())
610 {
611 case M2::ring_ZZp:
612 return false;
613 case M2::ring_ZZpFfpack:
614 return RP::promoter<ARingZZp, ARingZZpFFPACK>(R, S, fR, resultS);
615 default:
616 return false;
617 }
618 break;
619 case M2::ring_ZZpFfpack:
620 switch (S->ringID())
621 {
622 case M2::ring_ZZp:
623 return RP::promoter<ARingZZpFFPACK, ARingZZp>(R, S, fR, resultS);
624 case M2::ring_ZZpFfpack:
625 return RP::promoter<ARingZZpFFPACK, ARingZZpFFPACK>(
626 R, S, fR, resultS);
627 default:
628 return false;
629 }
630 case M2::ring_QQ:
631 switch (S->ringID())
632 {
633 case M2::ring_RR:
634 return RP::promoter<ARingQQ, ARingRR>(R, S, fR, resultS);
635 case M2::ring_RRR:
636 return RP::promoter<ARingQQ, ARingRRR>(R, S, fR, resultS);
637 case M2::ring_RRi:
638 return RP::promoter<ARingQQ, ARingRRi>(R, S, fR, resultS);
639 case M2::ring_CC:
640 return RP::promoter<ARingQQ, ARingCC>(R, S, fR, resultS);
641 case M2::ring_CCC:
642 return RP::promoter<ARingQQ, ARingCCC>(R, S, fR, resultS);
643 default:
644 return false;
645 }
646 case M2::ring_RR:
647 switch (S->ringID())
648 {
649 case M2::ring_RR:
650 return RP::promoter<ARingRR, ARingRR>(R, S, fR, resultS);
651 case M2::ring_RRR:
652 return RP::promoter<ARingRR, ARingRRR>(R, S, fR, resultS);
653 case M2::ring_RRi:
654 return RP::promoter<ARingRR, ARingRRi>(R, S, fR, resultS);
655 case M2::ring_CC:
656 return RP::promoter<ARingRR, ARingCC>(R, S, fR, resultS);
657 case M2::ring_CCC:
658 return RP::promoter<ARingRR, ARingCCC>(R, S, fR, resultS);
659 default:
660 return false;
661 }
662 case M2::ring_RRR:
663 switch (S->ringID())
664 {
665 case M2::ring_RR:
666 return RP::promoter<ARingRRR, ARingRR>(R, S, fR, resultS);
667 case M2::ring_RRR:
668 return RP::promoter<ARingRRR, ARingRRR>(R, S, fR, resultS);
669 case M2::ring_RRi:
670 return RP::promoter<ARingRRR, ARingRRi>(R, S, fR, resultS);
671 case M2::ring_CC:
672 return RP::promoter<ARingRRR, ARingCC>(R, S, fR, resultS);
673 case M2::ring_CCC:
674 return RP::promoter<ARingRRR, ARingCCC>(R, S, fR, resultS);
675 default:
676 return false;
677 }
678 case M2::ring_RRi:
679 switch (S->ringID())
680 {
681 case M2::ring_RR:
682 return RP::promoter<ARingRRi, ARingRR>(R, S, fR, resultS);
683 case M2::ring_RRR:
684 return RP::promoter<ARingRRi, ARingRRR>(R, S, fR, resultS);
685 case M2::ring_RRi:
686 return RP::promoter<ARingRRi, ARingRRi>(R, S, fR, resultS);
687 default:
688 return false;
689 }
690 case M2::ring_CC:
691 switch (S->ringID())
692 {
693 case M2::ring_CC:
694 return RP::promoter<ARingCC, ARingCC>(R, S, fR, resultS);
695 case M2::ring_CCC:
696 return RP::promoter<ARingCC, ARingCCC>(R, S, fR, resultS);
697 default:
698 return false;
699 }
700 case M2::ring_CCC:
701 switch (S->ringID())
702 {
703 case M2::ring_CCC:
704 return RP::promoter<ARingCCC, ARingCCC>(R, S, fR, resultS);
705 case M2::ring_CC:
706 return RP::promoter<ARingCCC, ARingCC>(R, S, fR, resultS);
707 default:
708 return false;
709 }

716// given a natural map: R --> S = this,
717// 'lift' sets result_gR with an element which maps to gS, if possible.
718// true is returned iff this is possible.
719template <typename RingType>
720bool ConcreteRing<RingType>::lift(const Ring *R,
721 const ring_elem gS,
722 ring_elem &result_gR) const
723{
724 const Ring *S = this;
725
726 namespace RP = RingPromoter;
727 if (R == S)
728 {
729 result_gR = gS;
730 return true;
731 }
732 if (R == globalZZ)
733 {
734 printf("error!! lift called with no ZZ lifting method\n");
735 // MES:TODO!! WRITE ME
736 return true;
737 }
738 switch (R->ringID())
739 {
740 case M2::ring_ZZp:
741 switch (S->ringID())
742 {
743 case M2::ring_ZZp:
744 return false;
745 case M2::ring_ZZpFfpack:
746 return RP::lifter<ARingZZp, ARingZZpFFPACK>(R, S, result_gR, gS);
747 default:
748 return false;
749 }
750 break;
751 case M2::ring_ZZpFfpack:
752 switch (S->ringID())
753 {
754 case M2::ring_ZZp:
755 return RP::lifter<ARingZZpFFPACK, ARingZZp>(R, S, result_gR, gS);
756 case M2::ring_ZZpFfpack:
757 return RP::lifter<ARingZZpFFPACK, ARingZZpFFPACK>(
758 R, S, result_gR, gS);
759 default:
760 return false;
761 }
762 case M2::ring_RR:
763 switch (S->ringID())
764 {
765 case M2::ring_RR:
766 return RP::lifter<ARingRR, ARingRR>(R, S, result_gR, gS);
767 case M2::ring_RRR:
768 return RP::lifter<ARingRR, ARingRRR>(R, S, result_gR, gS);
769 case M2::ring_RRi:
770 return RP::lifter<ARingRR, ARingRRi>(R, S, result_gR, gS);
771 case M2::ring_CC:
772 return RP::lifter<ARingRR, ARingCC>(R, S, result_gR, gS);
773 case M2::ring_CCC:
774 return RP::lifter<ARingRR, ARingCCC>(R, S, result_gR, gS);
775 default:
776 return false;
777 }
778 case M2::ring_RRR:
779 switch (S->ringID())
780 {
781 case M2::ring_RR:
782 return RP::lifter<ARingRRR, ARingRR>(R, S, result_gR, gS);
783 case M2::ring_RRR:
784 return RP::lifter<ARingRRR, ARingRRR>(R, S, result_gR, gS);
785 case M2::ring_RRi:
786 return RP::lifter<ARingRRR, ARingRRi>(R, S, result_gR, gS);
787 case M2::ring_CC:
788 return RP::lifter<ARingRRR, ARingCC>(R, S, result_gR, gS);
789 case M2::ring_CCC:
790 return RP::lifter<ARingRRR, ARingCCC>(R, S, result_gR, gS);
791 default:
792 return false;
793 }
794 case M2::ring_CC:
795 switch (S->ringID())
796 {
797 case M2::ring_CC:
798 return RP::lifter<ARingCC, ARingCC>(R, S, result_gR, gS);
799 case M2::ring_CCC:
800 return RP::lifter<ARingCCC, ARingCC>(R, S, result_gR, gS);
801 default:
802 return false;
803 }
804 case M2::ring_CCC:
805 switch (S->ringID())
806 {
807 case M2::ring_CC:
808 return RP::lifter<ARingCC, ARingCCC>(R, S, result_gR, gS);
809 case M2::ring_CCC:
810 return RP::lifter<ARingCCC, ARingCCC>(R, S, result_gR, gS);
811 default:
812 return false;
813 }
814 default:
815#ifndef NDEBUG
816 fprintf(stderr,
817 "oh no: rings not in list\n, R->ringID()=%d S->ringID()=%d\n",
818 R->ringID(),
819 S->ringID());
820#endif
821 break;
822 };
823 return false;
824}
