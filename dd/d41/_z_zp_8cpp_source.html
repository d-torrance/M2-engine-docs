15bool Z_mod::initialize_Z_mod(int p)
16{
17 initialize_ring(p);
18 P = p;
19
20 declare_field();
21 int i, j, q, n;
22
23 if (P == 2)
24 _minus_one = 0;
25 else
26 _minus_one = (P - 1) / 2;
27
28 if (P == 2)
29 _prim_root = 1;
30 else
31 {
32 j = 1;
33 for (i = 2; (i < P && j < P - 1); i++)
34 for (q = i, j = 1; (q != 1 && j < P); q = (q * i) % P, j++)
35 ;
36 _prim_root = i - 1;
37 }
38
39 // cerr << "z_mod_p: creating table for P = " << P << endl;
40 _log_table = newarray_atomic(int, P);
41 _exp_table = newarray_atomic(int, P);
42 for (i = 0, n = 1; i < P - 1; i++, n = (n * _prim_root) % P)
43 {
44 _log_table[n] = i; // i = log_(base _prim_root)(n)
45 _exp_table[i] = n; // n = (_prim_root)^i
46 }
47 _ZERO = P - 1;
48 _exp_table[_ZERO] = 0;
49 _log_table[0] = _ZERO;
50
51 _P1 = P - 1;
52
53 zeroV = from_long(0);
54 oneV = from_long(1);
55 minus_oneV = from_long(-1);
56
57 coeffR = new CoefficientRingZZp(P, _log_table, _exp_table);
58 aringZZp = new M2::ARingZZp(P); // WARNING: this uses that the primitive
59 // element is the SAME as computed above!!
60 // Remove this warning once Z_mod is not in existence any longer.
61 return true;
62}

250ring_elem Z_mod::power(const ring_elem f, int n) const
251{
252 if (f.get_int() == _ZERO) {
253 if (n < 0)
254 throw exc::division_by_zero_error();
255 else if (n == 0)
256 return ring_elem(0); // this is the element one in this ring. (P-1) is the zero element...
257 return ring_elem(_ZERO);
258 }
259 int m = (f.get_int() * n) % _P1;
260 if (m < 0) m += _P1;
261 return ring_elem(m);
262}
263ring_elem Z_mod::power(const ring_elem f, mpz_srcptr n) const
264{
265 if (f.get_int() == _ZERO) {
266 if (mpz_sgn(n)<0) throw exc::division_by_zero_error();
267 else if (mpz_sgn(n) == 0)
268 return ring_elem(0);
269 return ring_elem(_ZERO);
270 }
271 int n1 = RingZZ::mod_ui(n, _P1);
272 int m = (f.get_int() * n1) % _P1;
273 if (m < 0) m += _P1;
274 return ring_elem(m);
275}

277ring_elem Z_mod::invert(const ring_elem f) const
278{
279 int a = f.get_int();
280 if (a == _ZERO) throw exc::division_by_zero_error();
281 if (a == 0) return 0; // this is the case f == ONE
282 return ring_elem(P - 1 - a);
283}

285ring_elem Z_mod::divide(const ring_elem f, const ring_elem g) const
286{
287 if (g.get_int() == _ZERO) throw exc::division_by_zero_error();
288 if (f.get_int() == _ZERO) return ring_elem(_ZERO);
