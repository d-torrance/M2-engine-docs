10 bool almostEqual(const RingCCC *R, int nbits, ring_elem a, ring_elem b)
11 {
12  mpfr_t epsilon;
13  mpfr_init2(epsilon, 100);
14  mpfr_set_ui_2exp(epsilon, 1, -nbits, MPFR_RNDN);
15 
16  ring_elem f = R->subtract(a, b);
17  auto f1 = BIGCC_RE(f);
18  bool re_is_zero = (mpfr_cmpabs(f1, epsilon) < 0);
19  auto f2 = BIGCC_IM(f);
20  bool im_is_zero = (mpfr_cmpabs(f2, epsilon) < 0);
21 
22  bool ret = re_is_zero && im_is_zero;
23  mpfr_clear(epsilon);
24  return ret;
25 }
