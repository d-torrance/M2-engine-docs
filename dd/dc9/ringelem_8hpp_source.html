113/* Implements an iterator for linked list-based multi-termed structs
114 *
115 * For example, the functions begin(Nterm*) and end(Nterm*) return
116 * a TermIterator<Nterm> object which makes for(Nterm& t : f) work. */
117template<typename T>
118struct TermIterator
119{
120 T* p;
121
122 TermIterator(): p(nullptr) {}
123 TermIterator(T* ptr): p(ptr) {}
124 TermIterator(ring_elem ptr): p(ptr) {}
125
126 TermIterator& operator++() { p = p->next; return *this; }
127
128 T const& operator*() const { return *p; }
129 T& operator*() { return *p; }
130 T const* operator->() const { return p; }
131 T* operator->() { return p; }
132
133 bool operator==(TermIterator const& rhs) const { return p == rhs.p; }
134 bool operator!=(TermIterator const& rhs) const { return p != rhs.p; }
135};
