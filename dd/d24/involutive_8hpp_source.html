108  template <typename MonomType>
109  const Matrix* BooleanInvolutiveBasis<MonomType>::ToMatrix() const
110  {
111  MatrixConstructor matrixConstructor(PRing->make_FreeModule(1), 0);
112  const Monoid* monoid = PRing->getMonoid();
113  const ring_elem coefficientUnit = PRing->getCoefficients()->one();
114  monomial tmpRingMonomial = monoid->make_one();
115 
116  for (typename std::list<Polynom<MonomType>*>::const_iterator currentPolynom = GBasis.begin();
117  currentPolynom != GBasis.end();
118  ++currentPolynom)
119  {
120  if (!*currentPolynom)
121  {
122  continue;
123  }
124 
125  ring_elem currentRingPolynomial;
126 
127  for (const MonomType* currentMonom = &(**currentPolynom).Lm();
128  currentMonom;
129  currentMonom = currentMonom->Next)
130  {
131  exponents_t currentExponent = newarray_atomic_clear(int, MonomType::GetDimIndepend());
132  typename std::set<typename MonomType::Integer> variablesSet = currentMonom->GetVariablesSet();
133 
134  for (typename std::set<typename MonomType::Integer>::const_iterator currentVariable = variablesSet.begin();
135  currentVariable != variablesSet.end();
136  ++currentVariable)
137  {
138  currentExponent[*currentVariable] = 1;
139  }
140 
141  monoid->from_expvector(currentExponent, tmpRingMonomial);
142  freemem(currentExponent);
143 
144  ring_elem tmpRingPolynomial = PRing->make_flat_term(coefficientUnit, tmpRingMonomial);
145  PRing->add_to(currentRingPolynomial, tmpRingPolynomial);
146  }
147 
148  matrixConstructor.append(PRing->make_vec(0, currentRingPolynomial));
149  }
150 
151  return matrixConstructor.to_matrix();
152  }

154  template <typename MonomType>
155  void BooleanInvolutiveBasis<MonomType>::FillInitialSet(const Matrix* matrix, std::list<Polynom<MonomType>*>& initialSet) const
156  {
157  const Monoid* monoid = PRing->getMonoid();
158  typename MonomType::Integer independ = MonomType::GetDimIndepend();
159 
160  //construct Polynom for every column in matrix
161  for (int column = 0; column < matrix->n_cols(); ++column)
162  {
163  vec polynomVector = matrix->elem(column);
164  if (!polynomVector)
165  {
166  continue;
167  }
168 
169  Polynom<MonomType>* currentPolynom = new Polynom<MonomType>();
170 
171  for (Nterm& currentTerm : polynomVector->coeff)
172  {
173  exponents_t monomVector = newarray_atomic(int, independ);
174  monoid->to_expvector(currentTerm.monom, monomVector);
175 
176  //construct Monom for every term
177  MonomType* currentMonom = new MonomType();
178  if (!currentMonom)
179  {
180  freemem(monomVector);
181  throw std::string("BIBasis::BooleanInvolutiveBasis::FillInitialSet(): got NULL instead of new monom.");
182  }
183 
184  for (typename MonomType::Integer currentVariable = 0; currentVariable < independ; ++currentVariable)
185  {
186  if (monomVector[currentVariable])
187  {
188  *currentMonom *= currentVariable;
189  }
190  }
191 
192  *currentPolynom += *currentMonom;
193  freemem(monomVector);
194  delete currentMonom;
195  }
196 
197  initialSet.push_back(currentPolynom);
198  }

201  template <typename MonomType>
202  Polynom<MonomType>* BooleanInvolutiveBasis<MonomType>::NormalForm(const Triple<MonomType>* triple) const
203  {
204  /* As far as currentTriple can't be 0 (checked in QSET and TSET),
205  * no need to check for NULL pointer.
206  */
207 
208  const Triple<MonomType>* involutiveDivisor = 0;
209  Polynom<MonomType>* originalForm = 0;
210  Polynom<MonomType>* normalForm = new Polynom<MonomType>();
211 
212  if (triple->GetVariable() == -1)
213  {
214  originalForm = new Polynom<MonomType>(*triple->GetPolynom());
215  }
216  else
217  {
218  originalForm = new Polynom<MonomType>(*triple->GetWeakAncestor()->GetPolynom());
219  (*originalForm) *= triple->GetVariable();
220  }
221 
222  while (!originalForm->IsZero())
223  {
224  involutiveDivisor = IntermediateBasis.Find(originalForm->Lm());
225  while (involutiveDivisor)
226  {
227  originalForm->HeadReduction(*involutiveDivisor->GetPolynom());
228  if (!originalForm->IsZero())
229  {
230  involutiveDivisor = IntermediateBasis.Find(originalForm->Lm());
231  }
232  else
233  {
234  involutiveDivisor = 0;
235  }
236  }
237 
238  if (!originalForm->IsZero())
239  {
240  (*normalForm) += originalForm->Lm();
241  originalForm->RidOfLm();
242  }
243  }
244 
245  delete originalForm;
246  return normalForm;

251  , const std::list<Polynom<MonomType>*>& set

278  template <typename MonomType>
279  Polynom<MonomType>* BooleanInvolutiveBasis<MonomType>::Reduce(Polynom<MonomType>* polynom
280  , const std::list<Polynom<MonomType>*>& set
281  , bool toGroebner) const
282  {
283  if (!polynom)
284  {
285  return 0;
286  }
287 
288  Polynom<MonomType>* result = new Polynom<MonomType>();
289  const Polynom<MonomType>* currentReducer = 0;
290 
291  while (!polynom->IsZero())
292  {
293  currentReducer = FindDivisor(polynom, set, toGroebner);
294  while (currentReducer)
295  {
296  polynom->Reduction(*currentReducer);
297  currentReducer = FindDivisor(polynom, set, toGroebner);
298  }
299  if (!polynom->IsZero())
300  {
301  (*result) += polynom->Lm();
302  polynom->RidOfLm();
303  }
304  }
305 
306  polynom = result;
307  return result;

310  template <typename MonomType>
311  void BooleanInvolutiveBasis<MonomType>::ReduceSet(bool toGroebner)
312  {
313  std::list<Polynom<MonomType>*> tmpPolySet;
314  GBasis.sort(PointerMoreComparator<Polynom<MonomType> >());
315 
316  while (!GBasis.empty())
317  {
318  Polynom<MonomType>* currentPolynom = GBasis.front();
319  GBasis.pop_front();
320  currentPolynom = Reduce(currentPolynom, tmpPolySet, toGroebner);
321 
322  if (currentPolynom && !currentPolynom->IsZero())
323  {
324  const MonomType& hLm = currentPolynom->Lm();
325  typename std::list<Polynom<MonomType>*>::iterator iteratorTmpPolySet = tmpPolySet.begin();
326  while (iteratorTmpPolySet != tmpPolySet.end())
327  {
328  if ((**iteratorTmpPolySet).Lm().IsDivisibleBy(hLm))
329  {
330  GBasis.push_back(*iteratorTmpPolySet);
331  iteratorTmpPolySet = tmpPolySet.erase(iteratorTmpPolySet);
332  }
333  else
334  {
335  ++iteratorTmpPolySet;
336  }
337  }
338  tmpPolySet.push_back(currentPolynom);
339  }
340  }
341 
342  unsigned tmpPolySetSize = static_cast<unsigned int>(tmpPolySet.size());
343  for (unsigned i = 0; i < tmpPolySetSize; ++i)
344  {
345  Polynom<MonomType>* currentPolynom = tmpPolySet.front();
346  tmpPolySet.pop_front();
347  currentPolynom = Reduce(currentPolynom, tmpPolySet, toGroebner);
348  if (!currentPolynom || currentPolynom->IsZero())
349  {
350  tmpPolySetSize--;
351  }
352  else
353  {
354  tmpPolySet.push_back(currentPolynom);
355  }
356  }
357 
358  GBasis = tmpPolySet;
359  }

391  template <typename MonomType>
392  void BooleanInvolutiveBasis<MonomType>::ConstructInvolutiveBasis()
393  {
394  typename TSet<MonomType>::Iterator tit(IntermediateBasis.Begin());
395  Polynom<MonomType>* newNormalForm = 0;
396  Triple<MonomType>* currentTriple = 0;
397 
398  while (!ProlongationsSet.Empty())
399  {
400  currentTriple = ProlongationsSet.Get();
401  newNormalForm = NormalForm(currentTriple);
402  /* As far as currentTriple can't be 0 (checked in QSET and TSET),
403  * NormalForm can't return 0.
404  */
405 
406  std::set<typename MonomType::Integer> currentNmpSet;
407  const Triple<MonomType>* currentAncestor = 0;
408  if (!newNormalForm->IsZero() && newNormalForm->Lm() == currentTriple->GetPolynomLm())
409  {
410  currentNmpSet = currentTriple->GetNmp();
411  currentAncestor = currentTriple->GetAncestor();
412  if (currentAncestor == currentTriple)
413  {
414  currentAncestor = 0;
415  }
416  }
417  delete currentTriple;
418 
419  if (!newNormalForm->IsZero())
420  {
421  std::list<Triple<MonomType>*> newProlongations;
422  tit = IntermediateBasis.Begin();
423 
424  while (tit != IntermediateBasis.End())
425  {
426  if ((**tit).GetPolynomLm().IsTrueDivisibleBy(newNormalForm->Lm()))
427  {
428  ProlongationsSet.DeleteDescendants(*tit);
429  newProlongations.push_back(*tit);
430  tit = IntermediateBasis.Erase(tit);
431  }
432  else
433  {
434  ++tit;
435  }
436  }
437 
438  IntermediateBasis.PushBack(new Triple<MonomType>(newNormalForm, currentAncestor, currentNmpSet, 0, -1));
439  if (!newNormalForm->Degree())
440  {
441  return;
442  }
443 
444  IntermediateBasis.CollectNonMultiProlongations(--IntermediateBasis.End(), newProlongations);
445  ProlongationsSet.Insert(newProlongations);
446  }
447  else
448  {
449  delete newNormalForm;
450  }
451  }
452  }
