113 void NCBasis::basis0()
114 {
115  if (system_interrupted()) return;
116 
117  // order of events:
118  // 1. check if the degree is greater than hi_degree. If so, exit.
119  // 2. check if a suffix of the current word is a pattern in the WordTable. If so, exit.
120  // 3. if the degree is >= lo_degree, then call insert.
121  // 4. if the degree is equal to hi_degree, then exit (no recursion necessary)
122  // 5. for each variable, append the variable to the state monomial, and make recursive call.
123 
124  Word tmpWord(mMonomial.data(),mMonomial.data() + mCurrentIndex + 1);
125  int notUsed;
126 
127  if (mHiHeft != -1 and mCurrentHeftValue > mHiHeft) return;
128 
129  if (mWordTable->isSuffix(tmpWord,notUsed)) return;
130 
131  if (mCurrentHeftValue >= mLoHeft) insert();
132 
133  if (mHiHeft != -1 and mCurrentHeftValue == mHiHeft) return;
134 
135  // this ensures that we do not move past allocated memory in the loop below
136  if (mMonomial.size() <= mCurrentIndex + 1) mMonomial.push_back(0);
137 
138  mCurrentIndex++;
139  for (int i = 0; i < mVariables.size(); ++i)
140  {
141  mMonomial[mCurrentIndex] = mVariables[i];
142  mCurrentHeftValue += mVariableHefts[i];
143 
144  basis0();
145 
146  mCurrentHeftValue -= mVariableHefts[i];
147 
148  if (mLimit == 0) return;
149  }
150  mCurrentIndex--;
151 }
