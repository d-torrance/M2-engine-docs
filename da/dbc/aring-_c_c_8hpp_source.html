180  void invert(ElementType& res, const ElementType& a) const
181  // we silently assume that a != 0. If it is, result is set to a^0, i.e. 1
182  {
183  ElementType result;

185  {
186  double p = a.im / a.re;
187  double denom = a.re + p * a.im;

273  void divide(ElementType& res,
274  const ElementType& a,
275  const ElementType& b) const
276  {
277  RealElementType p, denom; // double
278 
279  ElementType result;
280  if (fabs(b.re) >= fabs(b.im))
281  {
282  p = b.im / b.re;
283  denom = b.re + p * b.im;
284  result.re = (a.re + p * a.im) / denom;
285  result.im = (a.im - p * a.re) / denom;
286  }
287  else
288  {
289  p = b.re / b.im;
290  denom = b.im + p * b.re;
291  result.re = (a.im + p * a.re) / denom;
292  result.im = (p * a.im - a.re) / denom;
293  }

307  void power(ElementType& result, const ElementType& a, int n) const
308  {
309  ElementType curr_pow;

311  set_from_long(result, 1);
312  if (n == 0)
313  {
314  }
