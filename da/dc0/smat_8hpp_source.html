355 template <typename CoeffRing>
356 void SMat<CoeffRing>::vec_interchange_rows(sparsevec *&v,
357  size_t i,
358  size_t j) const
359 {
360  sparsevec *p;
361  if (i == j) return;
362  if (v == 0) return;
363  if (i < j)
364  {
365  size_t tmp = i;
366  i = j;
367  j = tmp;
368  }
369  // So now i > j.
370  sparsevec head;
371  head.next = v;
372  sparsevec *vec1;
373  sparsevec *vec2;
374  for (p = &head; p->next != 0; p = p->next)
375  if (p->next->row <= i) break;
376  vec1 = p;
377  for (; p->next != 0; p = p->next)
378  if (p->next->row <= j) break;
379  vec2 = p;
380  if (vec1->next != 0 && vec1->next->row == i)
381  {
382  if (vec2->next != 0 && vec2->next->row == j)
383  {
384  std::swap(vec1->next->coeff, vec2->next->coeff);
385  return;
386  }
387  }
388  else if (vec2->next != 0 && vec2->next->row == j)
389  {
390  sparsevec *tmp = vec1;
391  vec1 = vec2;
392  vec2 = tmp;
393  j = i; // Used below.
394  }
395  else
396  return;
397 
398  sparsevec *tmp = vec1->next;
399  if (vec2 != tmp)
400  {
401  vec1->next = tmp->next;
402  tmp->next = vec2->next;
403  vec2->next = tmp;
404  }
405  tmp->row = j;
406  v = head.next;
407 }

516 // v := v+w, w := 0
517 {

519  if (v == 0)
520  {
521  v = w;
522  w = 0;
523  return;
524  }
525  sparsevec head;
526  sparsevec *result = &head;
527  while (true)
528  if (v->row < w->row)
529  {
530  result->next = w;
531  result = result->next;
532  w = w->next;
533  if (w == 0)
534  {
535  result->next = v;
536  v = head.next;
537  return;
538  }
539  }
540  else if (v->row > w->row)
541  {
542  result->next = v;
543  result = result->next;
544  v = v->next;
545  if (v == 0)
546  {
547  result->next = w;
548  v = head.next;
549  w = 0;
550  return;
551  }
552  }
553  else
554  {
555  sparsevec *tmv = v;
556  sparsevec *tmw = w;
557  v = v->next;
558  w = w->next;
559  ring().add(tmv->coeff, tmv->coeff, tmw->coeff);
560  if (ring().is_zero(tmv->coeff))
561  {
562  vec_remove_node(tmv);
563  }
564  else
565  {
566  result->next = tmv;
567  result = result->next;
568  }
569  vec_remove_node(tmw);
570  if (w == 0)
571  {
572  result->next = v;
573  v = head.next;
574  return;
575  }
576  if (v == 0)
577  {
578  result->next = w;

586 template <typename CoeffRing>
587 void SMat<CoeffRing>::vec_row_op(sparsevec *&v,
588  size_t r1,
589  const elem &a,
590  size_t r2) const
591 // row(r1 in v) := row(r1 in v) + a * row(r2 in v)
592 {
593  sparsevec *p;
594  sparsevec *vec2 = 0;
595  for (p = v; p != 0; p = p->next)
596  if (p->row == r2)
597  {
598  vec2 = p;
599  break;
600  }
601  if (vec2 == 0) return;
602  elem c;
603  ring().init(c);
604  ring().set_zero(c);
605  ring().mult(c, vec2->coeff, a);
606  if (ring().is_zero(c))
607  {
608  ring().clear(c);
609  return; // nothing to change
610  }
611  // Now add c to the r1'th row of v
612  sparsevec head;
613  head.next = v;
614  for (p = &head; p->next != 0; p = p->next)
615  if (p->next->row <= r1) break;
616  if (p->next == 0 || p->next->row < r1)
617  {
618  // Make a new node
619  sparsevec *w = vec_new();
620  w->next = p->next;
621  w->row = r1;
622  ring().init_set(w->coeff, c);
623  p->next = w;
624  }
625  else
626  {
627  ring().add(p->next->coeff, p->next->coeff, c);
628  if (ring().is_zero(p->next->coeff))
629  {
630  sparsevec *tmp = p->next;
631  p->next = tmp->next;
632  vec_remove_node(tmp);
633  }
634  }
635  v = head.next;
636  ring().clear(c);
637 }

639 template <typename CoeffRing>
640 void SMat<CoeffRing>::vec_row_op2(sparsevec *&v,
641  size_t r1,
642  size_t r2,
643  const elem &a1,
644  const elem &a2,
645  const elem &b1,
646  const elem &b2) const
647 // row(r1 in v) := a1 * row(r1 in v) + a2 * row(r2 in v)
648 // row(r2 in v) := b1 * row(r1 in v) + b2 * row(r2 in v) (RHS refers to previous
649 // values)
650 {
651  // v[row r1] = a1 * v[r1] + a2 * v[r2]
652  // v[row r2] = b1 * v[r1] + b2 * v[r2]
653  elem e1, e2, c1, c2, c3, c4;
654 
655  ring().init(c1);
656  ring().init(c2);
657  ring().init(c3);
658  ring().init(c4);
659  ring().init(e1);
660  ring().init(e2);
661  ring().set_zero(c1);
662  ring().set_zero(c2);
663  ring().set_zero(c3);
664  ring().set_zero(c4);
665  ring().set_zero(e1);
666  ring().set_zero(e2);
667  bool r1_nonzero = vec_get_entry(v, r1, e1);
668  bool r2_nonzero = vec_get_entry(v, r2, e2);
669  if (!r1_nonzero && !r2_nonzero)
670  {
671  ring().clear(c1);
672  ring().clear(c2);
673  ring().clear(c3);
674  ring().clear(c4);
675  ring().clear(e1);
676  ring().clear(e2);
677  return;
678  }
679  if (r1_nonzero)
680  {
681  ring().mult(c1, a1, e1);
682  ring().mult(c3, b1, e1);
683  }
684  if (r2_nonzero)
685  {
686  ring().mult(c2, a2, e2);
687  ring().mult(c4, b2, e2);
688  }
689 
690  ring().add(c1, c1, c2);
