98 ResPolynomialConstructor(const ResPolyRing& R) : mRing(R)

102
103 ~ResPolynomialConstructor() {
104 mRing.vectorArithmetic().deallocateElementArray(coeffs);

106 void appendMonicTerm(res_packed_monomial monom)
107 {
108 monoms.push_back(monom); // a pointer
109 mRing.vectorArithmetic().pushBackOne(coeffs);
110 }
111
112 void pushBackTerm(res_packed_monomial monom)
113 {
114 monoms.push_back(monom); // a pointer
115 }
116
117 ElementArray& coefficientInserter() { return coeffs; }
118 void setPoly(ResPolynomial& result)
119 {
120 ncalls++;
121 result.len = static_cast<int>(mRing.vectorArithmetic().size(coeffs));
122 std::swap(result.coeffs, coeffs);
123 result.monoms.resize(result.len * mRing.monoid().max_monomial_size());
124
125 // copy monoms: not pointers, actual monoms
126 res_monomial_word* monomptr = result.monoms.data();
127 for (int i = 0; i < result.len; i++)
128 {
129 mRing.monoid().copy(monoms[i], monomptr);
130 monomptr += mRing.monoid().monomial_size(monoms[i]);
131 }
132 }
133
134 static void setPolyFromArrays(ResPolynomial& result,
135 int len,
136 ElementArray& coeffs,
137 std::vector<res_monomial_word>& monoms)
138 {
139 ncalls_fromarray++;
140 result.len = len;
141 result.coeffs.swap(coeffs);
142 std::swap(result.monoms, monoms);
143 }
144};
