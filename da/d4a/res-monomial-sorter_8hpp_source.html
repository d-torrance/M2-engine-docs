63class ResMonomialSorter
64{
65private:
66 const Monoid& mMonoid;
67 const ResMonoid& mResMonoid;
68 const ResSchreyerOrder& mSchreyerOrder;
69 const std::vector<res_packed_monomial>& mColumns;
70
71 long mNumComparisons;
72 memt::Arena mArena;
73 std::vector<int*> mMonoms; // each monom: [tiebreaker, basecomp, followed by totalmon]
74 std::vector<int> mPositions;
75public:
76 ResMonomialSorter(const Monoid& M,
77 const ResMonoid& resMonoid,
78 const ResSchreyerOrder& S, // order at level-1 in free res
79 const std::vector<res_packed_monomial>& columns // at level.
80 ) :
81 mMonoid(M),
82 mResMonoid(resMonoid),
83 mSchreyerOrder(S),
84 mColumns(columns),
85 mNumComparisons(0)
86 {
87 }
88
89 void setMonoms()
90 {
91 for (int i=0; i<mColumns.size(); i++)
92 {
93 std::pair<int*, int*> mon = mArena.allocArrayNoCon<int>(mMonoid.monomial_size() + 2);
94 toMonomial(mColumns[i], mon);
95 mMonoms.push_back(mon.first);
96 }
97 }
98
99 bool ordered()
100 {
101 setMonoms();
102 MonomialSorterObject C(mMonoid, mMonoms);
103 for (int i=1; i<mMonoms.size(); i++)
104 {
105 if (not C(i-1,i)) return false;
106 }
107 return true;
108 }
109
110 std::vector<int> sort()
111 {
112 setMonoms();
113
114 std::vector<int> result;
115
116 for (int i=0; i<mColumns.size(); i++)
117 result.push_back(i);
118
119 MonomialSorterObject C(mMonoid, mMonoms);
120 C.resetNumComparisons();
121
122 std::stable_sort(result.begin(), result.end(), C);
123
124 mNumComparisons = C.numComparisons();
125 return result;
126 }
127
128 long numComparisons() const { return mNumComparisons; }
129private:
130 void toMonomial(res_packed_monomial mon, std::pair<int*,int*> resultAlreadyAllocateds)
131 {
132 int comp, comp2;
133 int nvars = mMonoid.n_vars();
134 std::pair<int*, int*> exp = mArena.allocArrayNoCon<int>(nvars);
135 std::pair<int*, int*> exp2 = mArena.allocArrayNoCon<int>(nvars);
136 mResMonoid.to_expvector(mon, exp.first, comp);
137 mResMonoid.to_expvector(mSchreyerOrder.mTotalMonom[comp], exp2.first, comp2);
138 exponents::mult(nvars, exp.first, exp2.first, exp2.first);
139 auto p = resultAlreadyAllocateds.first;
140 *p++ = mSchreyerOrder.mTieBreaker[comp];
141 *p++ = comp2;
142 mMonoid.from_expvector(exp2.first, p);
143 mArena.freeTop(exp2.first); // note: can only pop one at a time from an mt::Arena!
144 mArena.freeTop(exp.first);
145 }
146};
