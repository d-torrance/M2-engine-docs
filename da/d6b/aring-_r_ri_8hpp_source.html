226 void power(ElementType &result, const ElementType &a, int n) const
227 {
228 if (n >= 2)
229 {
230 if (n%2 == 0)
231 {
232 ElementType b;
233 init(b);
234 power(b,a,n/2);
235 mult(result,b,b);
236 }
237 else
238 {
239 ElementType b;
240 init(b);
241 power(b,a,n-1);
242 mult(result,a,b);
243 }
244 }
245 else if (n == 1)
246 mpfi_set(&result,&a);
247 else if (n == 0)
248 mpfi_set_si(&result,1);
249 else if (n<0)
250 throw 20;
251 }
