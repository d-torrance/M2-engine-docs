16 ResolutionComputation *ResolutionComputation::choose_res(
17  const Matrix *m,
18  M2_bool resolve_cokernel,
19  int max_level,
20  M2_bool use_max_slanted_degree,
21  int max_slanted_degree,
22  int algorithm,
23  int strategy)
24 {
25  // The following modification is because some algorithms do not work if
26  // max_level is 0.
27  // github issue (crash, #368).
28  if (max_level <= 0) max_level = 1;
29 
30  const Ring *R = m->get_ring();
31  ResolutionComputation *C = nullptr;
32  int origsyz;
33  // First, we need to check that m is homogeneous, and that
34  // the heft values of the variables are all positive.
35  // All of these algorithms also assume that R is a polynomial ring.
36 
37  const M2FreeAlgebraOrQuotient *NCP = R->cast_to_M2FreeAlgebraOrQuotient();
38  if (NCP != nullptr)
39  {
40  if (M2_gbTrace > 0) emit_line("NC resolution");
41  C = createNCRes(m, max_level, strategy);
42  return C;
43  }
44  const PolynomialRing *P = R->cast_to_PolynomialRing();
45  if (P == nullptr)
46  {
47  ERROR("engine resolution strategies all require a polynomial base ring");
48  return nullptr;
49  }
50  const Ring* K = P->getCoefficientRing();
51  if (K->get_precision() != 0)
52  {
53  ERROR("free resolutions over polynomial rings with RR or CC coefficients not yet implemented");
54  return nullptr;
55  }
56  if (!P->getMonoid()->primary_degrees_of_vars_positive())
57  {
58  ERROR(
59  "engine resolution strategies all require a Heft vector which is "
60  "positive for all variables");
61  return nullptr;
62  }
63  if (algorithm < 4 and !m->is_homogeneous())
64  {
65  ERROR("engine resolution strategies require a homogeneous module");
66  return nullptr;
67  }
68 
69  switch (algorithm)
70  {
71  case 1:
72  if (!resolve_cokernel)
73  {
74  ERROR(
75  "resolution Strategy=>1 cannot resolve a cokernel with a given "
76  "presentation: use Strategy=>2 or Strategy=>3 instead");
77  return nullptr;
78  }
79  if (!R->is_commutative_ring())
80  {
81  ERROR(
82  "use resolution Strategy=>2 or Strategy=>3 for non commutative "
83  "polynomial rings");
84  return nullptr;
85  }
86  if (M2_gbTrace > 0) emit_line("resolution Strategy=>1");
87  C = new res_comp(m, max_level, strategy);
88  break;
89  case 0:
90  if (!resolve_cokernel)
91  {
92  ERROR(
93  "resolution Strategy=>0 cannot resolve a cokernel with a given "
94  "presentation: use Strategy=>2 or Strategy=>3 instead");
95  return nullptr;
96  }
97  if (!R->is_commutative_ring())
98  {
99  ERROR(
100  "use resolution Strategy=>2 or Strategy=>3 for non commutative "
101  "polynomial rings");
102  return nullptr;
103  }
104  if (M2_gbTrace > 0) emit_line("resolution Strategy=>0");
105  C = new res2_comp(
106  m, max_level, use_max_slanted_degree, max_slanted_degree, strategy);
107  break;
108  case 2:
109  origsyz = m->n_cols();
110  if (M2_gbTrace > 0) emit_line("resolution Strategy=>2");
111  C = new gbres_comp(m, max_level + 1, origsyz, strategy);
112  break;
113  case 3:
114  origsyz = m->n_cols();
115  if (M2_gbTrace > 0) emit_line("resolution Strategy=>3");
116  C = new gbres_comp(
117  m, max_level + 1, origsyz, strategy | STRATEGY_USE_HILB);
118  break;
119  case 4:
120  case 5:
121  if (!resolve_cokernel)
122  {
123  ERROR(
124  "resolution Strategy=>4 cannot resolve a cokernel with a given "
125  "presentation: use Strategy=>2 or Strategy=>3 instead");
126  return nullptr;
127  }
128  if (!P->is_skew_commutative() and !R->is_commutative_ring())
129  {
130  ERROR(
131  "use resolution Strategy=>2 or Strategy=>3 for non commutative "
132  "polynomial rings");
133  return nullptr;
134  }
135  if (M2_gbTrace > 0) emit_line("resolution Strategy=>4 (res-f4)");
136  C = createF4Res(m, max_level, strategy);
137  if (C == nullptr) return nullptr;
138  break;
139  }
140  if (C == nullptr)
141  {
142  ERROR("unknown resolution algorithm");
143  return nullptr;
144  }
145  intern_res(C);
146  return C;
147 }

155 M2_arrayint ResolutionComputation::betti_make(int lo,
156  int hi,
157  int len,
158  int *bettis)
159 {
160  int d, lev;
161  int hi1 = hi + 1;
162  int len1 = len + 1;
163 
164  // Reset 'hi1' to reflect the top degree that occurs
165  for (d = hi; d >= lo; d--)
166  {
167  for (lev = 0; lev <= len; lev++)
168  if (bettis[lev + (len + 1) * (d - lo)] > 0)
169  {
170  hi1 = d;
171  break;
172  }
173  if (hi1 <= hi) break;
174  }
175  if (hi1 > hi) hi1 = hi;
176 
177  // Reset 'len1' to reflect the top level that occurs
178  for (lev = len; lev >= 0; lev--)
179  {
180  for (d = lo; d <= hi1; d++)
181  if (bettis[lev + (len + 1) * (d - lo)] > 0)
182  {
183  len1 = lev;
184  break;
185  }
186  if (len1 <= len) break;
187  }
188  if (len1 > len) len1 = len;
189 
190  int totallen = (hi1 - lo + 1) * (len1 + 1);
191  M2_arrayint result = M2_makearrayint(3 + totallen);
192 
193  result->array[0] = lo;
194  result->array[1] = hi1;
195  result->array[2] = len1;
196 
197  int next = 3;
198  for (d = lo; d <= hi1; d++)
199  for (lev = 0; lev <= len1; lev++)
200  result->array[next++] = bettis[lev + (len + 1) * (d - lo)];
201 
202  return result;
203 }
