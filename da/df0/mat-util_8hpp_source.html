15 {
16  // Assumption: Mat is either DMat<RingType> or SMat<RingType>, in that it
17  // defines the following:
18  // Mat::ElementType
19  // A.ring()
20  // A.numRows(), A.numColumns()
21  // A.entry(r,c)
22  // and A.ring().elem_text_out
23 
24  size_t nrows = A.numRows();
25  size_t ncols = A.numColumns();
26  buffer* p = new buffer[nrows];
27  size_t r;
28  for (size_t c = 0; c < ncols; c++)
29  {
30  size_t maxcount = 0;
31  for (r = 0; r < nrows; r++)
32  {
33  const typename Mat::ElementType& a = A.entry(r, c);
34  if (!A.ring().is_zero(a))
35  A.ring().elem_text_out(p[r], a, true, false, false);
36  else
37  p[r] << ".";
38  if (p[r].size() > maxcount) maxcount = p[r].size();
39  }
40  for (r = 0; r < nrows; r++)
41  for (size_t k = maxcount + 1 - p[r].size(); k > 0; k--) p[r] << ' ';
42  }
43  for (r = 0; r < nrows; r++)
44  {
45  p[r] << '\0';
46  char* s = p[r].str();

48  }
49  delete[] p;
50 }
