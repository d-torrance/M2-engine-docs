14bool almostEqual(const M2::ARingRRR& R,
15 int nbits,
16 const M2::ARingRRR::ElementType& a,
17 const M2::ARingRRR::ElementType& b)
18{
19 mpfr_t epsilon;
20 mpfr_init2(epsilon, R.get_precision());
21 mpfr_set_ui_2exp(epsilon, 1, -nbits, MPFR_RNDN);
22
23 M2::ARingRRR::ElementType c;
24 R.init(c);
25 R.subtract(c, a, b);
26 bool ret = mpfr_cmpabs(&c, epsilon) < 0;
27
28 R.clear(c);
29 mpfr_clear(epsilon);
30 return ret;
31}

167TEST(ARingRRR, axioms)
168{
169 M2::ARingRRR R(100);
170 ARingElementGenerator<M2::ARingRRR> gen(R);
171 M2::ARingRRR::ElementType a, b, c, d, e;
172 R.init(a);
173 R.init(b);
174 R.init(c);
175 R.init(d);
176 R.init(e);
177 for (int i = 0; i < ntrials; i++)
178 {
179 gen.nextElement(a);
180 gen.nextElement(b);
181 gen.nextElement(c);
182 // Test commutativity
183 // test: a*b = b*a
184 // test: a+b == b+a
185 R.add(d, a, b);
186 R.add(e, b, a);
187 EXPECT_TRUE(almostEqual(R, 98, d, e));
188 R.mult(d, a, b);
189 R.mult(e, b, a);
190 EXPECT_TRUE(almostEqual(R, 98, d, e));
191
192 // Test associativity
193 // test: a+(b+c) == (a+b)+c
194 // test: a*(b*c) == (a*b)*c
195 R.add(e, b, c);
196 R.add(d, a, e); // a+(b+c)
197 R.add(e, a, b);
198 R.add(e, e, c); // (a+b)+c
199 EXPECT_TRUE(almostEqual(R, 94, d, e));
200 R.mult(e, b, c);
201 R.mult(d, a, e); // a*(b*c)
202 R.mult(e, a, b);
203 R.mult(e, e, c); // (a*b)*c
204 EXPECT_TRUE(almostEqual(R, 94, d, e));
205
206 // Test distributivity
207 // test: a*(b+c) == a*b + a*c
208 R.add(e, b, c);
209 R.mult(d, a, e); // a*(b+c)
210 R.mult(b, a, b);
211 R.mult(c, a, c);
212 R.add(e, b, c); // a*b + a*c
213 EXPECT_TRUE(almostEqual(R, 94, d, e));
214 }
215 R.clear(e);
216 R.clear(d);
217 R.clear(c);
218 R.clear(b);
219 R.clear(a);
220}
