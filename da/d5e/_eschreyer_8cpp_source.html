71 int GBKernelComputation::calc()
72 {
73  // First find the skeleton
74  for (int i = 0; i < gb.size(); i++) new_pairs(i);
75 
76  // Debug code
77  GBMatrix *mm = new GBMatrix(G);
78  for (int p = 0; p < syzygies.size(); p++)
79  mm->append(GR->gbvector_copy(syzygies[p]));
80  buffer o;

82  if (M2_gbTrace >= 5)
83  {
84  o << "skeleton = " << newline;
85  m->text_out(o);
86  emit(o.str());

88 
89  // Sort the skeleton now?
90 
91  // Now reduce each one of these elements
92  for (int j = 0; j < syzygies.size(); j++)
93  {
94  gbvector *v = s_pair(syzygies[j]);

130 {
131  const gc_vector<gbvector*> &g = m->elems;
132  int i;
133  int *components = newarray_atomic_clear(int, F->rank());

145  last->next = t;

150  }
151  for (i = 0; i < F->rank(); i++) mi.push_back(new MonomialIdeal(R));

155 void GBKernelComputation::new_pairs(int i)
156 // Create and insert all of the pairs which will have lead term 'gb[i]'.
157 // This also places 'in(gb[i])' into the appropriate monomial ideal
158 {
159  gc_vector<Bag*> elems;
160  gc_vector<int> vp; // This is 'p'.
161  gc_vector<int> thisvp;
162 
163  if (SF)
164  {
165  monomial PAIRS_mon = ALLOCATE_MONOMIAL(monom_size);
166  SF->schreyer_down(gb[i]->monom, gb[i]->comp - 1, PAIRS_mon);
167  M->to_varpower(PAIRS_mon, vp);
168  }
169  else
170  M->to_varpower(gb[i]->monom, vp);
171 
172  // First add in syzygies arising from exterior variables
173  // At the moment, there are none of this sort.
174 
175  if (R->is_skew_commutative())
176  {
177  exponents_t find_pairs_exp = newarray_atomic(int, M->n_vars());
178 
179  varpower::to_expvector(M->n_vars(), vp.data(), find_pairs_exp);
180  for (int w = 0; w < R->n_skew_commutative_vars(); w++)
181  if (find_pairs_exp[R->skew_variable(w)] > 0)
182  {
183  thisvp.resize(0);
184  varpower::var(w, 1, thisvp);
185  Bag *b = new Bag(static_cast<void *>(nullptr), thisvp);
186  elems.push_back(b);
187  }
188 
189  freemem(find_pairs_exp);
190  }
191 
192  // Second, add in syzygies arising from the base ring, if any
193  // The baggage of each of these is NULL
194  if (R->is_quotient_ring())
195  {
196  const MonomialIdeal *Rideal = R->get_quotient_monomials();
197  for (Bag& a : *Rideal)
198  {
199  // Compute (P->quotient_ideal->monom : p->monom)
200  // and place this into a varpower and Bag, placing
201  // that into 'elems'
202  thisvp.resize(0);
203  varpower::quotient(a.monom().data(), vp.data(), thisvp);
204  if (varpower::is_equal(a.monom().data(), thisvp.data()))
205  continue;
206  Bag *b = new Bag(static_cast<void *>(nullptr), thisvp);
207  elems.push_back(b);
208  }
209  }
210 
211  // Third, add in syzygies arising from previous elements of this same level
212  // The baggage of each of these is their corresponding res2_pair
213 
214  MonomialIdeal *mi_orig = mi[gb[i]->comp - 1];
215  for (Bag& a : *mi_orig)
216  {
217  Bag *b = new Bag();
218  varpower::quotient(a.monom().data(), vp.data(), b->monom());
219  elems.push_back(b);
220  }
221 
222  // Make this monomial ideal, and then run through each minimal generator
223  // and insert into the proper degree. (Notice that sorting does not
224  // need to be done yet: only once that degree is about to begin.
225 
226  mi_orig->insert_minimal(new Bag(i, vp));
227 
228  MonomialIdeal *new_mi = new MonomialIdeal(R, elems);
229 
230  monomial m = M->make_one();
231  for (Bag& a : *new_mi)
232  {
233  M->from_varpower(a.monom().data(), m);
234  M->mult(m, gb[i]->monom, m);
235 
236  gbvector *q = make_syz_term(
237  GR->get_flattened_coefficients()->from_long(1), m, i + 1);
238  syzygies.push_back(q);
239  }
240 }

246 bool GBKernelComputation::find_ring_divisor(const_exponents exp,
247  const gbvector *&result)
248 // If 'exp' is divisible by a ring lead term, then 1 is returned,
249 // and result is set to be that ring element.
250 // Otherwise 0 is returned.
251 {

260 int GBKernelComputation::find_divisor(const MonomialIdeal *this_mi,
261  const_exponents exp,
262  int &result)
263 {
264  // Find all the possible matches, use some criterion for finding the best...
265  gc_vector<Bag*> bb;
266  this_mi->find_all_divisors(exp, bb);
267  int ndivisors = bb.size();
268  if (ndivisors == 0) return 0;
269  result = bb[0]->basis_elem();
270  // Now search through, and find the best one. If only one, just return it.
271  if (M2_gbTrace >= 5)
272  if (this_mi->size() > 1)
273  {
274  buffer o;
275  o << ":" << this_mi->size() << "." << ndivisors << ":";
276  emit(o.str());
277  }
278  if (ndivisors == 1)
279  {
280  if (this_mi->size() == 1)
281  n_ones++;
282  else
283  n_unique++;
284  return 1;
285  }
286  n_others++;
287 
288  int lowest = result;
289  for (int i = 1; i < ndivisors; i++)
290  {
291  int p = bb[i]->basis_elem();
292  if (p < lowest) lowest = p;
293  }
294  result = lowest;
295  return ndivisors;
296 }

298 gbvector *GBKernelComputation::s_pair(gbvector *gsyz)
299 {
300  gbvector *result = nullptr;
301  monomial si = M->make_one();
302  for (gbvector *f = gsyz; f != nullptr; f = f->next)
303  {
304  SG->schreyer_down(f->monom, f->comp - 1, si);
305  gbvector *h = GR->mult_by_term(F, gb[f->comp - 1], f->coeff, si, 0);
306  wipe_unneeded_terms(h);
307  GR->gbvector_add_to(F, result, h);
308  }
309  M->remove(si);
310  return result;

314 {

317  exponents_t exp = newarray_atomic(int, GR->n_vars());

322  {

328  {

331  }

334  // Want to dump this term

352 void GBKernelComputation::reduce(gbvector *&f, gbvector *&fsyz)
353 {
354  exponents_t REDUCE_exp = ALLOCATE_EXPONENTS(exp_size);
355  monomial REDUCE_mon = ALLOCATE_MONOMIAL(monom_size);
356 
357  const Ring *gbringK = GR->get_flattened_coefficients();
358  ring_elem one = gbringK->from_long(1);
359  gbvector *lastterm = fsyz; // fsyz has only ONE term.
360  const gbvector *r;
361  int q;
362 
363  int nhits = 0;
364  int nremoved = 0;
365  int max_len = GR->gbvector_n_terms(f);
366 
367  int count = 0;
368  if (M2_gbTrace >= 4) emit_wrapped(",");
369 
370  while (f != nullptr)
371  {
372  if (SF)
373  {
374  SF->schreyer_down(f->monom, f->comp - 1, REDUCE_mon);
375  M->to_expvector(REDUCE_mon, REDUCE_exp);
376  }
377  else
378  M->to_expvector(f->monom, REDUCE_exp);
379  if (find_ring_divisor(REDUCE_exp, r))
380  {
381  ring_elem u, v;
382  // Subtract off f, leave fsyz alone
383  M->divide(f->monom, r->monom, REDUCE_mon);
384  gbringK->syzygy(f->coeff, r->coeff, u, v);
385  gbvector *h = GR->mult_by_term(F, r, v, REDUCE_mon, f->comp);
386  GR->gbvector_add_to(F, f, h);
387  if (!gbringK->is_equal(u, one))
388  {
389  GR->gbvector_mult_by_coeff_to(fsyz, u);
390  GR->gbvector_mult_by_coeff_to(f, u);
391  gbringK->remove(u);
392  gbringK->remove(v);
393  }
394  total_reduce_count++;
395  count++;
396  }
397  else if (find_divisor(mi[f->comp - 1], REDUCE_exp, q))
398  {
399  ring_elem u, v;
400  gbringK->syzygy(f->coeff, gb[q]->coeff, u, v);
401  M->divide(f->monom, gb[q]->monom, REDUCE_mon);
402  gbvector *h = GR->mult_by_term(F, gb[q], v, REDUCE_mon, 0);
403  if (!gbringK->is_equal(u, one))
404  {
405  GR->gbvector_mult_by_coeff_to(fsyz, u);
406  GR->gbvector_mult_by_coeff_to(f, u);
407  }
408  int n1 = GR->gbvector_n_terms(h);
409  wipe_unneeded_terms(h);
410  int n2 = GR->gbvector_n_terms(h);
411  nremoved += (n1 - n2);
412  lastterm->next = make_syz_term(v, f->monom, q + 1); // grabs v.
413  lastterm = lastterm->next;
414  gbringK->remove(u);
415  gbringK->remove(v);
416  int n3 = GR->gbvector_n_terms(f);
417  GR->gbvector_add_to(F, f, h);
418  int n4 = GR->gbvector_n_terms(f);
419  if (n4 > max_len) max_len = n4;
420  nhits +=
421  (n2 + n3 - n4 -
422  2); // the -2 is to avoid counting the lead term cancellations.
423  total_reduce_count++;
424  count++;
425  }
426  else
427  {
428  // To get here is an ERROR!
429  f = f->next; // Just to not have an infinite loop
430  emit_line(
431  "error in Schreyer reduction: element does not reduce to zero!");
432  }
433  }
434 
435  if (M2_gbTrace >= 4)
436  {
437  buffer o;
438  o << count;
439  o << "[" << max_len << " " << nhits << " " << nremoved << "]";
440  emit_wrapped(o.str());
441  }
442 }
