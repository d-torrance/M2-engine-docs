282 void power(ElementType& result, const ElementType& a, int n) const
283 {
284 if (n < 0)

293 void power_mpz(ElementType& result, const ElementType& a, mpz_srcptr n) const
294 {
295 if (mpz_sgn(n) < 0)
296 invert(result, a);
297 else
298 copy(result, a);
299
300 mpz_t abs_n;
301 mpz_init(abs_n);
302 mpz_abs(abs_n, n);
303
304 fmpz_t fn;
305 fmpz_init_set_readonly(fn, abs_n);
306 fq_zech_pow(&result, &result, fn, mContext);
307 fmpz_clear_readonly(fn);
308 mpz_clear(abs_n);
309 }
