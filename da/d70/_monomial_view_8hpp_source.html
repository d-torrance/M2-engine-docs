50 inline static bool monomialDivides(const MonomialView& divisor,
51 const MonomialView& divisee)
52 {
53 if (divisor.size() > divisee.size()) return false;
54 auto divisorIt = divisor.begin();
55 auto diviseeIt = divisee.begin();
56 auto divisorEnd = divisor.end();
57 auto diviseeEnd = divisee.end();
58 while (divisorIt != divisorEnd)
59 {
60 auto divisorVar = divisorIt.var();
61 while (true)
62 {
63 if (diviseeIt == diviseeEnd) return false;
64 if (diviseeIt.var() > divisorVar) return false;
65 if (diviseeIt.var() == divisorVar) break;
66 ++diviseeIt;
67 }
68 // at this point, both diviseeIt.var() and divisorIt.var() are equal
69 if (divisorIt.power() > diviseeIt.power()) return false;
70 ++divisorIt;
71 ++diviseeIt;
72 }
73 return true;
74 }
