199  void add(elem &result, elem a, elem b) const
200  {
201  if (a == 0)
202  result = b;
203  else if (b == 0)
204  result = a;
205  else
206  {
207  int n = a - b;
208  if (n > 0)
209  {
210  if (n == mGF.minusOne())
211  result = 0;
212  else
213  result = modulus_add(b, mGF.oneTable(n), mGF.orderMinusOne());
214  }
215  else if (n < 0)
216  {
217  if (-n == mGF.minusOne())
218  result = 0;
219  else
220  result = modulus_add(a, mGF.oneTable(-n), mGF.orderMinusOne());
221  }
222  else
223  {
224  if (mGF.characteristic() == 2)
225  result = 0;
226  else
227  result =
228  modulus_add(a, mGF.oneTable(mGF.one()), mGF.orderMinusOne());
