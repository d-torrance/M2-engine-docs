32void GaussElimComputation::insert(gm_elem *p)
33{
34 if (p->f == nullptr)
35 {
36 if (p->fsyz != nullptr && collect_syz)
37 {
38 syz_list.push_back(p->fsyz);
39 n_syz++;
40 }
41 freemem(p);
42 }
43 else
44 {
45 ring_elem leadinv = R->invert(p->f->coeff);
46 R->mult_vec_to(p->f, leadinv, false);
47 R->mult_vec_to(p->fsyz, leadinv, false);
48 p->nterms = R->n_nonzero_terms(p->f);
49 int i = p->f->comp;
50 if (gb_list[i] == nullptr)
51 {
52 gb_list[i] = p;
53 n_gb++;
54 }
55 else
56 {
57 if (p->nterms < gb_list[i]->nterms)
58 {
59 gm_elem *q = p;
60 p = gb_list[i];
61 gb_list[i] = q;
62 }
63 p->next = reduce_list[i];
64 reduce_list[i] = p;
65 }
66 }
67}

130void GaussElimComputation::reduce(gm_elem *&p, gm_elem *q)
131{
132 // MES: rewrite.
133 // Reduce q by p. This is a single step reduction.
134 // The element is then inserted further down.
135
136 ring_elem c1 = p->f->coeff;
137 ring_elem c2 = q->f->coeff;
138 ring_elem d2 = R->negate(c2);
139 vec v1 = R->mult_vec(c1, q->f);
140 vec v2 = R->mult_vec(d2, p->f);
141 vec s1 = R->mult_vec(c1, q->fsyz);
142 vec s2 = R->mult_vec(d2, p->fsyz);
143 R->remove_vec(q->f);
144 R->remove_vec(q->fsyz);
145 R->remove(d2);
146 R->add_vec_to(v1, v2);
147 R->add_vec_to(s1, s2);
148 q->f = v1;
149 q->fsyz = s1;
150}

152void GaussElimComputation::reduce(vec &f)
153{
154 vecterm head;
155 vecterm *result = &head;
156
157 while (f != nullptr)
158 {
159 int r = f->comp;
160 if (gb_list[r] != nullptr)
161 {
162 // Reduce w.r.t. this term
163 ring_elem c = f->coeff;
164 c = R->negate(c);
165 vec g = R->mult_vec(c, gb_list[r]->f);
166 R->add_vec_to(f, g);
167 }
168 else
169 {
170 result->next = f;
171 f = f->next;
172 result = result->next;
173 }
174 }
175
176 result->next = nullptr;
177 f = head.next;
178}
179void GaussElimComputation::reduce(vec &f, vec &fsyz, bool tail_only)
180{
181 if (f == nullptr) return;
182 vecterm head;
183 vecterm *result = &head;
184
185 if (tail_only)
186 {
187 // Don't reduce the head term.
188 result->next = f;
189 f = f->next;
190 result = result->next;
191 }
192 while (f != nullptr)
193 {
194 int r = f->comp;
195 if (gb_list[r] != nullptr)
196 {
197 // Reduce w.r.t. this term
198 ring_elem c = f->coeff;
199 c = R->negate(c);
200 vec gsyz = R->mult_vec(c, gb_list[r]->fsyz);
201 vec g = R->mult_vec(c, gb_list[r]->f);
202 R->add_vec_to(f, g);
203 R->add_vec_to(fsyz, gsyz);
204 }
205 else
206 {
207 result->next = f;
208 f = f->next;
209 result = result->next;
210 }
211 }
212
213 result->next = nullptr;
214 f = head.next;
215}
216void GaussElimComputation::start_computation()
217{
218 if (status() == COMP_DONE) return;
219 for (; row >= 0; row--)
220 {
221 if (gb_list[row] == nullptr) continue;
222 while (reduce_list[row] != nullptr)
223 {
224 gm_elem *p = reduce_list[row];
225 reduce_list[row] = p->next;
226 p->next = nullptr;
227 reduce(gb_list[row], p); // replaces p
228 if (M2_gbTrace >= 3)
229 {
230 if (p->f == nullptr)
231 {
232 if (p->fsyz == nullptr)
233 emit_wrapped("o");
234 else
235 emit_wrapped("z");
236 }
237 else
238 emit_wrapped("r");
239 }
240 else
241 {
242 }
243 insert(p);
244 n_pairs++;
245 if (system_interrupted())
246 {
247 set_status(COMP_INTERRUPTED);
248 return;
249 }
250 if (n_pairs == stop_.pair_limit)
251 {
252 set_status(COMP_DONE_PAIR_LIMIT);
253 return;
254 }
255 if (n_syz == stop_.syzygy_limit)
256 {
257 set_status(COMP_DONE_SYZYGY_LIMIT);
258 return;
259 }
260 }
261 }
262 // Now auto reduce these
263 for (int r = 1; r < gens->n_rows(); r++)
264 {
265 if (gb_list[r] == nullptr) continue;
266 reduce(gb_list[r]->f, gb_list[r]->fsyz, true);
267 }
268 if (M2_gbTrace >= 10)
269 {
270 buffer o;
271 text_out(o);
272 emit(o.str());
273 }
274 set_status(COMP_DONE);
275}

343const Matrix /* or null */ *GaussElimComputation::matrix_remainder(
344 const Matrix *m)
345{

348 ERROR("encountered different rings");

350 }
351 if (m->n_rows() != gens->rows()->rank())
352 {
353 ERROR("expected matrices to have same number of rows");
354 return nullptr;
355 }
356 MatrixConstructor mat_remainder(m->rows(), m->cols(), m->degree_shift());
357 for (int i = 0; i < m->n_cols(); i++)

367M2_bool GaussElimComputation::matrix_lift(
368 const Matrix *m,
369 const Matrix /* or null */ **result_remainder,
370 const Matrix /* or null */ **result_quotient)
371{
372 if (m->get_ring() != R)
373 {
374 ERROR("encountered different rings");
375 *result_remainder = nullptr;
376 *result_quotient = nullptr;
377 return false;
378 }
379 if (m->n_rows() != gens->rows()->rank())
380 {
381 ERROR("expected matrices to have same number of rows");
382 *result_remainder = nullptr;
383 *result_quotient = nullptr;
384 return false;
385 }
386 MatrixConstructor mat_remainder(m->rows(), m->cols(), m->degree_shift());
387 MatrixConstructor mat_quotient(Fsyz, m->cols(), nullptr);
388 bool all_zeroes = true;
389 for (int i = 0; i < m->n_cols(); i++)
390 {
391 vec f = R->copy_vec(m->elem(i));
392 vec fsyz = nullptr;
393
394 reduce(f, fsyz);
395 R->negate_vec_to(fsyz);
396 if (f != nullptr) all_zeroes = false;
397 mat_remainder.set_column(i, f);
398 mat_quotient.set_column(i, fsyz);
399 }
400 *result_remainder = mat_remainder.to_matrix();
401 *result_quotient = mat_quotient.to_matrix();
402 return all_zeroes;
403}

405int GaussElimComputation::contains(const Matrix *m)
406// Return -1 if every column of 'm' reduces to zero.
407// Otherwise return the index of the first column that
408// does not reduce to zero.
409{
410 if (m->get_ring() != R)
411 {
412 ERROR("encountered different ring");
413 return -1;
414 }
415 // Reduce each column of m one by one.
416 for (int i = 0; i < m->n_cols(); i++)
417 {
418 vec f = R->copy_vec(m->elem(i));
419 reduce(f);
420 if (f != nullptr)
421 {
422 R->remove_vec(f);
423 return i;
424 }
425 }
426 return -1;
427}
