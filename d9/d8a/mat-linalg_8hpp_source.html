331 template <typename RT>
332 void mult(const DMat<RT>& A, const DMat<RT>& B, DMat<RT>& result_product)
333 {
334  // printf("entering dmat mult\n");
335  typedef typename RT::ElementType ElementType;
336  typedef typename DMat<RT>::ConstIterator ConstIterator;
337 
338  assert(A.numColumns() == B.numRows());
339  assert(A.numRows() == result_product.numRows());
340  assert(B.numColumns() == result_product.numColumns());
341 
342  ElementType* result = result_product.array();
343 
344  typename RT::Element tmp(A.ring());
345  // WARNING: this routine expects the result matrix to be in ROW MAJOR ORDER
346  for (size_t i = 0; i < A.numRows(); i++)
347  for (size_t j = 0; j < B.numColumns(); j++)
348  {
349  ConstIterator i1 = A.rowBegin(i);
350  ConstIterator iend = A.rowEnd(i);
351  ConstIterator j1 = B.columnBegin(j);
352 
353  while (i1 != iend)
354  {
355  A.ring().mult(tmp, *i1, *j1);
356  A.ring().add(*result, *result, tmp);
357  ++i1;
358  ++j1;
359  }
360  result++;
361  }
362 }

486 M2_arrayintOrNull LUincremental(std::vector<size_t>& P, DMat<RT>& LU, const DMat<RT>& v, int m)
487 {
488  size_t n = LU.numRows();
489 
490  // copy permuted v to m-th column of LU
491  // TODO: change to iter
492  for (size_t j = 0; j < n; j++)
493  LU.ring().set(LU.entry(j, m), v.entry(P[j], 0));
494 
495  // reduce the m-th column of LU and forward solve
496  DMat<RT> x{LU.ring(), n, 1};
497  triangularSolve(LU, x, m, 1);
498  // place solution of forward solve in U
499  // TODO: change to iter
500  for (size_t i = 0; i < m; i++)
501  LU.ring().set(LU.entry(i, m), x.entry(i, 0));
502 
503  // look for a pivot in L
504  int pivotPosition = -1;
505  // TODO: change to iter
506  for (size_t j = m; j < n; j++)
507  if (!LU.ring().is_zero(LU.entry(j, m)))
508  {
509  pivotPosition = j;
510  break;
511  }
512  // if no pivot found, return
513  if (pivotPosition == -1) return stdvector_to_M2_arrayint(P);
514  // otherwise swap rows and update P
515  MatElementaryOps<DMat<RT>>::interchange_rows(LU, pivotPosition, m);
516  std::swap(P[pivotPosition], P[m]);
517 
518  // scale column of L
519  // TODO: change to iter
520  for (int j = m + 1; j < n; j++)
521  LU.ring().divide(LU.entry(j, m), LU.entry(j, m), LU.entry(m, m));
522 
523  return stdvector_to_M2_arrayint(P);
524 }
