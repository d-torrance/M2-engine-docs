331template <typename RT>
332void mult(const DMat<RT>& A, const DMat<RT>& B, DMat<RT>& result_product)
333{
334 // printf("entering dmat mult\n");
335 typedef typename RT::ElementType ElementType;
336 typedef typename DMat<RT>::ConstIterator ConstIterator;
337
338 assert(A.numColumns() == B.numRows());
339 assert(A.numRows() == result_product.numRows());
340 assert(B.numColumns() == result_product.numColumns());
341
342 ElementType* result = result_product.array();
343
344 typename RT::Element tmp(A.ring());
345 // WARNING: this routine expects the result matrix to be in ROW MAJOR ORDER
346 for (size_t i = 0; i < A.numRows(); i++)
347 for (size_t j = 0; j < B.numColumns(); j++)
348 {
349 ConstIterator i1 = A.rowBegin(i);
350 ConstIterator iend = A.rowEnd(i);
351 ConstIterator j1 = B.columnBegin(j);
352
353 while (i1 != iend)
354 {
355 A.ring().mult(tmp, *i1, *j1);
356 A.ring().add(*result, *result, tmp);
357 ++i1;
358 ++j1;
359 }
360 result++;
361 }
362}

371template <typename RT>
372void subtractMultipleTo(DMat<RT>& C, const DMat<RT>& A, const DMat<RT>& B)
373// C = C - A*B
374{
375 typedef typename RT::ElementType ElementType;
376 typedef typename DMat<RT>::ConstIterator ConstIterator;
377
378 assert(A.numColumns() == B.numRows());
379 assert(A.numRows() == C.numRows());
380 assert(B.numColumns() == C.numColumns());
381
382 ElementType* result = C.array();
383
384 typename RT::Element tmp(A.ring());
385 // WARNING: this routine expects the result matrix to be in ROW MAJOR ORDER
386 for (size_t i = 0; i < A.numRows(); i++)
387 for (size_t j = 0; j < B.numColumns(); j++)
388 {
389 ConstIterator i1 = A.rowBegin(i);
390 ConstIterator iend = A.rowEnd(i);
391 ConstIterator j1 = B.columnBegin(j);
392
393 while (i1 != iend)
394 {
395 A.ring().mult(tmp, *i1, *j1);
396 A.ring().subtract(*result, *result, tmp);
397 ++i1;
398 ++j1;
399 }
400 result++;
401 }

431void triangularSolve(DMat<RT>& Lv, DMat<RT>& x, int m, int strategy)
432{

443 {

446 x.ring().negate(a, a);

450 break;
451 case 1:
452 // TODO: change to iter
453 for (size_t i = 0; i < m; i++)

459 }
460 break;
461 case 2:
462 // TODO: change to iter
463 for (size_t i = 1; i < m + 1; i++)
464 {
465 auto& a = x.entry(m - i, 0);
466 x.ring().divide(a, Lv.entry(m - i, m), Lv.entry(m - i, m - i));
467 x.ring().negate(a, a);
468 MatElementaryOps<DMat<RT>>::column_op(Lv, m, a, m - i);
469 x.ring().negate(a, a);
470 }
471 break;
472 case 3:
473 // TODO: change to iter
474 for (size_t i = 1; i < m + 1; i++)
475 {
476 auto& a = x.entry(m - i, 0);

478 MatElementaryOps<DMat<RT>>::column_op(Lv, m, a, m - i);
479 x.ring().negate(a, a);
480 }

485template <typename RT>
486M2_arrayintOrNull LUincremental(std::vector<size_t>& P, DMat<RT>& LU, const DMat<RT>& v, int m)
487{
488 size_t n = LU.numRows();
489
490 // copy permuted v to m-th column of LU
491 // TODO: change to iter
492 for (size_t j = 0; j < n; j++)
493 LU.ring().set(LU.entry(j, m), v.entry(P[j], 0));
494
495 // reduce the m-th column of LU and forward solve
496 DMat<RT> x{LU.ring(), n, 1};
497 triangularSolve(LU, x, m, 1);
498 // place solution of forward solve in U
499 // TODO: change to iter
500 for (size_t i = 0; i < m; i++)
501 LU.ring().set(LU.entry(i, m), x.entry(i, 0));
502
503 // look for a pivot in L
504 int pivotPosition = -1;
505 // TODO: change to iter
506 for (size_t j = m; j < n; j++)
507 if (!LU.ring().is_zero(LU.entry(j, m)))
508 {
509 pivotPosition = j;
510 break;
511 }
512 // if no pivot found, return
513 if (pivotPosition == -1) return stdvector_to_M2_arrayint(P);
514 // otherwise swap rows and update P
515 MatElementaryOps<DMat<RT>>::interchange_rows(LU, pivotPosition, m);
516 std::swap(P[pivotPosition], P[m]);
517
518 // scale column of L
519 // TODO: change to iter
520 for (int j = m + 1; j < n; j++)
521 LU.ring().divide(LU.entry(j, m), LU.entry(j, m), LU.entry(m, m));
522
523 return stdvector_to_M2_arrayint(P);
524}

1043 fmpz_mat_entry(m2, r, c));

1045 fmpz_mat_clear(m2);
1046 return nullity;
