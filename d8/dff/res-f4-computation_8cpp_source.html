25ResolutionComputation* createF4Res(const Matrix* groebnerBasisMatrix,
26 int max_level,
27 int strategy,
28 int numThreads,
29 bool parallelizeByDegree)
30{
31 // We expect the following to hold:
32 // the ring of groebnerBasisMatrix is a PolynomialRing, but not:
33 // quotient ring
34 // Weyl algebra
35 // We assume also that the matrix is homogeneous.
36 // If any of these are incorrect, an error message is provided, and
37 // null is returned.
38 const PolynomialRing* origR =
39 groebnerBasisMatrix->get_ring()->cast_to_PolynomialRing();
40 if (origR == nullptr)
41 {
42 ERROR("expected polynomial ring");
43 return nullptr;
44 }
45 if (origR->is_quotient_ring())
46 {
47 ERROR("cannot use res(...,FastNonminimal=>true) for quotient rings");
48 return nullptr;
49 }
50 if (origR->is_weyl_algebra())
51 {
52 ERROR("cannot use res(...,FastNonminimal=>true) over Weyl algebras");
53 return nullptr;
54 }
55 if (origR->is_solvable_algebra())
56 {
57 ERROR(
58 "cannot use res(...,FastNonminimal=>true) over non-commutative "
59 "algebras");
60 return nullptr;
61 }
62 // if (!groebnerBasisMatrix->is_homogeneous())
63 // {
64 // ERROR(
65 // "cannot use res(...,FastNonminimal=>true) with inhomogeneous input");
66 // return nullptr;
67 // }
68 // if (origR->getMonoid()->get_degree_ring()->n_vars() != 1)
69 // {
70 // ERROR("expected singly graded with positive degrees for the variables");
71 // return nullptr;
72 // }
73 if (!origR->getMonoid()->primary_degrees_of_vars_positive())
74 {
75 ERROR("expected the degree of each variable to be positive");
76 return nullptr;
77 }
78 // Still to check:
79 // (a) coefficients are ZZ/p, for p in range.
80
81 const Ring* K = origR->getCoefficients();
82 auto mo = origR->getMonoid()->getMonomialOrdering(); // mon ordering
83 auto motype = MonomialOrderingType::Weights;
84 if (moIsLex(mo))
85 motype = MonomialOrderingType::Lex;
86 else if (moIsGRevLex(mo))
87 motype = MonomialOrderingType::GRevLex;
88
89 auto MI = new ResMonoid(origR->n_vars(),
90 origR->getMonoid()->getPrimaryDegreeVector(),
91 origR->getMonoid()->getFirstWeightVector(),
92 motype);
93 ResPolyRing* R;
94 if (origR->is_skew_commutative())
95 {
96 R = new ResPolyRing(K, MI, origR->getMonoid(), &(origR->getSkewInfo()));
97 }
98 else
99 {
100 R = new ResPolyRing(K, MI, origR->getMonoid());
101 }
102 auto result = new F4ResComputation(origR, R, groebnerBasisMatrix, max_level, numThreads, parallelizeByDegree);
103
104 // Set level 0
105 // take the columns of the matrix, and insert them into mComp
106 const FreeModule* F = groebnerBasisMatrix->rows();
107 int maxdeg = 0;
108 if (F->rank() > 0)
109 {
110 maxdeg = F->primary_degree(0);
111 for (int j = 1; j < F->rank(); j++)
112 if (maxdeg < F->primary_degree(j)) maxdeg = F->primary_degree(j);
113 }
114
115 // Set level 0
116 // take the info from F, place it into mComp
117 SchreyerFrame& frame = result->frame();
118 for (int i = 0; i < F->rank(); i++)
119 {
120 res_packed_monomial elem =
121 frame.monomialBlock().allocate(MI->max_monomial_size());
122 MI->one(i, elem);
123 frame.insertLevelZero(elem, F->primary_degree(i), maxdeg);
124 }
125 frame.endLevel();
126
127 // At this point, we want to sort the columns of groebnerBasisMatrix.
128 Matrix* leadterms = groebnerBasisMatrix->lead_term();
129 M2_arrayint pos = leadterms->sort(1 /* ascending degree */,
130 -1 /* descending monomial order */);
131
132 std::vector<ResPolynomial> input_polys;
133 for (int i = 0; i < groebnerBasisMatrix->n_cols(); i++)
134 {
135 ResPolynomial f;
136 ResF4toM2Interface::from_M2_vec(*R, F, groebnerBasisMatrix->elem(i), f);
137 input_polys.emplace_back(std::move(f));
138 }
139
140 // Set level 1.
141 for (int j = 0; j < F->rank(); j++)
142 {
143 // Only insert the ones whose lead monomials are in component j:
144 for (int i = 0; i < pos->len; i++)
145 {
146 int loc = pos->array[i];
147 ResPolynomial& f = input_polys[loc];
148 if (f.len == 0) continue;
149 if (MI->get_component(f.monoms.data()) != j) continue;
150 res_packed_monomial elem =
151 frame.monomialBlock().allocate(MI->max_monomial_size());
152 MI->copy(f.monoms.data(), elem);
153 // the following line grabs f.
154 if (!frame.insertLevelOne(
155 elem, groebnerBasisMatrix->cols()->primary_degree(loc), f))
156 {
157 ERROR(
158 "input polynomials/vectors were computed in a non-compatible "
159 "monomial order");
160 // TODO: clean up.
161 return nullptr;
162 }
163 }
164 }
165 frame.endLevel();
166 // frame.show(0);
167 // Remove matrix:
168 delete leadterms;
169
170 return result;
171}

211M2_arrayint F4ResComputation::minimal_betti(M2_arrayint slanted_degree_limit,
212 M2_arrayint length_limit)
213{
214 bool stop_after_degree = (slanted_degree_limit->len == 1);
215 int top_slanted_degree = slanted_degree_limit->array[0];
216 int new_length_limit = (length_limit->len == 1 ? length_limit->array[0]
217 : frame().maxLevel() - 1);
218
219 BettiDisplay B = frame().minimalBettiNumbers(
220 stop_after_degree, top_slanted_degree, new_length_limit);
221 return B.getBetti();
222}
