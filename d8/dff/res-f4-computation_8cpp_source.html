25 ResolutionComputation* createF4Res(const Matrix* groebnerBasisMatrix,
26  int max_level,
27  int strategy)
28 {
29  // We expect the following to hold:
30  // the ring of groebnerBasisMatrix is a PolynomialRing, but not:
31  // quotient ring
32  // Weyl algebra
33  // We assume also that the matrix is homogeneous.
34  // If any of these are incorrect, an error message is provided, and
35  // null is returned.
36  const PolynomialRing* origR =
37  groebnerBasisMatrix->get_ring()->cast_to_PolynomialRing();
38  if (origR == 0)
39  {
40  ERROR("expected polynomial ring");
41  return nullptr;
42  }
43  if (origR->is_quotient_ring())
44  {
45  ERROR("cannot use res(...,FastNonminimal=>true) for quotient rings");
46  return nullptr;
47  }
48  if (origR->is_weyl_algebra())
49  {
50  ERROR("cannot use res(...,FastNonminimal=>true) over Weyl algebras");
51  return nullptr;
52  }
53  if (origR->is_solvable_algebra())
54  {
55  ERROR(
56  "cannot use res(...,FastNonminimal=>true) over non-commutative "
57  "algebras");
58  return nullptr;
59  }
60  // if (!groebnerBasisMatrix->is_homogeneous())
61  // {
62  // ERROR(
63  // "cannot use res(...,FastNonminimal=>true) with inhomogeneous input");
64  // return nullptr;
65  // }
66  // if (origR->getMonoid()->get_degree_ring()->n_vars() != 1)
67  // {
68  // ERROR("expected singly graded with positive degrees for the variables");
69  // return nullptr;
70  // }
71  if (!origR->getMonoid()->primary_degrees_of_vars_positive())
72  {
73  ERROR("expected the degree of each variable to be positive");
74  return nullptr;
75  }
76  // Still to check:
77  // (a) coefficients are ZZ/p, for p in range.
78 
79  const Ring* K = origR->getCoefficients();
80  ResGausser* KK = ResGausser::newResGausser(K);
81  if (KK == 0)
82  {
83  ERROR(
84  "cannot use res(...,FastNonminimal=>true) with this type of "
85  "coefficient ring");
86  return nullptr;
87  }
88 
89  auto mo = origR->getMonoid()->getMonomialOrdering(); // mon ordering
90  auto motype = MonomialOrderingType::Weights;
91  if (moIsLex(mo))
92  motype = MonomialOrderingType::Lex;
93  else if (moIsGRevLex(mo))
94  motype = MonomialOrderingType::GRevLex;
95 
96  auto MI = new ResMonoid(origR->n_vars(),
97  origR->getMonoid()->getPrimaryDegreeVector(),
98  origR->getMonoid()->getFirstWeightVector(),
99  motype);
100  ResPolyRing* R;
101  if (origR->is_skew_commutative())
102  {
103  R = new ResPolyRing(KK, MI, origR->getMonoid(), &(origR->getSkewInfo()));
104  }
105  else
106  {
107  R = new ResPolyRing(KK, MI, origR->getMonoid());
108  }
109  auto result = new F4ResComputation(origR, R, groebnerBasisMatrix, max_level);
110 
111  // Set level 0
112  // take the columns of the matrix, and insert them into mComp
113  const FreeModule* F = groebnerBasisMatrix->rows();
114  int maxdeg = 0;
115  if (F->rank() > 0)
116  {
117  maxdeg = F->primary_degree(0);
118  for (int j = 1; j < F->rank(); j++)
119  if (maxdeg < F->primary_degree(j)) maxdeg = F->primary_degree(j);
120  }
121 
122  // Set level 0
123  // take the info from F, place it into mComp
124  SchreyerFrame& frame = result->frame();
125  for (int i = 0; i < F->rank(); i++)
126  {
127  res_packed_monomial elem =
128  frame.monomialBlock().allocate(MI->max_monomial_size());
129  MI->one(i, elem);
130  frame.insertLevelZero(elem, F->primary_degree(i), maxdeg);
131  }
132  frame.endLevel();
133 
134  // At this point, we want to sort the columns of groebnerBasisMatrix.
135  Matrix* leadterms = groebnerBasisMatrix->lead_term();
136  M2_arrayint pos = leadterms->sort(1 /* ascending degree */,
137  -1 /* descending monomial order */);
138 
139  std::vector<ResPolynomial> input_polys;
140  for (int i = 0; i < groebnerBasisMatrix->n_cols(); i++)
141  {
142  ResPolynomial f;
143  ResF4toM2Interface::from_M2_vec(*R, F, groebnerBasisMatrix->elem(i), f);
144  input_polys.emplace_back(std::move(f));
145  }
146 
147  // Set level 1.
148  for (int j = 0; j < F->rank(); j++)
149  {
150  // Only insert the ones whose lead monomials are in component j:
151  for (int i = 0; i < pos->len; i++)
152  {
153  int loc = pos->array[i];
154  ResPolynomial& f = input_polys[loc];
155  if (f.len == 0) continue;
156  if (MI->get_component(f.monoms.data()) != j) continue;
157  res_packed_monomial elem =
158  frame.monomialBlock().allocate(MI->max_monomial_size());
159  MI->copy(f.monoms.data(), elem);
160  // the following line grabs f.
161  if (!frame.insertLevelOne(
162  elem, groebnerBasisMatrix->cols()->primary_degree(loc), f))
163  {
164  ERROR(
165  "input polynomials/vectors were computed in a non-compatible "
166  "monomial order");
167  // TODO: clean up.
168  return nullptr;
169  }
170  }
171  }
172  frame.endLevel();
173  // frame.show(0);
174  // Remove matrix:
175  delete leadterms;
176 
177  return result;
178 }
