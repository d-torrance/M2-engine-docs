49// PointArray
50class PointArray
51{
52 public:
53 using RealVector = std::vector<double>;
54 using Weight = double;
55 PointArray(Weight epsilon, const RealVector& weights)
56 : mEpsilon(epsilon), mWeights(weights)
57 {
58 }
59 PointArray(Weight epsilon, int n) : mEpsilon(epsilon)
60 {
61 double s = 0;
62 for (int i = 0; i < n; i++)
63 {
64 double r = rand();
65 s += r;
66 mWeights.push_back(r);
67 }
68 for (int i = 0; i < n; i++) mWeights[i] /= s;
69 }
70 virtual ~PointArray() {
71 // std::cerr << "entering ~PointArray()" << std::endl;
72 }
73 int lookup_or_append(const RealVector& a)
74 {
75 Weight w = weight(a);
76 int p = lookup(a);
77 if (p != -1) return p;
78 if (mMap.find(w) != mMap.end()) return -1;
79 int ret = static_cast<int>(mPoints.size());
80 mPoints.push_back(a);
81 mMap.insert(std::pair<Weight, int>(w, ret));
82 return ret;
83 }
84 int lookup(const RealVector& a) const
85 {
86 Weight w = weight(a);
87 auto le = left(w);
88 auto ri = right(w);
89 for (auto i = le; i != ri; ++i)
90 if (are_same(mPoints[i->second], a)) return i->second;
91 return -1;
92 }
93 Weight weight(const RealVector& a) const
94 {
95 Weight w = 0;
96 auto n = mWeights.size();
97 assert(n == a.size());
98 for (int i = 0; i < n; ++i) w += mWeights[i] * a[i];
99 return w;
100 }
101 bool are_same(const RealVector& a, const RealVector& b) const
102 {
103 auto n = a.size();
104 assert(n == b.size());
105 for (int i = 0; i < n; ++i)
106 if (fabs(a[i] - b[i]) >= mEpsilon) return false;
107 return true;
108 }
109 void text_out(buffer& o) const
110 {
111 o << "PointArray( " << mPoints.size() << " points: ";
112 for (auto i = mMap.begin(); i != mMap.end(); ++i) o << i->second << " ";
113 o << ")" << newline;
114 }
115
116 private:
117 std::map<Weight, int> mMap;
118 std::vector<RealVector> mPoints;
119 Weight mEpsilon; // tolerance
120 RealVector mWeights; // random positive numbers summing up to 1
121
122 decltype(mMap)::const_iterator left(Weight key) const
123 {
124 return mMap.lower_bound(key - mEpsilon);
125 }
126 decltype(mMap)::const_iterator right(Weight key) const
127 {
128 return mMap.upper_bound(key + mEpsilon);
129 }
130};

250inline complex complex::operator*(complex c)
251{
252 complex tmp;
253 tmp.real = (real * c.real) - (imag * c.imag);
254 tmp.imag = (real * c.imag) + (imag * c.real);
255 return tmp;
256}
257
258inline complex complex::operator/(complex c)
259{
260 double div = (c.real * c.real) + (c.imag * c.imag);
261 complex tmp;
262 tmp.real = (real * c.real) + (imag * c.imag);
263 tmp.real /= div;
264 tmp.imag = (imag * c.real) - (real * c.imag);
265 tmp.imag /= div;
266 return tmp;
267}

269inline complex complex::getconjugate() const
270{
271 complex tmp;
272 tmp.real = this->real;
273 tmp.imag = this->imag * -1;
274 return tmp;
275}

277inline complex complex::getreciprocal()
278{
279 complex t;
280 t.real = real;
281 t.imag = imag * -1;
282 double div;
283 div = (real * real) + (imag * imag);
284 t.real /= div;
285 t.imag /= div;
286 return t;
287}
288
289inline double complex::getmodulus()
290{
291 double z;
292 z = (real * real) + (imag * imag);
293 z = sqrt(z);
294 return z;
295}

298inline double complex::getimaginary() const { return imag; }
299inline bool complex::operator==(complex c)
300{
301 return (real == c.real) && (imag == c.imag) ? 1 : 0;
302}

566 // inline functions needed by track
567 void evaluate_slpHxt(int n, const complex* x0t0, complex* Hxt)
568 {
569 if (is_projective)
570 {
571 complex* SxS = newarray_atomic(complex, (n + 1) * (n - 1));
572 complex* TxT = newarray_atomic(complex, (n + 1) * (n - 1));
573 const complex *x0 = x0t0, *t0 = x0t0 + n;
574 const double t = (*(double*)t0) * bigT; // t0 should be real
575 slpSxS->evaluate(n, x0, SxS);
576 slpTxT->evaluate(n, x0, TxT);
577 double c = cos(t), s = sin(t),
578 sqrt_one_minus_productST_2 = sqrt(1 - productST * productST);
579 double mS = c - productST * s / sqrt_one_minus_productST_2;
580 double mT = s / sqrt_one_minus_productST_2;
581 int j, i;
582 for (j = 0; j < n - 1; j++)
583 for (i = 0; i < n; i++)
584 *(Hxt + i * n + j) =
585 *(SxS + i * (n - 1) + j) * mS + *(TxT + i * (n - 1) + j) * mT;
586 j = n - 1; // last column
587 for (i = 0; i < n; i++) *(Hxt + i * n + j) = (x0 + i)->getconjugate();
588 i = n; // last row = H_t
589 mS = -s - productST * c / sqrt_one_minus_productST_2;
590 mT = c / sqrt_one_minus_productST_2;
591 for (j = 0; j < n - 1; j++)
592 {
593 complex tt =
594 *(SxS + i * (n - 1) + j) * mS + *(TxT + i * (n - 1) + j) * mT;
595 *(Hxt + i * n + j) = tt;
596 }
597 *(Hxt + n * n + n - 1) = complex(0.); // last row and column
598 freemem(SxS);
599 freemem(TxT);
600 }
601 else
602 slpHxt->evaluate(n + 1, x0t0, Hxt);
603 }
