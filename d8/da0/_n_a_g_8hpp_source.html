50 // PointArray
51 class PointArray
52 {
53  public:
54  using RealVector = std::vector<double>;
55  using Weight = double;
56  PointArray(Weight epsilon, const RealVector& weights)
57  : mEpsilon(epsilon), mWeights(weights)
58  {
59  }
60  PointArray(Weight epsilon, int n) : mEpsilon(epsilon)
61  {
62  double s = 0;
63  for (int i = 0; i < n; i++)
64  {
65  double r = rand();
66  s += r;
67  mWeights.push_back(r);
68  }
69  for (int i = 0; i < n; i++) mWeights[i] /= s;
70  }
71  virtual ~PointArray() {
72  // std::cerr << "entering ~PointArray()" << std::endl;
73  }
74  int lookup_or_append(const RealVector& a)
75  {
76  Weight w = weight(a);
77  int p = lookup(a);
78  if (p != -1) return p;
79  if (mMap.find(w) != mMap.end()) return -1;
80  int ret = static_cast<int>(mPoints.size());
81  mPoints.push_back(a);
82  mMap.insert(std::pair<Weight, int>(w, ret));
83  return ret;
84  }
85  int lookup(const RealVector& a) const
86  {
87  Weight w = weight(a);
88  auto le = left(w);
89  auto ri = right(w);
90  for (auto i = le; i != ri; ++i)
91  if (are_same(mPoints[i->second], a)) return i->second;
92  return -1;
93  }
94  Weight weight(const RealVector& a) const
95  {
96  Weight w = 0;
97  auto n = mWeights.size();
98  assert(n == a.size());
99  for (int i = 0; i < n; ++i) w += mWeights[i] * a[i];
100  return w;
101  }
102  bool are_same(const RealVector& a, const RealVector& b) const
103  {
104  auto n = a.size();
105  assert(n == b.size());
106  for (int i = 0; i < n; ++i)
107  if (fabs(a[i] - b[i]) >= mEpsilon) return false;
108  return true;
109  }
110  void text_out(buffer& o) const
111  {
112  o << "PointArray( " << mPoints.size() << " points: ";
113  for (auto i = mMap.begin(); i != mMap.end(); ++i) o << i->second << " ";
114  o << ")" << newline;
115  }
116 
117  private:
118  std::map<Weight, int> mMap;
119  std::vector<RealVector> mPoints;
120  Weight mEpsilon; // tolerance
121  RealVector mWeights; // random positive numbers summing up to 1
122 
123  decltype(mMap)::const_iterator left(Weight key) const
124  {
125  return mMap.lower_bound(key - mEpsilon);
126  }
127  decltype(mMap)::const_iterator right(Weight key) const
128  {
129  return mMap.upper_bound(key + mEpsilon);
130  }
131 };

566  *slpTxT; // slps for S and T, needed if is_projective

570  double maxDegreeTo3halves; // max(degree of equation)^{3/2}
571  // inline functions needed by track
572  void evaluate_slpHxt(int n, const complex* x0t0, complex* Hxt)
573  {
574  if (is_projective)
575  {
576  complex* SxS = newarray_atomic(complex, (n + 1) * (n - 1));
577  complex* TxT = newarray_atomic(complex, (n + 1) * (n - 1));
578  const complex *x0 = x0t0, *t0 = x0t0 + n;
579  const double t = (*(double*)t0) * bigT; // t0 should be real
580  slpSxS->evaluate(n, x0, SxS);
581  slpTxT->evaluate(n, x0, TxT);
582  double c = cos(t), s = sin(t),
583  sqrt_one_minus_productST_2 = sqrt(1 - productST * productST);
584  double mS = c - productST * s / sqrt_one_minus_productST_2;
585  double mT = s / sqrt_one_minus_productST_2;
586  int j, i;
587  for (j = 0; j < n - 1; j++)
588  for (i = 0; i < n; i++)
589  *(Hxt + i * n + j) =
590  *(SxS + i * (n - 1) + j) * mS + *(TxT + i * (n - 1) + j) * mT;
591  j = n - 1; // last column
592  for (i = 0; i < n; i++) *(Hxt + i * n + j) = (x0 + i)->getconjugate();
593  i = n; // last row = H_t
594  mS = -s - productST * c / sqrt_one_minus_productST_2;
595  mT = c / sqrt_one_minus_productST_2;
596  for (j = 0; j < n - 1; j++)
597  {
598  complex tt =
599  *(SxS + i * (n - 1) + j) * mS + *(TxT + i * (n - 1) + j) * mT;
600  *(Hxt + i * n + j) = tt;
601  }
602  *(Hxt + n * n + n - 1) = complex(0.); // last row and column
603  freemem(SxS);
604  freemem(TxT);
605  }
606  else
607  slpHxt->evaluate(n + 1, x0t0, Hxt);
608  }
609  void evaluate_slpHxtH(int n, const complex* x0t0, complex* HxtH)
610  {
611  if (is_projective)
612  ERROR("not implemented");
613  else
614  slpHxtH->evaluate(n + 1, x0t0, HxtH);
615  }
616  void evaluate_slpHxH(int n, const complex* x0t0, complex* HxH)
617  {
618  if (is_projective)
619  {
620  complex* SxS = newarray_atomic(complex, (n + 1) * (n - 1));
621  complex* TxT = newarray_atomic(complex, (n + 1) * (n - 1));
622  const complex *x0 = x0t0, *t0 = x0t0 + n;
623  const double t = (*(double*)t0) * bigT; // t0 should be real
624  slpSxS->evaluate(n, x0, SxS);
625  slpTxT->evaluate(n, x0, TxT);
626  double c = cos(t), s = sin(t),
627  sqrt_one_minus_productST_2 = sqrt(1 - productST * productST);
628  double mS = c - productST * s / sqrt_one_minus_productST_2;
629  double mT = s / sqrt_one_minus_productST_2;
630  int j, i;
631  for (j = 0; j < n - 1; j++)
632  for (i = 0; i <= n; i++)
633  *(HxH + i * n + j) =
634  *(SxS + i * (n - 1) + j) * mS + *(TxT + i * (n - 1) + j) * mT;
