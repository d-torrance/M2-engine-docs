196void NCF4::updateOverlaps(const Poly * toAdd)
197{
198 std::vector<Overlap> newLeftOverlaps;
199 std::vector<Overlap> newRightOverlaps;
200
201 Word newLeadWord = freeAlgebra().lead_word(*toAdd);
202
203 // the word table insert places the right overlaps into newOverlaps
204 mWordTable.insert(newLeadWord,newRightOverlaps);

220auto NCF4::overlapHeft(Overlap o) const -> int
221// overlap: of a pair of words v, w, v = a*s, w = s*b, returns the
222// heft degree of a*s*b.
223// o = triple (index of left GB element, pos, index of right GB element,
224// pos is the location in left GB element where s starts.
225{
226 Word tmpL = mWordTable[std::get<0>(o)];
227 Word tmpR = mWordTable[std::get<2>(o)];
228 int len_of_s = tmpL.size() - std::get<1>(o);
229 return freeAlgebra().monoid().wordHeft(tmpL) +
230 freeAlgebra().monoid().wordHeft(tmpR, len_of_s);
231}

249auto NCF4::isOverlapNecessary(const Overlap& o) -> bool
250{
251 // this function tests if the lead word of the overlap polynomial
252 // of o is a multiple of another pattern in the word table.
253
254 // need to be careful, however, since an overlap lead word is trivially
255 // a multiple of the words used to build it. These possibilities must be discarded
256 bool retval;
257 Word w;
258
259 // not optimal. Should pass information to wordTable for checking.
260 w = createOverlapLeadWord(o);
261 retval = !mWordTable.isNontrivialSuperword(w, std::get<0>(o), std::get<2>(o));
262 return retval;
263}

265auto NCF4::checkOldOverlaps(Word& newLeadWord) -> void
266{
267 // this function flags any previous overlaps that properly
268 // contain newLeadWord as no longer necessary.
269
270 // the newest overlaps are not yet added at this point
271 std::vector<int> startIndices;
272
273 for (auto& p : mOverlapTable.overlapMap())
274 {
275 for (auto& o : p.second)
276 {
277 if (not std::get<3>(o)) continue;
278 if (std::get<1>(o) == -1) continue;
279 int overlapLen = std::get<1>(o) + mWordTable[std::get<2>(o)].size();
280 if (overlapLen <= newLeadWord.size()) continue;
281 startIndices.clear();
282 // this is not optimal. Should pass info to word table to avoid creation of another word
283 auto w = createOverlapLeadWord(o);
284 WordTable::subwordPositions(newLeadWord,w,startIndices);
285 for (auto j : startIndices)
286 {
287 if (j != 0 or j != w.size() - newLeadWord.size())
288 {
289 if (M2_gbTrace >= 3)
290 std::cout << "Reduction avoided using lazy 2nd criterion." << std::endl;
291 std::get<3>(o) = false;
292 break;
293 }
294 }
295 }
296 }

299Word NCF4::createOverlapLeadWord(const Overlap& o)
300{
301 // this function adds the return value to mMemoryBlock, so should only be used
302 // when running a GB since it will be subsequently cleared.
303 Word left(mWordTable[std::get<0>(o)].begin(),
304 mWordTable[std::get<0>(o)].begin()+std::get<1>(o));
305 Word right = mWordTable[std::get<2>(o)];
306 int sz = left.size() + right.size();
307 auto rg = mMonomialSpace.allocateArray<int>(sz);
308 std::copy(left.begin(),left.end(),rg.first);
309 std::copy(right.begin(),right.end(),rg.first+left.size());
310 return Word(rg.first, rg.second);
311}

391void NCF4::preRowsFromOverlap(const Overlap& o)
392{
393 // o = (gbLeftIndex, overLapPos, gbRightIndex).
394 // BUT: if overlapPos < 0, then gbRightIndex is also < 0 (and is ignored), and gbLeftIndex
395 // refers to a generator, and we add to mOverlapsTodo in only (1,gbLeftIndex,1).
396 // where 1 = empty word.
397
398 int gbLeftIndex = std::get<0>(o);
399 int overlapPos = std::get<1>(o);
400 int gbRightIndex = std::get<2>(o);
401
402 if (overlapPos < 0)
403 {
404 // Sneaky trick: a PreRow with index a < 0 refers to generator with index -a-1
405 mOverlapsTodo.emplace_back(PreRow {Word(),
406 - gbLeftIndex - 1,
407 Word(),
408 OverlapPreRow});
409 return;
410 }
411
412 // LM(gbLeft) = x^a x^b
413 // LM(gbRight) = x^b x^c
414 // overlapPos = starting position of x^b in gbLeft.
415 // one prerow will be: (1, gbLeftIndex, x^c)
416 // another prerow will be: (x^a, gbRightIndex, 1)
417
418 Word leadWordLeft = mWordTable[gbLeftIndex];
419 Word leadWordRight = mWordTable[gbRightIndex];
420 int overlapLen = leadWordLeft.size() - overlapPos;
421
422 Word suffix2 {}; // trivial word
423 Word prefix2(leadWordLeft.begin(), leadWordLeft.begin() + overlapPos);
424
425 Word suffix1(leadWordRight.begin() + overlapLen, leadWordRight.end());
426 Word prefix1 {}; // trivial word
427
428 // need to add in the lead monomial to the mColumnMonomials list now
429 // so they know which row reduces them
430 Word newWord = freeAlgebra().monoid().wordProductAsWord(leadWordLeft, suffix1, mMonomialSpace);
431
432 // This overlap may already have lead term in table.
433 // Only have to add it in if it is not yet present.
434 auto it = mColumnMonomials.find(newWord);
435 if (it == mColumnMonomials.cend())
436 mColumnMonomials.emplace(newWord, std::make_pair(-1,-1));
437
438 // it *matters* which one is a reducer and which one is an overlap.
439 // this is due to how the word table lookup works -- it searches them
440 // in the order that they were entered into the word table, which may
441 // not be sorted in term order.
442 if (gbLeftIndex > gbRightIndex)
443 {
444 mReducersTodo.emplace_back(PreRow {prefix2,
445 gbRightIndex,
446 suffix2,
447 ReducerPreRow});
448 mOverlapsTodo.emplace_back(PreRow {prefix1,
449 gbLeftIndex,
450 suffix1,
451 OverlapPreRow});
452 }
453 else
454 {
455 mReducersTodo.emplace_back(PreRow {prefix1,
456 gbLeftIndex,
457 suffix1,
458 ReducerPreRow});
459 mOverlapsTodo.emplace_back(PreRow {prefix2,
460 gbRightIndex,
461 suffix2,
462 OverlapPreRow});
463 }
464}

466void NCF4::buildF4Matrix(const std::deque<Overlap>& overlapsToProcess)
467{
468 matrixReset();
469
470 mColumnMonomials.rehash(pow(2, ceil(log2(mPreviousColumnMonomials.size()))));
471
472 for (auto o : overlapsToProcess)
473 {
474 auto stillValid = std::get<3>(o);
475 if (stillValid) preRowsFromOverlap(o);
476 }
477
478 // can't do this loop as a range-based for loop since we are adding to it
479 // during the for loop
480 // process each element in mReducersTodo (false indicates a reducer row)
481 for (int i=0 ; i < mReducersTodo.size(); ++i)
482 processPreRow(mReducersTodo[i],mRows); // this often adds new elements to mReducersTodo
483
484 int numReducersAtFirst = mReducersTodo.size();
485
486 // process the overlap rows (true indicates an overlap row)
487 for (auto over : mOverlapsTodo)
488 processPreRow(over,mOverlaps); // this often adds new elements to mReducersTodo
489
490 // can't do this loop as a range-based for loop since we are adding
491 // to mReducersTodo during the for loop
492 for (int i=numReducersAtFirst ; i < mReducersTodo.size(); ++i)
493 processPreRow(mReducersTodo[i],mRows); // this often adds new elements to mReducersTodo
494
495 // Now we move the overlaps into mRows, and set mFirstOverlap.
496 // double range, so use a for loop
497 mFirstOverlap = mRows.size();
498 for (int i=0; i < mOverlapsTodo.size(); ++i)
499 {
500 mRows.emplace_back(mOverlaps[i]);
501 mReducersTodo.emplace_back(mOverlapsTodo[i]);
502 }
503}

506// since it is not as fast as the serial version due

508void NCF4::parallelBuildF4Matrix(const std::deque<Overlap>& overlapsToProcess)
509{
510 matrixReset();
511
512 mColumnMonomials.rehash(pow(2, ceil(log2(mPreviousColumnMonomials.size()))));
513
514 for (auto o : overlapsToProcess)
515 {
516 auto stillValid = std::get<3>(o);
517 if (stillValid) preRowsFromOverlap(o);
518 }
519
520 struct ThreadData {
521 RowsVector rowsVector;
522 MemoryBlock* memoryBlock;
523 };
524
525 mtbb::enumerable_thread_specific<ThreadData> threadData([&](){
526 mtbb::queuing_mutex::scoped_lock myColumnLock(mColumnMutex);
527 ThreadData data;
528 data.memoryBlock = new MemoryBlock;
529 mMemoryBlocks.push_back(data.memoryBlock);
530 return data;
531 });
532
533 // can't do this loop as a range-based for loop since we are adding to it
534 // during the for loop
535 // process each element in mReducersTodo
536
537 mtbb::parallel_for_each(mReducersTodo.begin(), mReducersTodo.end(),
538 [&](const PreRow& prerow, PreRowFeeder& feeder)
539 {
540 auto& data = threadData.local();
541 processPreRow(prerow,
542 data.rowsVector,
543 *data.memoryBlock,
544 &feeder);
545 });
546
547 // combine the thread local rows into mRows
548 for (const auto& data : threadData)
549 {
550 mRows.reserve(mRows.size() + data.rowsVector.size());
551 std::move(data.rowsVector.begin(),
552 data.rowsVector.end(),
553 std::back_inserter(mRows));
554 }
555
556 // WARNING: The feeder doesn't actually add things to mReducersTodo
557 // so in this algorithm there is now a disconnect between the
558 // sizes of mReducersTodo and mRows.
559
560 int numReducersAtFirst = mReducersTodo.size();
561
562 // this can be a parallel_for
563 for (auto over : mOverlapsTodo)
564 processPreRow(over,mOverlaps); // this often adds new elements to mReducersTodo
565
566 // this must (eventually) be a parallel_for_each
567 // can't do this loop as a range-based for loop since we are adding
568 // to mReducersTodo during the for loop
569 for (int i=numReducersAtFirst ; i < mReducersTodo.size(); ++i)
570 processPreRow(mReducersTodo[i],mRows); // this often adds new elements to mReducersTodo
571
572 // Now we move the overlaps into mRows, and set mFirstOverlap.
573 mFirstOverlap = mRows.size();
574 for (int i=0; i < mOverlapsTodo.size(); ++i)
575 {
576 mRows.emplace_back(mOverlaps[i]);
577 mReducersTodo.emplace_back(mOverlapsTodo[i]);
578 }

581void NCF4::processPreRow(PreRow r,
582 RowsVector& rowsVector,
583 MemoryBlock& memoryBlock,
584 PreRowFeeder* feeder)
585{
586 // note: left and right should be the empty word if gbIndex < 0 indicating
587 // an input polynomial.
588 Word left = r.left;
589 int gbIndex = r.preRowIndex;
590 Word right = r.right;
591 PreRowType preRowType = r.preRowType;
592
593 if (M2_gbTrace >= 100)
594 std::cout << "Processing PreRow: ("
595 << left << "," << gbIndex << "," << right << ")"
596 << std::endl;
597
598 // construct the elem corresponding to this prerow
599 // it will either be:
600 // a multiple of a previously reduced row (if prevReducer)
601 // an element of the input (if gbIndex < 0 and not prevReducer)
602 // a multiple of a gb element (if gbIndex >= 0 and not prevReducer)
603 const Poly* elem;
604 Poly* tempelem;
605 if (preRowType == PreviousReducerPreRow)
606 {
607 // in this case, we construct the poly locally for processing in this
608 // function. We will destroy it at the end, as the new monomials for reduction
609 // are created and inserted into mMonomialSpace
610 tempelem = new Poly;
611 reducedRowToPoly(tempelem,mPreviousRows,mPreviousColumns,gbIndex);
612 elem = tempelem;
613 }
614 else
615 {
616 if (gbIndex < 0)
617 elem = mInput[-gbIndex-1];
618 else
619 elem = mGroebner[gbIndex];
620 }
621
622 // loop through all monomials of the product
623 // for each monomial:
624 // (the below steps are now in processMonomInPreRow)
625 // is its prefix or suffix the lead term of a row from the previous degree?
626 // if so: insert the information from this row in the appropriate places
627 // if not: is the monomial in the hash table?
628 // if so: return the column index into the component for the new row.
629 // if not: insert it, and return the new column index into same place
630 // and place this monomial into mColumns.
631 // and search for divisor for it.
632 //
633 int numTerms = elem->numTerms();
634
635 auto wordRange = memoryBlock.allocateArray<Word>(numTerms);
636 auto columnRange = memoryBlock.allocateArray<int>(numTerms);
637 // int* componentAlloc = (int*)mMemoryPool.malloc(numTerms*sizeof(int));
638 // Range<int> componentRange(componentAlloc,componentAlloc + numTerms);
639 // for (auto& i : componentRange) i = 0;
640
641 Word* nextColWord = wordRange.first;
642
643 for (auto i = elem->cbegin(); i != elem->cend(); ++i)
644 {
645 Word mid;
646 freeAlgebra().monoid().wordFromMonom(mid,i.monom());
647 Word w = freeAlgebra().monoid().wordProductAsWord(left,mid,right,memoryBlock);
648 processWordInPreRow(w,feeder);
649 *nextColWord = w;
650 ++nextColWord;
651 }
652
653 // this memory is stored in rowsVector and cleaned up later.
654 ElementArray coeffs = mVectorArithmetic->elementArrayFromContainer(elem->getElementArray());
655
656 // delete the Poly created for prevReducer case, if necessary.
657 if (preRowType == PreviousReducerPreRow)
658 {
659 mFreeAlgebra.clear(*tempelem);
660 delete tempelem;
661 }
662
663 // add the processed row to the appropriate list
664 rowsVector.emplace_back(Row {coeffs, columnRange, wordRange});
665}

676void NCF4::processWordInPreRow(Word& w,
677 PreRowFeeder* feeder)
678{
679 const auto it = mColumnMonomials.find(w);
680 if (it == mColumnMonomials.end())
681 {
682 // if we are here, this monomial is not yet accounted for in the matrix
683 // so, check if the monomial is a multiple of a lead term of a gb element
684
685 // insert column information into mColumnMonomials,
686 // which provides us search (and eventually) indexing of columns
687 mColumnMonomials.emplace(w,std::make_pair(-1,-1));
688 auto divresult = findDivisor(w);
689 if (divresult.first)
690 {
691 // if m is a divisor of a gb element (or a left/right
692 // variable multiple of a previous reducer), add that
693 // multiple of the GB element to mReducersToDo for
694 // processing
695
696 // if feeder is nullptr, then this is a computation on a list
697 // not being added to during the loop.
698 // e.g. either F4 serial or overlaps processing
699 // otherwise, we must use the feeder to add more work.
700 if (feeder == nullptr)
701 mReducersTodo.push_back(divresult.second);
702 else
703 feeder->add(divresult.second);
704 }
705 }
706}

708std::pair<bool, NCF4::PreRow> NCF4::findDivisor(Word word)
709{
710 Word newword;
711
712 // look in previous F4 matrix for Monom before checking mWordTable
713
714 auto usePreviousSuffix = findPreviousReducerSuffix(word);
715 if (usePreviousSuffix.first)
716 {
717 // here, we use a multiple of a previously reduced row
718 Word tmpWord;
719 if (word.size() != 0)
720 tmpWord.init(word.begin(), word.begin() + 1);
721 return std::make_pair(true, PreRow {tmpWord,
722 usePreviousSuffix.second,
723 Word(),
724 PreviousReducerPreRow});
725 }
726 auto usePreviousPrefix = findPreviousReducerPrefix(word);
727 if (usePreviousPrefix.first)
728 {
729 // here, we use a multiple of a previously reduced row
730 Word tmpWord;
731 if (word.size() != 0)
732 tmpWord.init(word.end()-1, word.end());
733 return std::make_pair(true, PreRow {Word(),
734 usePreviousPrefix.second,
735 tmpWord,
736 PreviousReducerPreRow});
737 }
738
739 // if we are here, then the Monom does not have a prefix/suffix
740 // that was processed previously.
741
742 // look in mWordTable for Monom
743 std::pair<int,int> divisorInfo;
744
745 // TODO: for certain inputs (e.g. Monoms that we *know* are of
746 // the form xm or mx for m a standard monomial and x a
747 // variable, one needs only to check prefixes or
748 // suffixes, not all subwords. Not sure how to utilize that
749 // just yet.
750
751 bool found = mWordTable.subword(word, divisorInfo);
752
753 // if newword = x^a x^b x^c, with x^b in the word table, then:
754 // divisorInfo.first = index of the GB element with x^b as lead monomial.
755 // divisorInfo.second = position of the start of x^b in newword
756 // (that is, the length of x^a).
757 if (not found)
758 return std::make_pair(false, PreRow {Word(), 0, Word(), ReducerPreRow});
759 // if found, then return this information to caller
760 Word prefix = Word(word.begin(), word.begin() + divisorInfo.second);
761 Word divisorWord = mWordTable[divisorInfo.first];
762 Word suffix = Word(word.begin() + divisorInfo.second + divisorWord.size(),
763 word.end());
764 return std::make_pair(true, PreRow {prefix, divisorInfo.first, suffix, ReducerPreRow});
765}

767std::pair<bool,int> NCF4::findPreviousReducerPrefix(const Word& w)
768{
769 // build the largest proper prefix of m and look it up in previous
770 // mColumnMonomials table. we will return (true, row index) if
771 // found and (false, -1) if not.
772
773 // note that just because the second coordinate of an
774 // mColumnMonomials entry is -1 does not mean there isn't a reducer
775 // in the row -- in this case it is a new GB element and already
776 // added to the mGroebner list. In this case, we will be using that
777 // entry anyway so no need to make the additional effort to find it
778 // in the mRows table.
779
780 // get prefix of m as a monom
781 // search for prefix in mPreviousColumnMonomials
782 // if it.second.second != -1, then return (true,it.second.first)
783 // else return (false, -1)
784
785 // if the monom is the empty monomial, return false
786 if (w.size() == 0) return std::make_pair(false,-1);
787
788 std::pair<bool,int> retval;
789
790 Word prefix(w.begin(),w.end()-1);
791 const auto it = mPreviousColumnMonomials.find(prefix);
792 if (it == mPreviousColumnMonomials.end()) // not in table
793 retval = std::make_pair(false,-1);
794 else
795 {
796 int colNum = (*it).second.first;
797 if (mPreviousColumns[colNum].pivotRow == -1) // in column table and not a reducer monomial
798 retval = std::make_pair(false,-1);
799 else // in table and a reducer monomial
800 retval = std::make_pair(true,mPreviousColumns[colNum].pivotRow);
801 }
802 return retval;
803}

806std::pair<bool,int> NCF4::findPreviousReducerSuffix(const Word& w)
807{
808 // basically the same as above, except with suffixes
809
810 // get suffix of m
811 // search for suffix in mPreviousColumnMonomials
812 // if it.second.second != -1, then return (true,it.second.first)
813 // else return (false, -1)
814 if (w.size() == 0) return std::make_pair(false,-1);
815
816 std::pair<bool,int> retval;
817
818 Word suffix(w.begin()+1,w.end());
819 const auto it = mPreviousColumnMonomials.find(suffix);
820 if (it == mPreviousColumnMonomials.end()) //not in table
821 retval = std::make_pair(false,-1);
822 else
823 {
824 int colNum = (*it).second.first;
825 if (mPreviousColumns[colNum].pivotRow == -1) // in column table and not a reducer monomial
826 retval = std::make_pair(false,-1);
827 else // in table and a reducer monomial
828 retval = std::make_pair(true,mPreviousColumns[colNum].pivotRow);
829 }
830 return retval;

833// Besides sorting the columns (using 'perm'), this also sets the
834// pivot rows of each column index (in the new sorted order).
835void NCF4::labelAndSortF4Matrix()
836{
837 size_t sz = mColumnMonomials.size();
838 std::vector<int> columnIndices;
839 std::vector<Word> tempWords;
840
841 tempWords.reserve(sz);
842 columnIndices.resize(sz);
843 std::iota(columnIndices.begin(), columnIndices.end(), 0);
844
845 // store all the column monomials in a temporary to sort them
846 // and also set the reducer column for them on the same pass
847 for (auto& i : mColumnMonomials)
848 tempWords.emplace_back(i.first);
849
850 // create the monomial sorter object
851 MonomSort<std::vector<Word>> monomialSorter(&freeAlgebra().monoid(),&tempWords);
852 // stable sort was here before, but this sort is based on a total ordering
853 // with no ties so we can use an unstable (and hence parallel!) sort.
854 if (mIsParallel)
855 mtbb::parallel_sort(columnIndices.begin(),columnIndices.end(),monomialSorter);
856 else
857 std::stable_sort(columnIndices.begin(),columnIndices.end(),monomialSorter);
858
859 auto applyLabelingColumns = [&](const mtbb::blocked_range<int>& r) {
860 for (auto count = r.begin(); count != r.end(); ++count)
861 {
862 auto& val = mColumnMonomials[tempWords[columnIndices[count]]];
863 val.first = count;
864 mColumns[count].word = tempWords[columnIndices[count]];
865 mColumns[count].pivotRow = -1;
866 }
867 };
868
869 // apply the sorted labeling to the columns
870 mColumns.resize(sz);
871 if (mIsParallel)
872 mtbb::parallel_for(mtbb::blocked_range<int>{0,(int)sz}, applyLabelingColumns);
873 else
874 applyLabelingColumns(mtbb::blocked_range<int>{0,(int)sz});
875
876 auto applyLabelingRows = [&](const mtbb::blocked_range<int>& r) {
877 for (auto i = r.begin(); i != r.end(); ++i)
878 {
879 auto& comps = mRows[i].columnIndices;
880 auto& words = mRows[i].columnWords;
881 // sets the pivot row in the column if this is a reducer row
882 if (i < mFirstOverlap)
883 {
884 mColumns[mColumnMonomials[words[0]].first].pivotRow = i;
885 mColumnMonomials[words[0]].second = i;
886 }
887 for (int j = 0; j < words.size(); ++j)
888 comps[j] = mColumnMonomials[words[j]].first;
889 }
890 };
891
892 // now fix the column labels in the rows and set pivot rows in columns
893 if (mIsParallel)
894 mtbb::parallel_for(mtbb::blocked_range<int>{0,(int)mRows.size()},applyLabelingRows);
895 else
896 applyLabelingRows(mtbb::blocked_range<int>{0,(int)mRows.size()});
897}

899// both reduceF4Row and parallelReduceF4Row call this function
900// with the appropriate mutexes inserted for lock.
901// reduceF4Row: tbb::null_mutex (a do-nothing mutex)
902// parallelReduceF4Row: tbb::queuing_mutex (for now)
903template<typename LockType>
904void NCF4::generalReduceF4Row(int index,
905 int first,
906 int firstcol,
907 NCF4Stats &ncF4Stats,
908 ElementArray& dense,
909 bool updateColumnIndex,
910 LockType& lock)
911{
912 int sz = mRows[index].columnIndices.size();
913 //assert(sz > 0); this may be zero when autoreducing the new gb elements
914 if (sz == 0) return;
915 if (sz == 1 && firstcol != -1) return;
916
917 int last = mRows[index].columnIndices[sz-1];
918
919 mVectorArithmetic->fillDenseArray(dense,
920 mRows[index].coeffVector,
921 mRows[index].columnIndices);
922
923 do {
924 int pivotrow = mColumns[first].pivotRow;
925 if (pivotrow >= 0)
926 {
927 ncF4Stats.numCancellations++;
928 mVectorArithmetic->denseCancelFromSparse(dense,
929 mRows[pivotrow].coeffVector,
930 mRows[pivotrow].columnIndices);
931 // last component in the row corresponding to pivotrow
932 int last1 = mRows[pivotrow].columnIndices.cend()[-1];
933 last = (last1 > last ? last1 : last);
934 }
935 else if (firstcol == -1)
936 {
937 firstcol = first;
938 }
939 first = mVectorArithmetic->denseNextNonzero(dense, first+1, last);
940 } while (first <= last);
941
942 // we have to free mRows[index] information because we are about to overwrite it
943 // how can I free mRows[index].columnIndices? it wasn't the last block allocated on mMonomialSpace...
944 mVectorArithmetic->safeDenseToSparse(dense,
945 mRows[index].coeffVector,
946 mRows[index].columnIndices,
947 firstcol,
948 last,
949 mMonomialSpace,
950 lock);
951 if (mVectorArithmetic->size(mRows[index].coeffVector) > 0)
952 {
953 mVectorArithmetic->makeMonic(mRows[index].coeffVector);
954 // don't do this in the parallel version
955 if (updateColumnIndex) mColumns[firstcol].pivotRow = index;
956 }
957}

959void NCF4::parallelReduceF4Matrix()
960{
961 NCF4Stats ncF4Stats;
962 using threadLocalDense_t = mtbb::enumerable_thread_specific<ElementArray>;
963 using threadLocalStats_t = mtbb::enumerable_thread_specific<NCF4Stats>;
964
965 // create a dense array for each thread
966 threadLocalDense_t threadLocalDense([&]() {
967 return mVectorArithmetic->allocateElementArray(mColumnMonomials.size());
968 });
969 auto denseVector = mVectorArithmetic->allocateElementArray(mColumnMonomials.size());
970
971 threadLocalStats_t threadLocalStats([&]() {
972 return NCF4Stats();
973 });
974
975 // access the number of allowable threads this way.
976 //std::cout << "M2 Number of Threads: " << getAllowableThreads() << std::endl;
977
978 // reduce each overlap row by mRows.
979
980 mtbb::queuing_mutex lock;
981 mtbb::parallel_for(mtbb::blocked_range<int>{mFirstOverlap,(int)mRows.size()},
982 [&](const mtbb::blocked_range<int>& r)
983 {
984 threadLocalDense_t::reference my_dense = threadLocalDense.local();
985 threadLocalStats_t::reference my_threadStats = threadLocalStats.local();
986 for (auto i = r.begin(); i != r.end(); ++i) {
987 parallelReduceF4Row(i,
988 mRows[i].columnIndices[0],
989 -1,
990 my_threadStats,
991 my_dense,
992 lock);
993 my_threadStats.numRows++;
994 }
995 });
996
997 int numThreads = 0;
998 for (auto tlStats : threadLocalStats)
999 {
1000 ++numThreads;
1001 if (M2_gbTrace >= 2)
1002 {
1003 std::cout << "numCancellations for this thread: " << tlStats.numCancellations << std::endl;
1004 std::cout << "numRows for this thread: " << tlStats.numRows << std::endl;
1005 }
1006 ncF4Stats.numCancellations += tlStats.numCancellations;
1007 }
1008
1009 // sequentially perform one more pass to reduce the spair rows down
1010 for (auto i = mFirstOverlap; i < mRows.size(); ++i)
1011 reduceF4Row(i,
1012 mRows[i].columnIndices[0],
1013 -1,
1014 ncF4Stats,
1015 denseVector);
1016
1017 // interreduce the matrix with respect to these overlaps.
1018 // This needs to be sequential as well
1019 for (auto i = mRows.size(); i > mFirstOverlap; --i)
1020 reduceF4Row(i-1,
1021 mRows[i-1].columnIndices[1],
1022 mRows[i-1].columnIndices[0],
1023 ncF4Stats,
1024 denseVector);
1025
1026 for (auto tlDense : threadLocalDense)
1027 mVectorArithmetic->deallocateElementArray(tlDense);
1028
1029 mVectorArithmetic->deallocateElementArray(denseVector);
1030 if (M2_gbTrace >= 2)
1031 {
1032 std::cout << "Number of cancellations: " << ncF4Stats.numCancellations << std::endl;
1033 std::cout << "Number of threads used: " << numThreads << std::endl;
1034 }
1035}
