391 void NCF4::preRowsFromOverlap(const Overlap& o)
392 {
393  // o = (gbLeftIndex, overLapPos, gbRightIndex).
394  // BUT: if overlapPos < 0, then gbRightIndex is also < 0 (and is ignored), and gbLeftIndex
395  // refers to a generator, and we add to mOverlapsTodo in only (1,gbLeftIndex,1).
396  // where 1 = empty word.
397 
398  int gbLeftIndex = std::get<0>(o);
399  int overlapPos = std::get<1>(o);
400  int gbRightIndex = std::get<2>(o);
401 
402  if (overlapPos < 0)
403  {
404  // Sneaky trick: a PreRow with index a < 0 refers to generator with index -a-1
405  mOverlapsTodo.emplace_back(PreRow {Word(),
406  - gbLeftIndex - 1,
407  Word(),
408  OverlapPreRow});
409  return;
410  }
411 
412  // LM(gbLeft) = x^a x^b
413  // LM(gbRight) = x^b x^c
414  // overlapPos = starting position of x^b in gbLeft.
415  // one prerow will be: (1, gbLeftIndex, x^c)
416  // another prerow will be: (x^a, gbRightIndex, 1)
417 
418  Word leadWordLeft = mWordTable[gbLeftIndex];
419  Word leadWordRight = mWordTable[gbRightIndex];
420  int overlapLen = leadWordLeft.size() - overlapPos;
421 
422  Word suffix2 {}; // trivial word
423  Word prefix2(leadWordLeft.begin(), leadWordLeft.begin() + overlapPos);
424 
425  Word suffix1(leadWordRight.begin() + overlapLen, leadWordRight.end());
426  Word prefix1 {}; // trivial word
427 
428  // need to add in the lead monomial to the mColumnMonomials list now
429  // so they know which row reduces them
430  Word newWord = freeAlgebra().monoid().wordProductAsWord(leadWordLeft, suffix1, mMonomialSpace);
431 
432  // This overlap may already have lead term in table.
433  // Only have to add it in if it is not yet present.
434  auto it = mColumnMonomials.find(newWord);
435  if (it == mColumnMonomials.cend())
436  mColumnMonomials.emplace(newWord, std::make_pair(-1,-1));
437 
438  // it *matters* which one is a reducer and which one is an overlap.
439  // this is due to how the word table lookup works -- it searches them
440  // in the order that they were entered into the word table, which may
441  // not be sorted in term order.
442  if (gbLeftIndex > gbRightIndex)
443  {
444  mReducersTodo.emplace_back(PreRow {prefix2,
445  gbRightIndex,
446  suffix2,
447  ReducerPreRow});
448  mOverlapsTodo.emplace_back(PreRow {prefix1,
449  gbLeftIndex,
450  suffix1,
451  OverlapPreRow});
452  }
453  else
454  {
455  mReducersTodo.emplace_back(PreRow {prefix1,
456  gbLeftIndex,
457  suffix1,
458  ReducerPreRow});
459  mOverlapsTodo.emplace_back(PreRow {prefix2,
460  gbRightIndex,
461  suffix2,
462  OverlapPreRow});
463  }

581 void NCF4::processPreRow(PreRow r,
582  RowsVector& rowsVector,
583  MemoryBlock& memoryBlock,
584  PreRowFeeder* feeder)
585 {
586  // note: left and right should be the empty word if gbIndex < 0 indicating
587  // an input polynomial.
588  Word left = r.left;
589  int gbIndex = r.preRowIndex;
590  Word right = r.right;
591  PreRowType preRowType = r.preRowType;
592 
593  if (M2_gbTrace >= 100)
594  std::cout << "Processing PreRow: ("
595  << left << "," << gbIndex << "," << right << ")"
596  << std::endl;
597 
598  // construct the elem corresponding to this prerow
599  // it will either be:
600  // a multiple of a previously reduced row (if prevReducer)
601  // an element of the input (if gbIndex < 0 and not prevReducer)
602  // a multiple of a gb element (if gbIndex >= 0 and not prevReducer)
603  const Poly* elem;
604  Poly* tempelem;
605  if (preRowType == PreviousReducerPreRow)
606  {
607  // in this case, we construct the poly locally for processing in this
608  // function. We will destroy it at the end, as the new monomials for reduction
609  // are created and inserted into mMonomialSpace
610  tempelem = new Poly;
611  reducedRowToPoly(tempelem,mPreviousRows,mPreviousColumns,gbIndex);
612  elem = tempelem;
613  }
614  else
615  {
616  if (gbIndex < 0)
617  elem = mInput[-gbIndex-1];
618  else
619  elem = mGroebner[gbIndex];
620  }
621 
622  // loop through all monomials of the product
623  // for each monomial:
624  // (the below steps are now in processMonomInPreRow)
625  // is its prefix or suffix the lead term of a row from the previous degree?
626  // if so: insert the information from this row in the appropriate places

708 std::pair<bool, NCF4::PreRow> NCF4::findDivisor(Word word)
709 {
710  Word newword;
711 
712  // look in previous F4 matrix for Monom before checking mWordTable
713 
714  auto usePreviousSuffix = findPreviousReducerSuffix(word);
715  if (usePreviousSuffix.first)
716  {
717  // here, we use a multiple of a previously reduced row
718  Word tmpWord;
719  if (word.size() != 0)
720  tmpWord.init(word.begin(), word.begin() + 1);
721  return std::make_pair(true, PreRow {tmpWord,
722  usePreviousSuffix.second,
723  Word(),
724  PreviousReducerPreRow});
725  }
726  auto usePreviousPrefix = findPreviousReducerPrefix(word);
727  if (usePreviousPrefix.first)
728  {
729  // here, we use a multiple of a previously reduced row

731  if (word.size() != 0)
732  tmpWord.init(word.end()-1, word.end());

746  // the form xm or mx for m a standard monomial and x a
747  // variable, one needs only to check prefixes or
748  // suffixes, not all subwords. Not sure how to utilize that
749  // just yet.

753  // if newword = x^a x^b x^c, with x^b in the word table, then:

757  if (not found)

767 std::pair<bool,int> NCF4::findPreviousReducerPrefix(const Word& w)
768 {
769  // build the largest proper prefix of m and look it up in previous
770  // mColumnMonomials table. we will return (true, row index) if
771  // found and (false, -1) if not.
772 
773  // note that just because the second coordinate of an
774  // mColumnMonomials entry is -1 does not mean there isn't a reducer
775  // in the row -- in this case it is a new GB element and already
776  // added to the mGroebner list. In this case, we will be using that
777  // entry anyway so no need to make the additional effort to find it
778  // in the mRows table.
779 
780  // get prefix of m as a monom
781  // search for prefix in mPreviousColumnMonomials
782  // if it.second.second != -1, then return (true,it.second.first)
783  // else return (false, -1)
784 
785  // if the monom is the empty monomial, return false
786  if (w.size() == 0) return std::make_pair(false,-1);
787 
788  std::pair<bool,int> retval;
789 
790  Word prefix(w.begin(),w.end()-1);
791  const auto it = mPreviousColumnMonomials.find(prefix);
792  if (it == mPreviousColumnMonomials.end()) // not in table
793  retval = std::make_pair(false,-1);
794  else
795  {
796  int colNum = (*it).second.first;
797  if (mPreviousColumns[colNum].pivotRow == -1) // in column table and not a reducer monomial
798  retval = std::make_pair(false,-1);
799  else // in table and a reducer monomial
800  retval = std::make_pair(true,mPreviousColumns[colNum].pivotRow);
801  }
802  return retval;
803 }

836 {

838  std::vector<int> columnIndices;
839  std::vector<Word> tempWords;
840 
841  tempWords.reserve(sz);
842  columnIndices.resize(sz);
843  std::iota(columnIndices.begin(), columnIndices.end(), 0);
844 
845  // store all the column monomials in a temporary to sort them
846  // and also set the reducer column for them on the same pass
847  for (auto& i : mColumnMonomials)
848  tempWords.emplace_back(i.first);
849 
850  // create the monomial sorter object
851  MonomSort<std::vector<Word>> monomialSorter(&freeAlgebra().monoid(),&tempWords);
852  // stable sort was here before, but this sort is based on a total ordering
853  // with no ties so we can use an unstable (and hence parallel!) sort.
854  if (mIsParallel)

860  for (auto count = r.begin(); count != r.end(); ++count)

867  };
