88template <typename Key>
89void F4MonomialLookupTableT<Key>::insert1(mi_node *&top,
90 const_varpower_monomial b,
91 Key k)
92{
93 count += 2;
94 mi_node **p = &top, *up = nullptr;
95 bool one_element = true;
96
97 for (index_varpower_monomial i = b; i.valid();)
98 {
99 one_element = false;
100 varpower_word insert_var = i.var();
101 varpower_word insert_exp;
102
103 if (*p == nullptr)
104 {
105 // make a new header node
106 *p = new_mi_node(insert_var, 0, up);
107 (*p)->header = (*p)->left = (*p)->right = *p;
108 }
109 else if ((*p)->var < insert_var)
110 {
111 // make a new layer
112 mi_node *header_node, *zero_node;
113 header_node = new_mi_node(insert_var, 0, up);
114 zero_node = new_mi_node(insert_var, 0, *p);
115
116 header_node->left = header_node->right = zero_node;
117 (*p)->down() = zero_node;
118 *p = header_node->header = zero_node->header = zero_node->left =
119 zero_node->right = header_node;
120 }
121
122 if ((*p)->var > insert_var)
123 {
124 insert_var = (*p)->var;
125 insert_exp = 0;
126 }
127 else
128 {
129 insert_exp = i.exponent();
130 ++i;
131 }
132
133 mi_node *q = (*p)->right;
134 while ((q != q->header) && (q->exp < insert_exp)) q = q->right;
135 if (q->exp != insert_exp)
136 {
137 mi_node *insert_node;
138
139 if (i.valid())
140 {
141 insert_node = new_mi_node(
142 insert_var, insert_exp, static_cast<mi_node *>(nullptr));
143 q->insert_to_left(insert_node);
144 q = insert_node;
145 }
146 else
147 {
148 insert_node = new_mi_node(insert_var, insert_exp, k);
149 q->insert_to_left(insert_node);
150 return;
151 }
152 }
153
154 up = q;
155 p = &(q->down());
156 }
157 if (one_element)
158 {
159 // insert a header node and a var/exp = 0/0 leaf
160 top = new_mi_node(0, 0, static_cast<mi_node *>(nullptr));
161 mi_node *leaf_node = new_mi_node(0, 0, k);
162 top->left = top->right = leaf_node;
163 top->header = leaf_node->header = leaf_node->left = leaf_node->right =
164 top;
165 }
166}

168template <typename Key>
169bool F4MonomialLookupTableT<Key>::find_one_divisor1(mi_node *mi,
170 const_ntuple_monomial exp,
171 Key &result_k) const
172// mi is the top: where to start looking
173{
174 if (mi == nullptr) return false;
175
176 mi_node *p = mi;
177
178 for (;;)
179 {
180 p = p->right;
181
182 if (p == p->header)
183 {
184 if ((p = p->down()) == nullptr) return false;
185 continue;
186 }
187
188 if (p->exp > exp[p->var])
189 {
190 if ((p = p->header->down()) == nullptr) return false;
191 continue;
192 }
193
194 if (p->tag == mi_node::leaf)
195 {
196 result_k = p->key();
197 return true;
198 }
199
200 p = p->down();
201 }
202}

237template <typename Key>
238void F4MonomialLookupTableT<Key>::update_expvector(
239 int topvar,
240 const_varpower_monomial m)
241{
242 int nvars = topvar + 1;
243 if (*m > 0 && m[1] >= nvars) nvars = static_cast<int>(m[1] + 1);
244 if (size_of_exp <= nvars)
245 {
246 // Increase size of exponent vector
247 freemem(exp0);
248 if (nvars > 2 * size_of_exp)
249 size_of_exp = nvars;
250 else
251 size_of_exp *= 2;
252
253 exp0 = newarray_atomic_clear(ntuple_word, size_of_exp);
254 }
255
256 int nparts = static_cast<int>(*m++);
257 for (int i = nparts; i > 0; i--, m += 2)
258 {
259 exp0[*m] = m[1];
260 }
261}

274template <typename Key>
275bool F4MonomialLookupTableT<Key>::find_one_divisor_vp(long comp,
276 const_varpower_monomial m,
277 Key &result_k) const
278{
279 if (comp >= mis.size()) return false;
280 mi_node *mi = mis[comp];
281 if (mi == nullptr) return false;
282
283 F4MonomialLookupTableT *me = const_cast<F4MonomialLookupTableT *>(this);
284 me->update_expvector(static_cast<int>(mi->var), m);
285 bool result = find_one_divisor1(mi, exp0, result_k);
286 me->reset_expvector(m);
287 return result;
288}

396void F4MonomialLookupTableT<Key>::do_node(mi_node *p,
397 int indent,
398 int disp) const
399{
400 buffer o;
401 int i;
402 assert(p->left != nullptr);
403 assert(p->right != nullptr);
404 assert(p->left->right == p);
405 assert(p->right->left == p);
406 if (disp)
407 {
408 for (i = 1; i <= indent; i++) o << ' ';
409 o << p->var << ' ' << p->exp;
410 }
411 if (p->tag == mi_node::leaf)
412 {
413 nleaves++;
414 if (disp) o << ' ' << p->key();
415 }
416 else if (p == p->header)
417 nlists++;
418 else
419 nnodes++;
420 emit_line(o.str());

424void F4MonomialLookupTableT<Key>::do_tree(mi_node *p,

426 int indent,
427 int disp) const
428{

437 q = q->right;

461template <typename Key>
462int F4MonomialLookupTableT<Key>::debug_check(mi_node *const p,
463 const mi_node *const up) const
464// Returns the number of leaves at tree with root p.
465// Make sure that the list header is constructed ok, that the
466// left/right pointers are ok on this level, that the
467// var, exp, values in this train are correct.
468// Then loop through, checking each node (recursively) and each leaf
469{
470 mi_node *q;
471 // First check the node 'p' itself
472 assert(p != nullptr);
473 assert(p->var >= 0);
474 if (up != nullptr) assert(p->var < up->var);
475 assert(p->header == p);
476 assert(p->tag == mi_node::node);
477 assert(p->down() == up);
478 assert(p->left != nullptr);
479 assert(p->right != nullptr);
480
481 // Now loop through each element in left/right chain, checking that
482 // v, e, left, right values are consistent.
483 for (q = p->left; q != p; q = q->left)
484 {
485 assert(q->left != nullptr);
486 assert(q->right != nullptr);
487 assert(q->header == p);
488 assert(q->right->left == q);
489 assert(q->left->right == q);
490 assert(q->var == p->var);
491 assert((q->right == p) || (q->exp < q->right->exp));
492 assert(q->exp >= 0);
493 }
494
495 // Now loop through again, this time descending into nodes
496 int c = 0;
497 for (q = p->right; q != p; q = q->right)
498 if (q->tag == mi_node::node)
499 c += debug_check(q->down(), q);
500 else
501 c++;
502 return c;
503}

536void minimalize_varpower_monomials(const VECTOR(varpower_monomial) & elems,
537 VECTOR(int) & result_minimals,
538 stash *mi_stash)
539{
540 std::vector<std::pair<int, int>> degs_and_indices;
541 int count = 0;
542 for (auto& b : elems)
543 {
544 int deg = varpower_monomials::simple_degree(b);
545 degs_and_indices.push_back(std::make_pair(deg, count));
546 ++count;
547 }
548 std::stable_sort(degs_and_indices.begin(), degs_and_indices.end());
549
550 F4MonomialLookupTableT<int> M(10, mi_stash); // 10 is simply a suggested value
551 for (auto p : degs_and_indices)
552 {
553 int k; // unused: we only care if there is a divisor, not which index it has
554 if (!M.find_one_divisor_vp(0, elems[p.second], k))
555 {
556 M.insert_minimal_vp(0, elems[p.second], 0);
557 result_minimals.push_back(p.second);
558 }
559 }
560}
