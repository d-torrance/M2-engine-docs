87 M2_arrayint IM2_FreeModule_get_degrees(const FreeModule *F)
88 {
89  const Ring *R = F->get_ring();
90  const Monoid *D = R->degree_monoid();
91  M2_arrayint result = M2_makearrayint(F->rank() * D->n_vars());
92  int next = 0;
93  exponents_t exp = newarray_atomic(int, D->n_vars());
94  for (int i = 0; i < F->rank(); i++)
95  {
96  D->to_expvector(F->degree(i), exp);
97  for (int j = 0; j < D->n_vars(); j++) result->array[next++] = exp[j];
98  }
99  freemem(exp);
100  return result;
