9bool almostEqual(const RingRRR *R, int nbits, ring_elem a, ring_elem b)
10{
11 mpfr_t epsilon;
12 mpfr_init2(epsilon, 100);
13 mpfr_set_ui_2exp(epsilon, 1, -nbits, MPFR_RNDN);
14
15 ring_elem c = R->subtract(a, b);
16 bool ret = mpfr_cmpabs(c.get_mpfr(), epsilon) < 0;
17
18 mpfr_clear(epsilon);
19 return ret;
20}

93TEST(RingRRR, axioms)
94{
95 RingRRR *R = RingRRR::create(std::make_unique<M2::ARingRRR>(100));
96 RingElementGenerator<RingRRR> gen(*R);
97 for (int i = 0; i < ntrials; i++)
98 {
99 ring_elem a = gen.nextElement();
100 ring_elem b = gen.nextElement();
101 ring_elem c = gen.nextElement();
102
103 // Test commutativity
104 // test: a*b = b*a
105 // test: a+b == b+a
106 ring_elem d = R->add(a, b);
107 ring_elem e = R->add(b, a);
108 EXPECT_TRUE(R->is_equal(d, e));
109 d = R->mult(a, b);
110 e = R->mult(b, a);
111 EXPECT_TRUE(almostEqual(R, 98, d, e));
112
113 // Test associativity
114 // test: a+(b+c) == (a+b)+c
115 // test: a*(b*c) == (a*b)*c
116 d = R->add(a, R->add(b, c));
117 e = R->add(R->add(a, b), c);
118 EXPECT_TRUE(almostEqual(R, 94, d, e));
119 d = R->mult(a, R->mult(b, c));
120 e = R->mult(R->mult(a, b), c);
121 EXPECT_TRUE(almostEqual(R, 94, d, e));
122
123 // Test distributivity
124 // test: a*(b+c) == a*b + a*c
125 d = R->mult(a, R->add(b, c));
126 e = R->add(R->mult(a, b), R->mult(a, c));
127 EXPECT_TRUE(almostEqual(R, 92, d, e));

159TEST(RingRRR, syzygy)
160{
161 // NOTE: RingRRR::syzygy, CCC::syzygy are not useful functions.
162 // Should we remove these tests, and the corresponding functions?
163 RingRRR *R = RingRRR::create(std::make_unique<M2::ARingRRR>(100));
164
165 RingElementGenerator<RingRRR> gen(*R);
166 for (int i = 0; i < ntrials; i++)
167 {
168 ring_elem u, v;
169 ring_elem a = gen.nextElement();
170 ring_elem b = gen.nextElement();
171 if (R->is_zero(b)) continue;
172
173 // special cases (note: b != 0 for rest of routine)
174 // syzygy(0,b) returns (1,0)
175 R->syzygy(R->zero(), b, u, v);
176 EXPECT_TRUE(R->is_equal(u, R->one()));
177 EXPECT_TRUE(R->is_equal(v, R->zero()));
178 // syzygy(a,1) returns (1,-a)
179 R->syzygy(a, R->one(), u, v);
180 EXPECT_TRUE(R->is_equal(u, R->one()));
181 EXPECT_TRUE(almostEqual(R, 98, v, R->negate(a)));
182 // syzygy(a,-1) returns (1,a)
183 R->syzygy(a, R->minus_one(), u, v);
184 EXPECT_TRUE(R->is_equal(u, R->one()));
185 EXPECT_TRUE(almostEqual(R, 98, v, a));
186 R->syzygy(a, b, u, v);
187 ring_elem result = R->add(R->mult(a, u), R->mult(b, v));
188 EXPECT_TRUE(almostEqual(R, 94, result, R->zero()));
