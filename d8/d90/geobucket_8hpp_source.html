66template <class FREEMODULETYPE, class VECTYPE>
67inline void geobucket<FREEMODULETYPE, VECTYPE>::add(VECTYPE p)
68{
69 mLead = -1;
70 int len = F->n_terms(p);
71 int i= 0;
72 while (len >= heap_size[i]) i++;
73 F->add_to(heap[i], p);
74 len = F->n_terms(heap[i]);
75 p = NULL;
76 while (len >= heap_size[i])
77 {
78 i++;
79 F->add_to(heap[i], heap[i-1]);
80 len = F->n_terms(heap[i]);
81 heap[i-1] = NULL;
82 }
83 if (i > top_of_heap)
84 top_of_heap = i;
85}

87template <class FREEMODULETYPE, class VECTYPE>
88inline const VECTYPE geobucket<FREEMODULETYPE, VECTYPE>::get_lead_term()
89{
90 int lead_so_far = -1;
91 for (int i=0; i <= top_of_heap; i++)
92 {
93 if (heap[i] == NULL) continue;
94 if (lead_so_far < 0)
95 {
96 lead_so_far = i;
97 continue;
98 }
99 int cmp = F->compare(heap[lead_so_far], heap[i]);
100 if (cmp == GT) continue;
101 if (cmp == LT)
102 {
103 lead_so_far = i;
104 continue;
105 }
106 // At this point we have equality
107 K->add_to(heap[lead_so_far]->coeff, heap[i]->coeff);
108 VECTYPE tmp = heap[i];
109 heap[i] = tmp->next;
110 tmp->next = NULL;
111 F->remove(tmp);
112
113 if (K->is_zero(heap[lead_so_far]->coeff))
114 {
115 // Remove, and start over
116 tmp = heap[lead_so_far];
117 heap[lead_so_far] = tmp->next;
118 tmp->next = NULL;
119 F->remove(tmp);
120 lead_so_far = -1;
121 i = -1;
122 }
123 }
124 mLead = lead_so_far;
125 if (lead_so_far < 0) return NULL;
126 VECTYPE result = heap[lead_so_far];
127 return result;
128}

130template <class FREEMODULETYPE, class VECTYPE>
131inline VECTYPE geobucket<FREEMODULETYPE, VECTYPE>::remove_lead_term()
132{
133 if (mLead < 0) get_lead_term();
134 if (mLead < 0) return NULL;
135 VECTYPE result = heap[mLead];
136 heap[mLead] = result->next;
137 result->next = NULL;
138 mLead = -1;
139 return result;
140}
