93  // the GB_list contains all of the remaining hm_elems's
94  // the 'next' field for these is the GB_list list.

124 hm_elem *HermiteComputation::merge(hm_elem *f, hm_elem *g)
125 {
126  if (g == nullptr) return f;
127  if (f == nullptr) return g;
128  hm_elem head;
129  hm_elem *result = &head;
130  hm_elem *h;
131  while (1) switch (compare_elems(f, g))
132  {
133  case 1:
134  result->next = g;
135  result = result->next;
136  g = g->next;
137  if (g == nullptr)
138  {
139  result->next = f;
140  return head.next;
141  }
142  break;
143  case 0:
144  // In this case, we remove one, and re-insert:
145  if (mpz_cmp(f->f->coeff.get_mpz(), g->f->coeff.get_mpz()) == 0)
146  {
147  vec f1 = globalZZ->copy_vec(f->f);
148  vec f2 = globalZZ->copy_vec(f->fsyz);
149  globalZZ->subtract_vec_to(g->f, f1);
150  globalZZ->subtract_vec_to(g->fsyz, f2);
151  }
152  else
153  {
154  vec f1 = globalZZ->copy_vec(f->f);
155  vec f2 = globalZZ->copy_vec(f->fsyz);
156  globalZZ->add_vec_to(g->f, f1);
157  globalZZ->add_vec_to(g->fsyz, f2);
158  }
159 
160  h = g;
161  // We need to reset the lead term
162  if (g->f != nullptr) mpz_abs(h->lead, g->f->coeff.get_mpz());
163  g = g->next;
164  insert(h);
165  if (g == nullptr)
166  {
167  result->next = f;
168  return head.next;
169  }
170  // Now fall through to merge f into the result:
171  case -1:
172  result->next = f;
173  result = result->next;
174  f = f->next;
175  if (f == nullptr)
176  {
177  result->next = g;
178  return head.next;
179  }
180  break;
181  }
182 }

184 void HermiteComputation::sort(hm_elem *&p)
185 {
186  // Sort in ascending absolute value of lead term
187  if (p == nullptr || p->next == nullptr) return;
188  hm_elem *p1 = nullptr;
189  hm_elem *p2 = nullptr;
190  while (p != nullptr)
191  {
192  hm_elem *tmp = p;
193  p = p->next;
194  tmp->next = p1;
195  p1 = tmp;
196 
197  if (p == nullptr) break;
198  tmp = p;
199  p = p->next;
200  tmp->next = p2;
201  p2 = tmp;
202  }
203 
204  sort(p1);
205  sort(p2);
206  p = merge(p1, p2);
207 }

209 void HermiteComputation::reduce(hm_elem *&p, hm_elem *q)
210 {
211  // compute (u,v) s.t. u lead(p) + v lead(q) = gcd
212  // set p <- u*p + v*q;
213  // set q <- lead(q)/gcd * p - lead(p)/gcd * q
214  // DOn't forget to also reset the 'lead' fields!
215  ring_elem u, v;
216  ring_elem g = globalZZ->gcd_extended(p->f->coeff, q->f->coeff, u, v);
217  ring_elem a = globalZZ->divide(q->f->coeff, g); // exact
218  ring_elem b = globalZZ->divide(p->f->coeff, g); // exact
219  globalZZ->negate_to(b);
220 
221  vec p1 = globalZZ->mult_vec(u, p->f);
222  vec p2 = globalZZ->mult_vec(v, q->f);
223  globalZZ->add_vec_to(p1, p2);
224 
225  vec syz1 = globalZZ->mult_vec(u, p->fsyz);
226  vec syz2 = globalZZ->mult_vec(v, q->fsyz);
227  globalZZ->add_vec_to(syz1, syz2);
228 
229  vec q1 = globalZZ->mult_vec(a, p->f);
230  vec q2 = globalZZ->mult_vec(b, q->f);
231  globalZZ->add_vec_to(q1, q2);
232 
233  vec qsyz1 = globalZZ->mult_vec(a, p->fsyz);
234  vec qsyz2 = globalZZ->mult_vec(b, q->fsyz);
235  globalZZ->add_vec_to(qsyz1, qsyz2);
236 
237  globalZZ->remove_vec(p->f);
238  globalZZ->remove_vec(q->f);
239  globalZZ->remove_vec(p->fsyz);
240  globalZZ->remove_vec(q->fsyz);
241  globalZZ->remove(a);
242  globalZZ->remove(b);
243  globalZZ->remove(g);
244  globalZZ->remove(u);
245  globalZZ->remove(v);
246 
247  // Now that the arithmetic has been done, put back into 'p', 'q':
248  p->f = p1;
249  p->fsyz = syz1;
250  mpz_set(p->lead, p->f->coeff.get_mpz());
251 
252  q->f = q1;
253  q->fsyz = qsyz1;
254  if (q->f != nullptr) mpz_abs(q->lead, q->f->coeff.get_mpz());
255 
256  insert(q);
257 }

259 void HermiteComputation::start_computation()
260 {
261  // ngb = stop[0]
262  // nsyz = stop[1]
263  // npairs = stop[2]
264  if (status() == COMP_DONE) return;
265  for (; row >= 0; row--)
266  {
267  hm_elem *p = initial[row];
268  if (p == nullptr) continue;
269  initial[row] = nullptr;
270  sort(p); // This can remove elements, inserting them back
271  while (p != nullptr && p->next != nullptr)
272  {
273  hm_elem *pnext = p->next->next;
274  reduce(p, p->next); // replaces p1, and re-inserts p2.
275  p->next = pnext;
276  }
277  // At this point, 'p' is the only remaining element with this lead term
278  // So insert it into GB_list
279  p->next = GB_list;
280  GB_list = p;
281  n_gb++;
282  }
283  // At this point, we are done, so reset initial[...] (it is all NULL right
284  // now)
285 
286  // Now let's auto reduce the result...
287  // We will reduce the GB elems
288  // one by one, placing them back into the 'initial' table.
289  // We use the following: the lead components of elements of GB_list are in
290  // increasing order
291 
292  for (hm_elem *p = GB_list; p != nullptr; p = p->next)
293  {
294  if (!globalZZ->is_positive(p->f->coeff))
295  {
296  vec f = globalZZ->negate_vec(p->f);
297  vec fsyz = globalZZ->negate_vec(p->fsyz);
298  globalZZ->remove_vec(p->f);
299  globalZZ->remove_vec(p->fsyz);
300  p->f = f;
301  p->fsyz = fsyz;
302  }
303  gb_reduce(p->f, p->fsyz);
304  initial[p->f->comp] = p;
305  }
306 
307  // for (hm_elem *p = GB_list; p != 0; p = p->next)
308  // initial[p->f->comp] = p;
309  set_status(COMP_DONE);
310 }

385 void HermiteComputation::gb_reduce(vec &f) const
386 {
387  // Reduce f so that each of its terms are < corresponding initial term
388  // (in absolute value).
389  vecterm head;
390  vecterm *result = &head;
391  head.next = nullptr;
392  while (f != nullptr)
393  {
394  int x = f->comp;
395  hm_elem *h = initial[x];
396  if (h != nullptr)
397  {
398  ring_elem v;
399  ring_elem rem =
400  globalZZ->remainderAndQuotient(f->coeff, h->f->coeff, v);
401  bool do_reduce = !globalZZ->is_zero(v);
402  if (do_reduce)
403  {
404  v = globalZZ->negate(v);
405  vec g = globalZZ->mult_vec(v, h->f);
406  globalZZ->add_vec_to(f, g);
407  }
408  if (globalZZ->is_zero(rem)) continue;
409  }
410  // The lead term stays
411  result->next = f;
412  f = f->next;
413  result = result->next;
414  result->next = nullptr;
415  continue;
416  }
417 
418  f = head.next;
419 }
