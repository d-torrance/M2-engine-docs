31void ARingQQGMP::elem_text_out(buffer& o,
32 const ElementType& a,
33 bool p_one,
34 bool p_plus,
35 bool p_parens) const
36{
37 char s[1000];
38 char* str;
39
40 bool is_neg = (mpq_sgn(&a) == -1);
41 bool one = is_pm_one(a);
42
43 size_t size = mpz_sizeinbase(mpq_numref(&a), 10) +
44 mpz_sizeinbase(mpq_denref(&a), 10) + 3;
45
46 char* allocstr = (size > 1000 ? newarray_atomic(char, size) : s);
47
48 if (!is_neg && p_plus) o << '+';
49 if (one)
50 {
51 if (is_neg) o << '-';
52 if (p_one) o << '1';
53 }
54 else
55 {
56 str = mpq_get_str(allocstr, 10, &a);
57 o << str;
58 }
59 if (size > 1000) freemem(allocstr);
60}
