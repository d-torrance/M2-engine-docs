454 bool RingElement::getSmallIntegerCoefficients(
455  std::vector<long> &result_coeffs) const
456 {
457  const PolynomialRing *R = get_ring()->cast_to_PolynomialRing();
458  const Ring *K = R->getCoefficientRing();
459  if (R == 0 || R->n_vars() != 1)
460  {
461  throw exc::engine_error(
462  "Expected a polynomial in a univariate polynomial ring");
463  return false; // Should not be needed
464  }
465 
466  if (is_zero())
467  {
468  result_coeffs.resize(0);
469  return true;
470  }
471  int lo, deg; // ignore lo, and deg == degree of the univariate polynomial f.
472  R->degree_of_var(0, get_value(), lo, deg);
473  result_coeffs.resize(deg + 1);
474  for (int i = 0; i <= deg; i++) result_coeffs[i] = 0;
475  int exp[1];
476  for (Nterm& t : get_value())
477  {
478  std::pair<bool, long> res = K->coerceToLongInteger(t.coeff);
479  if (not res.first)
480  {
481  // At this point, the answer is meaningless
482  result_coeffs.resize(0);
483  return false;
484  }
485  long coeff = res.second;
486 
487  R->getMonoid()->to_expvector(t.monom, exp);
488  assert(exp[0] >= 0);
489  assert(exp[0] <= deg);
490  result_coeffs[exp[0]] = coeff;
491  }
492  return true;
493 }
