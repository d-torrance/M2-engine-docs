48 void MarkedGB::add_marked_elems(const VECTOR(gbvector *) & leadterms0,
49  const VECTOR(POLY) & polys0,
50  bool auto_reduced)
51 {
52  // First sort these elements via increasing lex order (or monomial order?)
53  // Next insert minimal elements into T, and polys
54  const Monoid *M = originalR->getMonoid();
55 
56  leadterms = newarray(gbvector *, leadterms0.size());
57 
58  // Now loop through each element, and see if the lead monomial is in T.
59  // If not, add it in , and place element into 'polys'.
60 
61  for (int i = 0; i < leadterms0.size(); i++)
62  {
63  POLY h;
64  ring_elem junk;
65 
66  gbvector *f = polys0[i].f;
67  gbvector *inf = leadterms0[i];
68 
69  h.f = R->gbvector_copy(f);
70  h.fsyz = R->gbvector_copy(polys0[i].fsyz);
71 
72  gbvector *iinf = 0;
73  for (gbvector *t = h.f; t != 0; t = t->next)
74  if (inf->comp == t->comp && EQ == M->compare(inf->monom, t->monom))
75  {
76  iinf = t;
77  break;
78  }
79  if (!iinf)
80  {
81  ERROR("lead term does not appear in the polynomial!");
82  iinf = f;
83  }
84  leadterms[i] = iinf;
85 
86  exponents_t e = R->exponents_make();
87  R->gbvector_get_lead_exponents(F, iinf, e);
88 
89  // if (auto_reduced)
90  // remainder(h,false,junk); // This auto-reduces h. MES MES!! Maybe
91  // not for marked GB!!
92 
93  R->gbvector_remove_content(h.f, h.fsyz);
94 
95  T->insert(e, iinf->comp, i);
96  polys.push_back(h);
97  }
98 
99  auto_reduce();
100 }

112 void MarkedGB::marked_remainder(POLY &f,
113  bool use_denom,
114  ring_elem &denom,
115  gbvector *marked_lead_term)
116 // If marked_lead_term is not NULL, then it should be a pointer
117 // to a term in f.f. This term will not be reduced, and the new lead term will
118 // replace this one.
119 // (This could happen if the coefficient changes).
120 {
121  gbvector head;
122  gbvector *frem = &head;
123  frem->next = 0;
124  POLY h = f;
125  exponents_t EXP = ALLOCATE_EXPONENTS(R->exponent_byte_size());
126 
127  while (!R->gbvector_is_zero(h.f))
128  {
129  if (h.f != marked_lead_term)
130  {
131  R->gbvector_get_lead_exponents(F, h.f, EXP);
132  int x = h.f->comp;
133  int w = T->find_divisor(EXP, x);
134  if (w >= 0)
135  {
136  POLY g = polys[w];
137  R->gbvector_reduce_with_marked_lead_term(F,
138  Fsyz,
139  head.next,
140  h.f,
141  h.fsyz,
142  leadterms[w],
143  g.f,
144  g.fsyz,
145  use_denom,
146  denom);
147  continue;
148  }
149  }
150  frem->next = h.f;
151  frem = frem->next;
152  h.f = h.f->next;
153  frem->next = 0;
154  }
155 
156  h.f = head.next;
157  f.f = h.f;
158  f.fsyz = h.fsyz;
159  R->gbvector_sort(F, f.f);
160  R->gbvector_sort(Fsyz, f.fsyz);
161 }
