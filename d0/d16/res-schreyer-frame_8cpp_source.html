141 BettiDisplay SchreyerFrame::minimalBettiNumbers(bool stop_after_degree,
142  int top_slanted_degree,
143  int length_limit)
144 {
145  // The lo degree will be: mLoSlantedDegree.
146  // The highest slanted degree will either be mHiSlantedDegree, or
147  // top_slanted_degree (minimum of these two).
148  // The length we need to compute to is either maxLevel(), or length_limit+1.
149  // We set maxlevel to length_limit. We insist that length_limit <= maxLevel()
150  // - 2.
151  // Here is what needs to be computed:
152  // lo: . . . . . . .
153  // . . . . . . .
155  // Each dot in all rows other than 'hi' needs to have syzygies computed for
156  // it.
157  // if hi == mHiSlantedDegree, then we do NOT need to compute syzygies in this
158  // last row.
159  // else we need to compute syzygies in these rows, EXCEPT not at level
160  // maxlevel+1
161 
162  computeFrame();
163 
164  int top_degree; // slanted degree
165  if (stop_after_degree)
166  {
167  top_degree = std::min(top_slanted_degree, mHiSlantedDegree);
168  top_degree = std::max(mLoSlantedDegree, top_degree);
169  }
170  else
171  {
172  top_degree = mHiSlantedDegree;
173  }
174  // First: if length_limit is too low, extend the Frame
175  if (length_limit >= maxLevel())
176  {
177  std::cout << "WARNING: cannot extend resolution length" << std::endl;
178  length_limit = maxLevel() - 1;
179  // Extend the length of the Frame, change mMaxLength, possibly
180  // mHiSlantedDegree
181  // increase mComputationStatus if needed, mMinimalBetti, ...
182  // computeFrame()
183  }
184 
185 #if defined(WITH_TBB)
186  // build the dependency graph
187  if (mParallelizeByDegree)
188  {
189  //std::cout << "In dep graph" << std::endl;
190  mScheduler.execute([&] {
191  makeDependencyGraph(mDepGraph,length_limit+1,top_degree - mLoSlantedDegree+1,true);
192  mDepGraph.startComputation();
193  mDepGraph.waitForCompletion();
194  });
195  //std::cout << "Out dep graph" << std::endl;
196  }
197  // If this is not run, the 'computeRanks' calls below compute all of these elements
198 #endif
199 
200  // What needs to be computed?
201  // lodeg..hideg, level: 0..maxlevel. Note: need to compute at level
202  // maxlevel+1 in order to get min betti numbers at
203  // level maxlevel.
204  // Also note: if hideg is the highest degree that occurs in the frame, we do
205  // not need to compute any matrices for these.
206 
207  for (int deg = mLoSlantedDegree; deg <= top_degree - 1; deg++)
208  for (int lev = 1; lev <= length_limit + 1; lev++)
209  {
210  computeRank(deg, lev);
211  }
212 
213  for (int lev = 1; lev <= length_limit; lev++)
214  {
215  computeRank(top_degree, lev);
216  }
217 
218  if (M2_gbTrace >= 1)
219  {
220  std::cout << "displaying stats" << std::endl;
221  showMemoryUsage();
222  monoid().show();
223  std::cout << "total setPoly: " << ResPolynomialConstructor::ncalls << std::endl;
224  std::cout << "total setPolyFromArray: "
225  << ResPolynomialConstructor::ncalls_fromarray << std::endl;
226  std::cout << "total ~ResPolynomial: " << ResPolynomial::npoly_destructor << std::endl;
227 
228  std::cout << "total time for make matrix: " << timeMakeMatrix
229  << std::endl;
230  std::cout << "total time for sort matrix: " << timeSortMatrix
231  << std::endl;
232  std::cout << "total time for reorder matrix: " << timeReorderMatrix
233  << std::endl;
234  std::cout << "total time for gauss matrix: " << timeGaussMatrix
235  << std::endl;
236  std::cout << "total time for clear matrix: " << timeClearMatrix
237  << std::endl;
238  std::cout << "total time for reset hash table: " << timeResetHashTable
239  << std::endl;
240  std::cout << "total time for computing ranks: " << timeComputeRanks
241  << std::endl;
242  std::cout << "total time for computing sparse ranks: " << timeComputeSparseRanks
243  << std::endl;
244  }
245 
246  BettiDisplay B(mBettiMinimal); // copy
247  B.resize(mLoSlantedDegree, top_degree, length_limit);

470 component_index SchreyerFrame::computeIdealQuotient(int lev,
471  component_index begin,
472  component_index elem)
473 {

477  res_packed_monomial m = monomial(lev, elem);
478  std::vector<PreElement*> elements;
479  if (ring().isSkewCommutative())

493  elements.push_back(vp);
494  }

496  }
497  for (component_index i = begin; i < elem; i++)
498  elements.push_back(createQuotientElement(monomial(lev, i), m));

535 component_index SchreyerFrame::computeNextLevel()
536 {
537  if (currentLevel() == 1) return 0;
538  if (currentLevel() >= mFrame.mLevels.size()) return 0;
539  // std::cout << "computeNextLevel: level = " << currentLevel() << std::endl;
540  // loop through all the elements at level currentLevel()-2
541  int level0 = currentLevel() - 2;
542  int level1 = level0 + 1;
543  component_index n_elems_added = 0;
544  for (auto i = level(level0).begin(); i != level(level0).end(); ++i)
545  {
546  component_index begin = (*i).mBegin;
547  component_index end = (*i).mEnd;
548  for (component_index i = begin; i < end; ++i)
549  {
550  auto& elem = level(level1)[i];
551  elem.mBegin = n_elems_added;
552  n_elems_added += computeIdealQuotient(level1, begin, i);
553  elem.mEnd = n_elems_added;
554  }
555  }
556  setSchreyerOrder(mCurrentLevel);
557  mCurrentLevel++;
558 
559  return n_elems_added;
560 }

562 void SchreyerFrame::setSchreyerOrder(int lev)
563 {
564  auto& myframe = level(lev);
565  auto& myorder = schreyerOrder(lev);
566  myorder.mTieBreaker.resize(myframe.size());
567  // std::cout << "setSchreyerOrder: entering, lev=" << lev << ", nelems=" <<
568  // myframe.size() << std::endl;
569  if (lev == 0)
570  {
571  for (component_index i = 0; i < myorder.mTieBreaker.size(); i++)
572  myorder.mTieBreaker[i] = i;
573  return;
574  }
575 
576  auto& prevorder = schreyerOrder(lev - 1);
577  long* tiebreakers = new long[myframe.size()];
578 
579  auto n_frame_elems = myframe.size();
580  for (component_index i = 0; i < n_frame_elems; i++)
581  {
582  component_index comp = monoid().get_component(myframe[i].mMonom);
583  tiebreakers[i] = i + n_frame_elems * prevorder.mTieBreaker[comp];
584  }
585  std::stable_sort(tiebreakers, tiebreakers + n_frame_elems);
586 
587  for (component_index i = 0; i < n_frame_elems; i++)
588  {
589  myorder.mTieBreaker[tiebreakers[i] % n_frame_elems] = i;
590  }
591  delete[] tiebreakers;
592  // std::cout << " setSchreyerOrder: exiting, lev=" << lev << std::endl;
593 }

595 void SchreyerFrame::insertBasic(int lev, res_packed_monomial monom, int degree)
596 {
597  assert(lev >= 1); // Should not be called with lev==0.
598  // if lev >= 2, then level(lev-1)[comp].(mBegin,mEnd) is set separately.
599  auto& myframe = level(lev);
600  long idx = myframe.size();
601  myframe.emplace_back(FrameElement(monom, degree));
602  auto& myelem = myframe[idx];
603  myelem.mSyzygy.coeffs = vectorArithmetic().allocateElementArray();
604  // The rest of this code simply sets the total monomial for the Schreyer order
605  // and should be moved out of here. (MES 3 Feb 2016)
606  auto& myorder = schreyerOrder(lev);
607  auto myTotalMonom = monomialBlock().allocate(monoid().max_monomial_size());
608  auto& prevorder = schreyerOrder(lev - 1);
609  component_index comp = monoid().get_component(myelem.mMonom);
610  monoid().unchecked_mult(
611  myelem.mMonom, prevorder.mTotalMonom[comp], myTotalMonom);
612  monoid().set_component(monoid().get_component(prevorder.mTotalMonom[comp]),
613  myTotalMonom);
614  myorder.mTotalMonom.push_back(myTotalMonom);
