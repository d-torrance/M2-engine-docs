122template <typename RT>
123SLEvaluator* SLEvaluatorConcrete<RT>::specialize(
124 const MutableMat<DMat<RT> >* parameters) const
125{
126 size_t nNewParams = parameters->n_rows();
127 if (parameters->n_cols() != 1 || nNewParams > nInputs)
128 throw exc::engine_error("1-column matrix expected; or #parameters > #vars");
129 SLEvaluatorConcrete<RT>* e = new SLEvaluatorConcrete<RT>(*this);
130 if (isCompiled) {
131 e->nParams = nParams + nNewParams;
132 e->nInputs -= nNewParams;
133 delete[] e->parametersAndInputs;
134 e->parametersAndInputs = new ElementType[e->nParams+e->nInputs];
135 std::copy(parametersAndInputs, parametersAndInputs+nParams, e->parametersAndInputs); // old parameters values
136 for (int i = 0; i < nNewParams; ++i)
137 ring().set(e->parametersAndInputs[nParams+i], parameters->getMat().entry(i, 0));
138 }
139 else {
140 for (int i = 0; i < nNewParams; ++i)
141 ring().set(e->values[varsPos[i]], parameters->getMat().entry(i, 0));
142 e->varsPos.erase(e->varsPos.begin(), e->varsPos.begin() + nNewParams);
143 e->nInputs = e->varsPos.size();
144 e->nParams = nParams + nNewParams;
145 }
146 return e;
147}

149template <typename RT>
150void SLEvaluatorConcrete<RT>::computeNextNode()
151{
152 ElementType& v = *vIt;
153 switch (*nIt++)
154 {
155 case SLProgram::MProduct:
156 ring().set_from_long(v, 1);
157 for (int i = 0; i < *numInputsIt; i++)
158 ring().mult(v, v, *(vIt + (*inputPositionsIt++)));
159 numInputsIt++;
160 break;
161 case SLProgram::MSum:
162 ring().set_zero(v);
163 for (int i = 0; i < *numInputsIt; i++)
164 ring().add(v, v, *(vIt + (*inputPositionsIt++)));
165 numInputsIt++;
166 break;
167 case SLProgram::Det:
168 {
169 int n = static_cast<int>(sqrt(*numInputsIt++));
170 DMat<RT> mat(ring(), n, n);
171 for (int i = 0; i < n; i++)
172 for (int j = 0; j < n; j++)
173 ring().set(mat.entry(i, j), *(vIt + (*inputPositionsIt++)));
174 DMatLinAlg<RT>(mat).determinant(v);
175 }
176 break;
177 case SLProgram::Divide:
178 ring().set(v, *(vIt + (*inputPositionsIt++)));
179 ring().divide(v, v, *(vIt + (*inputPositionsIt++)));
180 break;
181 default:
182 std::cerr << "unknown node type\n";
183 abort();
184 }
185}

187template <typename RT>
188bool SLEvaluatorConcrete<RT>::evaluate(const MutableMatrix* inputs,
189 MutableMatrix* outputs)
190{
191 auto inp = dynamic_cast<const MutableMat<DMat<RT> >*>(inputs);
192 auto out = dynamic_cast<MutableMat<DMat<RT> >*>(outputs);
193 if (inp == nullptr)
194 {
195 ERROR("inputs: expected a dense mutable matrix");
196 return false;
197 }
198 if (out == nullptr)
199 {
200 ERROR("outputs: expected a dense mutable matrix");
201 return false;
202 }
203 if (&ring() != &inp->getMat().ring())
204 {
205 ERROR("inputs are in a different ring");
206 return false;
207 }
208 if (&ring() != &out->getMat().ring())
209 {
210 ERROR("outputs are in a different ring");
211 return false;
212 }
213 return evaluate(inp->getMat(), out->getMat());
214}

216template <typename RT>
217bool SLEvaluatorConcrete<RT>::evaluate(const DMat<RT>& inputs,
218 DMat<RT>& outputs)
219{
220 if (nInputs != inputs.numRows() * inputs.numColumns()) {
221 ERROR(
222 "inputs: the number of inputs does not match the number of entries "
223 "in the inputs matrix");
224 std::cout << nInputs
225 << " != " << inputs.numRows() * inputs.numColumns()
226 << std::endl;
227 return false;
228 }
229 if (nOutputs != outputs.numRows() * outputs.numColumns()) {
230 ERROR(
231 "outputs: the number of outputs does not match the number of entries "
232 "in the outputs matrix");
233 std::cout << nOutputs << " != " << outputs.numRows()
234 << " * " << outputs.numColumns() << std::endl;
235 return false;
236 }
237
238 if (isCompiled) {
239 if(parametersAndInputs==nullptr) {
240 (*compiled_fn)(inputs.array(), outputs.array());
241 } else {
242 std::copy(inputs.array(),inputs.array()+nInputs,parametersAndInputs+nParams);
243 (*compiled_fn)(parametersAndInputs, outputs.array());
244 }
245 return true;
246 } else {
247 size_t i = 0;
248 for (size_t r = 0; r < inputs.numRows(); r++)
249 for (size_t c = 0; c < inputs.numColumns(); c++)
250 ring().set(values[varsPos[i++]], inputs.entry(r, c));
251 nIt = slp->mNodes.begin();
252 numInputsIt = slp->mNumInputs.begin();
253 inputPositionsIt = slp->mInputPositions.begin();
254 for (vIt = values.begin() + slp->inputCounter; vIt != values.end(); ++vIt)
255 computeNextNode();
256 i = 0;
257 for (size_t r = 0; r < outputs.numRows(); r++)
258 for (size_t c = 0; c < outputs.numColumns(); c++)
259 ring().set(outputs.entry(r, c), values[ap(slp->mOutputPositions[i++])]);
260 return true;
261 }

298template <typename RT>
299inline void norm2(const DMat<RT>& M,
300 size_t n,
301 typename RT::RealElementType& result)
302{
303 const auto& C = M.ring();
304 const auto& R = C.real_ring();
305 typename RT::RealRingType::Element c(R);
306 R.set_zero(result);
307 for (size_t i = 0; i < n; i++)
308 {
309 C.abs_squared(c, M.entry(0, i));
310 R.add(result, result, c);
311 }
312}

343template <typename RT>
344bool HomotopyConcrete<RT, FixedPrecisionHomotopyAlgorithm>::track(
345 const MutableMatrix* inputs,
346 MutableMatrix* outputs,
347 MutableMatrix* output_extras,
348 gmp_RR init_dt,
349 gmp_RR min_dt,
350 gmp_RR epsilon, // o.CorrectorTolerance,
351 int max_corr_steps,
352 gmp_RR infinity_threshold,
353 bool checkPrecision)
354{
355 std::chrono::steady_clock::time_point start =
356 std::chrono::steady_clock::now();
357 size_t solveLinearTime = 0, solveLinearCount = 0, evaluateTime = 0;
358
359 // std::cout << "inside
360 // HomotopyConcrete<RT,FixedPrecisionHomotopyAlgorithm>::track" << std::endl;
361 // double the_smallest_number = 1e-13;
362 const Ring* matRing = inputs->get_ring();
363 if (outputs->get_ring() != matRing)
364 {
365 ERROR("outputs and inputs are in different rings");
366 return false;
367 }
368 auto inp = dynamic_cast<const MutableMat<DMat<RT> >*>(inputs);
369 auto out = dynamic_cast<MutableMat<DMat<RT> >*>(outputs);
370 auto out_extras =
371 dynamic_cast<MutableMat<DMat<M2::ARingZZGMP> >*>(output_extras);
372 if (inp == nullptr)
373 {
374 ERROR("inputs: expected a dense mutable matrix");
375 return false;
376 }
377 if (out == nullptr)
378 {
379 ERROR("outputs: expected a dense mutable matrix");
380 return false;
381 }
382 if (out_extras == nullptr)
383 {
384 ERROR("output_extras: expected a dense mutable matrix");
385 return false;
386 }
387
388 auto& in = inp->getMat();
389 auto& ou = out->getMat();
390 auto& oe = out_extras->getMat();
391 size_t n_sols = in.numColumns();
392 size_t n = in.numRows() - 1; // number of x vars
393
394 if (ou.numColumns() != n_sols or ou.numRows() != n + 2)
395 {
396 ERROR("output: wrong shape");
397 return false;
398 }
399 if (oe.numColumns() != n_sols or oe.numRows() != 2)
400 {
401 ERROR("output_extras: wrong shape");
402 return false;
403 }
404
405 const RT& C = in.ring();
406 typename RT::RealRingType R = C.real_ring();
407
408 typedef typename RT::ElementType ElementType;
409 typedef typename RT::Element Element;
410 typedef typename RT::RealRingType::Element RealElement;
411 typedef typename RT::RealRingType::ElementType RealElementType;
412 typedef MatElementaryOps<DMat<RT> > MatOps;
413
414 RealElement t_step(R), min_step2(R), epsilon2(R), infinity_threshold2(R);
415 R.set_from_BigReal(t_step, init_dt); // initial step
416 R.set_from_BigReal(min_step2, min_dt);
417 R.mult(min_step2, min_step2, min_step2); // min_step^2
418 R.set_from_BigReal(epsilon2, epsilon);
419 int tolerance_bits = int(log2(fabs(R.coerceToDouble(epsilon2))));
420 R.mult(epsilon2, epsilon2, epsilon2); // epsilon^2
421 R.set_from_BigReal(infinity_threshold2, infinity_threshold);
422 R.mult(infinity_threshold2, infinity_threshold2, infinity_threshold2);
423 int num_successes_before_increase = 3;
424
425 RealElement t0(R), dt(R), one_minus_t0(R), dx_norm2(R), x_norm2(R), abs2dc(R);
426
427 // constants
428 RealElement one(R), two(R), four(R), six(R), one_half(R), one_sixth(R);
429 RealElementType& dt_factor = one_half;
430 R.set_from_long(one, 1);
431 R.set_from_long(two, 2);
432 R.set_from_long(four, 4);
433 R.set_from_long(six, 6);
434 R.divide(one_half, one, two);
435 R.divide(one_sixth, one, six);
436
437 Element c_init(C), c_end(C), dc(C), one_half_dc(C);
438
439 // think: x_0..x_(n-1), c
440 // c = the homotopy continuation parameter "t" upstair, varies on a (staight
441 // line) segment of complex plane (from c_init to c_end)
442 // t = a real running in the interval [0,1]
443
444 DMat<RT> x0c0(C, n + 1, 1);
445 DMat<RT> x1c1(C, n + 1, 1);
446 DMat<RT> xc(C, n + 1, 1);
447 DMat<RT> HxH(C, n, n + 1);
448 DMat<RT>& Hxt = HxH; // the matrix has the same shape: reuse memory
449 DMat<RT> LHSmat(C, n, n);
450 auto LHS = submatrix(LHSmat);
451 DMat<RT> RHSmat(C, n, 1);
452 auto RHS = submatrix(RHSmat);
453 DMat<RT> dx(C, n, 1);
454 DMat<RT> dx1(C, n, 1);
455 DMat<RT> dx2(C, n, 1);
456 DMat<RT> dx3(C, n, 1);
457 DMat<RT> dx4(C, n, 1);
458 DMat<RT> Jinv_times_random(C, n, 1);
459
460 ElementType& c0 = x0c0.entry(n, 0);
461 ElementType& c1 = x1c1.entry(n, 0);
462 ElementType& c = xc.entry(n, 0);
463 RealElementType& tol2 = epsilon2; // current tolerance squared
464 bool linearSolve_success;
465 for (size_t s = 0; s < n_sols; s++)
466 {
467 SolutionStatus status = PROCESSING;
468 // set initial solution and initial value of the continuation parameter
469 // for(size_t i=0; i<=n; i++)
470 // C.set(x0c0.entry(i,0), in.entry(i,s));
471 submatrix(x0c0) = submatrix(const_cast<DMat<RT>&>(in), 0, s, n + 1, 1);
472 C.set(c_init, c0);
473 C.set(c_end, ou.entry(n, s));
474
475 R.set_zero(t0);
476 bool t0equals1 = false;
477
478 // t_step is actually the initial (absolute) length of step on the
479 // interval [c_init,c_end]
480 // dt is an increment for t on the interval [0,1]
481 R.set(dt, t_step);
482 C.subtract(dc, c_end, c_init);
483 C.abs(abs2dc, dc); // don't want to create new temporary elts: reusing dc
484 // and abs2dc
485 R.divide(dt, dt, abs2dc);
486
487 int predictor_successes = 0;
488 int count = 0; // number of steps
489 // track the real segment (1-t)*c0 + t*c1, a\in [0,1]
490 while (status == PROCESSING and not t0equals1)
491 {
492 if (M2_numericalAlgebraicGeometryTrace > 3)
493 {
494 buffer o;
495 R.elem_text_out(o, t0, true, false, false);
496 std::cout << "t0 = " << o.str();
497 o.reset();
498 C.elem_text_out(o, c0, true, false, false);
499 std::cout << ", c0 = " << o.str() << std::endl;
500 }
501 R.subtract(one_minus_t0, one, t0);
502 if (R.compare_elems(dt, one_minus_t0) > 0)
503 {
504 R.set(dt, one_minus_t0);
505 t0equals1 = true;
506 C.subtract(dc, c_end, c0);
507 C.set(c1, c_end);
508 }
509 else
510 {
511 C.subtract(dc, c_end, c0);
512 C.mult(dc, dc, dt);
513 C.divide(dc, dc, one_minus_t0);
514 C.add(c1, c0, dc);
515 }
516
517 // PREDICTOR in: x0c0,dt
518 // out: dx
519 /* top-level code for Runge-Kutta-4
520 dx1 := solveHxTimesDXequalsMinusHt(x0,t0);
521 dx2 := solveHxTimesDXequalsMinusHt(x0+(1/2)*dx1*dt,t0+(1/2)*dt);
522 dx3 := solveHxTimesDXequalsMinusHt(x0+(1/2)*dx2*dt,t0+(1/2)*dt);
523 dx4 := solveHxTimesDXequalsMinusHt(x0+dx3*dt,t0+dt);
524 (1/6)*dt*(dx1+2*dx2+2*dx3+dx4)
525 */
526
527 C.mult(one_half_dc, dc, one_half);
528
529 // dx1
530 submatrix(xc) = submatrix(x0c0);
531 TIME(evaluateTime, mHxt.evaluate(xc, Hxt))
532
533 LHS = submatrix(Hxt, 0, 0, n, n);
534 RHS = submatrix(Hxt, 0, n, n, 1);
535 MatrixOps::negateInPlace(RHSmat);
536
537 TIME(solveLinearTime,
538 linearSolve_success =
539 MatrixOps::solveLinear(LHSmat, RHSmat, dx1));
540 solveLinearCount++;
541
542 // dx2
543 if (linearSolve_success)
544 {
545 submatrix(dx1) *= one_half_dc; // "dx1" := (1/2)*dx1*dt
546 submatrix(xc, 0, 0, n, 1) += submatrix(dx1);
547 C.add(c, c, one_half_dc);
548
549 TIME(evaluateTime, mHxt.evaluate(xc, Hxt))
550
551 LHS = submatrix(Hxt, 0, 0, n, n);
552 RHS = submatrix(Hxt, 0, n, n, 1);
553 MatrixOps::negateInPlace(RHSmat);
554
555 TIME(solveLinearTime,
556 linearSolve_success =
557 MatrixOps::solveLinear(LHSmat, RHSmat, dx2);)
558 solveLinearCount++;
559 }
560
561 // dx3
562 if (linearSolve_success)
563 {
564 submatrix(dx2) *= one_half_dc; // "dx2" := (1/2)*dx2*dt
565 submatrix(xc, 0, 0, n, 1) = submatrix(x0c0, 0, 0, n, 1);
566 submatrix(xc, 0, 0, n, 1) += submatrix(dx2);
567 // C.add(c,c,one_half_dc); // c should not change here??? or copy
568 // c two lines above???
569
570 TIME(evaluateTime, mHxt.evaluate(xc, Hxt));
571
572 LHS = submatrix(Hxt, 0, 0, n, n);
573 RHS = submatrix(Hxt, 0, n, n, 1);
574 MatrixOps::negateInPlace(RHSmat);
575
576 TIME(solveLinearTime,
577 linearSolve_success =
578 MatrixOps::solveLinear(LHSmat, RHSmat, dx3););
579 solveLinearCount++;
580 }
581
582 // dx4
583 if (linearSolve_success)
584 {
585 submatrix(dx3) *= dc; // "dx3" := dx3*dt
586 submatrix(xc) = submatrix(
587 x0c0); // sets c=c0 as well (not needed for dx2???,dx3)
588 submatrix(xc, 0, 0, n, 1) += submatrix(dx3);
589 C.add(c, c, dc);
590
591 TIME(evaluateTime, mHxt.evaluate(xc, Hxt));
592
593 LHS = submatrix(Hxt, 0, 0, n, n);
594 RHS = submatrix(Hxt, 0, n, n, 1);
595 MatrixOps::negateInPlace(RHSmat);
596
597 TIME(solveLinearTime,
598 linearSolve_success =
599 MatrixOps::solveLinear(LHSmat, RHSmat, dx4););
600 solveLinearCount++;
601 }
602
603 // "dx1" = .5*dx1*dt, "dx2" = .5*dx2*dt, "dx3" = dx3*dt
604 if (linearSolve_success)
605 {
606 submatrix(dx4) *= dc; // "dx4" = dx4*dt
607 submatrix(dx) =
608 submatrix(dx4); // dx = (1/6)*dt*(dx1+2*dx2+2*dx3+dx4)
609 // = (1/6)*(2*"dx1"+4*"dx2"+2*"dx3"+"dx4")
610 submatrix(dx1) *= two;
611 submatrix(dx) += dx1;
612 // submatrix(dx).addMultipleTo(two,dx1);
613 submatrix(dx2) *= four;
614 submatrix(dx) += dx2;
615 submatrix(dx3) *= two;
616 submatrix(dx) += dx3;
617 submatrix(dx) *= one_sixth;
618
619 // update x0c0
620 submatrix(x1c1) = submatrix(x0c0);
621 submatrix(x1c1, 0, 0, n, 1) += dx;
622 C.add(c1, c0, dc);
623 }
624
625 // CORRECTOR
626 bool is_successful;
627 int n_corr_steps = 0;
628 if (linearSolve_success)
629 {
630 do
631 {
632 n_corr_steps++;
633
634 TIME(evaluateTime, mHxH.evaluate(x1c1, HxH));
635
636 LHS = submatrix(HxH, 0, 0, n, n);
637 RHS = submatrix(HxH, 0, n, n, 1);
638 MatrixOps::negateInPlace(RHSmat);
639
640 TIME(solveLinearTime,
641 linearSolve_success =
642 MatrixOps::solveLinear(LHSmat, RHSmat, dx););
643 solveLinearCount++;
644
645 // x1 += dx
646 submatrix(x1c1, 0, 0, n, 1) += dx;
647
648 normSquared(submatrix(dx), dx_norm2);
649 normSquared(submatrix(x1c1, 0, 0, n, 1), x_norm2);
650
651 R.mult(x_norm2, x_norm2, tol2);
652 is_successful = R.compare_elems(dx_norm2, x_norm2) < 0;
653 }
654 while (not is_successful and n_corr_steps < max_corr_steps);
655 }
656 // std::cout << "past corrector loop...\n";
657 if (not linearSolve_success or not is_successful)
658 {
659 // predictor failure
660 predictor_successes = 0;
661 R.mult(dt, dt, dt_factor);
662 t0equals1 = false;
663 C.abs_squared(abs2dc, dc);
664 if (R.compare_elems(abs2dc, min_step2) < 0)
665 status = MIN_STEP_FAILED;
666 }
667 else
668 {
669 // predictor success
670 predictor_successes = predictor_successes + 1;
671 MatOps::setFromSubmatrix(x1c1, 0, n, 0, 0, x0c0); // x1=x0
672 R.add(t0, t0, dt); // increment t: so far only s was incremented
673 count++;
674 if (predictor_successes >= num_successes_before_increase)
675 {
676 predictor_successes = 0;
677 R.divide(dt, dt, dt_factor);
678 }
679 }
680
681 normSquared(submatrix(x0c0, 0, 0, n, 1),
682 x_norm2); // x_norm2 = ||x0||^2
683
684 if (not linearSolve_success)
685 status = SINGULAR;
686 else if (checkPrecision and not t0equals1)
687 { // precision check
688 mHxH.evaluate(x0c0, HxH);
689 MatOps::setFromSubmatrix(HxH, 0, n - 1, 0, n - 1, LHSmat); // Hx
690 // setRandomUnitVector(RHSmat,n);
691 for (int i = 0; i < n; i++) C.random(RHSmat.entry(i, 0));
692
693 TIME(solveLinearTime,
694 linearSolve_success = MatrixOps::solveLinear(
695 LHSmat, RHSmat, Jinv_times_random););
696 solveLinearCount++;
697
698 norm2(Jinv_times_random,
699 n,
700 dx_norm2); // this stands for ||J^{-1}||
702// ||J^{-1}|| should be multiplied by a factor
703// reflecting an estimate on the error of evaluation of J
704#define PRECISION_SAFETY_BITS 10
705 int more_bits = int(log2(fabs(R.coerceToDouble(dx_norm2)))) / 2;
706 int precision_needed = PRECISION_SAFETY_BITS + tolerance_bits + more_bits;
707 if (precision_needed<53) precision_needed = 53;
708 if (M2_numericalAlgebraicGeometryTrace > 3)
709 std::cout << "precision needed = " << precision_needed << " = "
710 << PRECISION_SAFETY_BITS << "(safety) + "
711 << tolerance_bits << "(tolerance) + "
712 << more_bits << "(additional)\n"
713 << "current precision = " << R.get_precision() << std::endl;
714 if (R.get_precision() < precision_needed)
715 status = INCREASE_PRECISION;
716 else if (R.get_precision() != 53 and
717 R.get_precision() > 2 * precision_needed)
718 status = DECREASE_PRECISION;
719 if (M2_numericalAlgebraicGeometryTrace > 3)
720 std::cout << "status = " << status << std::endl;
721 };
722
723 // infinity/origin checks
724 if (status == PROCESSING)
725 {
726 if (R.compare_elems(infinity_threshold2, x_norm2) < 0)
727 status = INFINITY_FAILED;
728 else
729 {
730 if (R.is_zero(x_norm2))
731 status = ORIGIN_FAILED;
732 else
733 {
734 R.divide(x_norm2, one, x_norm2); // 1/||x||^2
735 if (R.compare_elems(infinity_threshold2, x_norm2) < 0)
736 status = ORIGIN_FAILED;
737 }
738 }
739 }
740 }
741 // record the solution
742 // set initial solution and initial value of the continuation parameter
743 for (size_t i = 0; i <= n; i++) C.set(ou.entry(i, s), x0c0.entry(i, 0));
744 C.set(ou.entry(n + 1, s), dc); // store last increment attempted
745 if (status == PROCESSING) status = REGULAR;
746 oe.ring().set_from_long(oe.entry(0, s), status);
747 oe.ring().set_from_long(oe.entry(1, s), count);
748 }
749
750 std::chrono::steady_clock::time_point end = std::chrono::steady_clock::now();
751 if (M2_numericalAlgebraicGeometryTrace > 1)
752 {
753 std::cout << "-- track took "
754 << std::chrono::duration_cast<std::chrono::milliseconds>(end -
755 start)
756 .count()
757 << "ms.\n";
758 std::cout << "-- # of solveLinear calls = " << solveLinearCount
759 << std::endl;
760 std::cout << "-- time of solveLinear calls = " << solveLinearTime
761 << " ns." << std::endl;
762 std::cout << "-- time of evaluate calls = " << evaluateTime << " ns."
763 << std::endl;
764 }
765 return true;
766}
