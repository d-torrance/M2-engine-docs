125template <typename T>
126void testRingAxioms(const T* R, int ntrials)
127{
128 RingElementGenerator<T> gen(*R);
129 for (int i = 0; i < ntrials; i++)
130 {
131 ring_elem a = gen.nextElement();
132 ring_elem b = gen.nextElement();
133 ring_elem c = gen.nextElement();
134
135 // Test commutativity
136 // test: a*b = b*a
137 // test: a+b == b+a
138 ring_elem d = R->add(a, b);
139 ring_elem e = R->add(b, a);
140 EXPECT_TRUE(R->is_equal(d, e));
141 d = R->mult(a, b);
142 e = R->mult(b, a);
143 EXPECT_TRUE(R->is_equal(d, e));
144
145 // Test associativity
146 // test: a+(b+c) == (a+b)+c
147 // test: a*(b*c) == (a*b)*c
148 d = R->add(a, R->add(b, c));
149 e = R->add(R->add(a, b), c);
150 EXPECT_TRUE(R->is_equal(d, e));
151 d = R->mult(a, R->mult(b, c));
152 e = R->mult(R->mult(a, b), c);
153 EXPECT_TRUE(R->is_equal(d, e));
154
155 // Test distributivity
156 // test: a*(b+c) == a*b + a*c
157 d = R->mult(a, R->add(b, c));
158 e = R->add(R->mult(a, b), R->mult(a, c));
159 EXPECT_TRUE(R->is_equal(d, e));
