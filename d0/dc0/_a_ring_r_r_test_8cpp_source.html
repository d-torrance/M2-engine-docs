171 TEST(ARingRR, axioms)
172 {
173  M2::ARingRR R;
174  auto nbits = R.get_precision();
175  ARingElementGenerator<M2::ARingRR> gen(R);
176  M2::ARingRR::ElementType a, b, c, d, e;
177  R.init(a);
178  R.init(b);
179  R.init(c);
180  R.init(d);
181  R.init(e);
182  for (int i = 0; i < ntrials; i++)
183  {
184  gen.nextElement(a);
185  gen.nextElement(b);
186  gen.nextElement(c);
187  // Test commutativity
188  // test: a*b = b*a
189  // test: a+b == b+a
190  R.add(d, a, b);
191  R.add(e, b, a);
192  EXPECT_TRUE(almostEqual(R, nbits - 2, d, e));
193  R.mult(d, a, b);
194  R.mult(e, b, a);
195  EXPECT_TRUE(almostEqual(R, nbits - 2, d, e));
196 
197  // Test associativity
198  // test: a+(b+c) == (a+b)+c
199  // test: a*(b*c) == (a*b)*c
200  R.add(e, b, c);
201  R.add(d, a, e); // a+(b+c)
202  R.add(e, a, b);
203  R.add(e, e, c); // (a+b)+c
204  EXPECT_TRUE(almostEqual(R, nbits - 6, d, e));
205  R.mult(e, b, c);
206  R.mult(d, a, e); // a*(b*c)
207  R.mult(e, a, b);
208  R.mult(e, e, c); // (a*b)*c
209  EXPECT_TRUE(almostEqual(R, nbits - 6, d, e));
210 
211  // Test distributivity
212  // test: a*(b+c) == a*b + a*c
213  R.add(e, b, c);
214  R.mult(d, a, e); // a*(b+c)
215  R.mult(b, a, b);
216  R.mult(c, a, c);
217  R.add(e, b, c); // a*b + a*c
218  EXPECT_TRUE(almostEqual(R, nbits - 6, d, e));
219  }
220  R.clear(e);
221  R.clear(d);
222  R.clear(c);
223  R.clear(b);
