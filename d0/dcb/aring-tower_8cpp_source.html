98void ARingTower::clear(int level, ARingPolynomial &f) const
99// free all space associated to f, set f to 0.
100{
101 if (f == nullptr) return;
102 if (level == 0)
103 {
104 for (int i = 0; i <= f->deg; i++) mBaseRing.clear(f->coeffs[i]);
105 delete[] f->coeffs;
106 }
107 else
108 {
109 for (int i = 0; i <= f->deg; i++) clear(level - 1, f->polys[i]);
110 delete[] f->polys;
111 }
112 delete f;
113 f = nullptr;
114}

143void ARingTower::increase_capacity(int newdeg, ARingPolynomial &f) const
144{
145 assert(f != 0);
146 if (f->len <= newdeg)
147 {
148 ARingPolynomial *newelems = newarray(ARingPolynomial, newdeg + 1);
149 ARingPolynomial *fp = f->polys;
150 for (int i = 0; i <= f->deg; i++) newelems[i] = fp[i];
151 for (int i = f->deg + 1; i < newdeg + 1; i++) newelems[i] = nullptr;
152 delete[] fp;
153 f->polys = newelems;
154 f->len = newdeg + 1;
155 f->deg = newdeg;
156 }
157}

213void ARingTower::elem_text_out(buffer &o,
214 int level,
215 const ARingPolynomial f,
216 bool p_one,
217 bool p_plus,
218 bool p_parens) const
219{
220 // o << to_string(level, f);
221 if (f == nullptr)
222 {
223 o << "0";
224 return;
225 }
226
227 int nterms = n_nonzero_terms(level, f);
228 bool needs_parens = p_parens && (nterms >= 2);
229
230 if (needs_parens)
231 {
232 if (p_plus) o << '+';
233 o << '(';
234 p_plus = false;
235 }
236
237 bool one = is_one(level, f);
238
239 if (one)
240 {
241 if (p_plus) o << "+";
242 if (p_one) o << "1";
243 return;
244 }
245
246 const std::string &this_varname = varNames()[level];
247
248 if (level == 0)
249 {
250 bool firstterm = true;
251 for (int i = f->deg; i >= 0; i--)
252 if (f->coeffs[i] != 0)
253 {
254 if (!firstterm || p_plus) o << "+";
255 firstterm = false;
256 if (i == 0 || f->coeffs[i] != 1)
257 mBaseRing.elem_text_out(o, f->coeffs[i], p_one, p_plus, p_parens);
258 if (i > 0) o << this_varname;
259 if (i > 1) o << i;
260 }
261 if (needs_parens) o << ")";
262 }
263 else
264 {
265 bool firstterm = true;
266 for (int i = f->deg; i >= 0; i--)
267 if (f->polys[i] != nullptr)
268 {
269 bool this_p_parens = p_parens || (i > 0);
270
271 if (i == 0 || !is_one(level - 1, f->polys[i]))
272 elem_text_out(o,
273 level - 1,
274 f->polys[i],
275 p_one,
276 p_plus || !firstterm,
277 this_p_parens);
278 else if (p_plus || !firstterm)
279 o << "+";
280 if (i > 0) o << this_varname;
281 if (i > 1) o << i;
282
283 firstterm = false;
284 }
285 if (needs_parens) o << ")";
286 }

289ARingPolynomial ARingTower::var(int level, int v) const
290// make the variable v (but at level 'level')
291{
292 if (v > level) return nullptr;
293 int which = (v == 0 ? 1 : 0);
294 ARingPolynomial result =
295 alloc_poly_0(which); // TODO: check that this initializes elements to 0
296 result->coeffs[which] = 1;
297 for (int i = 1; i <= level; i++)
298 {
299 which = (i == v ? 1 : 0);
300 ARingPolynomial a = result;
301 result = alloc_poly_n(which);
302 result->polys[which] = a;
303 }
304 return result;
305}

307bool ARingTower::is_equal(int level, const ARingPolynomial f, const ARingPolynomial g) const
308{
309 if (f == nullptr)
310 {
311 if (g == nullptr) return true;
312 return false;
313 }
314 if (g == nullptr || f->deg != g->deg) return false;
315 if (level == 0)
316 {
317 BaseCoefficientType *fp = f->coeffs;
318 BaseCoefficientType *gp = g->coeffs;
319 for (int i = 0; i <= f->deg; i++)
320 if (fp[i] != gp[i]) return false;
321 return true;
322 }
323 // level > 0
324 ARingPolynomial *fp = f->polys;
325 ARingPolynomial *gp = g->polys;
326 for (int i = 0; i <= f->deg; i++)
327 if (!is_equal(level - 1, fp[i], gp[i])) return false;
328 return true;
329}

331void ARingTower::add_in_place(int level, ARingPolynomial &f, const ARingPolynomial g) const
332{
333 if (g == nullptr) return;
334 if (f == nullptr)
335 {
336 f = copy(level, g);
337 return;
338 }
339 int fdeg = f->deg;
340 int gdeg = g->deg;
341
342 increase_capacity(g->deg, f);
343 if (level == 0)
344 for (int i = 0; i <= gdeg; i++)
345 mBaseRing.add(f->coeffs[i], f->coeffs[i], g->coeffs[i]);
346 else
347 for (int i = 0; i <= gdeg; i++)
348 add_in_place(level - 1, f->polys[i], g->polys[i]);
349
350 if (gdeg > fdeg)
351 f->deg = gdeg;
352 else if (gdeg == fdeg)
353 reset_degree(f);
354}
