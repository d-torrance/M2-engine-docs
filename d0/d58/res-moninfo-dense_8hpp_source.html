92  bool from_expvector(res_const_ntuple_monomial e,
93  component_index comp,
94  res_packed_monomial result) const
95  {
96  // Pack the vector e[0]..e[nvars-1],comp. Create the hash value at the same
97  // time.
98  ncalls_from_expvector++;
99  result[0] = 0;
100  result[1] = comp;
101 
102  for (int i = 0; i < nvars; i++)
103  {
104  result[firstvar + i] = e[i];
105  if (e[i] > 0) result[0] += hashfcn[i] * e[i];
106  }
107 
108  const int* wt = weight_vectors.data();
109  for (int j = 0; j < nweights; j++, wt += nvars)
110  {
111  res_monomial_word val = 0;
112  for (int i = 0; i < nvars; i++)
113  {
114  auto a = e[i];
115  if (a > 0) val += a * wt[i];
116  }
117  result[2 + j] = val;
118  }
119  return true;
120  }

179  void from_varpower_monomial(res_const_varpower_monomial m,
180  component_index comp,
181  res_packed_monomial result) const
182  {
183  // 'result' must have enough space allocated
184  ncalls_from_varpower++;
185  result[0] = 0;
186  result[1] = comp;
187  for (int i = 2; i < nslots; i++)
188  {
189  result[i] = 0;
190  }
191  for (index_res_varpower_monomial j = m; j.valid(); ++j)
192  {
193  res_varpower_word v = j.var();
194  res_varpower_word e = j.exponent();
195  result[firstvar + v] = e;
196  if (e == 1)
197  result[0] += hashfcn[v];
198  else
199  result[0] += e * hashfcn[v];
200  }
201 
202  const int* wt = weight_vectors.data();
203  for (int j = 0; j < nweights; j++, wt += nvars)
204  {
205  res_monomial_word val = 0;
206  for (index_res_varpower_monomial i = m; i.valid(); ++i)
207  {
208  auto v = i.var();
209  auto e = i.exponent();
210  auto w = wt[v];
211  if (e == 1)
212  val += w;
213  else
214  val += w * e;
215  result[2 + j] = val;
216  }
217  }

270  bool divide(res_const_packed_monomial m,
271  res_const_packed_monomial n,
272  res_packed_monomial result) const
273  {
274  ncalls_divide++;
275  // First, divide monomials
276  // Then, if the division is OK, set the component, hash value and rest of
277  // the monomial
278  if (m[1] != n[1]) // components are not equal
279  return false;
280  res_const_packed_monomial m1 = m + nslots;
281  res_const_packed_monomial n1 = n + nslots;
282  res_packed_monomial result1 = result + nslots;
283  for (int i = nslots - 2; i > 0; i--)
284  {
285  res_varpower_word cmp = *--m1 - *--n1;
286  if (cmp < 0) return false;
287  *--result1 = cmp;
288  }
289  result[1] = 0; // the component of a division is in the ring (comp 0).
290  result[0] = m[0] - n[0]; // subtract hash codes
291  return true;
292  }
