7ring_elem sagbi::subduct(int numslots,
8 const PolyRing *R,
9 ring_elem a,
10 const RingMap *phi,
11 GBComputation *J)
12{
13 Nterm *f = a;
14 Nterm head;
15 Nterm *result = &head;
16 MatrixConstructor mat(R->make_FreeModule(1), 1);
17
18 while (f != nullptr)
19 {
20 Nterm *g = f;
21 f = f->next;
22 g->next = nullptr;
23
24 mat.set_entry(0, 0, g);
25 Matrix *m = mat.to_matrix();
26 const Matrix *n = J->matrix_remainder(m);
27 ring_elem g1 = n->elem(0, 0);
28 delete m;
29 delete n;
30
31 // Is g1 a monomial in the new variables?
32 if (R->in_subring(numslots, g1))
33 {
34 g->next = f;
35 f = g;
36 ring_elem phi_g1 = R->eval(phi, g1, 0);
37 ring_elem fr = f;
38 R->internal_subtract_to(fr, phi_g1);
39 f = fr;
40 }
41 else
42 {
43 result->next = g;
44 result = g;
45 }
46 }
47
48 result->next = nullptr;
49 return head.next;
50}

52Matrix *sagbi::subduct(int numparts, const Matrix *m, const RingMap *phi, GBComputation *J)
53{
54 MatrixConstructor result(m->rows(), m->cols());
55 const PolyRing *R = m->get_ring()->cast_to_PolyRing();
56 if (R == nullptr)
57 {
58 ERROR("expected polynomial ring");
59 return nullptr;
60 }
61 int nslots = R->getMonoid()->n_slots(numparts);
62 for (int i = 0; i < m->n_cols(); i++)
63 {
64 ring_elem a = m->elem(0, i);
65 ring_elem b = subduct(nslots, R, R->copy(a), phi, J);
66 result.set_entry(0, i, b);
67 }
68 return result.to_matrix();
69}

71ring_elem sagbi::subduct1(int numslots,
72 const PolyRing *T, // this is the tensor ring
73 const PolyRing *S, // this is the poly ring
74 ring_elem a,
75 const RingMap *inclusionAmbient,
76 const RingMap *fullSubstitution,
77 const RingMap *substitutionInclusion,
78 GBComputation *gbI,
79 GBComputation *gbReductionIdeal)
80{
81 Nterm *f = a;
82 ring_elem fr = f;
83 MatrixConstructor matT(T->make_FreeModule(1), 1);
84 MatrixConstructor matS(S->make_FreeModule(1), 1);
85 bool breakFlag = false;
86
87 while ((f != nullptr) && (breakFlag == false))
88 {
89 // tensorRingg = S#"inclusionAmbient" liftg
90 // tesnorRingg = gInT
91 ring_elem gInT = S->eval(inclusionAmbient,fr,0);
92
93 // This might be the wrong way to deal with this issue.
94 // I don't know what to do, however.
95 if(gInT != nullptr)
96 {
97 // tensorRingLTg = leadTerm tensorRingg
98 // tensorRingLTg = LTgInT
99 Nterm *LTgInT = gInT;
100 LTgInT->next = nullptr;
101
102 // h = tensorRingLTg % (inAIdeal)
103 // h = h1
104 matT.set_entry(0,0,LTgInT);
105 Matrix *m = matT.to_matrix();
106 const Matrix *n = gbReductionIdeal->matrix_remainder(m);
107 ring_elem h1 = n->elem(0,0);
108 delete m;
109 delete n;
110
111 ring_elem projectionh = T->eval(substitutionInclusion,h1,0);
112
113 if(projectionh != nullptr)
114 {
115 // hSub = (S#"fullSubstitution" h) % I
116 ring_elem hInS = T->eval(fullSubstitution,h1,0);
117 matS.set_entry(0,0,hInS);
118 Matrix *k = matS.to_matrix();
119 const Matrix *l = gbI->matrix_remainder(k);
120 ring_elem h1InS = l->elem(0,0);
121 delete k;
122 delete l;
123
124 S->internal_subtract_to(fr,h1InS);
125 f = fr;
126 }
127 else
128 breakFlag = true;
129 }
130 else
131 breakFlag = true;
132 }
133
134 return ring_elem(f);
135}

137Matrix *sagbi::subduct1(int numparts,
138 const Ring *rawT,
139 const Ring *rawS,
140 const Matrix *m,
141 const RingMap *inclusionAmbient,
142 const RingMap *fullSubstitution,
143 const RingMap *substitutionInclusion,
144 GBComputation *gbI,
145 GBComputation *gbReductionIdeal)
146{
147 MatrixConstructor result(m->rows(), m->cols());
148 const PolyRing *T = rawT->cast_to_PolyRing();
149 const PolyRing *S = rawS->cast_to_PolyRing();
150 if ((T == nullptr) || (S == nullptr))
151 {
152 ERROR("expected polynomial ring");
153 return nullptr;
154 }
155 int nslots = T->getMonoid()->n_slots(numparts);
156 for (int i = 0; i < m->n_cols(); i++)
157 {
158 ring_elem a = m->elem(0, i);
159 ring_elem b = subduct1(nslots-2, T, S, S->copy(a),
160 inclusionAmbient,fullSubstitution,substitutionInclusion,
161 gbI,gbReductionIdeal);
162 result.set_entry(0, i, b);
163 }
164
165 return result.to_matrix();
166}
