8  const PolyRing *R,

11  GBComputation *J)

14  Nterm head;
15  Nterm *result = &head;
16  MatrixConstructor mat(R->make_FreeModule(1), 1);
17 
18  while (f != nullptr)
19  {
20  Nterm *g = f;
21  f = f->next;
22  g->next = nullptr;
23

29  delete n;
30

39  f = fr;

41  else

46  }

120  ring_elem h1InS = l->elem(0,0);

122  delete l;
123 
124  S->internal_subtract_to(fr,h1InS);
125  f = fr;

132  }
