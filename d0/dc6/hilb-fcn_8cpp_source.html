43bool HilbertController::setDegree(int this_degree)
44{
45 // Recomputes Hilbert function, returns #elems expected in degree this_degree
46 // There should be NO elements expected in lower degrees than that.
47 // false is returned if either the computation was interrupted, or
48 // it was determined that the Hilbert function could not be correct
49 // (by expecting a negative number of elements).
50 if (!recomputeHilbertFunction()) return false;
51 hilb_n_in_degree = hilb_comp::coeff_of(hf_diff, this_degree);
52 if (error()) return false;
53 return true;
54}

56bool HilbertController::addMonomial(int *a, int comp)
57// return true if that was the last one expected
58{
59 monomial m = R->getMonoid()->make_one();
60 R->getMonoid()->from_expvector(a, m);
61 ring_elem r = R->make_flat_term(R->getCoefficientRing()->one(), m);
62 vec v = R->make_vec(comp - 1, r);
63 elems.push_back(v);
64 hilb_new_elems = true;
65 hilb_n_in_degree--;
66 return hilb_n_in_degree == 0;
67}

69bool HilbertController::recomputeHilbertFunction()
70{
71 if (hilb_new_elems)
72 {
73 // Recompute h, hf_diff
74 Matrix *M = make_lead_term_matrix();
75 RingElement *h = hilb_comp::hilbertNumerator(M);
76 if (h == nullptr) return false; // computation was interrupted
77 hf_diff = (*h) - (*hf_orig);
78 hilb_new_elems = false;
79 }
80 return true;
81}

83Matrix *HilbertController::make_lead_term_matrix()
84{
85 MatrixConstructor mat(F, INTSIZE(elems));
86 for (int i = 0; i < elems.size(); i++) mat.set_column(i, elems[i]);
87 return mat.to_matrix();
88}
