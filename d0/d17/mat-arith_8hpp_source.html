33template <typename MatType>
34struct SubMatrix
35{
36 MatType& matrix;
37 const long begin_row;
38 const long end_row;
39 const long begin_column;
40 const long end_column;
41
42 SubMatrix(MatType& m)
43 : matrix(m),
44 begin_row(0),
45 end_row(m.numRows()),
46 begin_column(0),
47 end_column(m.numColumns())
48 {
49 }
50
51 SubMatrix(MatType& m, long x, long y, long nrows, long ncols)
52 : matrix(m),
53 begin_row(x),
54 end_row(x + nrows),
55 begin_column(y),
56 end_column(y + ncols)
57 {
58 }
59
60 bool sameSize(const SubMatrix& b) const
61 {
62 return (end_row - begin_row == b.end_row - b.begin_row) and
63 (end_column - begin_column == b.end_column - b.begin_column);
64 }
65
66 void operator=(int x)
67 {
68 if (x == 0)
69 {
70 for (long rA = begin_row; rA < end_row; ++rA)
71 for (size_t cA = begin_column; cA < end_column; ++cA)
72 matrix.ring().set_zero(matrix.entry(rA, cA));
73 }
74 }
75
76 void operator=(SubMatrix<MatType> src)
77 {
78 assert(sameSize(src));
79 long rA = begin_row;
80 long rB = src.begin_row;
81 for (; rA < end_row; ++rA, ++rB)
82 {
83 long cA = begin_column;
84 long cB = src.begin_column;
85 for (; cA < end_column; ++cA, ++cB)
86 matrix.ring().set(matrix.entry(rA, cA), src.matrix.entry(rB, cB));
87 }
88 }
89
90 void operator+=(SubMatrix<MatType> src)
91 {
92 assert(sameSize(src));
93 long rA = begin_row;
94 long rB = src.begin_row;
95 for (; rA < end_row; ++rA, ++rB)
96 {
97 long cA = begin_column;
98 long cB = src.begin_column;
99 for (; cA < end_column; ++cA, ++cB)
100 {
101 auto& a = matrix.entry(rA, cA);
102 matrix.ring().add(a, a, src.matrix.entry(rB, cB));
103 }
104 }
105 }

137template <typename MatType>
138void normSquared(SubMatrix<MatType> M,
139 typename MatType::CoeffRing::RealElementType& result)
140{
141 const auto& C = M.matrix.ring();
142 const auto& R = C.real_ring();
143 typename MatType::CoeffRing::RealRingType::Element c(R);
144 R.set_zero(result);
145 for (long rA = M.begin_row; rA < M.end_row; ++rA)
146 for (long cA = M.begin_column; cA < M.end_column; ++cA)
147 {
148 C.abs_squared(c, M.matrix.entry(rA, cA));
149 R.add(result, result, c);
150 }
151}

165template <typename RT>
166bool isZero(const DMat<RT>& A)
167{
168 size_t len = A.numRows() * A.numColumns();
169 if (len == 0) return true;
170 for (auto t = A.array() + len - 1; t >= A.array(); t--)
171 if (!A.ring().is_zero(*t)) return false;
172 return true;
173}

175template <typename RT>
176bool isEqual(const DMat<RT>& A, const DMat<RT>& B)
177{
178 assert(&A.ring() == &B.ring());
179 if (B.numRows() != A.numRows()) return false;
180 if (B.numColumns() != A.numColumns()) return false;
181 size_t top = A.numRows() * A.numColumns();
182 auto elemsA = A.array();
183 auto elemsB = B.array();
184 for (size_t i = 0; i < top; i++)
185 if (!A.ring().is_equal(*elemsA++, *elemsB++)) return false;
186 return true;
187}

239template <typename RT>
240void transpose(const DMat<RT>& A, DMat<RT>& result)
241{
242 assert(&A != &result); // these cannot be aliased!
243 assert(result.numRows() == A.numColumns());
244 assert(result.numColumns() == A.numRows());
245 for (size_t c = 0; c < A.numColumns(); ++c)
246 {
247 auto i = A.columnBegin(c);
248 auto j = result.rowBegin(c);
249 auto end = A.columnEnd(c);
250 for (; i != end; ++i, ++j) A.ring().set(*j, *i);
251 }
252}

309void addMultipleTo(DMat<RT>& A,
310 MatrixWindow wA,
311 const typename RT::ElementType& c,
312 const DMat<RT>& B,
313 MatrixWindow wB)
314{
315 assert(wA.sameSize(wB));

318 long rB = wB.begin_row;
