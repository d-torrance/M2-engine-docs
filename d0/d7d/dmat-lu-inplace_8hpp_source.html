129template <class RingType>
130size_t DMatLUinPlace<RingType>::findPivot(size_t row, size_t col)
131{
132 // Look at elements A[row,col], A[row+1,col], ..., A[nrows-1, col]
133 // Return the index r s.y. abs(A[r,col]) is maximum over all of these
134
135 for (size_t i = row; i < mLU.numRows(); i++)
136 {
137 if (!ring().is_zero(mLU.entry(i, col))) return i;
138 }
139 return static_cast<size_t>(-1);
140}

142template <>
143inline size_t DMatLUinPlace<M2::ARingRRR>::findPivot(size_t row, size_t col)
144{
145 // Look at elements A[row,col], A[row+1,col], ..., A[nrows-1, col]
146 // Return the index r s.y. abs(A[r,col]) is maximum over all of these
147
148 M2::ARingRRR::Element largest(ring()), abs(ring());
149 size_t best_row_so_far = static_cast<size_t>(-1);
150
151 ring().set_zero(largest);
152 for (size_t i = row; i < mLU.numRows(); i++)
153 {
154 ring().abs(abs, mLU.entry(i, col));
155 if (ring().compare_elems(abs, largest) > 0)
156 {
157 best_row_so_far = i;
158 ring().set(largest, abs);
159 }
160 }
161 return best_row_so_far;
162}

188template <class RingType>
189void DMatLUinPlace<RingType>::computeLU()
190{
191 if (mIsDone) return;
192
193 // std::cout << "computing LU decomposition NAIVE version" << std::endl;
194 typename RingType::Element tmp(mLU.ring());
195
196 size_t col = 0; // current column we are working on
197 size_t row = 0; // current row we are working on
198 size_t nrows = mLU.numRows();
199 size_t ncols = mLU.numColumns();
200
201 while (col < ncols && row < nrows)
202 {
203 // printf("*** in naive row,col = (%ld, %ld) ***\n", row, col);
204 // debug_out();
205
206 // Step 1: Set the 'upper' values: (row,col)..(nrows-1,col)
207 for (size_t r = row; r < nrows; r++)
208 {
209 for (size_t i = 0; i < row; i++)
210 {
211 mLU.ring().mult(tmp, mLU.entry(r, i), mLU.entry(i, col));
212 mLU.ring().subtract(mLU.entry(r, col), mLU.entry(r, col), tmp);
213 }
214 }
215
216 // printf("after step 1\n");
217 // debug_out();
218
219 // Step 2: Find a pivot among the elements in step 1.
220 // If one: swap rows if needed
221 // If none, increment 'col', and continue at start of loop
222 size_t k = findPivot(row, col);
223 if (k == static_cast<size_t>(-1))
224 {
225 col = col + 1;
226 continue;
227 }
228 // printf("pivot is in row %ld\n", k);
229 std::swap(mPerm[row], mPerm[k]);
230 if (k != row)
231 {
232 MatElementaryOps<Mat>::interchange_rows(mLU, k, row);
233 mSign = !mSign;
234 }
235 mPivotColumns.push_back(col);
236 const ElementType& pivot = mLU.entry(row, col);
237
238 // printf("after step 2:\n");
239 // debug_out();
240
241 // Step 3A: Set the 'upper' elements in (row,col+1), ..., (row,ncols-1).
242 for (size_t c = col + 1; c < ncols; c++)
243 {
244 for (size_t i = 0; i < row; i++)
245 {
246 mLU.ring().mult(tmp, mLU.entry(row, i), mLU.entry(i, c));
247 mLU.ring().subtract(mLU.entry(row, c), mLU.entry(row, c), tmp);
248 }
249 }
250 // printf("after step 3A:\n");
251 // debug_out();
252
253 // Step 3B: Set the 'lower' elements in (row+1,row), ..., (nrows-1,row)
254 // from (row+1,col), ..., (nrows-1,col)
255 // This just means dividing then by the pivot
256 // except, if we have skipped columns for pivots, we must set these
257 // elements
258 // in column 'row', not 'col'...
259 // Step 3C: if row != col, then set these elements to 0:
260 // (row+1,col), ..., (nrows-1,col)
261 for (size_t r = row + 1; r < nrows; r++)
262 {
263 mLU.ring().divide(mLU.entry(r, row), mLU.entry(r, col), pivot);
264 if (row < col) ring().set_zero(mLU.entry(r, col));
265 }
266
267 // printf("after step 3B:\n");
268 // debug_out();
269
270 row++;
271 col++;
272 }
273
274 mIsDone = true;
275}

277template <>
278inline void DMatLUinPlace<M2::ARingRR>::computeLU()
279{
280 if (mIsDone) return;
281
282 int rows = static_cast<int>(mLU.numRows());
283 int cols = static_cast<int>(mLU.numColumns());
284 int info;
285 int min = (rows <= cols) ? rows : cols;
286
287 if (min == 0)
288 return;
289
290 // printf("entering DMatLUinPlace::computeLUNaive for RR\n");
291
292 int* perm = newarray_atomic(int, min);
293
294 double* copyA = newarray_atomic(double, mLU.numRows() * mLU.numColumns());
295
296 // place all elements of mLU, but in column major order.
297 double* p = copyA;
298 for (size_t c = 0; c < mLU.numColumns(); c++)
299 {
300 auto end = mLU.columnEnd(c);
301 for (auto a = mLU.columnBegin(c); a != end; ++a) *p++ = *a;
302 }
303
304 dgetrf_(&rows, &cols, copyA, &rows, perm, &info);
305
306 if (info < 0)
307 {
308 // First, clean up, then throw an exception
309 throw exc::engine_error("argument passed to dgetrf had an illegal value");
310 // return;
311 }
312
313 // Now copy back to row major order
314 p = copyA;
315 for (size_t c = 0; c < mLU.numColumns(); c++)
316 {
317 auto end = mLU.columnEnd(c);
318 for (auto a = mLU.columnBegin(c); a != end; ++a) *a = *p++;
319 }
320
321 // Now place the correct permutation into mPerm
322 for (int i = 0; i < min; i++)
323 {
324 int thisloc = perm[i] - 1;
325 if (i != thisloc)
326 {
327 mSign = not mSign;
328 size_t tmp = mPerm[thisloc];
329 mPerm[thisloc] = mPerm[i];
330 mPerm[i] = tmp;
331 }
332 }
333
334 LUUtil<RingType>::computePivotColumns(mLU, mPivotColumns);
335 mIsDone = true;
336
337 freemem(perm);
338 freemem(copyA);
339}

413template <class RingType>
414void LUUtil<RingType>::setUpperLower(const Mat& LU, Mat& lower, Mat& upper)
415{
416 size_t min = std::min(LU.numRows(), LU.numColumns());
417 lower.resize(LU.numRows(), min);
418 upper.resize(min, LU.numColumns());
419
420 // At this point, lower and upper should be zero matrices.
421 assert(MatrixOps::isZero(lower));
422 assert(MatrixOps::isZero(upper));
423
424 for (size_t c = 0; c < LU.numColumns(); c++)
425 {
426 if (c < min) LU.ring().set_from_long(lower.entry(c, c), 1);
427 for (size_t r = 0; r < LU.numRows(); r++)
428 {
429 if (r <= c)
430 LU.ring().set(upper.entry(r, c), LU.entry(r, c));
431 else if (c < lower.numRows())
432 {
433 LU.ring().set(lower.entry(r, c), LU.entry(r, c));
434 }
435 }
436 }
437}

439template <class RingType>
440void LUUtil<RingType>::computePivotColumns(const Mat& LU,
441 std::vector<size_t>& result_columns)
442{
443 result_columns.clear();
444 size_t thiscol = 0;
445 size_t thisrow = 0;
446 while (thisrow < LU.numRows() and thiscol < LU.numColumns())
447 {
448 if (not LU.ring().is_zero(LU.entry(thisrow, thiscol)))
449 {
450 result_columns.push_back(thiscol);
451 thisrow++;
452 }
453 thiscol++;
454 }
455}
