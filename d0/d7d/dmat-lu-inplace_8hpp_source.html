125 size_t DMatLUinPlace<RingType>::findPivot(size_t row, size_t col)
126 {
127  // Look at elements A[row,col], A[row+1,col], ..., A[nrows-1, col]
128  // Return the index r s.y. abs(A[r,col]) is maximum over all of these
129 
130  for (size_t i = row; i < mLU.numRows(); i++)
131  {
132  if (!ring().is_zero(mLU.entry(i, col))) return i;
133  }
134  return static_cast<size_t>(-1);

183 template <class RingType>
184 void DMatLUinPlace<RingType>::computeLU()
185 {
186  if (mIsDone) return;
187 
188  // std::cout << "computing LU decomposition NAIVE version" << std::endl;
189  typename RingType::Element tmp(mLU.ring());
190 
191  size_t col = 0; // current column we are working on
192  size_t row = 0; // current row we are working on
193  size_t nrows = mLU.numRows();
194  size_t ncols = mLU.numColumns();
195 
196  while (col < ncols && row < nrows)
197  {
198  // printf("*** in naive row,col = (%ld, %ld) ***\n", row, col);
199  // debug_out();
200 
201  // Step 1: Set the 'upper' values: (row,col)..(nrows-1,col)
202  for (size_t r = row; r < nrows; r++)
203  {
204  for (size_t i = 0; i < row; i++)
205  {
206  mLU.ring().mult(tmp, mLU.entry(r, i), mLU.entry(i, col));
207  mLU.ring().subtract(mLU.entry(r, col), mLU.entry(r, col), tmp);
208  }
209  }
210 
211  // printf("after step 1\n");
212  // debug_out();
213 
214  // Step 2: Find a pivot among the elements in step 1.
215  // If one: swap rows if needed
216  // If none, increment 'col', and continue at start of loop
217  size_t k = findPivot(row, col);
218  if (k == static_cast<size_t>(-1))
219  {
220  col = col + 1;
221  continue;
222  }
223  // printf("pivot is in row %ld\n", k);
224  std::swap(mPerm[row], mPerm[k]);
225  if (k != row)
226  {
227  MatElementaryOps<Mat>::interchange_rows(mLU, k, row);
228  mSign = !mSign;
229  }
230  mPivotColumns.push_back(col);
231  const ElementType& pivot = mLU.entry(row, col);
232 
233  // printf("after step 2:\n");
234  // debug_out();
235 
236  // Step 3A: Set the 'upper' elements in (row,col+1), ..., (row,ncols-1).
237  for (size_t c = col + 1; c < ncols; c++)
238  {
239  for (size_t i = 0; i < row; i++)
240  {
241  mLU.ring().mult(tmp, mLU.entry(row, i), mLU.entry(i, c));
242  mLU.ring().subtract(mLU.entry(row, c), mLU.entry(row, c), tmp);
243  }
244  }
245  // printf("after step 3A:\n");
246  // debug_out();
247 
248  // Step 3B: Set the 'lower' elements in (row+1,row), ..., (nrows-1,row)
249  // from (row+1,col), ..., (nrows-1,col)
250  // This just means dividing then by the pivot
251  // except, if we have skipped columns for pivots, we must set these
252  // elements
253  // in column 'row', not 'col'...
254  // Step 3C: if row != col, then set these elements to 0:
255  // (row+1,col), ..., (nrows-1,col)
256  for (size_t r = row + 1; r < nrows; r++)
257  {
258  mLU.ring().divide(mLU.entry(r, row), mLU.entry(r, col), pivot);
259  if (row < col) ring().set_zero(mLU.entry(r, col));
260  }
261 
262  // printf("after step 3B:\n");
263  // debug_out();
264 
265  row++;
266  col++;
267  }
268 
269  mIsDone = true;
270 }

273 inline void DMatLUinPlace<M2::ARingRR>::computeLU()
274 {
275  if (mIsDone) return;
276 
277  int rows = static_cast<int>(mLU.numRows());
278  int cols = static_cast<int>(mLU.numColumns());
279  int info;
280  int min = (rows <= cols) ? rows : cols;
281 
282  // printf("entering DMatLUinPlace::computeLUNaive for RR\n");
283 
284  int* perm = newarray_atomic(int, min);
285 
286  double* copyA = newarray_atomic(double, mLU.numRows() * mLU.numColumns());
287 
288  // place all elements of mLU, but in column major order.
289  double* p = copyA;
290  for (size_t c = 0; c < mLU.numColumns(); c++)
291  {
292  auto end = mLU.columnEnd(c);
293  for (auto a = mLU.columnBegin(c); a != end; ++a) *p++ = *a;
294  }
295 
296  dgetrf_(&rows, &cols, copyA, &rows, perm, &info);
297 
298  if (info < 0)
299  {
300  // First, clean up, then throw an exception
301  throw exc::engine_error("argument passed to dgetrf had an illegal value");
302  // return;
303  }
304 
305  // Now copy back to row major order
306  p = copyA;
307  for (size_t c = 0; c < mLU.numColumns(); c++)
308  {
309  auto end = mLU.columnEnd(c);
310  for (auto a = mLU.columnBegin(c); a != end; ++a) *a = *p++;
311  }
312 
313  // Now place the correct permutation into mPerm
314  for (int i = 0; i < min; i++)
315  {
316  int thisloc = perm[i] - 1;
317  if (i != thisloc)
318  {
319  mSign = not mSign;
320  size_t tmp = mPerm[thisloc];
321  mPerm[thisloc] = mPerm[i];
322  mPerm[i] = tmp;
323  }
324  }
325 
326  LUUtil<RingType>::computePivotColumns(mLU, mPivotColumns);
327  mIsDone = true;
328 
329  freemem(perm);
330  freemem(copyA);

404 {
405  size_t min = std::min(LU.numRows(), LU.numColumns());
406  lower.resize(LU.numRows(), min);
407  upper.resize(min, LU.numColumns());
408 
409  // At this point, lower and upper should be zero matrices.
410  assert(MatrixOps::isZero(lower));
411  assert(MatrixOps::isZero(upper));
412 
413  for (size_t c = 0; c < LU.numColumns(); c++)
414  {
415  if (c < min) LU.ring().set_from_long(lower.entry(c, c), 1);
416  for (size_t r = 0; r < LU.numRows(); r++)
417  {
418  if (r <= c)
419  LU.ring().set(upper.entry(r, c), LU.entry(r, c));
420  else if (c < lower.numRows())
421  {
422  LU.ring().set(lower.entry(r, c), LU.entry(r, c));
423  }
424  }
