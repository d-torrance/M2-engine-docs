88gmp_ZZorNull IM2_RingElement_to_Integer(const RingElement *a)
89/* If the ring of a is ZZ, or ZZ/p, this returns the underlying representation.
90 Otherwise, NULL is returned, and an error is given */
91{
92 const Ring *R = a->get_ring();
93 if (R->is_ZZ())
94 {
95 // Note: RingElement ZZ elements are on the "gc" side of the memory barrier, and
96 // are read only. Therefore we can just return the same value.
97 return a->get_value().get_mpz();
98 }
99 if (R->isFinitePrimeField())
100 {
101 mpz_ptr result = newitem(__mpz_struct);
102
103 std::pair<bool, long> res = R->coerceToLongInteger(a->get_value());
104 assert(res.first);
105
106 mpz_init_set_si(result, static_cast<int>(res.second));
107 mpz_reallocate_limbs(result);
108 return result;
109 }
110 ERROR("Expected ZZ or ZZ/p as base ring");
111 return nullptr;
112}

124gmp_RRorNull IM2_RingElement_to_BigReal(const RingElement *a)
125{
126 const Ring *R = a->get_ring();
127 gmp_RRmutable result;
128 const M2::ConcreteRing<M2::ARingRRR> *R1;
129
130 switch (R->ringID())
131 {
132 case M2::ring_RR:
133 result = getmemstructtype(gmp_RRmutable);
134 mpfr_init2(result, 53);
135 mpfr_set_d(result, a->get_value().get_double(), MPFR_RNDN);
136 return moveTo_gmpRR(result);
137 case M2::ring_RRR:
138 R1 =
139 dynamic_cast<const M2::ConcreteRing<M2::ARingRRR> *>(a->get_ring());
140 result = getmemstructtype(gmp_RRmutable);
141 mpfr_init2(result, R1->get_precision());
142 mpfr_set(result, a->get_value().get_mpfr(), MPFR_RNDN);
143 return moveTo_gmpRR(result);
144 default:
145 ERROR("expected an element of RRR");
146 return nullptr;
147 }

443const RingElement /* or null */ *IM2_RingElement_term(const Ring *R,
444 const RingElement *a,
445 const EngineMonomial *m)
446/* R must be a polynomial ring, and 'a' an element of the
447 coefficient ring of R. Returns a*m, if this is a valid
448 element of R. Returns NULL if not (with an error message). */
449{
450 try {
451 const PolynomialRing *P = R->cast_to_PolynomialRing();
452 if (P != nullptr)
453 {
454 int nvars0 = P->n_vars();
455 const PolynomialRing *K = a->get_ring()->cast_to_PolynomialRing();
456 if (K != nullptr && K != P->getCoefficients()) nvars0 -= K->n_vars();
457 exponents_t exp = newarray_atomic(int, nvars0);
458 varpower::to_expvector(nvars0, m->ints(), exp);
459 ring_elem val = P->make_logical_term(a->get_ring(), a->get_value(), exp);
460 return RingElement::make_raw(R,val);
461 }
462 auto Q = dynamic_cast<const M2FreeAlgebraOrQuotient *>(R);
463 if (Q != nullptr)
464 {
465 if (Q->coefficientRing() != a->get_ring())
466 {
467 ERROR("wrong coefficient ring");
468 return nullptr;
469 }
470 return RingElement::make_raw(Q,
471 Q->makeTerm(a->get_value(),
472 m->ints()));
473 }
474 ERROR("requires a polynomial ring");
475 return nullptr;
476 }
477 catch (exc::engine_error& e) {
478 ERROR(e.what());
479 return nullptr;
480 }
481}

587 const RingElement *f)
588/* Return an array of RingElement's, each having pure weight, and sorted by
589 strictly increasing weight value. The wt vector values must fit into
590 a word length integer. */
591{
592 try
593 {
594 const PolynomialRing *P = f->get_ring()->cast_to_PolynomialRing();

605 result->len = static_cast<int>(relems_len);
606 for (int i = 0; i < result->len; i++)
607 result->array[i] = RingElement::make_raw(P, relems[i]);
608 freemem(relems);
609 return result;
610 } catch (const exc::engine_error& e)
611 {
612 ERROR(e.what());
613 return nullptr;

640void convolve(const PolyRing *R,
641 const VECTOR(ring_elem) & input_relems,
642 VECTOR(ring_elem) & output_relems,
643 int convolve_type)
644{
645 const Ring *K = R->getCoefficientRing();
646 ring_elem invn;
647 size_t n =
648 input_relems.size() - 1; // array is 0..n, with the 0 part being ignored.
649 // first set [1, e1] := [1, h1].
650 output_relems[0] = input_relems[0];
651 for (int i = 1; i <= n; i++)
652 {
653 // ASSUMPTION: input_relems[i] is either a variable or - of a variable
654 ring_elem result = R->copy(input_relems[i]);
655 if (convolve_type == 2) R->mult_coeff_to(K->from_long(-i), result);
656 for (int j = i - 1; j >= 1; --j)
657 {
658 ring_elem hr;
659 Nterm *g = input_relems[j];
660 if (g != nullptr)
661 {
662 hr.poly_val =
663 R->mult_by_term(output_relems[i - j], g->coeff, g->monom);
664 R->internal_add_to(result, hr);
665 }
666 }
667 if (convolve_type == 1)
668 {
669 invn = K->invert(K->from_long(i));
670 R->mult_coeff_to(invn, result);
671 }
672 output_relems[i] = result;
673 }
674}

676engine_RawRingElementArrayOrNull rawConvolve(engine_RawRingElementArray H,
677 int convolve_type)
678{
679 // convolve_type == 1: E_n = E_(n-1) * H_1 - E(n-2) * H_2 + ... + (-1)^n * H_n
680 // convolve_type == 2: n * E_n = E_(n-1) * H_1 - E(n-2) * H_2 + ... + (-1)^n *
681 // H_n
682 // others of interest? These are good for translating between symmetric
683 // functions of type E,H,P
684
685 // H_0 is ignored... (OR: should we instead shift everything: first entry of H
686 // is denoted H_1...?)
687
688 try
689 {
690 int len = H->len;
691 if (len <= 1)
692 {
693 ERROR("expected ring element array of length at least 2");
694 return nullptr;
695 }
696 const PolyRing *P = H->array[1]->get_ring()->cast_to_PolyRing();
697 if (P == nullptr)
698 {
699 ERROR("expected a polynomial ring");
700 return nullptr;
701 }
702 VECTOR(ring_elem) input_relems(len);
703 VECTOR(ring_elem) output_relems(len);
704 for (int i = 0; i < len; i++) input_relems[i] = H->array[i]->get_value();
705 convolve(P, input_relems, output_relems, convolve_type);
706
707 engine_RawRingElementArray result =
708 getmemarraytype(engine_RawRingElementArray, len);
709 result->len = len;
710 for (int i = 0; i < result->len; i++)
711 result->array[i] = RingElement::make_raw(P, output_relems[i]);
712
713 return result;
714 } catch (const exc::engine_error& e)
715 {
716 ERROR(e.what());
717 return nullptr;
718 }
719}

721const RingElement /* or null */ *rawGetPart(const M2_arrayint wts,
722 const RingElement *f,
723 M2_bool lobound_given,
724 M2_bool hibound_given,
725 long lobound,
726 long hibound)
727/* Return the sum of all of the terms t of f, which satisfy: lobound <= wt.t <=
728 hibound,
729 where, if lobound_given is false, then lobound is -infinity, and if
730 hibound_given
731 is false, then hibound is infinity. */
732{
733 try
734 {
735 const PolynomialRing *P = f->get_ring()->cast_to_PolynomialRing();
736 if (P == nullptr)
737 {
738 ERROR("expected a polynomial");
739 return nullptr;
740 }
741 ring_elem g = P->get_part(M2_arrayint_to_stdvector<int>(wts),
742 f->get_value(),
743 lobound_given,
744 hibound_given,
745 lobound,
