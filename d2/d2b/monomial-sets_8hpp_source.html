6 // This file contains classes for keeping sets of monomials.
7 // There are two types of monomials:
8 // a. fixed size (and that size is passed in as a parameter)
9 // b. variable size, in which case the first int is the total length (so the range is [m, m + *m).
10 //
11 // Here are some assumptions about these monomials:
12 // 1. each monomial is a contiguous sequence of int's.
13 // 2. each monomial is hashed using all entries of the monomial. Equality is equality of all elements.
14 // (and size, if variable length).

27 class MonomialMemorySpace
28 {
29 public:
30  MonomialMemorySpace() : mCount(0) {}
31 
32  // returns a range [begin, end), allocated in the block, such that end-begin == size.
33  std::pair<int*,int*> alloc(int size) { mCount++; return mArena.allocArrayNoCon<int>(size); }
34 
35  int* allocate(int size) { auto result = alloc(size); return result.first; }
36 
37  // if m is the begin part of the last range allocated here, then this pops that.
38  // Afterwards, one can pop the next, etc. If they are popped out of order though,
39  // undefined behavior results.
40  void popLastAlloc(int* m) { mCount--; mArena.freeTop(m); }
41 
42  // If m is a pointer into the last monomial (i.e. >= a.first <= a.second), this resets the monomial
43  // to be [a.first, m). The public interface of memt::Arena should perhaps allow this use, but it does not.
44  // However, the code works fine, unless debugging is turned on. I might need to modify that code, and
45  // document that this can be done.
46  // Reasoning for wanting/using this: in the case when monomials are not fixed size, one can allocate
47  // the maximum size it can be, compute it in place, then shrink the monomial.
48  // UNSAFE. Use of this function requires taking care.
49  void shrinkLastAlloc(int* m) { mArena.freeTop(m); }
50 
51  size_t size() const { return mCount; }
52 
53  void freeAllAllocs() { mArena.freeAllAllocs(); }
54 
55  void freeAllAllocsAndBackingMemory() { mArena.freeAllAllocsAndBackingMemory(); }
56 private:
57  size_t mCount;
58  memt::Arena mArena;
59 };

205  // insert: two versions: depending on whether we need to copy the pointer.
206  // decide if monom is in the set: if so, (equiv monom, false) is returned.
207  // If it is not in the set, it is copied into the memory block, and (equiv monom, true) is returned.
208  // Note: the only pointers this function returns (as the first part of the pair) are pointers into
209  // the memory block.
210  std::pair<const int*, bool> findOrInsert(const int* monom)
211  {
212  std::pair<int*, int*> mon { mMonomialMemory.alloc(elementSize()) };
213  std::copy(monom, monom + elementSize(), mon.first);
214  assert(mon.second - mon.first >= elementSize());
215  return findOrInsertTopInternedMonomial(mon.first);
216  }
217 
218  // assumption: 'monom' is currently the top (last) monomial on mMonomialMemory.
219  // If the monomial is found in the set, then monom is popped off the memory block.
220  std::pair<const int*, bool> findOrInsertTopInternedMonomial(int*& monom)
221  {
222  // if the monomial is found in the set, then monom is popped
223  auto result = mSet.findOrInsert(monom);
224  if (not result.second) // i.e. monomial was not needed.
