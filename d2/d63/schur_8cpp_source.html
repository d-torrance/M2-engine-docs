42void tableau::fill(int *lamb, int *pp)
43// Fill the skew tableau p\lambda with 1..nboxes
44// starting at top right, moving left and then down
45// row by row.
46{
47 int i, j;
48 p = pp;
49 lambda = lamb;
50
51 int next = 1;
52 for (i = 1; p[i] != 0; i++)
53 {
54 int a = lambda[i];
55 for (j = p[i]; j > a; j--)
56 {
57 xloc[next] = i;
58 yloc[next++] = j;
59 }
60 }
61}

117void SchurRing::to_partition(const int *m, int *exp) const
118// exp[1]..exp[nvars] are set
119{
120 exponents_t EXP1 = ALLOCATE_EXPONENTS(exp_size); // 0..nvars-1
121 // remark: exp_size is defined in an ancestor class of SchurRing
122 M_->to_expvector(m, EXP1);
123 exp[nvars_] = EXP1[nvars_ - 1];
124 for (int i = nvars_ - 1; i >= 1; i--) exp[i] = exp[i + 1] + EXP1[i - 1];
125}
126void SchurRing::from_partition(const int *exp, int *m) const
127{
128 exponents_t EXP1 = ALLOCATE_EXPONENTS(exp_size); // 0..nvars-1
129 EXP1[nvars_ - 1] = exp[nvars_];
130 for (int i = nvars_ - 1; i > 0; i--) EXP1[i - 1] = exp[i] - exp[i + 1];
131 M_->from_expvector(EXP1, m);
132}

134void SchurRing::bounds(int &lo, int &hi)
135{
136 int i, k;
137 int x = _SMfilled.xloc[_SMcurrent];
138 int y = _SMfilled.yloc[_SMcurrent];
139
140 // First set the high bound, using info from the "one to the right"
141 // in the reverse lex filled skew tableau.
142
143 if (y == _SMfilled.p[x]) // There is not one to the right
144 {
145 hi = nvars_;
146 for (k = 1; k <= nvars_; k++)
147 if (_SMtab.p[k] == 0)
148 {
149 hi = k;
150 break;
151 }
152 }
153 else // note that the case _SMcurrent==1 will be handled
154 { // in the previous statement.
155 hi = _SMtab.xloc[_SMcurrent - 1];
156 }
157
158 // Now we set the lo bound, using info from the "one above"
159
160 if (x == 1 || y <= _SMfilled.lambda[x - 1])
161 lo = 1; // There is not one above
162 else
163 {
164 int above = _SMcurrent - _SMfilled.p[x] + _SMfilled.lambda[x - 1];
165 int xabove = _SMtab.xloc[above];
166 int yabove = _SMtab.yloc[above];
167 for (i = xabove + 1; i <= hi; i++)
168 if (_SMtab.p[i] < yabove) break;
169 lo = i;
170 }
171}

173void SchurRing::SM()
174{
175 int lo, hi;
176
177 if (_SMcurrent == _SMfinalwt)
178 {
179 // partition is to be output
180 Nterm *f = new_term();
181 f->coeff = K_->from_long(1);
182 // fprintf(stderr, "partition: ");
183 // for (int i=1; i <= nvars_; i++)
184 // fprintf(stderr, " %d", _SMtab.p[i]);
185 // fprintf(stderr, "\n");
186 from_partition(_SMtab.p, f->monom);
187 f->next = _SMresult;
188 _SMresult = f;
189 return;
190 }
191
192 _SMcurrent++;
193 bounds(lo, hi);
194 int this_one = LARGE_NUMBER; // larger than any entry of _SMtab
195 int last_one;
196 for (int i = lo; i <= hi; i++)
197 {
198 last_one = this_one;
199 this_one = _SMtab.p[i];
200 if (last_one > this_one)
201 {
202 _SMtab.p[i]++;
203 _SMtab.xloc[_SMcurrent] = i;
204 _SMtab.yloc[_SMcurrent] = _SMtab.p[i];
205 SM();
206 _SMtab.p[i]--;
207 }
208 }
209 _SMcurrent--;
210}

212Nterm *SchurRing::skew_schur(int *lambda, int *p)
213{
214 _SMcurrent = 0;
215
216 _SMfinalwt = 0;
217 for (int i = 1; p[i] != 0; i++) _SMfinalwt += (p[i] - lambda[i]);
218
219 _SMtab.wt = _SMfinalwt;
220 _SMtab.resize(_SMfinalwt);
221 _SMfilled.resize(_SMfinalwt);
222 _SMfilled.fill(lambda, p);
223 _SMresult = nullptr;
224 SM();
225 ring_elem result = _SMresult;
226 _SMresult = nullptr;
227 return result;
228}

230ring_elem SchurRing::mult_monomials(const int *m, const int *n)
231{
232 int i;
233
234 exponents_t a_part = ALLOCATE_EXPONENTS(sizeof(int) * (nvars_ + 1));
235 exponents_t b_part = ALLOCATE_EXPONENTS(sizeof(int) * (nvars_ + 1));
236 exponents_t lambda = ALLOCATE_EXPONENTS(2 * sizeof(int) * (nvars_ + 1));
237 exponents_t p = ALLOCATE_EXPONENTS(2 * sizeof(int) * (nvars_ + 1));
238
239 // First: obtain the partitions
240 to_partition(m, a_part);
241 to_partition(n, b_part);
242
243 // Second: make the skew partition
244 int a = b_part[1];
245 for (i = 1; i <= nvars_ && a_part[i] != 0; i++)
246 {
247 p[i] = a + a_part[i];
248 lambda[i] = a;
249 }
250 int top = i - 1;
251 for (i = 1; i <= nvars_ && b_part[i] != 0; i++)
252 {
253 p[top + i] = b_part[i];
254 lambda[top + i] = 0;
255 }
256 p[top + i] = 0;
257 lambda[top + i] = 0;
258
259 // Call the SM() algorithm
260 return skew_schur(lambda, p);
261}

263ring_elem SchurRing::mult_by_term(const ring_elem f,
264 const ring_elem c,
265 const int *m) const
266{
267 // return c*m*f
268 ring_elem result = ZERO_RINGELEM;
269 for (Nterm& t : f)
270 {
271 ring_elem a = K_->mult(c, t.coeff);
272 ring_elem g = const_cast<SchurRing *>(this)->mult_monomials(t.monom, m);
273 for (Nterm& s : g)
274 {
275 ring_elem b = K_->mult(a, s.coeff);
276 s.coeff = b;
277 }
278 Nterm *gt = g;
279 sort(gt);
280 g = gt;
281 add_to(result, g);
282 }
283 return result;

311void SchurRing::dimension(const int *exp, mpz_t result) const
312// exp: 0..nvars_
313// Return in 'result' the dimension of the irreducible
314// GL(nvars_) representation having highest weight
315// 'exp'
316{
317 int i, j;
318
319 mpz_set_ui(result, 1);
320 for (i = 1; i < nvars_; i++)
321 for (j = i + 1; j <= nvars_; j++)
322 if (exp[i] != exp[j]) mpz_mul_ui(result, result, exp[i] - exp[j] + j - i);
323
324 for (i = 1; i < nvars_; i++)
325 for (j = i + 1; j <= nvars_; j++)
326 if (exp[i] != exp[j]) mpz_fdiv_q_ui(result, result, j - i);
327}

329ring_elem SchurRing::dimension(const ring_elem f) const
330{
331 exponents_t EXP = ALLOCATE_EXPONENTS(sizeof(int) * (nvars_ + 1));
332 ring_elem result = K_->from_long(0);
333 mpz_t dim;
334 mpz_init(dim);
335 for (Nterm& t : f)
336 {
337 to_partition(t.monom, EXP);
338 dimension(EXP, dim);
339 ring_elem h = K_->from_int(dim);
340 ring_elem h2 = K_->mult(t.coeff, h);
341 K_->add_to(result, h2);
342 K_->remove(h);
343 }
344 mpz_clear(dim);
345 return result;
346}

348void SchurRing::elem_text_out(buffer &o,
349 const ring_elem f,
350 bool p_one,
351 bool p_plus,
352 bool p_parens) const
353{
354 exponents_t EXP = ALLOCATE_EXPONENTS(sizeof(int) * (nvars_ + 1));
355 int n = n_terms(f);
356
357 bool needs_parens = p_parens && (n > 1);
358 if (needs_parens)
359 {
360 if (p_plus) o << '+';
361 o << '(';
362 p_plus = false;
363 }
364
365 p_one = false;
366 for (Nterm& t : f)
367 {
368 int isone = M_->is_one(t.monom);
369 p_parens = !isone;
370 K_->elem_text_out(o, t.coeff, p_one, p_plus, p_parens);
371 to_partition(t.monom, EXP);
372 o << "{" << EXP[1];
373 for (int i = 2; i <= nvars_ && EXP[i] != 0; i++) o << "," << EXP[i];
374 o << "}";
375 p_plus = true;
376 }
377 if (needs_parens) o << ')';
378}
