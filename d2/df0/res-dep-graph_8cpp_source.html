137void makeDependencyGraph(DependencyGraph &G, int nlevels, int nslanted_degrees, bool doMinimalBetti)
138{
139 // Create the nodes
140 for (int sldeg=0; sldeg < nslanted_degrees; ++sldeg)
141 for (int lev=0; lev < nlevels; ++lev)
142 {
143 int newIndex = G.addVertex(lev,sldeg);
144 if (newIndex != getIndex(lev,sldeg,nlevels,nslanted_degrees))
145 std::cout << "ERROR: Index of created vertex does not match expected index." << std::endl;
146 if (G.getVertex(newIndex).mLevel != lev or G.getVertex(newIndex).mSlantedDegree != sldeg)
147 std::cout << "ERROR: Level or Slanted Degree of Node not the expected value." << std::endl;
148 }
149
150 // Add the fill matrix edges
151 for (int sldeg=0; sldeg < nslanted_degrees; ++sldeg)
152 for (int lev=0; lev < nlevels; ++lev)
153 {
154 if (lev > 0)
155 G.addFillMatrixEdge(getIndex(lev-1,sldeg,nlevels,nslanted_degrees),
156 getIndex(lev,sldeg,nlevels,nslanted_degrees));
157 if (sldeg > 0)
158 G.addFillMatrixEdge(getIndex(lev,sldeg-1,nlevels,nslanted_degrees),
159 getIndex(lev,sldeg,nlevels,nslanted_degrees));
160 if ((lev < nlevels-1) && doMinimalBetti)
161 {
162 G.addMinimalBettiEdge(lev,sldeg,nlevels,nslanted_degrees);
163 }
164 }
165}
