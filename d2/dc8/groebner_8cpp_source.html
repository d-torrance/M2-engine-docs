262Computation /* or null */ *rawStartComputation(Computation *C)
263/* start or continue the computation */
264{
265 try
266 {
267 clear_emit_size();
268 C->start_computation();
269
270 if (M2_gbTrace == 15)
271 {
272 ComputationStatusCode ret = C->status();
273 switch (ret)
274 {
275 case COMP_DONE_DEGREE_LIMIT:
276 emit_line("computation stopped at degree limit");
277 break;
278 case COMP_DONE:
279 emit_line("computation of GB completed");
280 break;
281 case COMP_DONE_PAIR_LIMIT:
282 emit_line("computation stopped at pair limit");
283 break;
284 case COMP_NEED_RESIZE:
285 case COMP_ERROR:
286 case COMP_INTERRUPTED:
287 case COMP_NOT_STARTED:
288 case COMP_INITIAL_STOP:
289 case COMP_DONE_LENGTH_LIMIT:
290 case COMP_DONE_SYZYGY_LIMIT:
291 case COMP_DONE_GB_LIMIT:
292 case COMP_DONE_SYZ_LIMIT:
293 case COMP_DONE_CODIM:
294 case COMP_DONE_MIN_GENS:
295 case COMP_DONE_STEPS:
296 case COMP_DONE_SUBRING_LIMIT:
297 case COMP_COMPUTING:
298 case COMP_OVERFLOWED:
299 emit_line("computation stopped for some good reason");
300 break;
301 default:
302 emit_line("incorrect status code encountered");
303 break;
304 }
305 }
306
307 return error() ? nullptr : C;
308 } catch (const exc::engine_error& e)
309 {
310 ERROR(e.what());
311 return nullptr;
312 }

712 computedStr, charac, nvars, inputMatrix->n_rows());
713 mgb::IdealStreamChecker<decltype(computed)> checked(computed);
714
715 matrixToStream(inputMatrix, checked);
716
717 std::cout << "result: " << std::endl;
718 std::cout << computedStr.str() << std::endl;

745// TODO: The following (in x-monoid.cpp) needs to be put into a header file.
746extern bool monomialOrderingToMatrix(
747 const struct MonomialOrdering &mo,
748 std::vector<int> &mat,
749 bool &base_is_revlex,
750 int &component_direction, // -1 is Down, +1 is Up, 0 is not present
751 int &component_is_before_row // -1 means: at the end. 0 means before the
752 // order.
753 // and r means considered before row 'r' of the matrix.
754 );
755
756const Matrix *rawMGB(
757 const Matrix *inputMatrix,
758 int reducer,
759 int spairGroupSize, // a value of 0 means let the algorithm choose
760 int nthreads,
761 M2_string logging)
762{
763 try
764 {
765 const Ring *R = inputMatrix->get_ring();
766 const PolyRing *P = R->cast_to_PolyRing();
767 if (P == nullptr)
768 {
769 ERROR("expected a polynomial ring");
770 return nullptr;
771 }
772 if (nthreads < 0)
773 {
774 ERROR("mgb: expected a non-negative number of threads");
775 return nullptr;
776 }
777 if (P->characteristic() > std::numeric_limits<int>::max())
778 {
779 ERROR("characteristic is too large for mathic gb computation");
780 return nullptr;
781 }
782 if (P->characteristic() == 0)
783 {
784 ERROR(
785 "characteristic for mathic gb computation must be a prime "
786 "number");
787 return nullptr;
788 }
789 if (not P->getCoefficientRing()->isFinitePrimeField())
790 {
791 ERROR("coefficients for mathic gb computation must be a prime field");
792 }
793 int charac = static_cast<int>(P->characteristic());
794 int nvars = P->n_vars();
795 MGBCallback callback;
796 mgb::GroebnerConfiguration configuration(
797 charac, nvars, inputMatrix->n_rows());
798
799 const auto reducerType = reducer == 0
800 ? mgb::GroebnerConfiguration::ClassicReducer
801 : mgb::GroebnerConfiguration::MatrixReducer;
802 configuration.setReducer(reducerType);
803 configuration.setMaxSPairGroupSize(spairGroupSize);
804 configuration.setMaxThreadCount(nthreads);
805 std::string log((char *)logging->array, logging->len);
806 configuration.setLogging(log.c_str());
807 configuration.setCallback(&callback);
808
809 // Now set the monomial ordering info
810 std::vector<int> mat;
811 bool base_is_revlex = true;
812 int component_direction = 1;
813 int component_is_before_row = -1;
814 if (!monomialOrderingToMatrix(*P->getMonoid()->getMonomialOrdering(),
815 mat,
816 base_is_revlex,
817 component_direction,
818 component_is_before_row))
819 {
820 ERROR(
821 "monomial ordering is not appropriate for Groebner basis "

884{

888 {

901const Matrix* polyListToMatrix(const M2FreeAlgebraOrQuotient* A,
902 const PolyL& elems,
903 int numrows,
904 int numcols)
905{
906 if (elems.size() != numrows*numcols)
907 ERROR("Number of elements in list does not match matrix size.");
908 MatrixConstructor mat(A->make_FreeModule(numrows), numcols);
909 for (auto i = 0; i < elems.size(); ++i)
910 {
911 int curCol = i % numcols;
912 int curRow = (i - curCol) / numcols;
913 ring_elem a = const_cast<Nterm*>(reinterpret_cast<const Nterm*>(elems[i]));
914 mat.set_entry(curRow, curCol, a);
915 }
916 mat.compute_column_degrees();

951const Matrix* rawNCReductionTwoSided(const Matrix* toBeReduced, const Matrix* reducerMatrix)
952{
953 const Ring* R = toBeReduced->get_ring();
954 if (R != reducerMatrix->get_ring())
955 {
956 ERROR("expected matrices to be over the same ring");
957 return nullptr;
958 }
959 const M2FreeAlgebra* A = R->cast_to_M2FreeAlgebra();
960 if (A != nullptr and reducerMatrix->n_rows() == 1)
961 {
962 auto outRows = toBeReduced->n_rows();
963 auto outCols = toBeReduced->n_cols();
964 auto reducees = matrixToPolyList(A, toBeReduced);
965 auto reducers = matrixToPolyList(A, reducerMatrix);

967 G.initReductionOnly();
968 auto result = G.twoSidedReduction(reducees);
969 return polyListToMatrix(A, result, outRows, outCols); // consumes the Poly's in result.
970 }
971 ERROR("expected a matrix over a noncommutative algebra");
972 return nullptr;
