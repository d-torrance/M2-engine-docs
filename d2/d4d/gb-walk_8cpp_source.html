142 void GBWalker::start_computation()
143 {
144  if (stop_.always_stop) return; // don't change status
145 
146  for (;;) switch (state)
147  {
148  case STATE_compute_w:
149  if (!compute_next_w())
150  {
151  // We are done!
152  state = STATE_done;
153  set_status(COMP_DONE);
154  return;
155  }
156  inwwG = G->get_parallel_lead_terms(ww);
157  gb_inwwG = make_gb(inwwG);
158  state = STATE_do_gb;
159  case STATE_do_gb:
160  // Now compute the GB object. If not interrupted, go on:
161  gb_inwwG->start_computation();
162  if (gb_inwwG->status() == COMP_INTERRUPTED)
163  {
164  set_status(COMP_INTERRUPTED);
165  return;
166  }
167  next_to_reduce = 0;
168  state = STATE_reduce;
169  case STATE_reduce:
170  while (next_to_reduce < 0) // TODO: consider the top of the loop
171  {
172  H = G->matrix_remainder(
173  gb_inwwG->get_gb()); // Not quite: need to subtract...
174  next_to_reduce++;
175  }
176  state = STATE_autoreduce;
177  case STATE_autoreduce:
178  G->remove_gb();
179  delete G;
180  G1 = static_cast<MarkedGB *>(
181  GBDeclared::create(gb_inwwG->get_initial(-1), H, H, 0, 0));
182  state = STATE_compute_w;
183  case STATE_done:
184  set_status(COMP_DONE);
185  return;
186  }
187 }
