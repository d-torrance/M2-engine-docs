15bool Tower::initialize(long charac0,
16 M2_ArrayString names0,
17 const VECTOR(ring_elem) & extensions)
18{
19 initialize_ring(charac0);
20 declare_field();
21
22 names = names0;
23 nvars = names->len;
24 level = nvars - 1;
25
26 // Translate extensions to poly's
27 if (extensions.size() == 0)
28 {
29 D = DRing::create(charac0, nvars, nullptr);
30 }
31 else
32 {
33 TowerPolynomial* exts = new TowerPolynomial[extensions.size()];
34 for (int i = 0; i < extensions.size(); i++)
35 exts[i] = reinterpret_cast<TowerPolynomial>(extensions[i].poly_val);
36 D = DRing::create(charac0, nvars, exts);
37 delete[] exts;
38 }
39
40 zeroV = from_long(0);
41 oneV = from_long(1);
42 minus_oneV = from_long(-1);
43
44 return true;
45}

88M2_arrayint Tower::support(const ring_elem a) const
89{
90 TowerPolynomial f1 = TOWER_POLYNOMIAL_VAL(a);
91 std::vector<int> max_degs;
92 D->degrees_of_vars(f1, max_degs);
93 int nelems = 0;
94 for (size_t i = 0; i < max_degs.size(); i++)
95 if (max_degs[i] > 0) nelems++;
96 M2_arrayint result = M2_makearrayint(nelems);
97 int next = 0;
98 for (size_t i = 0; i < max_degs.size(); i++)
99 if (max_degs[i] > 0) result->array[next++] = static_cast<int>(i);
100 return result;
101}

259class TowerEvaluator : public DPolyTraverser
260{
261 const RingMap *map;
262 const Ring *K;
263 SumCollector *H;
264
265 TowerPolynomial f;
266 int first_var;
267 const Ring *target;
268 gc_vector<int> vp;
269 int nvars;
270
271 virtual bool viewTerm(long coeff, const_exponents exp)
272 {
273 // translate exp to varpower
274 // map->eval_term
275 // either:
276 // H->add, or target->add_to
277 vp.resize(0);
278 varpower::from_expvector(nvars, exp, vp);
279 ring_elem c = K->from_long(coeff);
280 ring_elem a = map->eval_term(K, c, vp.data(), first_var, nvars);
281 H->add(a);
282 return true;
283 }
284
285 public:
286 TowerEvaluator(const Tower *T,
287 const RingMap *map0,
288 const ring_elem f0,
289 int first_var0)
290 : DPolyTraverser(T->D),
291 map(map0),
292 f(TOWER_POLYNOMIAL_VAL(f0)),
293 first_var(first_var0),
294 nvars(T->n_vars())
295 {
296 target = map->get_ring();
297 H = target->make_SumCollector();
298 const PolynomialRing *P = target->cast_to_PolynomialRing();
299 K = (P == nullptr ? target : P->getCoefficients());
300 }
301
302 virtual ~TowerEvaluator() { delete H; }
303 ring_elem getValue()
304 {
305 traverse(f);
306 return H->getValue();
307 }
308};

444ring_elem Tower::translate(const PolynomialRing *R, ring_elem fR) const
445{
446 // create a TowerPolynomial in the Tower T, return it.
447
448 const Monoid *M = R->getMonoid();
449 const Ring *K = R->getCoefficients();
450 int nvars = R->n_vars();
451 TowerPolynomial result = nullptr;
452 exponents_t exp = new int[nvars];
453 for (Nterm& t : fR)
454 {
455 M->to_expvector(t.monom, exp);
456 std::pair<bool, long> res = K->coerceToLongInteger(t.coeff);
457 assert(res.first);
458 int c1 = static_cast<int>(res.second);
459
460 D->add_term(result, c1, exp);
461 }
462 delete[] exp;
463 return TOWER_RINGELEM(result);
464}
