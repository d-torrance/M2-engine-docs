41 int32_t rawRandomInt(int32_t max)
42 /* generate a random number in the range 0..max-1 */
43 {
44  if (max <= 0) return 0;
45  int32_t k = RandomSeed / IQ;
46  RandomSeed =
47  IA * (RandomSeed - k * IQ) - IR * k; /* Schrage algorithm to compute
48  idum = (IA*idum) mod IM */
49  if (RandomSeed < 0) RandomSeed += IM;
50  return RandomSeed % max;
51 }
