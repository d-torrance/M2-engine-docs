41int32_t rawRandomInt(int32_t max)
42/* generate a random number in the range 0..max-1 */
43{
44 if (max <= 0) return 0;
45 int32_t k = RandomSeed / IQ;
46 RandomSeed =
47 IA * (RandomSeed - k * IQ) - IR * k; /* Schrage algorithm to compute
48 idum = (IA*idum) mod IM */
49 if (RandomSeed < 0) RandomSeed += IM;
50 return RandomSeed % max;
51}

53gmp_ZZ rawRandomInteger(gmp_ZZ maxN)
54/* if height is the null pointer, use the default height */
55{
56 mpz_ptr result = getmemstructtype(mpz_ptr);

70gmp_QQ rawRandomQQ(gmp_ZZ height)
71/* returns random a/b, where 1 <= b <= height, 1 <= a <= height */
72/* if height is the null pointer, use the default height */
73{
74 mpq_ptr result = getmemstructtype(mpq_ptr);
75 mpq_init(result);

77 mpz_urandomm(mpq_numref(result), state, height);
