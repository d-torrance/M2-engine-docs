85 void matrixToStream(const Matrix* M, T& stream)
86 {
87  const Ring* R = M->get_ring();
88  const PolyRing* P = R->cast_to_PolyRing();
89  assert(P != 0);
90  const Ring* KK = P->getCoefficientRing();
91  size_t nvars = P->n_vars();
92  size_t ncols = M->n_cols();
93  int charac = static_cast<int>(P->characteristic());
94  assert(charac > 0);
95  exponents_t exp =
96  ALLOCATE_EXPONENTS(EXPONENT_BYTE_SIZE(nvars)); // allocated on stack
97  stream.idealBegin(ncols);
98  Matrix::iterator i(M);
99  for (int c = 0; c < ncols; c++)
100  {
101  i.set(c);
102  // We need the length of this column, in number of monomials
103  size_t nterms = 0;
104  for (; i.valid(); i.next())
105  {
106  Nterm* t = i.entry();
107  for ([[maybe_unused]] Nterm& s : t) nterms++;
108  }
109  stream.appendPolynomialBegin(nterms);
110 
111  i.set(c);
112  // Now we process each column, sending it to the stream
113  for (; i.valid(); i.next())
114  {
115  Nterm* t = i.entry();
116  for (Nterm& s : t)
117  {
118  P->getMonoid()->to_expvector(s.monom, exp);
119  stream.appendTermBegin(i.row());
120  for (size_t j = 0; j < nvars; j++)
121  if (exp[j] != 0) stream.appendExponent(j, exp[j]);
122  std::pair<bool, long> b = KK->coerceToLongInteger(s.coeff);
123  assert(b.first);
124  int a = static_cast<int>(
125  b.second); // This will fit, as the charac fits into an int
126  if (a < 0) a += charac;
127  stream.appendTermDone(a);
128  }
129  }
130  stream.appendPolynomialDone();
131  }
132  stream.idealDone();
133 }
