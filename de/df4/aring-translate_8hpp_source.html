453 inline bool mylift(const ARingRR& R,
454  const ARingRRi& S,
455  ARingRR::ElementType& result_gR,
456  const ARingRRi::ElementType& gS)
457 {
458  ARingRRR T(S.get_precision());
459  ARingRRR::Element gT(T);
460  auto gS1 = const_cast<ARingRRi::ElementType&>(gS);
461  S.midpoint(gT,gS1);
462  bool liftstep = mylift(R,T,result_gR,gT);
463  S.diameter(gT,gS1);
464  return liftstep && T.is_zero(gT);
465 }

495 // ZZ/p --> ZZ/p. 9 versions NONE OF THESE.
496 // instead:
497 // (1) lift to ZZ (int version?)
498 // (2) promote to the version of ZZ/p.
499 
500 // ZZ/p --> GF(p^n)
501 
502 // GF(p^m) --> GF(p^n), where m|n, also switch GF type.
503 
504 // The following are all essentially from_BigRational
505 // QQ --> RR
506 // QQ --> RRR
507 // QQ --> CC
508 // QQ --> CCC
509 
510 // really, there are:
511 // RR --> RRR (from double)
512 // RR --> CC (imag part = 0)
513 // CC --> CCC (from doubles)
514 // RRR --> RRR (change precision)
515 // RRR --> CCC (imag part = 0)
516 // CCC --> CCC (change precision)
517 
518 // RRR --> RR Use lift.
519 // CCC --> CC Use lift.
520 
521 // RRR --> RR (truncate)
522 // RR --> RRR (make new precision)
523 // RRR --> RRR (change precision)
524 // RR --> CC (imag part = 0)
525 // RRR --> CC (RRR --> RR, and imag part = 0)
526 // CCC --> CC (truncate)
527 // RR --> CCC
528 // RRR --> CCC
529 // CC --> CCC
530 // CCC --> CCC
