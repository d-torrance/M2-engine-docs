455inline bool mylift(const ARingRR& R,
456 const ARingRRi& S,
457 ARingRR::ElementType& result_gR,
458 const ARingRRi::ElementType& gS)
459{
460 ARingRRR T(S.get_precision());
461 ARingRRR::Element gT(T);
462 auto gS1 = const_cast<ARingRRi::ElementType&>(gS);
463 S.midpoint(gT,gS1);
464 bool liftstep = mylift(R,T,result_gR,gT);
465 S.diameter(gT,gS1);
466 return liftstep && T.is_zero(gT);
467}

497// ZZ/p --> ZZ/p. 9 versions NONE OF THESE.
498// instead:
499// (1) lift to ZZ (int version?)
500// (2) promote to the version of ZZ/p.
501
502// ZZ/p --> GF(p^n)
503
504// GF(p^m) --> GF(p^n), where m|n, also switch GF type.
505
506// The following are all essentially from_BigRational
507// QQ --> RR
508// QQ --> RRR
509// QQ --> CC
510// QQ --> CCC
511
512// really, there are:
513// RR --> RRR (from double)
514// RR --> CC (imag part = 0)
515// CC --> CCC (from doubles)
516// RRR --> RRR (change precision)
517// RRR --> CCC (imag part = 0)
518// CCC --> CCC (change precision)
519
520// RRR --> RR Use lift.
521// CCC --> CC Use lift.
522
523// RRR --> RR (truncate)
524// RR --> RRR (make new precision)
525// RRR --> RRR (change precision)
526// RR --> CC (imag part = 0)
527// RRR --> CC (RRR --> RR, and imag part = 0)
528// CCC --> CC (truncate)
529// RR --> CCC
530// RRR --> CCC
531// CC --> CCC
532// CCC --> CCC
