158 TEST(ARingCC, axioms)
159 {
160  M2::ARingCC C;
161  auto nbits = C.get_precision();
162  ARingElementGenerator<M2::ARingCC> gen(C);
163  M2::ARingCC::ElementType a, b, c, d, e;
164  C.init(a);
165  C.init(b);
166  C.init(c);
167  C.init(d);
168  C.init(e);
169  for (int i = 0; i < ntrials; i++)
170  {
171  gen.nextElement(a);
172  gen.nextElement(b);
173  gen.nextElement(c);
174  // Test commutativity
175  // test: a*b = b*a
176  // test: a+b == b+a
177  C.add(d, a, b);
178  C.add(e, b, a);
179  EXPECT_TRUE(almostEqual(C, nbits - 2, d, e));
180  C.mult(d, a, b);
181  C.mult(e, b, a);
182  EXPECT_TRUE(almostEqual(C, nbits - 2, d, e));
183 
184  // Test associativity
185  // test: a+(b+c) == (a+b)+c
186  // test: a*(b*c) == (a*b)*c
187  C.add(e, b, c);
188  C.add(d, a, e); // a+(b+c)
189  C.add(e, a, b);
190  C.add(e, e, c); // (a+b)+c
191  EXPECT_TRUE(almostEqual(C, nbits - 6, d, e));
192  C.mult(e, b, c);
193  C.mult(d, a, e); // a*(b*c)
194  C.mult(e, a, b);
195  C.mult(e, e, c); // (a*b)*c
196  EXPECT_TRUE(almostEqual(C, nbits - 6, d, e));
197 
198  // Test distributivity
199  // test: a*(b+c) == a*b + a*c
200  C.add(e, b, c);
201  C.mult(d, a, e); // a*(b+c)
202  C.mult(b, a, b);
203  C.mult(c, a, c);
204  C.add(e, b, c); // a*b + a*c
205  EXPECT_TRUE(almostEqual(C, nbits - 6, d, e));
206  }
207  C.clear(e);
208  C.clear(d);
209  C.clear(c);
210  C.clear(b);
211  C.clear(a);
