57void ARingZZ::syzygy(const ElementType& a,
58 const ElementType& b,
59 ElementType& x,
60 ElementType& y) const
61{
62 assert(!is_zero(b));
63 // First check the special cases a = 0, b = 1, -1. Other cases: use gcd.
64 if (is_zero(a))
65 {
66 set_from_long(x, 1);
67 set_zero(y);
68 return;
69 }
70 if (fmpz_cmp_ui(&b, 1) == 0)
71 {
72 set_from_long(x, 1);
73 negate(y, a);
74 return;
75 }
76 if (fmpz_cmp_si(&b, -1) == 0)
77 {
78 set_from_long(x, 1);
79 set(y, a);
80 return;
81 }
82 ElementType g;
83 init(g);
84 fmpz_gcd(&g, &a, &b);
85 divide(y, a, g);
86 divide(x, b, g);
87 if (fmpz_sgn(&x) > 0)
88 negate(y, y);
89 else
90 negate(x, x);
91 clear(g);
92}
