15 friend bool operator<(const Pair &pair1, const Pair &pair2)
16 {

18 {

22 {
23 return false;
24 }
25 else
26 {

28 {
29 return true;
30 }

33 return false;
34 }

42 public:
43 int i;
44 int j;
45 bool good;
46 brMonomial lcm; // lcm of LTs of fi and fj
47
48 Pair(int a, int b, const IntermediateBasis &F)
49 {
50 if (a < b)
51 {
52 i = a;
53 j = b;
54 }
55 else if (b < a)
56 {
57 i = b;
58 j = a;
59 }
60 else
61 {
62 throw "Invalid numbers in Pair";
63 }
64 IntermediateBasis::const_iterator end = F.end();
65 if (F.find(j) == end || (i >= 0 && F.find(i) == end))
66 { // fi or fj are not in F anymore
67 lcm = 0; // constant 1
68 good = false;
69 }
70 else
71 {
72 if (i < 0)
73 { // working with field polynomial

76 }
77 else

79 unsigned long a = F.find(i)->second.LT();

83 }
84 }

102 // i < j
103 FunctionPair(const Pair &pair, const IntermediateBasis &F, int n)
104 {

106 int j = pair.j;
107
108 IntermediateBasis::const_iterator end = F.end();
109 if (F.find(j) == end || (i >= 0 && F.find(i) == end))

131// generate list of index pairs for given intermediate basis
132// first insert all pairs with FPs, then insert pairs of all other polynomials
133// the list of indices was ordered by increasingly
134Pairs makeList(const IntermediateBasis &F, int n)
135{
136 Pairs B;
137 Pairs::iterator position = B.begin();
138 IntermediateBasis::const_iterator end = F.end();
139 for (IntermediateBasis::const_iterator iter = F.begin(); iter != end; ++iter)
140 {
141 int j = iter->first;
142 for (int i = -n; i < 0; i++)
143 {
144 Pair pair = Pair(i, j, F);
145 position = B.insert(position, pair);
146 }
147 for (int i = 0; i < j; i++)
148 {
149 Pair pair = Pair(i, j, F);
150 if (pair.good)
151 {
152 position = B.insert(position, pair);
153 }
154 }
155 }
156 return B;
157}

190// return true if both functions with indices of pair are in the intermediate
191// basis and their S polynomial should be computed
192bool isGoodPair(const Pair &pair,
193 const IntermediateBasis &F,
194 const Pairs &B,
195 int n)
196{
197 FunctionPair fp = FunctionPair(pair, F, n);
198 if (!fp.good)
199 {
200 return false;
201 }
202
203 // both polynomials are monomials, so their S polynomial reduces to 0
204 if (fp.g->size() == 1 && fp.f->size() == 1)
205 {
206 // cout << "m ";
207 return false;
208 }
209
210 brMonomial g = fp.g->LT();
211 brMonomial f = fp.f->LT();
212 if (BRP::isRelativelyPrime(g, f))
213 {
214 return false;
215 }
216
217 int i = pair.i;
218 int j = pair.j;
219
220 brMonomial lcm = pair.lcm;
221 IntermediateBasis::const_iterator end = F.end();
222 for (IntermediateBasis::const_iterator it = F.begin(); it != end; ++it)
223 {
224 int k = it->first;
225 const BRP *K = &(it->second);
226
227 if ((k != i && k != j && BRP::isDivisibleBy(lcm, K->LT()) &&
228 !inList(i, k, B, F) && !inList(j, k, B, F)))
229 {
230 return false;
231 }
232 }
233
234 // cout << "good pair ";
235 return true;
236}

238// compute S polynomial for a pair
239BRP sPolynomial(const Pair &pair, const IntermediateBasis &F, int n)
240{
241 FunctionPair fp = FunctionPair(pair, F, n);
242 if (!fp.good)
243 {
244 return BRP();
245 }
246
247 if (pair.i < 0)
248 {
249 // fp.g = x_i
250 // f = ax + b
251 BRP b = (fp.f)->remainder(*fp.g);
252 return b * *fp.g + b;
253 }
254 brMonomial f = fp.f->LT();
255 brMonomial g = fp.g->LT();
256 brMonomial lcm = f | g;
257 return *fp.f * (lcm ^ f) + *fp.g * (lcm ^ g);
258}

291// Reduce the leading term of f one step with the first polynomial g_i in the
292// intermediate basis that satisfies isLeadingReducibleBy(f,g_i)
293bool reduceLt(BRP &f,
294 const IntermediateBasis &F,
295 const IntermediateBasis::const_iterator itF)
296{
297 bool ret = false; // true if anything was reduced
298 IntermediateBasis::const_iterator it;
299 IntermediateBasis::const_iterator end = F.end();
300 while (!f.isZero() && (it = findDivisor(f, F, itF)) != end)
301 {
302 ret = true;
303 cancelLeadTerm(f, it->second);
304 }
305 return ret;

376void interreduction(IntermediateBasis &F)
377{
378 bool changesHappened = true;
379 IntermediateBasis::iterator end = F.end();
380 unsigned long numChanged = 0;
381 while (changesHappened)
382 {
383 changesHappened = false;
384 for (IntermediateBasis::iterator it = F.begin(); it != end;)
385 {
386 if (reduce(it->second, F, it))

390 if (it->second.isZero())

392 F.erase(it++);
393 }

396 ++it;
397 }
398 changesHappened = true;
399 }

401 {
402 ++it;
403 }
