360template <typename T>
361void testPower(const T& R, int ntrials)
362{
363 // test the following: (x=generator of the finite field, q = card of field)
364 // check: x^i != x, for 2 <= i <= characteristic-??
365 // x^q == x
366 // x^(q-1) == 1
367 // x^(-1) * x == 1
368 // x^(-2) * x^2 == 1
369
370 // a^2 == a*a, for various a
371 // a^3 == a*a*a
372 // a^0 == 1, what if a == 0?
373 // 1^n == 1, various n
374 ARingElementGenerator<T> gen(R);
375 typename T::ElementType a, b, c, d, one;
376 int q = static_cast<int>(R.cardinality());
377 R.init(one);
378 R.init(a);
379 R.init(b);
380 R.init(c);
381 R.init(d);
382 R.set_from_long(one, 1);
383 for (int i = 0; i < ntrials; i++)
384 {
385 gen.nextElement(a);
386 gen.nextElement(b);
387
388 R.power(c, a, q);
389 EXPECT_TRUE(R.is_equal(c, a)); // test a^q == a
390
391 if (R.is_zero(a)) continue;
392
393 R.power(c, a, q - 1);
394 EXPECT_TRUE(R.is_equal(c, one)); // test a^(q-1) == 1
395
396 R.power(c, a, -1); // test a^-1 * a == 1
397 R.mult(c, a, c);
398 EXPECT_TRUE(R.is_equal(c, one));
399
400 R.power(c, a, -2); // test a^-2 * a^3 == a
401 R.power(d, a, 3);
402 R.mult(d, c, d);
403 EXPECT_TRUE(R.is_equal(d, a));
404 }
405 R.clear(a);
406 R.clear(b);
407 R.clear(c);
408 R.clear(d);
