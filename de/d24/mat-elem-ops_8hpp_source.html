33 static size_t lead_row(const Mat& mat, size_t col)
34 /* returns the largest index row which has a non-zero value in column 'col'.
35 returns -1 if the column has no non-zero entries */
36 {
37 size_t row = mat.numRows();
38 while (row != 0)
39 {
40 --row;
41 if (!mat.ring().is_zero(mat.entry(row, col))) return row;
42 }
43 return static_cast<size_t>(-1);
44 }

46 static size_t lead_row(const Mat& mat, size_t col, ElementType& result)
47 /* returns the largest index row which has a non-zero value in column 'col'.
48 Also sets result to be the entry at this index.
49 returns -1 if the column is 0, or if col is out of range
50 No error is flagged. */
51 {
52 size_t row = mat.numRows();
53 while (row != 0)
54 {
55 --row;
56 if (!mat.ring().is_zero(mat.entry(row, col)))
57 {
58 mat.ring().set(result, mat.entry(row, col));
59 return row;
60 }
61 }
62 return static_cast<size_t>(-1);
63 }

71 static void interchange_rows(Mat& mat, size_t i, size_t j)
72 /* swap rows: row(i) <--> row(j) */
73 {
74 assert(i < mat.numRows());
75 assert(j < mat.numRows());
76 if (i == j) return;
77
78 auto ii = mat.rowBegin(i);
79 auto jj = mat.rowBegin(j);
80 auto end = mat.rowEnd(i);
81 for (; ii != end; ++ii, ++jj) mat.ring().swap(*ii, *jj);
82 }

92 auto jj = mat.columnBegin(j);
93 auto end = mat.columnEnd(i);

184 static void op2by2(const RT& ring,
185 Iterator& loc1,
186 ConstIterator& end,
187 Iterator& loc2,
188 const ElementType& a1,
189 const ElementType& a2,
190 const ElementType& b1,
191 const ElementType& b2)
192 {
193 Element f1(ring), f2(ring), g1(ring), g2(ring);
194 ring.set_zero(f1);
195 ring.set_zero(f2);
196 ring.set_zero(g1);
197 ring.set_zero(g2);
198 for (; loc1 != end; ++loc1, ++loc2)
199 {
200 ring.mult(f1, a1, *loc1);
201 ring.mult(f2, a2, *loc2);
202 ring.mult(g1, b1, *loc1);
203 ring.mult(g2, b2, *loc2);
204
205 ring.add(f1, f1, f2);
206 ring.add(g1, g1, g2);
207 ring.set(*loc1, f1);
208 ring.set(*loc2, g1);
209 }

279 static bool row_permute(Mat& mat, size_t start_row, M2_arrayint perm)
280 {
281 // We copy one row to another location for each cycle in 'perm' of length >
282 // 1.
283 size_t nrows_to_permute = perm->len;
284 std::unique_ptr<bool[]> done(new bool[nrows_to_permute]);
285 for (size_t i = 0; i < nrows_to_permute; i++) done[i] = true;
286 for (size_t i = 0; i < nrows_to_permute; i++)
287 {
288 size_t j = perm->array[i];
289 if (!done[j])
290 {
291 ERROR("expected permutation");
292 return false;
293 }
294 done[j] = false;
295 }
296 typename RT::ElementArray tmp(mat.ring(), mat.numColumns());
297 size_t next = 0;
298
299 while (next < nrows_to_permute)
300 {
301 if (done[next] || perm->array[next] == next)
302 {
303 next++;
304 }
305 else
306 {
307 // store row 'next' into tmp
308 auto loc1 = mat.rowBegin(start_row + next);
309 auto end1 = mat.rowEnd(start_row + next);
310 copy_from_iter(mat.ring(), tmp.data(), tmp.data()+mat.numColumns(), loc1);
311
312 size_t r = next;
313 for (;;)
314 {
315 // copy row perm[r] to row r
316 auto loc2 = mat.rowBegin(start_row + perm->array[r]);
317 loc1 = mat.rowBegin(start_row + r);
318 end1 = mat.rowEnd(start_row + r);
319
320 copy_from_iter(mat.ring(), loc1, end1, loc2);
321 done[r] = true;
322 size_t next_r = perm->array[r];
323 if (next_r == next) break; // and so r is the previous one
324 r = perm->array[r];
325 }
326 // Now copy tmp back
327 // TODO this is even more evil than the above
328 copy_from_iter(mat.ring(), loc1, end1, tmp.data());
329 done[r] = true;
330 }
331 }
332 return true;
333 }

467 // An internal function for reduceby_pivots
468 static void perform_reduction(Mat& M,
469 size_t r,
470 size_t c,
471 size_t nr,
472 size_t nc,
473 int pivot_type)
474 // Subroutine of reduce_pivots()
475 // pivot_type: 1 means pivot is 1, -1 means pivot is -1, 0 means pivot is unit
476 {
477 // Flip rows r, nr
478 // Flip cols c, nc
479 // Use (nr,nc) location to remove all terms in columns 0..nc-1
480 // and in row nr.
481 // Replace column nc with all zeros, except 1 in nr row.
482
483 Element pivot(M.ring()), coef(M.ring()), f(M.ring()), zero(M.ring()),
484 one(M.ring());
485 M.ring().set_from_long(zero, 0);
486 M.ring().set_from_long(one, 1);
487
488 interchange_columns(M, c, nc);
489 interchange_rows(M, r, nr);
490 long pivotrow = lead_row(M, nc, pivot);
491 if (pivot_type == -1) // pivot is -1
492 scale_column(M, nc, pivot);
493 else if (pivot_type == 0)
494 divide_column(M, nc, pivot);
495 for (int i = 0; i < nc; i++)
496 {
497 pivotrow = lead_row(M, i, coef);
498 if (pivotrow < 0) continue;
499 if (pivotrow == nr)
500 {
501 // Do the reduction
502 M.ring().negate(f, coef);
503 column_op(M, i, f, nc);
504 }
505 }
506
507 scale_column(M, nc, zero);
508 setEntry(M, nr, nc, one);
509 }

512 static void reduce_by_pivots(Mat& M)
513 {
514 if (M.numRows() == 0 or M.numColumns() == 0) return;
515 size_t nr = M.numRows() - 1;
516 size_t nc = M.numColumns() - 1;
517
518 Element one(M.ring()), minus_one(M.ring());
519 M.ring().set_from_long(one, 1);
520 M.ring().set_from_long(minus_one, -1);
521
522 // After using the pivot element, it is moved to [nrows-1,ncols-1]
523 // and nrows and ncols are decremented.
524
525 for (size_t i = 0; i <= nc; i++)
526 {
527 auto p = M.columnBegin(i);
528 auto p_end = M.columnEnd(i);
529 for (size_t j = 0; p != p_end; ++p, ++j)
530 {
531 if (M.ring().is_zero(*p)) continue;
532 int pivot_type = 0;
533 if (M.ring().is_equal(one, *p))
534 pivot_type = 1;
535 else if (M.ring().is_equal(minus_one, *p))
536 pivot_type = -1;
537 if (pivot_type != 0)
538 {
539 // printf("before reduction: j=%lu i=%lu:\n",j,i);
540 // displayMat(M);
541 perform_reduction(M, j, i, nr--, nc--, pivot_type);
542 // printf("after reduction: j=%lu i=%lu:\n",j,i);
543 // displayMat(M);
544 if (nr == static_cast<size_t>(-1) or
545 nc == static_cast<size_t>(-1))
546 return;
547 // restart loop with the (new) column i
548 i = -1;
549 break;
550 }
551 }
552 }
553
554 // Now search for other possible pivots
555 for (size_t i = 0; i <= nc; i++)
556 {
557 auto p = M.columnBegin(i);
558 auto p_end = M.columnEnd(i);
559 for (size_t j = 0; p != p_end; ++p, ++j)
560 {
561 if (M.ring().is_zero(*p)) continue;
562 if (!M.ring().is_unit(*p)) continue;
563 int pivot_type = 0;
564 if (M.ring().is_equal(one, *p))
565 pivot_type = 1;
566 else if (M.ring().is_equal(minus_one, *p))
567 pivot_type = -1;
568
569 // printf("before general reduction: j=%lu i=%lu:\n",j,i);
570 // displayMat(M);
571 perform_reduction(M, j, i, nr--, nc--, pivot_type);
572 // printf("after general reduction: j=%lu i=%lu:\n",j,i);
573 // displayMat(M);
574 if (nr == static_cast<size_t>(-1) or nc == static_cast<size_t>(-1))
575 return;
576 // restart loop with the (new) column i
577 i = -1;
578 break;
579 }
580 }

687 // and nrows and ncols are decremented.
688
689 MutableMatrix::iterator *p = M->begin();
690 for (int i=0; i<=nc; i++)
691 {
692 p->set(i);
693 for (; p->valid(); p->next())
694 {
695 int pivot_type = 0;
696 ring_elem coef;
697 p->copy_ring_elem(coef);
698 if (K->is_equal(one, coef))
699 pivot_type = 1;
700 else if (K->is_equal(minus_one, coef))
701 pivot_type = -1;
702 if (pivot_type != 0)
703 {
704 perform_reduction(M, static_cast<int>(p->row()), i, nr, nc, pivot_type);
705 if (nr == 0 or nc == 0) return;
706 --nr;
707 --nc;
708 // restart loop with the (new) column i
709 i = -1;
710 break;
711 }
712 }
713 }
714
715 // Now search for other possible pivots
716 for (int i=0; i<=nc; i++)
717 {
718 p->set(i);
719 for (; p->valid(); p->next())
720 {
721 ring_elem coef;
722 p->copy_ring_elem(coef);
723 if (!K->is_unit(coef)) continue;
724 int pivot_type = 0;
725 if (K->is_equal(one, coef))
726 pivot_type = 1;
727 else if (K->is_equal(minus_one, coef))
728 pivot_type = -1;
729
730 perform_reduction(M, static_cast<int>(p->row()), i, nr, nc, pivot_type);
731 if (nr == 0 or nc == 0) return;
732 --nr;
733 --nc;
734 // restart loop with the (new) column i
735 i = -1;
736 break;
737 }
738 }
739}
