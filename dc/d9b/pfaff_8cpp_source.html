57 ring_elem PfaffianComputation::calc_pfaff(size_t *r, int p2)
58 // Compute the pfaffian of the (skew symmetric)
59 // minor with rows and columns r[0]..r[p2-1].
60 // assumption: p2 is an even number.
61 {
62  if (p2 == 2) return M->elem(static_cast<int>(r[0]), static_cast<int>(r[1]));
63  ring_elem result = R->from_long(0);
64 
65  bool negate = true;
66  for (int i = p2 - 2; i >= 0; i--)
67  {
68  std::swap(r[i], r[p2 - 2]);
69  negate = !negate;
70  ring_elem g =
71  M->elem(static_cast<int>(r[p2 - 2]), static_cast<int>(r[p2 - 1]));
72  if (R->is_zero(g))
73  {
74  R->remove(g);
75  continue;
76  }
77  ring_elem h = calc_pfaff(r, p2 - 2);
78  ring_elem gh = R->mult(g, h);
79  R->remove(g);
80  R->remove(h);
81  if (negate)
82  R->subtract_to(result, gh);
83  else
84  R->add_to(result, gh);
85  }
86 
87  // pulling out the columns has disordered r. Fix it.
88 
89  size_t temp = r[p2 - 2];
90  for (size_t i = p2 - 2; i > 0; i--) r[i] = r[i - 1];
91  r[0] = temp;
92 
93  return result;
94 }
