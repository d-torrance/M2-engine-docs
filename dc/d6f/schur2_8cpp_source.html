48void tableau2::fill(int *lamb, int *pp) // FLAG: should be const, schur_word..
49// Fill the skew tableau2 p\lambda with 1..nboxes
50// starting at top right, moving left and then down
51// row by row.
52{
53 int i, j;
54 p = pp; // FLAG: why is this here?
55 lambda = lamb; // FLAG: why is this here?
56
57 int next = 1;
58 for (i = 1; i < p[0]; i++)
59 {
60 int a = lambda[i];
61 for (j = p[i]; j > a; j--)
62 {
63 xloc[next] = i;
64 yloc[next++] = j;
65 }
66 }

84template<typename T>
85static inline void hash_combine(size_t& seed, const T& val)
86{
87// the typical implementation
88 seed ^= std::hash<T>()(val) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
89}

91unsigned int SchurRing2::computeHashValue(const ring_elem a) const
92{
93// assuming a normal form: distinct monomials in the linear order introduced by compare_partitions()
94
95 const auto& coeffs = a.get_schur_poly()->coeffs;
96 const auto& monoms = a.get_schur_poly()->monoms;
97
98 size_t seed = 95864398; // using previous M2's constant hash value
99
100 for(auto it=coeffs.begin(); it!=coeffs.end(); ++it)
101 hash_combine(seed, coefficientRing->computeHashValue(*it));
102 for(auto it=monoms.begin(); it!=monoms.end(); ++it)
103 hash_combine(seed, *it);
104
105 return static_cast<unsigned>(seed);
106}

155bool SchurRing2::is_valid_partition(M2_arrayint part, bool set_error) const
156{
157 if (nvars >= 0 && part->len > nvars)
158 {
159 if (set_error) ERROR("expected a partition of size at most %d\n", nvars);
160 return false;
161 }
162 for (int i = 1; i < part->len; i++)
163 if (part->array[i - 1] < part->array[i])
164 {
165 if (set_error) ERROR("expected a non-increasing sequence of integers");
166 return false;
167 }
168 if (part->len > 0 && part->array[part->len - 1] < 0)
169 {
170 if (set_error) ERROR("expected nonnegative integers only");
171 return false;
172 }
173 return true;
174}

176static int last_nonzero(M2_arrayint part)
177{
178 for (int i = part->len - 1; i >= 0; i--)
179 if (part->array[i] != 0) return i;
180 return -1;
181}
182ring_elem SchurRing2::from_partition(M2_arrayint part) const
183{
184 schur_poly *f = new schur_poly;
185 f->coeffs.push_back(coefficientRing->one());
186 int len = last_nonzero(part) + 1;
187 f->monoms.push_back(len + 1);
188 for (int i = 0; i < len; i++) f->monoms.push_back(part->array[i]);
189 return ring_elem(f);
190}

200void SchurRing2::elem_text_out(buffer &o,
201 const ring_elem f,
202 bool p_one,
203 bool p_plus,
204 bool p_parens) const
205{
206 const schur_poly *g = f.get_schur_poly();
207 size_t n = g->size();
208
209 bool needs_parens = p_parens && (n > 1);
210 if (needs_parens)
211 {
212 if (p_plus) o << '+';
213 o << '(';
214 p_plus = false;
215 }
216
217 p_one = false;
218 for (schur_poly::iterator i = g->begin(); i != g->end(); ++i)
219 {
220 const_schur_partition part = i.getMonomial();
221 int len = *part++;
222 int isone = (len == 1); // the empty partition
223 p_parens = !isone;
224 coefficientRing->elem_text_out(
225 o, i.getCoefficient(), p_one, p_plus, p_parens);
226 o << "{";
227 for (int j = 0; j < len - 1; j++)
228 {
229 if (j > 0) o << ",";
230 o << part[j];
231 }
232 o << "}";
233 p_plus = true;
234 }
235 if (needs_parens) o << ')';
236}

251bool SchurRing2::is_equal(const ring_elem f, const ring_elem g) const
252{
253 const schur_poly *f1 = f.get_schur_poly();
254 const schur_poly *g1 = g.get_schur_poly();
255 if (f1->size() != g1->size()) return false;
256 if (f1->monoms.size() != g1->monoms.size()) return false;
257
258 VECTOR(schur_word)::const_iterator m_f = f1->monoms.begin();
259 VECTOR(schur_word)::const_iterator m_g = g1->monoms.begin();
260 for (; m_f != f1->monoms.end(); ++m_f, ++m_g)
261 if (*m_f != *m_g) return false;
262
263 VECTOR(ring_elem)::const_iterator c_f = f1->coeffs.begin();
264 VECTOR(ring_elem)::const_iterator c_g = g1->coeffs.begin();
265 for (; c_f != f1->coeffs.end(); ++c_f, ++c_g)
266 if (!coefficientRing->is_equal(*c_f, *c_g)) return false;
267
268 return true;
269}

341int SchurRing2::compare_partitions(const_schur_partition a,
342 const_schur_partition b) const
343{
344 int len = a[0];
345 if (b[0] < len) len = b[0];
346 for (int i = 1; i < len; i++)
347 {
348 int cmp = a[i] - b[i];
349 if (cmp < 0) return LT;
350 if (cmp > 0) return GT;
351 }
352 int cmp = a[0] - b[0];
353 if (cmp < 0) return LT;
354 if (cmp > 0) return GT;
355 return EQ;
356}

358int SchurRing2::compare_elems(const ring_elem f, const ring_elem g) const
359{
360// assuming the monomials are sorted in the linear order on the partitions
361// see SchurRing2::compare_partitions
362
363 auto f_it = f.get_schur_poly()->begin(),
364 f_end = f.get_schur_poly()->end();
365 auto g_it = g.get_schur_poly()->begin(),
366 g_end = g.get_schur_poly()->end();
367
368 for(; f_it!=f_end && g_it!=g_end; ++f_it, ++g_it) {
369 auto cmp = compare_partitions(f_it.getMonomial(), g_it.getMonomial());
370 if(cmp) return cmp;
371 }
372
373 return (f_it!=f_end)-(g_it!=g_end); // LT, EQ or GT
374}

376bool SchurRing2::promote_coeffs(const SchurRing2 *Rf,
377 const ring_elem f,
378 ring_elem &resultRE) const
379{
380 // Assumption in use: Rf (ring of f) is a Schur ring, with coeff ring coeffRf
381 const schur_poly *f1 = f.get_schur_poly();
382 schur_poly *result = new schur_poly;
383
384 for (schur_poly::iterator i = f1->begin(); i != f1->end(); ++i)
385 {
386 if (i.getMonomial()[0] - 1 > nvars) continue;
387 ring_elem a;
388 if (!coefficientRing->promote(
389 Rf->getCoefficientRing(), i.getCoefficient(), a))
390 {
391 delete result;
392 resultRE = from_long(0);
393 return false;
394 }
395 result->appendTerm(a, i.getMonomial());
396 }
397 resultRE = ring_elem(result);
398 return true;
399}

424bool SchurRing2::promote(const Ring *Rf,
425 const ring_elem f,
426 ring_elem &result) const
427{
428 // Cases:
429 // 1. Rf is ZZ
430 // 2. Rf is coefficientRing
431 // 3. Rf is another SchurRing2
432
433 if (Rf == globalZZ)
434 {
435 from_coeff(Rf->promote(globalZZ, f, result));
436 return true;
437 }
438 else if (Rf == coefficientRing)
439 {
440 result = from_coeff(f);
441 return true;
442 }
443 else
444 {
445 const SchurRing2 *Sf = Rf->cast_to_SchurRing2();
446 if (Sf != nullptr)
447 {
448 if (coefficientRing == Sf->getCoefficientRing())
449 {
450 result = truncate(f);
451 return true;
452 }
453
454 return promote_coeffs(Sf, f, result);
455 }
456 }
457 return false;
458}
459bool SchurRing2::lift(const Ring *Rg,
460 const ring_elem f,
461 ring_elem &result) const
462{
463 const schur_poly *f1 = f.get_schur_poly();
464 if (Rg == coefficientRing || Rg == globalZZ)
465 {
466 if (get_scalar(f1, result))
467 {
468 if (Rg == globalZZ)
469 return coefficientRing->lift(globalZZ, result, result);
470 return true;
471 }
472 }
473 else
474 {
475 const SchurRing2 *Sg = Rg->cast_to_SchurRing2();
476 if (Sg != nullptr)
477 {
478 if (coefficientRing == Sg->getCoefficientRing())
479 {
480 result = Sg->truncate(f);
481 return true;
482 }
483
484 return lift_coeffs(Sg, f, result);
485 }
486 }
487 return false;

489ring_elem SchurRing2::negate(const ring_elem f) const
490{
491 if (is_zero(f)) return f;
492 const schur_poly *f1 = f.get_schur_poly();
493 schur_poly *result = new schur_poly;
494
495 for (VECTOR(ring_elem)::const_iterator i = f1->coeffs.begin();
496 i != f1->coeffs.end();
497 ++i)
498 result->coeffs.push_back(coefficientRing->negate(*i));
499
500 result->monoms.insert(
501 result->monoms.end(), f1->monoms.begin(), f1->monoms.end());
502 return ring_elem(result);
503}

505ring_elem SchurRing2::truncate(const ring_elem f) const
506// assumption: f is a Schur poly over another Schur ring, with the SAME coeff
507// ring
508// each term is copied over, if the number of elements in the partition is <=
509// n_vars()
510{
511 if (is_zero(f)) return f;
512 const schur_poly *f1 = f.get_schur_poly();
513 schur_poly *result = new schur_poly;
514
515 for (schur_poly::iterator i = f1->begin(); i != f1->end(); ++i)
516 {
517 if (i.getMonomial()[0] - 1 > nvars) continue;
518 result->appendTerm(i.getCoefficient(), i.getMonomial());
519 }
520 return ring_elem(result);
521}

523ring_elem SchurRing2::add(const ring_elem f, const ring_elem g) const
524{
525 if (is_zero(f)) return g;
526 if (is_zero(g)) return f;
527 const schur_poly *f1 = f.get_schur_poly();
528 const schur_poly *g1 = g.get_schur_poly();
529
530 schur_poly *result = new schur_poly;
531
532 schur_poly::iterator i = f1->begin();
533 schur_poly::iterator j = g1->begin();
534 schur_poly::iterator iend = f1->end();
535 schur_poly::iterator jend = g1->end();
536
537 bool done = false;
538 while (!done)
539 {
540 int cmp = compare_partitions(i.getMonomial(), j.getMonomial());
541 switch (cmp)
542 {
543 case LT:
544 result->appendTerm(j.getCoefficient(), j.getMonomial());
545 ++j;
546 if (j == jend)
547 {
548 result->append(i, iend);
549 done = true;
550 }
551 break;
552 case GT:
553 result->appendTerm(i.getCoefficient(), i.getMonomial());
554 ++i;
555 if (i == iend)
556 {
557 result->append(j, jend);
558 done = true;
559 }
560 break;
561 case EQ:
562 ring_elem c =
563 coefficientRing->add(i.getCoefficient(), j.getCoefficient());
564 if (!coefficientRing->is_zero(c))
565 result->appendTerm(c, i.getMonomial());
566 ++j;
567 ++i;
568 if (j == jend)
569 {
570 result->append(i, iend);
571 done = true;
572 }
573 else
574 {
575 if (i == iend)
576 {
577 result->append(j, jend);
578 done = true;
579 }
580 }
581 break;
582 }
583 }
584 return ring_elem(result);
585}

593schur_poly *SchurRing2::mult_by_coefficient(ring_elem a,
594 const schur_poly *f) const
595{
596 schur_poly *result = new schur_poly;
597
598 VECTOR(ring_elem)::iterator c_result;
599 for (VECTOR(ring_elem)::const_iterator c_f = f->coeffs.begin();
600 c_f != f->coeffs.end();
601 ++c_f)
602 result->coeffs.push_back(coefficientRing->mult(a, *c_f));
603
604 result->monoms.insert(
605 result->monoms.end(), f->monoms.begin(), f->monoms.end());
606
607 return result;
608}
609ring_elem SchurRing2::mult(const ring_elem f, const ring_elem g) const
610{
611 ring_elem resultRE;
612 const schur_poly *f1 = f.get_schur_poly();
613 const schur_poly *g1 = g.get_schur_poly();
614 ring_elem a;
615 if (get_scalar(f1, a))
616 {
617 return ring_elem(mult_by_coefficient(a, g1));
618 // In this case, do a simple multiplication
619 }
620 else if (get_scalar(g1, a))
621 {
622 return ring_elem(mult_by_coefficient(a, f1));
623 }
624 else
625 {
626 // use the poly heap
627 schur_poly_heap H(this);
628 for (schur_poly::iterator i = f1->begin(); i != f1->end(); ++i)
629 for (schur_poly::iterator j = g1->begin(); j != g1->end(); ++j)
630 {
631 ring_elem c =
632 coefficientRing->mult(i.getCoefficient(), j.getCoefficient());
633 ring_elem r = const_cast<SchurRing2 *>(this)->mult_terms(
634 i.getMonomial(), j.getMonomial());
635 H.add(ring_elem(mult_by_coefficient(c, r.get_schur_poly())));
636 }
637 return H.value();
638 }
639}

641void to_varpower(const_schur_partition a, gc_vector<int>& result)
642{
643 int len = a[0];
644 result.resize(2 * len);
645 int *result_vp = result.data();
646 int *orig_result_vp = result_vp;
647 result_vp++;
648
649 if (len > 1)
650 {
651 int v = a[1];
652 int e = 1;
653
654 for (int i = 2; i < len; i++)
655 {
656 if (v == a[i])
657 e++;
658 else
659 {
660 *result_vp++ = v;
661 *result_vp++ = e;
662 v = a[i];
663 e = 1;
664 }
665 }
666 *result_vp++ = v;
667 *result_vp++ = e;
668 }
669
670 int newlen = static_cast<int>(result_vp - orig_result_vp);
671 *orig_result_vp = newlen;
672 result.resize(newlen);
673}

675engine_RawArrayPairOrNull SchurRing2::list_form(const Ring *coeffR,
676 const ring_elem f) const
677{
678 if (coeffR != coefficientRing)
679 {
680 ERROR("expected coefficient ring of Schur ring");
681 return nullptr;
682 }
683 const schur_poly *f1 = f.get_schur_poly();
684 int n = static_cast<int>(f1->size()); // this is here because the lengths of
685 // arrays for M3 front end use int as
686 // length field.
687 engine_RawMonomialArray monoms =
688 GETMEM(engine_RawMonomialArray, sizeofarray(monoms, n));
689 engine_RawRingElementArray coeffs =
690 GETMEM(engine_RawRingElementArray, sizeofarray(coeffs, n));
691 monoms->len = n;
692 coeffs->len = n;
693 engine_RawArrayPair result = newitem(struct engine_RawArrayPair_struct);
694 result->monoms = monoms;
695 result->coeffs = coeffs;
696
697 // Loop through the terms
698 gc_vector<int> vp;
699 schur_poly::iterator i = f1->begin();
700 for (int next = 0; next < n; ++i, ++next)
701 {
702 coeffs->array[next] =
703 RingElement::make_raw(coefficientRing, i.getCoefficient());
704 to_varpower(i.getMonomial(), vp);
705 monoms->array[next] = EngineMonomial::make(vp.data());
706 vp.resize(0);
707 }
708 return result;
709}

739void SchurRing2::SMbounds(int &lo, int &hi)
740{
741 int i, k;
742 int x = SMfilled.xloc[SMcurrent];
743 int y = SMfilled.yloc[SMcurrent];
744
745 // First set the high bound, using info from the "one to the right"
746 // in the reverse lex filled skew tableau.
747
748 if (y == SMfilled.p[x]) // There is not one to the right
749 {
750 hi = SMmaxrows;
751 for (k = 1; k <= SMmaxrows; k++)
752 if (SMtab.p[k] == 0)
753 {
754 hi = k;
755 break;
756 }
757 }
758 else // note that the case SMcurrent==1 will be handled
759 { // in the previous statement.
760 hi = SMtab.xloc[SMcurrent - 1];
761 }
762
763 // Now we set the lo bound, using info from the "one above"
764
765 if (x == 1 || y <= SMfilled.lambda[x - 1])
766 lo = 1; // There is not one above
767 else
768 {
769 int above = SMcurrent - SMfilled.p[x] + SMfilled.lambda[x - 1];
770 int xabove = SMtab.xloc[above];
771 int yabove = SMtab.yloc[above];
772 for (i = xabove + 1; i <= hi; i++)
773 if (SMtab.p[i] < yabove) break;
774 lo = i;
775 }
776}

786void SchurRing2::SM()
787{
788 int lo, hi;
789
790 if (SMcurrent == SMfinalwt)
791 {
792 // partition is to be output
793 SMsetPartitionLength(SMtab.p, SMmaxrows);
794 SMappendTerm(SMtab.p);
795 return;
796 }
797
798 SMcurrent++;
799 SMbounds(lo, hi);
800 int this_one = LARGE_NUMBER; // larger than any entry of SMtab: SMfinalwt+1
801 // should work...
802 int last_one;
803 for (int i = lo; i <= hi; i++)
804 {
805 last_one = this_one;
806 this_one = SMtab.p[i];
807 if (last_one > this_one)
808 {
809 SMtab.p[i]++;
810 SMtab.xloc[SMcurrent] = i;
811 SMtab.yloc[SMcurrent] = SMtab.p[i];
812 SM();
813 SMtab.p[i]--;
814 }
815 }
816 SMcurrent--;
817}

827ring_elem SchurRing2::skew_schur(const_schur_partition lambda,
828 const_schur_partition p)
829{
830 SMcurrent = 0;
831
832 SMfinalwt = 0;
833 for (int i = 1; i < p[0]; i++) SMfinalwt += p[i];
834 for (int i = 1; i < lambda[0]; i++) SMfinalwt -= lambda[i];
835 SMmaxrows = p[0] - 1; // this is the number of elements in the partition p
836 if (nvars != -1 && SMmaxrows > nvars) SMmaxrows = nvars;
837
838 delete[] SMtab
839 .p; // FLAG: should use gc for this? or not use it, but not both!
840 SMtab.p = new int[SMmaxrows + 1]; // FLAG: should use gc for this? or not
841 // use it, but not both!
842 for (int i = 0; i <= SMmaxrows; i++) SMtab.p[i] = 0;
843
844 SMtab.wt = SMfinalwt;
845 SMtab.resize(SMfinalwt);
846 SMfilled.resize(SMfinalwt);
847
848 // lambda and p should not be modified in the following call
849 SMfilled.fill(const_cast<schur_partition>(lambda),
850 const_cast<schur_partition>(
851 p)); // FLAG:fill should take const arguments...
852 lambda++; // FLAG: why is this here?
853 p++; // FLAG: why is this here?
854 SM();
855 return SMheap->value(); // resets itself back to new
856}

858ring_elem SchurRing2::mult_terms(const_schur_partition a,
859 const_schur_partition b)
860{
861 int maxsize = (a[0] - 1 + b[0] - 1) + 1; // this is the max number of
862 // elements in the output partition,
863 // plus one
864
865 schur_partition lambda = ALLOCATE_EXPONENTS(sizeof(schur_word) * maxsize);
866 schur_partition p = ALLOCATE_EXPONENTS(sizeof(schur_word) * maxsize);
867
868 // Second: make the skew partition (note: r,s>=1)
869 // this is: if a = r+1 a1 a2 ... ar
870 // b = s+1 b1 b2 ... bs
871 // p is:
872 // (r+s+1) b1+a1 b1+a2 ... b1+ar b1 b2 ... bs
873 // lambda is:
874 // (r+1) b1 b1 ... b1 0 0 ... 0
875
876 int r = a[0] - 1;
877 int s = b[0] - 1;
878 int c = b[1];
879
880 assert(r + s + 1 == maxsize);
881
882 for (int i = 1; i <= r; i++)
883 {
884 assert(i < maxsize);
885 p[i] = c + a[i];
886 lambda[i] = c;
887 }
888 for (int i = r + 1; i < r + s + 1; i++)
889 {
890 assert(i < maxsize);
891 p[i] = b[i - r];
892 lambda[i] = 0;
893 }
894 p[0] = r + s + 1;
895 lambda[0] = r + 1;
896 return skew_schur(lambda, p);
897}
