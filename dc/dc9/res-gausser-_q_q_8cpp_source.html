120 void ResGausserQQ::sparseCancel(CoefficientVector r,
121  CoefficientVector sparse,
122  ComponentIndex* comps,
123  CoefficientVector result_loc) const
124 // dense += c * sparse, where c is chosen to cancel column comps[0].
125 // ASSUMPTION: the lead coeff of 'sparse' is 1 or -1 (in the field)
126 // The value of c is appended to result_
127 {
128  // r += a*sparse
129  // ASSUMPTIONS:
130  // len > 0,
131  // sparse[0] = 1 or -1 (in the field)
132  // where a is
133  // r[comps[0]], if sparse[0]==1
134  // -r[comps[0]], if sparse[0]==-1
135  // r = [...., b, .....]
136  // sparse = [1,...] then a = -b
137  // sparse = [-1,...] then a = b
138 
139  auto& vec = coefficientVector(r);
140  auto elems = vec.data();
141  auto& svec = coefficientVector(sparse);
142  auto sparseelems = svec.data();
143  ComponentIndex len = static_cast<ComponentIndex>(svec.size());
144 
145  auto& result = coefficientVector(result_loc);
146 
147  // Basically, over ZZ/p, we are doing: r += a*sparse,
148  // where sparse is monic, and a is -r.coeffs[*comps].
149 
150  n_dense_row_cancel++;
151  n_subtract_multiple += len;
152  FieldElement a = elems[*comps];
153  if (sparseelems[0].mDouble <
154  0) // should be minus_one, since it is either 1 or -1.
155  {
156  a.mDouble = -a.mDouble;
157  Kp1.negate(a.mMod1, a.mMod1);
158  }
159  for (ComponentIndex i = len; i > 0; i--)
160  {
161  FieldElement& sparse = *sparseelems++;
162  FieldElement& result = elems[*comps++];
163  result.mDouble = result.mDouble - a.mDouble * sparse.mDouble;
164  Kp1.subtract_multiple(result.mMod1, a.mMod1, sparse.mMod1);
165  int newsize = a.mDenominatorSize + sparse.mDenominatorSize;
166  if (newsize > result.mDenominatorSize) result.mDenominatorSize = newsize;
167  }
168  a.mDouble = -a.mDouble;
169  Kp1.negate(a.mMod1, a.mMod1);
170  if (a.mDenominatorSize > mMaxDenominatorSize)
171  {
172  mMaxDenominatorSize = a.mDenominatorSize;
173  std::cout << "coeff with den = " << a.mDenominatorSize << ": ";
174  out(std::cout, a);
175  std::cout << std::endl;
176  }
177  result.push_back(a);
178 }
