112 void ResGausserQQHybrid::subtract_multiple_to_element(
113  FieldElement& result,

115  const FieldElement& b) const
116 {
117  result.mDouble = result.mDouble - a.mDouble * b.mDouble;
118  Kp1.subtract_multiple(result.mMod1, a.mMod1, b.mMod1);
119  Kp2.subtract_multiple(result.mMod2, a.mMod2, b.mMod2);

121  int newsize = a.mDenominatorSize + b.mDenominatorSize;
122  if (newsize > result.mDenominatorSize) result.mDenominatorSize = newsize;
123  if (result.mIsPresent) numAdditions++;
124  result.mIsPresent = true;

246 void ResGausserQQHybrid::sparseCancel(CoefficientVector r,
247  CoefficientVector sparse,
248  ComponentIndex* comps,
249  CoefficientVector result_loc) const
250 // dense += c * sparse, where c is chosen to cancel column comps[0].
251 // ASSUMPTION: the lead coeff of 'sparse' is 1 or -1 (in the field)
252 // The value of c is appended to result_
253 {
254  // r += a*sparse
255  // ASSUMPTIONS:
256  // len > 0,
257  // sparse[0] = 1 or -1 (in the field)
258  // where a is
259  // r[comps[0]], if sparse[0]==1
260  // -r[comps[0]], if sparse[0]==-1
261  // r = [...., b, .....]
262  // sparse = [1,...] then a = -b
263  // sparse = [-1,...] then a = b
264 
265  auto& vec = coefficientVector(r);
266  auto elems = vec.data();
267  auto& svec = coefficientVector(sparse);
268  auto sparseelems = svec.data();
269  ComponentIndex len = static_cast<ComponentIndex>(svec.size());
270 
271  auto& result = coefficientVector(result_loc);
272 
273  // Basically, over ZZ/p, we are doing: r += a*sparse,
274  // where sparse is monic, and a is -r.coeffs[*comps].
275 
276  n_dense_row_cancel++;
277  n_subtract_multiple += len;
278  FieldElement a;
279  init_element(a);
280  copy_element(a, elems[*comps]);
281  if (sparseelems[0].mDouble <
282  0) // should be minus_one, since it is either 1 or -1.
283  {
284  negate_element(a);
285  }
286  from_long_element(elems[*comps], 0);
287  comps++;
288  sparseelems++;
289  for (ComponentIndex i = len - 1; i > 0; i--)
290  {
291  FieldElement& sparse = *sparseelems++;
292  FieldElement& result = elems[*comps++];
293  subtract_multiple_to_element(result, a, sparse);
294  }
295  negate_element(a);
296  if (a.mDenominatorSize > mMaxDenominatorSize)
297  {
298  mMaxDenominatorSize = a.mDenominatorSize;
299  if (M2_gbTrace >= 1)
300  {
301  std::cout << "coeff with den = " << a.mDenominatorSize << ": ";
302  out(std::cout, a);
303  std::cout << std::endl;
304  }
305  }
306  result.emplace_back(std::move(a));
307 }
