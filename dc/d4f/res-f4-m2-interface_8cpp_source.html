95 vec ResF4toM2Interface::to_M2_vec(const ResPolyRing& R,
96  const ResPolynomial& f,
97  const FreeModule* F)
98 {
99  const PolynomialRing* origR = F->get_ring()->cast_to_PolynomialRing();
100  const Monoid* M = origR->getMonoid();
101 
102  monomial m1 = M->make_one();
103 
104  Nterm** comps = newarray(Nterm*, F->rank());
105  Nterm** last = newarray(Nterm*, F->rank());
106  for (int i = 0; i < F->rank(); i++)
107  {
108  comps[i] = nullptr;
109  last[i] = nullptr;
110  }
111 
112  int* exp = new int[M->n_vars()];
113 
114  const res_monomial_word* w = f.monoms.data();
115  for (int i = 0; i < f.len; i++)
116  {
117  component_index comp;
118  R.monoid().to_expvector(w, exp, comp);
119  w = w + R.monoid().monomial_size(w);
120  M->from_expvector(exp, m1);
121  ring_elem a =
122  R.vectorArithmetic().ringElemFromElementArray(f.coeffs,i);
123  Nterm* g = origR->make_flat_term(a, m1);
124  g->next = nullptr;
125  if (last[comp] == nullptr)
126  {
127  comps[comp] = g;
128  last[comp] = g;
129  }
130  else
131  {
132  last[comp]->next = g;
133  last[comp] = g;
134  }
135  }
136  vec result = nullptr;
137  for (int i = 0; i < F->rank(); i++)
138  {
139  if (comps[i] != nullptr)
140  {
141  vec v = origR->make_vec(i, comps[i]);
142  origR->add_vec_to(result, v);
143  comps[i] = nullptr;
144  last[i] = nullptr;
145  }
146  }
147 
148  delete[] exp;
149  return result;
150 }

152 FreeModule* ResF4toM2Interface::to_M2_freemodule(const PolynomialRing* R,
153  SchreyerFrame& C,
154  int lev)
155 {
156  FreeModule* result = new FreeModule(R, 0, true);
157  if (lev < 0 or lev > C.maxLevel())
158  {
159  return result;
160  }
161  const Monoid* M = R->getMonoid();
162  auto& thislevel = C.level(lev);
163  const ResSchreyerOrder& S = C.schreyerOrder(lev);
164  int* exp = new int[M->n_vars()];
165  for (auto i = 0; i < thislevel.size(); ++i)
166  {
167  int d[1];
168  d[0] = thislevel[i].mDegree;
169  monomial deg = M->degree_monoid()->make_one();
170  M->degree_monoid()->from_expvector(d, deg);
171  // Now grab the Schreyer info
172  // unpack to exponent vector, then repack into monoid element
173  monomial totalmonom = M->make_one();
174  component_index comp;
175  C.monoid().to_expvector(S.mTotalMonom[i], exp, comp);
176  M->from_expvector(exp, totalmonom);
177  result->append_schreyer(
178  deg, totalmonom, static_cast<int>(S.mTieBreaker[i]));
179  }
180  delete[] exp;
181  return result;
182 }

184 FreeModule* ResF4toM2Interface::to_M2_freemodule(const PolynomialRing* R,
185  const FreeModule* F,
186  SchreyerFrame& C,
187  int lev)
188 // The input F must be the original freemodule of level=0.
189 // assumption: lev >= 0.
190 {
191  if (lev < 0 or lev > C.maxLevel())
192  {
193  ERROR("expected level in the range %d..%d",1,C.maxLevel());
194  return nullptr;
195  }
196  FreeModule* result = new FreeModule(R, 0, true);
197  const Monoid* M = R->getMonoid();
198  auto& thislevel = C.level(lev);
199  const ResSchreyerOrder& S = C.schreyerOrder(lev);
200  int* exp = new int[M->n_vars()];
201  monomial deg1 = M->degree_monoid()->make_one();
202  for (auto i = 0; i < thislevel.size(); ++i)
203  {
204  component_index comp;
205  C.monoid().to_expvector(S.mTotalMonom[i], exp, comp);
206  monomial deg = M->degree_monoid()->make_new(F->degree(comp)); // resulting degree of this element
207  M->degree_of_expvector(exp, deg1);
208  M->degree_monoid()->mult(deg, deg1, deg);
209  // Now grab the Schreyer info
210  // unpack to exponent vector, then repack into monoid element
211  monomial totalmonom = M->make_one();
212  M->from_expvector(exp, totalmonom);
213  result->append_schreyer(
214  deg, totalmonom, static_cast<int>(S.mTieBreaker[i]));
215  }
216  delete[] exp;
217  M->degree_monoid()->remove(deg1);
218  return result;
219 }

291  {
292  component_index comp;

294  w = w + C.ring().monoid().monomial_size(w);

310  }

321 MutableMatrix* ResF4toM2Interface::to_M2_MutableMatrix(SchreyerFrame& C,
322  const Ring* K,
323  int lev,
324  int degree)
325 {
326  // The ring K should be the coefficient ring of the poly ring of C,
327  // OR: if the coefficient ring is QQ, then it can be RR, or a finite field.
328 
329  // Now we loop through the elements of degree 'degree' at level 'lev'
330  auto& thislevel = C.level(lev);
331  int n = 0;
332  for (auto p = thislevel.begin(); p != thislevel.end(); ++p)
333  {
334  if (p->mDegree == degree) n++;
335  }
336 
337  auto& prevlevel = C.level(lev - 1);
338  int* newcomps = new int[prevlevel.size()];
339  int nextcomp = 0;
340  for (int i = 0; i < prevlevel.size(); i++)
341  if (prevlevel[i].mDegree == degree)
342  newcomps[i] = nextcomp++;
343  else
344  newcomps[i] = -1;
345 
346  // create the mutable matrix
347  MutableMatrix* result = MutableMatrix::zero_matrix(K, nextcomp, n, true);
348  // Now loop through the elements at thislevel,
349  // and for each, loop through the terms of mSyzygy.
350  // if the component x satisfies newcomps[x] >= 0, then place
351  // this coeff into the mutable matrix.
352  int col = 0;
353 
354  for (auto p = thislevel.begin(); p != thislevel.end(); ++p)
355  {
356  if (p->mDegree != degree) continue;
357  auto& f = p->mSyzygy;
358  auto end = ResPolynomialIterator(C.ring(), f, 1);
359  auto i = ResPolynomialIterator(C.ring(), f);
360  for (; i != end; ++i)
361  {
362  long comp = C.monoid().get_component(i.monomial());
363  if (newcomps[comp] >= 0)
364  {
365  ring_elem a = C.ring().vectorArithmetic().ringElemFromElementArray(
366  f.coeffs, i.coefficient_index());
367  result->set_entry(newcomps[comp], col, a);
368  }
369  }
370  ++col;
371  }
372 
373  delete[] newcomps;
374  return result;
375 }

384 {
385 public:
386  DegreeZeroMapGenerator(SchreyerFrame& C, int slanted_degree, int lev)
387  : mSchreyerFrame(C),
388  mThisLevel(C.level(lev)),
389  mDegree(slanted_degree+lev),
390  mLevel(lev),
391  mNumRows(0),
392  mNumColumns(0)
393  {
394  if (lev <= 0 or lev > C.maxLevel())
395  {
396  return;
397  }
398  int degree = slanted_degree + lev;
399  for (auto p = mThisLevel.begin(); p != mThisLevel.end(); ++p)
400  {
401  if (p->mDegree == degree) mNumColumns++;
402  }
403 
404  auto& prevlevel = C.level(lev - 1);
405  mComponentTranslation.resize(prevlevel.size());
406  for (int i = 0; i < prevlevel.size(); i++)
407  {
408  if (prevlevel[i].mDegree == mDegree)
409  mComponentTranslation[i] = mNumRows++;
410  else
411  mComponentTranslation[i] = -1;
412  }
413  }

418 
419  int numColumns() const { return mNumColumns; }

421  long numNonzero() const
422  {
423  long nnonzeros = 0;

425  for (auto p = thislevel.begin(); p != thislevel.end(); ++p)
426  {

436  nnonzeros++;
437  }
438  }
439  return nnonzeros;
