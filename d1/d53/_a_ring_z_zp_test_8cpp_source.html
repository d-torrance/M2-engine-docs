46 void testCoerceToLongInteger(const RT& R)
47 {
48  long top1 = (R.characteristic() > 10000 ? 10000 : R.characteristic());
49  long bottom2 = (R.characteristic() > 10000 ? R.characteristic() - 10000 : 0);
50  for (long i = 0; i < top1; i++)
51  {
52  typename RT::ElementType a;
53  R.init(a);
54  R.set_from_long(a, i);
55  long b = R.coerceToLongInteger(a);
56  if (b < 0) b += R.characteristic();
57  EXPECT_EQ(b, i);
58  }
59  for (long i = bottom2; i < R.characteristic(); i++)
60  {
61  typename RT::ElementType a;
62  R.init(a);
63  R.set_from_long(a, i);
64  long b = R.coerceToLongInteger(a);
65  if (b < 0) b += R.characteristic();

67  }
