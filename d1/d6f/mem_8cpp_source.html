48 void stash::chop_slab()
49 {
50  // grab a new slab, and chop it into element_size pieces, placing them
51  // onto the free list.
52 
53  slab *new_slab = new slab;
54  new_slab->next = slabs;
55  slabs = new_slab;
56 
57  // Time to chop it up.
