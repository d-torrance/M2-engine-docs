182 void denseCancelFromSparse(ElementArray& dense,
183 const ElementArray& sparse,
184 const Range<const int>& comps) const
185 {
186 // ASSUMPTION: svec[0] == 1.
187 auto& dvec = * elementArray(dense);
188 auto& svec = * elementArray(sparse);
189
190 typename RingType::Element a(*mRing,dvec[comps[0]]);
191 for (int i=1; i < comps.size(); ++i)
192 mRing->subtract_multiple(dvec[comps[i]], a, svec[i]);
193 mRing->set_zero(dvec[comps[0]]);

196 void denseCancelFromSparse(ElementArray& dense,
197 const ElementArray& sparse,
198 const Range<const int>& comps,
199 ElementArray& result_multiplier) const
200 {
201 // ASSUMPTION: svec[0] == 1, comps.size() >= 1
202 auto& dvec = * elementArray(dense);
203 auto& svec = * elementArray(sparse);
204 auto& mults = * elementArray(result_multiplier);
205
206 FieldElement b;
207 mRing->init(b);
208 mRing->set(b, dvec[comps[0]]);
209
210 FieldElement one;
211 mRing->init(one);
212 mRing->set_from_long(one, 1);
213 if (not mRing->is_equal(svec[0], one)) // should be minus_one
214 mRing->negate(b, b);
215
216 for (int i=1; i < comps.size(); ++i)
217 {
218 mRing->subtract_multiple(dvec[comps[i]], b, svec[i]);
219 mStats.incrementNumAdditions();
220 }
221 mRing->set_zero(dvec[comps[0]]);
222
223 mRing->negate(b, b);
224 mults.push_back(b); // this transfers ownership of b.
225 }

261 template<typename LockType>
262 void safeDenseToSparse(ElementArray& dense,
263 ElementArray& sparse, // output value: sets this value
264 Range<int>& comps, // output value: sets comps
265 int first,
266 int last,
267 MemoryBlock& monomialSpace,
268 LockType& lock) const
269 {
270 auto& dvec = * elementArray(dense);
271
272 int len = 0;
273
274 // first can be -1 if the row is zero. in this case, we should
275 // not be accessing dense[i] for i negative.
276 for (int i = first; i >= 0 and i <= last; i++)
277 if (not mRing->is_zero(dvec[i])) len++;
278
279 comps = monomialSpace.safeAllocateArray<int,LockType>(len,lock);
280 deallocateElementArray(sparse);
281 sparse = allocateElementArray(len);
282 auto& svec = * elementArray(sparse);
283
284 int next = 0;
285 for (int i = first; i >= 0 and i <= last; i++)
286 if (not mRing->is_zero(dvec[i]))
287 {
288 mRing->set(svec[next],dvec[i]);
289 comps[next] = i;
290 ++next;
291 mRing->set_zero(dvec[i]);
292 }
