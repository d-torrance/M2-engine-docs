28void GB_comp::initialize0(const Matrix *m,
29 int csyz,
30 int nsyz,
31 M2_arrayint gb_weights)
32{
33 int i;
34 const PolynomialRing *R = m->get_ring()->cast_to_PolynomialRing();
35 if (R == nullptr)
36 {
37 ERROR("ring is not a polynomial ring");
38 // MES: throw an error here.
39 assert(0);
40 }
41 originalR = R;
42 _GR = R->get_gb_ring();
43 weightInfo_ = new GBWeight(m->rows(), gb_weights);
44 _M = _GR->get_flattened_monoid();
45 _K = _GR->get_flattened_coefficients();
46
47 _spairs = new s_pair_heap(_M);
48 _gens = new s_pair_heap(_M);
49
50 if (nsyz < 0 || nsyz > m->n_cols()) nsyz = m->n_cols();
51 _n_rows_per_syz = nsyz;
52
53 _F = m->rows();
54
55 _ar_i = _ar_j = _np_i = -1;
56
57 _n_gb = _n_subring = 0;
58 _n_pairs_computed = _n_saved_gcd = 0;
59 _n_gens_left = 0;
60 _n_reductions = 0;
61
62 _collect_syz = csyz;
63 _is_ideal = (_F->rank() == 1 && csyz == 0);
64 if (_GR->is_weyl_algebra()) _is_ideal = false;
65
66 _use_hilb = false;
67 _hilb_new_elems = false;
68 _hilb_n_in_degree = 0;
69
70 _n_saved_hilb = 0;
71
72 // set local variables for certain time-critical routines
73
74 _this_degree = _F->lowest_primary_degree() - 1;
75
76 for (i = 0; i <= _F->rank(); i++)
77 {
78 // The 0th one is not used.
79 monideal_pair *p = new monideal_pair(originalR);
80 _monideals.push_back(p);
81 }
82}

84void GB_comp::initialize(const Matrix *m,
85 int csyz,
86 int nsyz,
87 M2_arrayint gb_weights,
88 int strat)
89{
90 int i;
91 _strategy = strat;
92
93 initialize0(m, csyz, nsyz, gb_weights);
94
95 _Fsyz = m->cols()->sub_space(_n_rows_per_syz);
96
97 _state = GB_COMP_NEWDEGREE;
98
99 for (i = 0; i < m->n_cols(); i++)
100 {
101 ring_elem denom;
102 gbvector *f = originalR->translate_gbvector_from_vec(_F, (*m)[i], denom);
103 s_pair *p = new_gen(i, f, denom);
104 if (p != nullptr)
105 {
106 _gens->insert(p);
107 _n_gens_left++;
108 }
109 }
110}

184s_pair *GB_comp::new_gen(int i, gbvector *f, ring_elem denom)
185{
186 gbvector *fsyz;
187
188 if (i < _n_rows_per_syz)
189 fsyz = _GR->gbvector_term(_Fsyz, denom, i + 1);
190 else
191 fsyz = _GR->gbvector_zero();
192
193 if (_GR->gbvector_is_zero(f))
194 {
195 if (!_GR->gbvector_is_zero(fsyz))
196 {
197 _syz.push_back(fsyz);
198 _n_syz++;
199 }
200 return nullptr;
201 }
202
203 s_pair *result = new s_pair;
204 result->next = nullptr;
205 result->syz_type = SPAIR_GEN;
206 result->degree = weightInfo_->gbvector_weight(f);
207 result->compare_num = 0;
208 result->first = nullptr;
209 result->second = nullptr;
210 result->f = f; /* NOTE THAT WE GRAB f */
211 result->fsyz = fsyz;
212
213 result->lcm = _M->make_new(result->f->monom);
214
215 return result;
216}

222int GB_comp::gb_sort_partition(int lo, int hi)
223{
224 gb_elem *pivot = _gb[lo];
225 const gbvector *pivot_elem = pivot->f;
226 int i = lo - 1;
227 int j = hi + 1;
228 for (;;)
229 {
230 do
231 {
232 j--;
233 }
234 while (_GR->gbvector_compare(_F, _gb[j]->f, pivot_elem) > 0);
235
236 do
237 {
238 i++;
239 }
240 while (_GR->gbvector_compare(_F, _gb[i]->f, pivot_elem) < 0);
241
242 if (i < j)
243 {
244 gb_elem *tmp = _gb[j];
245 _gb[j] = _gb[i];
246 _gb[i] = tmp;
247 }
248 else
249 return j;
250 }
251}

253void GB_comp::gb_sort(int lo, int hi)
254{
255 if (lo < hi)
256 {
257 int q = gb_sort_partition(lo, hi);
258 gb_sort(lo, q);
259 gb_sort(q + 1, hi);
260 }

263void GB_comp::find_pairs(gb_elem *p)
264// compute min gen set of {m | m lead(p) is in (p1, ..., pr, f1, ..., fs)}
265// (includes cases m * lead(p) = 0).
266// Returns a list of new s_pair's.
267{
268 gc_vector<Bag*> elems;
269 gc_vector<int> vplcm;
270 monomial find_pairs_m = _M->make_one();
271 monomial f_m = _M->make_one();
272 exponents_t find_pairs_exp = newarray_atomic(int, _M->n_vars());
273 int *find_pairs_lcm = newarray_atomic(int, _M->n_vars());
274
275 _GR->gbvector_get_lead_monomial(_F, p->f, f_m);
276 if (_GR->is_skew_commutative())
277 {
278 _M->to_expvector(f_m, find_pairs_exp);
279
280 for (int v = 0; v < _GR->n_skew_commutative_vars(); v++)
281 {
282 int w = _GR->skew_variable(v);
283 if (find_pairs_exp[w] == 0) continue;
284
285 find_pairs_exp[w]++;
286 _M->from_expvector(find_pairs_exp, find_pairs_lcm);
287 find_pairs_exp[w]--;
288
289 vplcm.resize(0);
290 _M->to_varpower(find_pairs_lcm, vplcm);
291 s_pair *q = new_var_pair(p, find_pairs_lcm);
292 elems.push_back(new Bag(q, vplcm));
293 }
294 }
295
296 // Add in syzygies arising from a base ring
297
298 if (originalR->is_quotient_ring())
299 {
300 for (int i = 0; i < originalR->n_quotients(); i++)
301 {
302 const gbvector *f = originalR->quotient_gbvector(i);
303 _M->lcm(f->monom, f_m, find_pairs_lcm);
304 vplcm.resize(0);
305 _M->to_varpower(find_pairs_lcm, vplcm);
306 s_pair *q = new_ring_pair(p, find_pairs_lcm);
307 elems.push_back(new Bag(q, vplcm));
308 }
309 }
310 // Add in syzygies arising as s-pairs
311 MonomialIdeal *mi1 = _monideals[p->f->comp]->mi;
312 for (Bag& a : *mi1)
313 {
314 _M->from_varpower(a.monom().data(), find_pairs_m);
315 _M->lcm(find_pairs_m, f_m, find_pairs_lcm);
316 vplcm.resize(0);
317 _M->to_varpower(find_pairs_lcm, vplcm);
318 s_pair *q =
319 new_s_pair(p,
320 reinterpret_cast<gb_elem *>(a.basis_ptr()),
321 find_pairs_lcm);
322 elems.push_back(new Bag(q, vplcm));
323 }
324
325 // Add 'p' to the correct monideal
326 gc_vector<int> vp;
327 _M->to_varpower(f_m, vp);
328 mi1->insert(new Bag(p, vp));
329
330 // Now minimalize these elements, and insert them into
331 // the proper degree.
332
333 VECTOR(Bag *) rejects;
334 MonomialIdeal mi(originalR, elems, rejects);
335 for (auto& b : rejects)
336 {
337 s_pair *q = reinterpret_cast<s_pair *>(b->basis_ptr());
338 remove_pair(q);
339 delete b;
340 }
341 for (Bag& a : mi)
342 {
343 s_pair *q = reinterpret_cast<s_pair *>(a.basis_ptr());
344 if (_is_ideal && q->syz_type == SPAIR_PAIR)
345 {
346 _M->gcd(q->first->f->monom, q->second->f->monom, find_pairs_m);
347 if (_M->is_one(find_pairs_m))
348 {
349 _n_saved_gcd++;
350 if (M2_gbTrace >= 8)
351 {
352 buffer o;
353 o << "removed pair[" << q->first->me << " " << q->second->me
354 << "]" << newline;
355 emit(o.str());
356 }
357 remove_pair(q);
358 }
359 else
360 _spairs->insert(q);
361 }
362 else
363 _spairs->insert(q);
364 }
365
366 // Remove the local variables
367 _M->remove(find_pairs_m);
368 _M->remove(f_m);
369 freemem(find_pairs_exp);
370 freemem(find_pairs_lcm);
371}

373void GB_comp::compute_s_pair(s_pair *p)
374{
375 if (p->f == nullptr)
376 {
377 monomial s = _M->make_one();
378 _M->divide(p->lcm, p->first->f->monom, s);
379
380 _GR->gbvector_mult_by_term(
381 _F, _Fsyz, _GR->one(), s, p->first->f, p->first->fsyz, p->f, p->fsyz);
382 if (p->syz_type == SPAIR_PAIR)
383 _GR->gbvector_reduce_lead_term(
384 _F, _Fsyz, nullptr, p->f, p->fsyz, p->second->f, p->second->fsyz);
385 _M->remove(s);
386 }
387}

389void GB_comp::gb_reduce(gbvector *&f, gbvector *&fsyz)
390{
391 if ((_strategy & STRATEGY_LONGPOLYNOMIALS) != 0)
392 {
393 gb_geo_reduce(f, fsyz);
394 return;
395 }
396 gbvector head;
397 gbvector *result = &head;
398 result->next = nullptr;
399
400 exponents_t div_totalexp = newarray_atomic(int, _M->n_vars());
401 int count = 0;
402 if (M2_gbTrace == 10)
403 {
404 buffer o;
405 o << "reducing ";
406 _GR->gbvector_text_out(o, _F, f);
407 emit_line(o.str());
408 }
409 while (f != nullptr)
410 {
411 Bag *b;
412 _GR->gbvector_get_lead_exponents(_F, f, div_totalexp);
413 if (originalR->is_quotient_ring() &&
414 originalR->get_quotient_monomials()->search_expvector(div_totalexp,
415 b))
416 {
417 const gbvector *g = originalR->quotient_gbvector(b->basis_elem());
418 _GR->gbvector_reduce_lead_term(_F, _Fsyz, head.next, f, fsyz, g, nullptr);
419 count++;
420 _n_reductions++;
421 }
422 else if (_monideals[f->comp]->mi_search->search_expvector(div_totalexp,
423 b))
424 {
425 gb_elem *q = reinterpret_cast<gb_elem *>(b->basis_ptr());
426 _GR->gbvector_reduce_lead_term(
427 _F, _Fsyz, head.next, f, fsyz, q->f, q->fsyz);
428 count++;
429 _n_reductions++;
430 if (M2_gbTrace == 10)
431 {
432 buffer o;
433 o << " reduced by ";
434 _GR->gbvector_text_out(o, _F, q->f);
435 o << newline;
436 o << " giving ";
437 _GR->gbvector_text_out(o, _F, f);
438 o << newline;
439 emit(o.str());
440 }
441 }
442 else
443 {
444 result->next = f;
445 f = f->next;
446 result = result->next;
447 result->next = nullptr;
448 }
449 }
450
451 if (M2_gbTrace >= 4)
452 {
453 buffer o;
454 o << "." << count;
455 emit_wrapped(o.str());
456 }
457 f = head.next;
458 freemem(div_totalexp);
459}

461void GB_comp::gb_geo_reduce(gbvector *&f, gbvector *&fsyz)
462{
463 gbvector head;
464 gbvector *result = &head;
465 result->next = nullptr;
466
467 exponents_t div_totalexp = newarray_atomic(int, _M->n_vars());
468 int count = 0;
469
470 gbvectorHeap fb(_GR, _F);
471 gbvectorHeap fsyzb(_GR, _Fsyz);
472 fb.add(f);
473 fsyzb.add(fsyz);
474 const gbvector *lead;
475 while ((lead = fb.get_lead_term()) != nullptr)
476 {
477 Bag *b;
478 _GR->gbvector_get_lead_exponents(_F, lead, div_totalexp);
479 if (originalR->is_quotient_ring() &&
480 originalR->get_quotient_monomials()->search_expvector(div_totalexp,
481 b))
482 {
483 const gbvector *g = originalR->quotient_gbvector(b->basis_elem());
484 _GR->reduce_lead_term_heap(_F,
485 _Fsyz,
486 lead,
487 div_totalexp, // are these two needed
488 head.next,
489 fb,
490 fsyzb,
491 g,
492 nullptr);
493 count++;
494 }
495 else if (_monideals[lead->comp]->mi_search->search_expvector(div_totalexp,
496 b))
497 {
498 gb_elem *q = reinterpret_cast<gb_elem *>(b->basis_ptr());
499 _GR->reduce_lead_term_heap(_F,
500 _Fsyz,
501 lead,
502 div_totalexp,
503 head.next,
504 fb,
505 fsyzb,
506 q->f,
507 q->fsyz);
508 count++;
509 }
510 else
511 {
512 result->next = fb.remove_lead_term();
513 result = result->next;
514 result->next = nullptr;
515 }
516 }
517
518 if (M2_gbTrace >= 4)
519 {
520 buffer o;
521 o << "." << count;
522 emit_wrapped(o.str());
523 }
524 f = head.next;
525
526 fsyz = fsyzb.value();
527 freemem(div_totalexp);
528}

530void GB_comp::flush_pairs(int deg)
531{
532 s_pair *p;
533 while ((p = _spairs->remove()) != nullptr)
534 if (p->degree != deg)
535 {
536 _spairs->put_back(p);
537 break;
538 }
539 else
540 {
541 _n_saved_hilb++;
542 remove_pair(p);
543 }
544 while ((p = _gens->remove()) != nullptr)
545 if (p->degree != deg)
546 {
547 _gens->put_back(p);
548 break;
549 }
550 else
551 {
552 _n_saved_hilb++;
553 remove_pair(p);
554 }
555}

557void GB_comp::gb_insert(gbvector *f, gbvector *fsyz, int ismin)
558{
559 monomial f_m = _M->make_one();
560 gb_elem *p = new gb_elem(f, fsyz, ismin);
561
562 _GR->gbvector_get_lead_monomial(_F, p->f, f_m);
563 _GR->gbvector_remove_content(p->f, p->fsyz);
564
565 if (ismin)
566 {
567 p->is_min = 1;
568 }
569
570 if (_M->in_subring(1, f_m)) _n_subring++;
571 // insert into p->f->comp->mi_search
572 gc_vector<int> vp;
573 _M->to_varpower(f_m, vp);
574 _monideals[p->f->comp]->mi_search->insert(new Bag(p, vp));
575 _n_gb++;
576 _gb.push_back(p);
577 _M->remove(f_m);
578
579 // Now we must be a bit careful about this next, but we only want one
580 // copy of a GB element, since the whole thing can be quite large.
581 // Just make sure that when the GB is deleted at the end, that the 'f'
582 // field of the gb_elem's is not removed.
583
584 if (_use_hilb)
585 {
586 _hilb_new_elems = true;
587 if (--_hilb_n_in_degree == 0) flush_pairs(_this_degree);
588 }
589}

591int GB_comp::s_pair_step()
592// If no s-pairs left in the current degree,
593// return SPAIR_DONE.
594// Otherwise, compute the current s-pair, reduce it, and
595// dispatch the result. Return one of the other SPAIR_*
596// values.
597{
598 s_pair *p = _spairs->remove();
599 if (p == nullptr) return SPAIR_DONE;
600 if (p->degree != _this_degree)
601 {
602 _spairs->put_back(p);
603 return SPAIR_DONE;
604 }
605
606 if (M2_gbTrace == 100)
607 {
608 // Traces the computation, in its way
609 emit("Computing spair ");
610 debug_out(p);
611 }
612 _n_pairs_computed++;
613 compute_s_pair(p);
614
615 gbvector *f = p->f;
616 gbvector *fsyz = p->fsyz;
617 p->f = nullptr;
618 p->fsyz = nullptr;
619 remove_pair(p);
620
621 gb_reduce(f, fsyz);
622 if (!_GR->gbvector_is_zero(f))
623 {
624 if (M2_gbTrace == 100)
625 {
626 buffer o;
627 o << " inserting GB element " << _n_gb;
628 _GR->gbvector_text_out(o, _F, f);
629 o << newline;
630 emit(o.str());
631 }
632 gb_insert(f, fsyz, 0);
633 return SPAIR_GB;
634 }
635 if (!_GR->gbvector_is_zero(fsyz))
636 {
637 if (_collect_syz)
638 {
639 // vec fsyzvec = _GR->gbvector_to_vec(_Fsyz,fsyz);
640 _syz.push_back(fsyz);
641 _n_syz++;
642 return SPAIR_SYZ;
643 }
644 else
645 _GR->gbvector_remove(fsyz);
646 }
647
648 return SPAIR_ZERO;
649}
650
651int GB_comp::gen_step()
652// If no gens left in the current degree,
653// return SPAIR_DONE.
654// Otherwise, compute the current s-pair, reduce it, and
655// dispatch the result. Return one of the other SPAIR_*
656// values.
657{
658 s_pair *p = _gens->remove();
659 if (p == nullptr) return SPAIR_DONE;
660 if (p->degree != _this_degree)
661 {
662 _gens->put_back(p);
663 return SPAIR_DONE;
664 }
665
666 _n_pairs_computed++;
667 _n_gens_left--;
668
669 compute_s_pair(p);
670
671 gbvector *f = p->f;
672 gbvector *fsyz = p->fsyz;
673 p->f = nullptr;
674 p->fsyz = nullptr;
675 remove_pair(p);
676
677 gb_reduce(f, fsyz);
678 if (!_GR->gbvector_is_zero(f))
679 {
680 gb_insert(f, fsyz, 1); // 1 = minimal generator
681 return SPAIR_MINGEN;
682 }
683
684 if (!_GR->gbvector_is_zero(fsyz))
685 {
686 if (_collect_syz)
687 {
688 // vec fsyzvec = _GR->gbvector_to_vec(_Fsyz,fsyz);
689 _syz.push_back(fsyz);
690 _n_syz++;
691 return SPAIR_SYZ;
692 }
693 else
694 _GR->gbvector_remove(fsyz);
695 }
696
697 return SPAIR_ZERO;
698}

700bool GB_comp::auto_reduce_step()
701// Using _ar_i, _ar_j, reduce the gb element _ar_i wrt _ar_j.
702// Increment _ar_i, _ar_j as needed. If done, return false.
703{
704 if (_ar_j >= _n_gb)
705 {
706 _ar_i++;
707 _ar_j = _ar_i + 1;
708 if (_ar_j >= _n_gb) return false;
709 }
710 // Now compute gb(i) := gb(i) - c gb(j), where
711 // c in(gb(j)) is a term in gb(i).
712 // Also compute change(i) -= c change(j).
713
714 _GR->gbvector_auto_reduce(_F,
715 _Fsyz,
716 _gb[_ar_i]->f,
717 _gb[_ar_i]->fsyz,
718 _gb[_ar_j]->f,
719 _gb[_ar_j]->fsyz);
720 _ar_j++;
721 return true;
722}

724bool GB_comp::new_pairs_step()
725// Compute the new s-pairs associated to the given gb element.
726// Increment '_np_i'. If done with all pairs in this
727// degree, return false.
728{
729 if (_np_i >= _n_gb) return false;
730 find_pairs(_gb[_np_i]);
731 _np_i++;
732 return true;
733}

737ComputationStatusCode GB_comp::computation_is_complete() const
738{
739 // This handles everything but _Stop.always, _Stop.degree_limit
740 if (_state == GB_COMP_DONE) return COMP_DONE;
741 if (stop_.basis_element_limit > 0 && _n_gb > stop_.basis_element_limit)
742 return COMP_DONE_GB_LIMIT;
743 if (stop_.syzygy_limit > 0 && _n_syz > stop_.syzygy_limit)
744 return COMP_DONE_SYZ_LIMIT;
745 if (stop_.pair_limit > 0 && _n_pairs_computed > stop_.pair_limit)
746 return COMP_DONE_PAIR_LIMIT;
747 if (stop_.just_min_gens && _n_gens_left == 0) return COMP_DONE_MIN_GENS;

792//---- state machine (roughly) for the computation ----
793
794void GB_comp::start_computation()
795{
796 ComputationStatusCode is_done = COMP_COMPUTING;
797
798 for (;;)
799 {
800 if (is_done != COMP_COMPUTING) break;
801 is_done = computation_is_complete();
802 if (is_done != COMP_COMPUTING) break;
803 if (system_interrupted())
804 {
805 is_done = COMP_INTERRUPTED;
806 break;
807 }
808
809 switch (_state)
810 {
811 case GB_COMP_NEWDEGREE:
812 if (_spairs->n_elems() == 0 && _gens->n_elems() == 0)
813 {
814 _state = GB_COMP_DONE;
815 is_done = COMP_DONE;
816 break;
817 }
818 _this_degree = next_degree();
819 if (stop_.stop_after_degree &&
820 _this_degree > stop_.degree_limit->array[0])
821 {
822 is_done = COMP_DONE_DEGREE_LIMIT;
823 break;
824 }
825
826 if (_use_hilb)
827 {
828 if (_hilb_new_elems)
829 {
830 // Recompute h, _hf_diff
831 RingElement *h = compute_hilbert_function();
832 if (h == nullptr)
833 {
834 is_done = COMP_INTERRUPTED;
835 break;
836 }
837 _hf_diff = (*h) - (*_hf_orig);
838 _hilb_new_elems = false;
839 }
840 _hilb_n_in_degree = hilb_comp::coeff_of(_hf_diff, _this_degree);
841 if (error())
842 {
843 is_done = COMP_ERROR;
844 break;
845 }
846 if (_hilb_n_in_degree == 0) flush_pairs(_this_degree);
847 }
848 if (M2_gbTrace >= 1)
849 {
850 buffer o;
851 o << '{' << _this_degree << '}';
852 o << '(';
853 if (_use_hilb) o << _hilb_n_in_degree << ',';
854 o << _spairs->n_elems() << ')';
855 emit_wrapped(o.str());
856 }
857
858 // Set state information for auto reduction, new pairs
859 _ar_i = _n_gb;
860 _ar_j = _ar_i + 1;
861 _np_i = _n_gb;
862 _state = GB_COMP_S_PAIRS;
863 break;
864
865 case GB_COMP_S_PAIRS:
866 if (M2_gbTrace < 2)
867 {
868 if (s_pair_step() == SPAIR_DONE) _state = GB_COMP_GENS;
869 }
870 else
871 switch (s_pair_step())
872 {
873 case SPAIR_MINGEN:
874 emit_wrapped("g");
875 break;
876 case SPAIR_GB:
877 emit_wrapped("m");
878 break;
879 case SPAIR_SYZ:
880 emit_wrapped("z");
881 break;
882 case SPAIR_ZERO:
883 emit_wrapped("o");
884 break;
885 case SPAIR_DONE:
886 _state = GB_COMP_GENS;
887 break;
888 default:
889 emit_wrapped("ERROR");
890 break;
891 }
892 break;
893
894 case GB_COMP_GENS:
895 if (M2_gbTrace < 2)
896 {
897 if (gen_step() == SPAIR_DONE) _state = GB_COMP_AUTO_REDUCE;
898 }
899 else
900 switch (gen_step())
901 {
902 case SPAIR_MINGEN:
903 emit_wrapped("g");
904 break;
905 case SPAIR_SYZ:
906 emit_wrapped("z");
907 break;
908 case SPAIR_ZERO:
909 emit_wrapped("o");
910 break;
911 case SPAIR_DONE:
912 _state = GB_COMP_AUTO_REDUCE;
913 break;
914 default:
915 emit_wrapped("ERROR");
916 break;
917 }
918 break;
919
920 case GB_COMP_AUTO_REDUCE:
921 if (!auto_reduce_step())
922 {
923 _state = GB_COMP_NEWPAIRS;
924 if ((_strategy & STRATEGY_SORT) != 0)
925 {
926 gb_sort(
927 _np_i,
928 _n_gb - 1); // This is the range of elements to sort.
929 }
930 for (int j = _np_i; j < _n_gb; j++) _gb[j]->me = j;
931 }
932 break;
933
934 case GB_COMP_NEWPAIRS:
935 if (!new_pairs_step()) _state = GB_COMP_NEWDEGREE;
936 break;
937
938 case GB_COMP_DONE:
939 break;
940
941 case GB_COMP_NEED_RESIZE:
942 is_done = COMP_NEED_RESIZE;
943 break;
944 }
945 }
946 if (M2_gbTrace >= 1) emit_line("");
947 if (M2_gbTrace >= 4)
948 {
949 buffer o;
950 o << "Number of gb elements = " << _n_gb << newline;
951 o << "Number of gcd=1 pairs = " << _n_saved_gcd << newline;
952 o << "Number of pairs computed = " << _n_pairs_computed << newline;
953 o << "Number of reductions = " << _n_reductions << newline;
954 emit(o.str());
955 }
956
957 // _GR->memstats();
958 set_status(is_done);
959}

1097M2_bool GB_comp::matrix_lift(const Matrix *m,
1098 const Matrix /* or null */ **result_remainder,
1099 const Matrix /* or null */ **result_quotient)
1100{
1101 if (m->get_ring() != originalR)
1102 {
1103 ERROR("expected matrix over the same ring");
1104 *result_remainder = nullptr;
1105 *result_quotient = nullptr;
1106 return false;
1107 }
1108 if (m->n_rows() != _F->rank())
1109 {
1110 ERROR("expected matrices to have same number of rows");
1111 *result_remainder = nullptr;
1112 *result_quotient = nullptr;
1113 return false;
1114 }
1115 start_computation();
1116 MatrixConstructor mat_remainder(m->rows(), m->cols(), m->degree_shift());
1117 MatrixConstructor mat_quotient(_Fsyz, 0);
1118 bool all_zeroes = true;
1119 for (int i = 0; i < m->n_cols(); i++)
1120 {
1121 ring_elem denom;
1122 gbvector *f = originalR->translate_gbvector_from_vec(_F, (*m)[i], denom);
1123 gbvector *fsyz = _GR->gbvector_zero();
1124
1125 gb_reduce(f, fsyz);
1126 if (f != nullptr) all_zeroes = false;
1127
1128 vec fv = originalR->translate_gbvector_to_vec_denom(_F, f, denom);
1129 _K->negate_to(denom);
1130 vec fsyzv =
1131 originalR->translate_gbvector_to_vec_denom(_Fsyz, fsyz, denom);
1132 mat_remainder.set_column(i, fv);
1133 mat_quotient.append(fsyzv);
1134 }
1135 *result_remainder = mat_remainder.to_matrix();
1136 *result_quotient = mat_quotient.to_matrix();
1137 return all_zeroes;
1138}

1165 }
